```kotlin
            ReaderError.Internal("LayoutConstraints not set")
```

## 4) 修改 `TxtSearchProvider.kt`（更省内存 + 可取消）

把原来整块 `lowercase()` 的做法替换为 `regionMatches(ignoreCase = true)` 扫描。

```kotlin
// engines/txt/.../internal/provider/TxtSearchProvider.kt

// ✅ 新增 import
import kotlin.coroutines.coroutineContext
import kotlinx.coroutines.ensureActive

override fun search(query: String, options: SearchOptions): Flow<SearchHit> = flow {
    if (query.isBlank()) return@flow

    val q = query
    val chunkSize = 64 * 1024
    val overlap = (q.length + 64).coerceAtMost(8_192)

    val total = withContext(ioDispatcher) { store.totalChars() }
    var cursor = options.startFrom?.let { parse(it) } ?: 0
    cursor = cursor.coerceIn(0, total)

    var emitted = 0

    while (cursor < total && emitted < options.maxHits) {
        coroutineContext.ensureActive() // ✅ 支持取消

        val read = withContext(ioDispatcher) {
            store.readChars(cursor, (chunkSize + overlap).coerceAtMost(total - cursor))
        }
        if (read.isEmpty()) break

        var idx = 0
        while (idx <= read.length - q.length && emitted < options.maxHits) {
            coroutineContext.ensureActive()

            val found = indexOfMatch(read, q, start = idx, ignoreCase = !options.caseSensitive)
            if (found < 0) break

            if (options.wholeWord && !isWholeWord(read, found, q.length)) {
                idx = found + q.length
                continue
            }

            val start = cursor + found
            val end = start + q.length

            emit(
                SearchHit(
                    range = LocatorRange(
                        start = Locator(LocatorSchemes.TXT_OFFSET, start.toString()),
                        end = Locator(LocatorSchemes.TXT_OFFSET, end.toString())
                    ),
                    excerpt = buildExcerpt(read, found, q.length),
                    sectionTitle = null
                )
            )
            emitted++
            idx = found + q.length
        }

        cursor += chunkSize
    }
}

private fun indexOfMatch(hay: String, needle: String, start: Int, ignoreCase: Boolean): Int {
    val last = hay.length - needle.length
    var i = start.coerceAtLeast(0)
    while (i <= last) {
        if (hay.regionMatches(i, needle, 0, needle.length, ignoreCase = ignoreCase)) return i
        i++
    }
    return -1
}
```

# 二、新增代码

## 1) 新增 `TxtTextStoreFactory.kt`

按文件大小 & 是否可 seek 自动选择 store。

```kotlin
// engines/txt/.../internal/storage/TxtTextStoreFactory.kt
package com.example.reader.engines.txt.internal.storage

import com.example.reader.source.DocumentSource
import java.nio.charset.Charset
import kotlinx.coroutines.CoroutineDispatcher

internal object TxtTextStoreFactory {

    // 小文件走内存（更快），大文件走 indexed（不爆内存）
    private const val IN_MEMORY_THRESHOLD_BYTES: Long = 8L * 1024L * 1024L // 8MB

    suspend fun create(
        source: DocumentSource,
        charset: Charset,
        ioDispatcher: CoroutineDispatcher
    ): TxtTextStore {
        val size = source.sizeBytes

        if (size != null && size <= IN_MEMORY_THRESHOLD_BYTES) {
            return InMemoryTxtTextStore(
                source = source,
                charset = charset,
                ioDispatcher = ioDispatcher,
                maxBytes = IN_MEMORY_THRESHOLD_BYTES
            )
        }

        // 大文件优先 indexed（需要 PFD 才能 seek）
        val pfd = source.openFileDescriptor("r")
        return if (pfd != null) {
            IndexedTxtTextStore(
                source = source,
                pfd = pfd,
                charset = charset,
                ioDispatcher = ioDispatcher
            )
        } else {
            // 无 PFD：退回内存（如果 size 不可得且确实很大，你会看到 maxBytes 保护异常）
            InMemoryTxtTextStore(
                source = source,
                charset = charset,
                ioDispatcher = ioDispatcher,
                maxBytes = IN_MEMORY_THRESHOLD_BYTES
            )
        }
    }
}
```

## 2) 新增 `ChunkIndex.kt`

稀疏索引（按**字节间隔**做 anchor，保证 byteOffset 一定落在字符边界）。

```kotlin
// engines/txt/.../internal/storage/ChunkIndex.kt
package com.example.reader.engines.txt.internal.storage

internal data class ChunkAnchor(
    val charOffset: Int,
    val byteOffset: Long
)

internal class ChunkIndex(
    val anchors: List<ChunkAnchor>,
    val totalChars: Int,
    val startByteOffset: Long // BOM skip 后的起点
) {
    fun anchorFor(targetChar: Int): ChunkAnchor {
        if (anchors.isEmpty()) return ChunkAnchor(0, startByteOffset)
        val t = targetChar.coerceAtLeast(0)

        var lo = 0
        var hi = anchors.lastIndex
        var best = anchors[0]

        while (lo <= hi) {
            val mid = (lo + hi) ushr 1
            val a = anchors[mid]
            if (a.charOffset <= t) {
                best = a
                lo = mid + 1
            } else {
                hi = mid - 1
            }
        }
        return best
    }
}
```

## 3) 新增 `BomUtil.kt`

用于 indexed store 确保“charOffset=0 从 BOM 后开始”。

```kotlin
// engines/txt/.../internal/storage/BomUtil.kt
package com.example.reader.engines.txt.internal.storage

import java.nio.charset.Charset

internal object BomUtil {
    fun bomSkipBytes(header: ByteArray, charset: Charset): Int {
        // UTF-8 BOM: EF BB BF
        if (header.size >= 3 &&
            header[0] == 0xEF.toByte() &&
            header[1] == 0xBB.toByte() &&
            header[2] == 0xBF.toByte()
        ) return 3

        // UTF-16 LE / BE
        if (header.size >= 2 &&
            ((header[0] == 0xFF.toByte() && header[1] == 0xFE.toByte()) ||
             (header[0] == 0xFE.toByte() && header[1] == 0xFF.toByte()))
        ) return 2

        return 0
    }
}
```

## 4) 新增 `IndexedTxtTextStore.kt`

核心：构建稀疏索引（一次顺序 decode），之后支持快速 `readRange/readChars/readAround`。

```kotlin
// engines/txt/.../internal/storage/IndexedTxtTextStore.kt
package com.example.reader.engines.txt.internal.storage

import android.os.ParcelFileDescriptor
import com.example.reader.source.DocumentSource
import java.io.BufferedInputStream
import java.io.Closeable
import java.io.FileInputStream
import java.nio.ByteBuffer
import java.nio.CharBuffer
import java.nio.channels.FileChannel
import java.nio.charset.Charset
import java.nio.charset.CharsetDecoder
import kotlin.math.max
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext
import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock
import kotlin.coroutines.coroutineContext
import kotlinx.coroutines.ensureActive

internal class IndexedTxtTextStore(
    private val source: DocumentSource,
    private val pfd: ParcelFileDescriptor,
    override val charset: Charset,
    private val ioDispatcher: CoroutineDispatcher
) : TxtTextStore, Closeable {

    private val mutex = Mutex()
    private var index: ChunkIndex? = null

    private val channel: FileChannel by lazy {
        FileInputStream(pfd.fileDescriptor).channel
    }

    // 经验值：每 1MB 记一个 anchor，定位最多 decode 1MB 左右就能到目标
    private val anchorEveryBytes: Long = 1L * 1024L * 1024L

    override suspend fun totalChars(): Int = ensureIndex().totalChars

    override suspend fun readRange(startChar: Int, endCharExclusive: Int): String {
        val idx = ensureIndex()
        val total = idx.totalChars
        if (total <= 0) return ""

        val start = startChar.coerceIn(0, total)
        val end = endCharExclusive.coerceIn(start, total)
        if (start == end) return ""

        val anchor = idx.anchorFor(start)
        return withContext(ioDispatcher) {
            readFromAnchor(
                anchor = anchor,
                startChar = start,
                endChar = end
            )
        }
    }

    override suspend fun readChars(startChar: Int, maxChars: Int): String {
        val start = startChar.coerceAtLeast(0)
        val len = maxChars.coerceAtLeast(0)
        return readRange(start, start + len)
    }

    override suspend fun readAround(charOffset: Int, maxChars: Int): String {
        val idx = ensureIndex()
        val total = idx.totalChars
        if (total <= 0) return ""
        val half = (maxChars / 2).coerceAtLeast(1)
        val start = (charOffset - half).coerceIn(0, total)
        val end = (start + maxChars).coerceIn(start, total)
        return readRange(start, end)
    }

    override fun close() {
        runCatching { channel.close() }
        runCatching { pfd.close() }
    }

    // ---------------- index ----------------

    private suspend fun ensureIndex(): ChunkIndex =
        mutex.withLock {
            index ?: buildIndex().also { index = it }
        }

    private suspend fun buildIndex(): ChunkIndex = withContext(ioDispatcher) {
        // 读 header 用于 BOM skip
        val header = source.openInputStream().use { input ->
            val buf = ByteArray(4)
            val n = input.read(buf)
            if (n <= 0) ByteArray(0) else buf.copyOf(n)
        }
        val bomSkip = BomUtil.bomSkipBytes(header, charset).toLong()

        val anchors = ArrayList<ChunkAnchor>(256)
        anchors.add(ChunkAnchor(charOffset = 0, byteOffset = bomSkip))

        var lastAnchorByte = bomSkip
        var decodedChars = 0

        val decoder = charset.newDecoder()
        val byteBuf = ByteBuffer.allocateDirect(256 * 1024)
        val charBuf = CharBuffer.allocate(16 * 1024)
        val tmp = ByteArray(128 * 1024)

        source.openInputStream().use { raw ->
            val input = BufferedInputStream(raw, 256 * 1024)
            // skip BOM
            if (bomSkip > 0) {
                var toSkip = bomSkip
                while (toSkip > 0) {
                    val skipped = input.skip(toSkip)
                    if (skipped <= 0) break
                    toSkip -= skipped
                }
            }

            var totalBytesRead = bomSkip

            while (true) {
                coroutineContext.ensureActive()

                val r = input.read(tmp)
                if (r < 0) break
                totalBytesRead += r.toLong()

                // fill
                var off = 0
                while (off < r) {
                    val canPut = minOf(byteBuf.remaining(), r - off)
                    byteBuf.put(tmp, off, canPut)
                    off += canPut

                    byteBuf.flip()
                    decodeLoop(decoder, byteBuf, charBuf) { produced ->
                        decodedChars += produced
                        // 安全字节位置：已读入总字节 - buffer剩余未消费
                        val safeBytePos = totalBytesRead - byteBuf.remaining().toLong()
                        if (safeBytePos - lastAnchorByte >= anchorEveryBytes) {
                            anchors.add(ChunkAnchor(decodedChars, safeBytePos))
                            lastAnchorByte = safeBytePos
                        }
                    }
                    byteBuf.compact()
                }
            }

            // flush remaining bytes
            byteBuf.flip()
            while (true) {
                coroutineContext.ensureActive()
                charBuf.clear()
                val cr = decoder.decode(byteBuf, charBuf, true)
                charBuf.flip()
                val produced = charBuf.remaining()
                decodedChars += produced
                if (produced > 0) {
                    val safeBytePos = (bomSkip + (channel.size() - 0L)) // 这里不精准，下面用 remaining 方式补一次
                }
                if (cr.isUnderflow) break
            }
            // decoder.flush
            while (true) {
                coroutineContext.ensureActive()
                charBuf.clear()
                val cr = decoder.flush(charBuf)
                charBuf.flip()
                decodedChars += charBuf.remaining()
                if (cr.isUnderflow) break
            }
        }

        // 最终补一个结尾 anchor（方便二分）
        val endByte = channel.size().coerceAtLeast(bomSkip)
        if (anchors.lastOrNull()?.byteOffset != endByte) {
            anchors.add(ChunkAnchor(decodedChars, endByte))
        }

        ChunkIndex(
            anchors = anchors,
            totalChars = decodedChars,
            startByteOffset = bomSkip
        )
    }

    private inline fun decodeLoop(
        decoder: CharsetDecoder,
        byteBuf: ByteBuffer,
        charBuf: CharBuffer,
        onProduced: (Int) -> Unit
    ) {
        while (true) {
            charBuf.clear()
            val cr = decoder.decode(byteBuf, charBuf, false)
            charBuf.flip()
            val produced = charBuf.remaining()
            if (produced > 0) onProduced(produced)
            if (cr.isUnderflow) break
        }
    }

    // ---------------- read ----------------

    private suspend fun readFromAnchor(anchor: ChunkAnchor, startChar: Int, endChar: Int): String {
        val decoder = charset.newDecoder()
        val byteBuf = ByteBuffer.allocateDirect(256 * 1024)
        val charBuf = CharBuffer.allocate(16 * 1024)

        channel.position(anchor.byteOffset)

        var currentChar = anchor.charOffset
        val sb = StringBuilder((endChar - startChar).coerceAtMost(64_000))

        fun consumeChars(seq: CharSequence) {
            if (currentChar >= endChar) return
            val produced = seq.length
            val segStart = max(0, startChar - currentChar)
            val segEnd = minOf(produced, endChar - currentChar)
            if (segStart < segEnd) {
                sb.append(seq.subSequence(segStart, segEnd))
            }
            currentChar += produced
        }

        while (currentChar < endChar) {
            coroutineContext.ensureActive()

            byteBuf.clear()
            val read = channel.read(byteBuf)
            if (read < 0) break
            byteBuf.flip()

            while (byteBuf.hasRemaining() && currentChar < endChar) {
                charBuf.clear()
                val cr = decoder.decode(byteBuf, charBuf, false)
                charBuf.flip()
                if (charBuf.hasRemaining()) consumeChars(charBuf)
                if (cr.isUnderflow) break
            }
        }

        // end-of-input flush（防止最后一个字符丢失）
        byteBuf.clear()
        byteBuf.flip()
        charBuf.clear()
        decoder.decode(byteBuf, charBuf, true)
        charBuf.flip()
        if (charBuf.hasRemaining() && currentChar < endChar) consumeChars(charBuf)

        charBuf.clear()
        decoder.flush(charBuf)
        charBuf.flip()
        if (charBuf.hasRemaining() && currentChar < endChar) consumeChars(charBuf)

        return sb.toString()
    }
}
```

> 备注：这版索引 anchor 是“按字节间隔”，anchor byteOffset 一定来自 decoder 的安全边界（不会落在多字节字符中间），所以 seek 解码稳定。

# 三、你需要做的一个小检查

* `engines/txt` 模块要有 Android 依赖（因为用到 `ParcelFileDescriptor`、`FileInputStream`、`StaticLayout` 等）
