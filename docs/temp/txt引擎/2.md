下面是**在你现有代码基础上**补齐“专业版”缺口的实现：

* ✅ **SoftBreakIndex（持久化软换行索引）**：不再每页启发式判断，按索引 O(换行数) 应用，定位稳定、性能稳定
* ✅ **PageMap（持久化分页边界 + 后台补全）**：页码/总页数/跳转更准、下次打开复用
* ✅ **搜索加速索引（Trigram Bloom Block Index）**：大文件长关键字搜索显著加速（无误漏，只有少量误报回读校验）
* ✅ **标注 Decorations 渲染进 RenderPage**（TXT ReflowRange）
* ✅ **链接识别**（URL/Email），填充 `RenderPage.links`，并把 `capabilities.links=true`
* ✅ **更强的编码探测**（多候选评分，避免“UTF‑8 不合法就盲回退 GB18030”的粗糙）
* ✅ 同时给出必要的**最小改动点**（不重复你已有完整文件，只给新增文件 + 需要替换/插入的片段）

```kotlin
package com.example.reader.engines.txt.internal.util

import java.io.EOFException
import java.io.IOException
import java.io.RandomAccessFile

internal fun RandomAccessFile.writeVarLong(value: Long) {
    require(value >= 0) { "VarLong only supports non-negative values: $value" }
    var v = value
    while ((v and -128L) != 0L) {
        writeByte(((v and 0x7FL) or 0x80L).toInt())
        v = v ushr 7
    }
    writeByte(v.toInt())
}

internal fun RandomAccessFile.readVarLongOrNull(): Long? {
    var shift = 0
    var result = 0L
    try {
        while (shift < 64) {
            val b = readUnsignedByte()
            result = result or ((b and 0x7F).toLong() shl shift)
            if ((b and 0x80) == 0) return result
            shift += 7
        }
        throw IOException("VarLong too long")
    } catch (_: EOFException) {
        return null
    }
}

internal fun RandomAccessFile.writeStringUtf8(value: String) {
    val bytes = value.toByteArray(Charsets.UTF_8)
    require(bytes.size <= 0xFFFF) { "String too long for u16 length: ${bytes.size}" }
    writeShort(bytes.size)
    write(bytes)
}

internal fun RandomAccessFile.readStringUtf8(): String {
    val len = readShort().toInt() and 0xFFFF
    val bytes = ByteArray(len)
    readFully(bytes)
    return String(bytes, Charsets.UTF_8)
}
```

## 2.1 SoftBreakIndex：读取 + range 迭代

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/softbreak/SoftBreakIndex.kt
package com.example.reader.engines.txt.internal.softbreak

import com.example.reader.engines.txt.internal.open.TxtMeta
import java.io.Closeable
import java.io.File
import java.io.RandomAccessFile
import kotlin.math.max

internal class SoftBreakIndex private constructor(
    private val raf: RandomAccessFile,
    val version: Int,
    private val blockNewlines: Int,
    val lengthChars: Long,
    val newlineCount: Long,
    val sampleHash: String,
    private val blocks: List<BlockMeta>
) : Closeable {

    internal data class BlockMeta(
        val filePos: Long,
        val firstOffset: Long,
        val lastOffset: Long,
        val count: Int
    )

    companion object {
        private const val MAGIC = "SBX1"
        private const val VERSION = 1

        fun openIfValid(file: File, meta: TxtMeta): SoftBreakIndex? {
            if (!file.exists()) return null
            val raf = RandomAccessFile(file, "r")
            try {
                val magic = ByteArray(4).also { raf.readFully(it) }.toString(Charsets.US_ASCII)
                if (magic != MAGIC) return null.also { raf.close() }

                val version = raf.readInt()
                if (version != VERSION) return null.also { raf.close() }

                val blockNewlines = raf.readInt()
                val lengthChars = raf.readLong()
                val newlineCount = raf.readLong()
                val sampleHash = raf.readStringUtf8()
                val indexOffset = raf.readLong()

                if (lengthChars != meta.lengthChars) return null.also { raf.close() }
                if (sampleHash != meta.sampleHash) return null.also { raf.close() }

                raf.seek(indexOffset)
                val indexCount = raf.readInt()
                val blocks = ArrayList<BlockMeta>(indexCount)
                for (i in 0 until indexCount) {
                    val pos = raf.readLong()
                    val first = raf.readLong()
                    val last = raf.readLong()
                    val count = raf.readInt()
                    blocks.add(BlockMeta(pos, first, last, count))
                }

                return SoftBreakIndex(
                    raf = raf,
                    version = version,
                    blockNewlines = blockNewlines,
                    lengthChars = lengthChars,
                    newlineCount = newlineCount,
                    sampleHash = sampleHash,
                    blocks = blocks
                )
            } catch (t: Throwable) {
                runCatching { raf.close() }
                return null
            }
        }
    }

    /**
     * 遍历 [startChar, endChar) 内的所有 newline（包括 soft/hard），回调给调用方：
     * offset = '\n' 的全局 char offset
     * isSoft = true 表示渲染时应把 '\n' 显示成空格（长度不变）
     */
    fun forEachNewlineInRange(
        startChar: Long,
        endChar: Long,
        consumer: (offset: Long, isSoft: Boolean) -> Unit
    ) {
        if (blocks.isEmpty()) return
        val start = startChar.coerceAtLeast(0L)
        val end = endChar.coerceAtMost(lengthChars).coerceAtLeast(start)
        if (start >= end) return

        val idx = findFirstBlockByLastOffset(start)
        if (idx >= blocks.size) return

        for (i in idx until blocks.size) {
            val b = blocks[i]
            if (b.firstOffset >= end) break
            if (b.lastOffset < start) continue
            decodeBlockAndVisit(b, start, end, consumer)
        }
    }

    private fun findFirstBlockByLastOffset(start: Long): Int {
        var lo = 0
        var hi = blocks.size - 1
        var ans = blocks.size
        while (lo <= hi) {
            val mid = (lo + hi) ushr 1
            if (blocks[mid].lastOffset >= start) {
                ans = mid
                hi = mid - 1
            } else {
                lo = mid + 1
            }
        }
        return ans
    }

    private fun decodeBlockAndVisit(
        block: BlockMeta,
        start: Long,
        end: Long,
        consumer: (offset: Long, isSoft: Boolean) -> Unit
    ) {
        raf.seek(block.filePos)
        val count = raf.readInt()
        val firstOffset = raf.readLong()
        val flagsLen = (count + 7) / 8
        val flags = ByteArray(flagsLen)
        raf.readFully(flags)

        var offset = firstOffset
        for (i in 0 until count) {
            if (offset in start until end) {
                val soft = ((flags[i ushr 3].toInt() ushr (i and 7)) and 1) == 1
                consumer(offset, soft)
            }
            if (i < count - 1) {
                val delta = raf.readVarLongOrNull() ?: break
                offset = max(offset + delta, offset) // 防御：避免溢出/倒退
            }
        }
    }

    override fun close() {
        raf.close()
    }
}

// reuse util functions
private fun RandomAccessFile.readStringUtf8(): String =
    com.example.reader.engines.txt.internal.util.readStringUtf8(this)

private fun RandomAccessFile.readVarLongOrNull(): Long? =
    com.example.reader.engines.txt.internal.util.readVarLongOrNull(this)
```

> 上面用了 `readStringUtf8/readVarLongOrNull` 的转发，为了不重复 import；你也可以直接 `import ...` 然后改成扩展调用（更干净）。

## 2.2 SoftBreakIndexBuilder：构建索引（只在 hardWrapLikely=true 时构建）

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/softbreak/SoftBreakIndexBuilder.kt
package com.example.reader.engines.txt.internal.softbreak

import com.example.reader.engines.txt.internal.open.TxtBookFiles
import com.example.reader.engines.txt.internal.open.TxtMeta
import com.example.reader.engines.txt.internal.provider.ChapterDetector
import com.example.reader.engines.txt.internal.util.writeStringUtf8
import com.example.reader.engines.txt.internal.util.writeVarLong
import java.io.File
import java.io.FileInputStream
import java.io.RandomAccessFile
import java.nio.ByteBuffer
import java.nio.ByteOrder
import kotlin.math.max
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.withContext
import kotlin.coroutines.coroutineContext

internal object SoftBreakIndexBuilder {

    private const val MAGIC = "SBX1"
    private const val VERSION = 1
    private const val BLOCK_NEWLINES = 4096
    private const val MAX_TITLE_CHARS = 80

    private val strongEndPunct = setOf('。', '！', '？', '.', '!', '?', '…', ':', '：', ';', '；')
    private val detector = ChapterDetector()

    suspend fun buildIfNeeded(
        files: TxtBookFiles,
        meta: TxtMeta,
        ioDispatcher: CoroutineDispatcher
    ) = withContext(ioDispatcher) {
        // 非 hardwrap 的文本：不需要 softbreak 索引
        if (!meta.hardWrapLikely) return@withContext

        // 已存在且有效则跳过
        SoftBreakIndex.openIfValid(files.softBreakIdx, meta)?.close()?.also { return@withContext }

        files.bookDir.mkdirs()

        RandomAccessFile(files.softBreakLock, "rw").channel.use { lockCh ->
            lockCh.lock().use {
                // double-check after lock
                SoftBreakIndex.openIfValid(files.softBreakIdx, meta)?.close()?.also { return@withContext }

                val tmp = File(files.bookDir, "softbreak.idx.tmp").apply { if (exists()) delete() }
                val blocks = ArrayList<SoftBreakIndex.BlockMeta>(256)

                RandomAccessFile(tmp, "rw").use { raf ->
                    raf.setLength(0)

                    // header placeholder
                    raf.write(MAGIC.toByteArray(Charsets.US_ASCII))
                    raf.writeInt(VERSION)
                    raf.writeInt(BLOCK_NEWLINES)
                    raf.writeLong(meta.lengthChars)

                    val newlineCountPos = raf.filePointer
                    raf.writeLong(0L) // newlineCount placeholder

                    raf.writeStringUtf8(meta.sampleHash)

                    val indexOffsetPos = raf.filePointer
                    raf.writeLong(0L) // indexOffset placeholder

                    // block buffers
                    val offsets = LongArray(BLOCK_NEWLINES)
                    val flags = BooleanArray(BLOCK_NEWLINES)
                    var inBlock = 0

                    fun flushBlock() {
                        if (inBlock <= 0) return
                        val filePos = raf.filePointer
                        val first = offsets[0]
                        val last = offsets[inBlock - 1]

                        raf.writeInt(inBlock)
                        raf.writeLong(first)

                        // flags bytes first
                        val flagsLen = (inBlock + 7) / 8
                        val fb = ByteArray(flagsLen)
                        for (i in 0 until inBlock) {
                            if (flags[i]) {
                                val bi = i ushr 3
                                val mask = 1 shl (i and 7)
                                fb[bi] = (fb[bi].toInt() or mask).toByte()
                            }
                        }
                        raf.write(fb)

                        // deltas
                        var prev = first
                        for (i in 1 until inBlock) {
                            val d = offsets[i] - prev
                            raf.writeVarLong(d)
                            prev = offsets[i]
                        }

                        blocks.add(
                            SoftBreakIndex.BlockMeta(
                                filePos = filePos,
                                firstOffset = first,
                                lastOffset = last,
                                count = inBlock
                            )
                        )
                        inBlock = 0
                    }

                    // streaming scan u16
                    var global = 0L
                    var totalNewlines = 0L

                    var lineLen = 0
                    var leadingSpaces = 0
                    var seenNonSpace = false
                    var firstNonSpace: Char? = null
                    var lastNonSpace: Char? = null
                    val titleSb = StringBuilder(MAX_TITLE_CHARS)

                    data class LineInfo(
                        val len: Int,
                        val leadingSpaces: Int,
                        val firstNonSpace: Char?,
                        val lastNonSpace: Char?,
                        val isTitle: Boolean,
                        val startsWithBullet: Boolean
                    )

                    fun finishLine(): LineInfo {
                        val title = titleSb.toString().trim()
                        val isTitle = detector.isChapterTitle(title)
                        val startsWithBullet = firstNonSpace == '-' || firstNonSpace == '*' || firstNonSpace == '•'
                        val info = LineInfo(
                            len = lineLen,
                            leadingSpaces = leadingSpaces,
                            firstNonSpace = firstNonSpace,
                            lastNonSpace = lastNonSpace,
                            isTitle = isTitle,
                            startsWithBullet = startsWithBullet
                        )
                        // reset
                        lineLen = 0
                        leadingSpaces = 0
                        seenNonSpace = false
                        firstNonSpace = null
                        lastNonSpace = null
                        titleSb.setLength(0)
                        return info
                    }

                    fun isSoftBreak(line0: LineInfo, line1: LineInfo): Boolean {
                        val typical = meta.typicalLineLength.coerceAtLeast(40)

                        if (line0.len == 0 || line1.len == 0) return false
                        if (line0.isTitle || line1.isTitle) return false
                        if (line1.leadingSpaces >= 2) return false
                        if (line1.startsWithBullet) return false

                        if (line0.len < (typical * 0.60).toInt()) return false
                        if (line1.len < (typical * 0.30).toInt()) return false

                        val last = line0.lastNonSpace
                        if (last != null && strongEndPunct.contains(last)) return false

                        return true
                    }

                    data class Pending(val newlineOffset: Long, val line0: LineInfo)

                    var pending: Pending? = null
                    var lastLineInfo: LineInfo? = null

                    FileInputStream(files.contentU16).channel.use { ch ->
                        val buf = ByteBuffer.allocateDirect(128 * 1024).order(ByteOrder.LITTLE_ENDIAN)

                        while (true) {
                            coroutineContext.ensureActive()
                            buf.clear()
                            val read = ch.read(buf)
                            if (read <= 0) break
                            buf.flip()

                            val charBuf = buf.asCharBuffer()
                            while (charBuf.hasRemaining()) {
                                coroutineContext.ensureActive()

                                val c = charBuf.get()
                                when (c) {
                                    '\n' -> {
                                        val line1 = finishLine()
                                        val newlineOffset = global

                                        // classify pending newline (newline of previous line) now that we have line1
                                        pending?.let { p ->
                                            val soft = isSoftBreak(p.line0, line1)
                                            offsets[inBlock] = p.newlineOffset
                                            flags[inBlock] = soft
                                            inBlock++
                                            totalNewlines++
                                            if (inBlock == BLOCK_NEWLINES) flushBlock()
                                        }

                                        // current newline becomes new pending, to be decided by next line
                                        pending = Pending(newlineOffset, line1)
                                        lastLineInfo = line1
                                    }

                                    else -> {
                                        // collect title prefix
                                        if (titleSb.length < MAX_TITLE_CHARS) titleSb.append(c)

                                        if (!seenNonSpace) {
                                            if (c.isWhitespace()) {
                                                leadingSpaces++
                                            } else {
                                                seenNonSpace = true
                                                firstNonSpace = c
                                            }
                                        }
                                        if (!c.isWhitespace()) lastNonSpace = c
                                        lineLen++
                                    }
                                }
                                global++
                            }
                        }
                    }

                    // EOF: flush last pending as hard (not soft)
                    pending?.let { p ->
                        offsets[inBlock] = p.newlineOffset
                        flags[inBlock] = false
                        inBlock++
                        totalNewlines++
                    }
                    flushBlock()

                    // index table
                    val indexOffset = raf.filePointer
                    raf.writeInt(blocks.size)
                    for (b in blocks) {
                        raf.writeLong(b.filePos)
                        raf.writeLong(b.firstOffset)
                        raf.writeLong(b.lastOffset)
                        raf.writeInt(b.count)
                    }

                    // patch header
                    raf.seek(newlineCountPos)
                    raf.writeLong(totalNewlines)
                    raf.seek(indexOffsetPos)
                    raf.writeLong(indexOffset)
                }

                if (files.softBreakIdx.exists()) files.softBreakIdx.delete()
                tmp.renameTo(files.softBreakIdx)
            }
        }
    }
}
```

## 3.2 PageMapStore（append-only varint）

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/pagemap/PageMapStore.kt
package com.example.reader.engines.txt.internal.pagemap

import com.example.reader.engines.txt.internal.util.readStringUtf8
import com.example.reader.engines.txt.internal.util.readVarLongOrNull
import com.example.reader.engines.txt.internal.util.writeStringUtf8
import com.example.reader.engines.txt.internal.util.writeVarLong
import java.io.Closeable
import java.io.File
import java.io.RandomAccessFile

internal class PageMapStore private constructor(
    private val file: File,
    val profileId: String,
    val sampleHash: String,
    val lengthChars: Long,
    var completed: Boolean
) : Closeable {

    companion object {
        private const val MAGIC = "PMX1"
        private const val VERSION = 1
        private const val FLAG_COMPLETED = 1

        fun openOrCreate(file: File, profileId: String, sampleHash: String, lengthChars: Long): PageMapStore {
            file.parentFile?.mkdirs()

            if (!file.exists()) {
                RandomAccessFile(file, "rw").use { raf ->
                    raf.setLength(0)
                    raf.write(MAGIC.toByteArray(Charsets.US_ASCII))
                    raf.writeInt(VERSION)
                    raf.writeLong(lengthChars)
                    raf.writeInt(0) // flags
                    raf.writeStringUtf8(sampleHash)
                    raf.writeStringUtf8(profileId)
                    // body: first start absolute = 0
                    raf.writeVarLong(0L)
                }
            }

            RandomAccessFile(file, "r").use { raf ->
                val magic = ByteArray(4).also { raf.readFully(it) }.toString(Charsets.US_ASCII)
                require(magic == MAGIC) { "Invalid PageMap magic" }
                val version = raf.readInt()
                require(version == VERSION) { "Unsupported PageMap version: $version" }

                val len = raf.readLong()
                require(len == lengthChars) { "PageMap length mismatch: $len vs $lengthChars" }

                val flags = raf.readInt()
                val sh = raf.readStringUtf8()
                val pid = raf.readStringUtf8()
                require(sh == sampleHash) { "PageMap sampleHash mismatch" }
                require(pid == profileId) { "PageMap profile mismatch" }

                return PageMapStore(file, profileId, sampleHash, lengthChars, completed = (flags and FLAG_COMPLETED) != 0)
            }
        }

        fun loadStarts(file: File): LongArray {
            RandomAccessFile(file, "r").use { raf ->
                raf.skipBytes(4) // magic
                raf.readInt() // version
                raf.readLong() // length
                raf.readInt() // flags
                raf.readStringUtf8()
                raf.readStringUtf8()

                val starts = ArrayList<Long>(4096)
                val firstAbs = raf.readVarLongOrNull() ?: return longArrayOf(0L)
                var cur = firstAbs
                starts.add(cur)

                while (true) {
                    val d = raf.readVarLongOrNull() ?: break
                    cur += d
                    starts.add(cur)
                }
                return starts.toLongArray()
            }
        }
    }

    fun appendNextStartDelta(delta: Long) {
        require(delta > 0) { "delta must be >0" }
        RandomAccessFile(file, "rw").use { raf ->
            raf.seek(raf.length())
            raf.writeVarLong(delta)
        }
    }

    fun markCompleted() {
        if (completed) return
        completed = true
        RandomAccessFile(file, "rw").use { raf ->
            // flags is at offset: 4(magic)+4(ver)+8(len)=16
            raf.seek(16)
            val flags = raf.readInt()
            raf.seek(16)
            raf.writeInt(flags or FLAG_COMPLETED)
        }
    }

    override fun close() {
        // no-op (we open/close per op for simplicity & safety)
    }
}
```

```kotlin
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.launch
import kotlinx.coroutines.withContext
import kotlin.coroutines.coroutineContext

internal class PageMapManager(
    private val scope: CoroutineScope,
    private val ioDispatcher: CoroutineDispatcher,
    private val defaultDispatcher: CoroutineDispatcher,
    private val files: TxtBookFiles,
    private val meta: TxtMeta
) : Closeable {

    private var store: PageMapStore? = null
    private var starts: LongArray = longArrayOf(0L)
    private var buildJob: Job? = null
    private var profileId: String? = null

    fun currentProfileId(): String? = profileId
    fun completed(): Boolean = store?.completed == true
    fun totalPagesOrNull(): Int? = if (completed()) starts.size else null

    fun pageIndexForOffsetOrNull(offset: Long): Int? {
        if (starts.isEmpty()) return null
        val lastStart = starts.last()
        if (!completed() && offset > lastStart) return null

        val idx = floorIndex(starts, offset)
        return if (idx >= 0) idx else null
    }

    suspend fun attachProfile(
        paginator: TxtPaginator,
        config: RenderConfig.ReflowText,
        constraints: LayoutConstraints,
        softBreakVersion: Int,
        hardWrapEnabled: Boolean
    ) {
        val pid = PaginationProfileId.of(config, constraints, softBreakVersion, hardWrapEnabled)
        if (pid == profileId) return

        cancelBuild()
        profileId = pid

        val pmFile = File(files.bookDir, "pagemap_$pid.pmx")
        val pmStore = withContext(ioDispatcher) {
            PageMapStore.openOrCreate(pmFile, pid, meta.sampleHash, meta.lengthChars)
        }
        store = pmStore
        starts = withContext(ioDispatcher) { PageMapStore.loadStarts(pmFile) }.ifEmpty { longArrayOf(0L) }

        // 自动后台补全分页（专业版：不阻塞 UI）
        if (!pmStore.completed) {
            startBackgroundBuild(paginator, config, constraints)
        }
    }

    private fun startBackgroundBuild(
        paginator: TxtPaginator,
        config: RenderConfig.ReflowText,
        constraints: LayoutConstraints
    ) {
        val st = store ?: return
        buildJob = scope.launch(defaultDispatcher) {
            // 从最后一个已知 start 开始向后补全
            while (!st.completed) {
                coroutineContext.ensureActive()

                val last = starts.last()
                if (last >= meta.lengthChars) {
                    st.markCompleted()
                    break
                }

                val slice = paginator.pageAt(
                    startOffset = last,
                    config = config,
                    constraints = constraints,
                    policy = RenderPolicy(quality = RenderPolicy.Quality.DRAFT, allowCache = false, prefetchNeighbors = 0)
                )

                val next = slice.end
                if (next <= last) {
                    // 防御：避免死循环
                    st.markCompleted()
                    break
                }

                if (next >= meta.lengthChars) {
                    st.markCompleted()
                    break
                }

                // append boundary
                val delta = next - last
                withContext(ioDispatcher) { st.appendNextStartDelta(delta) }

                // 更新内存 starts
                starts = starts + next

                // 让出 CPU，避免抢占翻页线程
                delay(8)
            }
        }
    }

    fun cancelBuild() {
        buildJob?.cancel()
        buildJob = null
    }

    override fun close() {
        cancelBuild()
        store?.close()
    }

    private fun floorIndex(arr: LongArray, x: Long): Int {
        var lo = 0
        var hi = arr.size - 1
        var ans = -1
        while (lo <= hi) {
            val mid = (lo + hi) ushr 1
            val v = arr[mid]
            if (v <= x) {
                ans = mid
                lo = mid + 1
            } else {
                hi = mid - 1
            }
        }
        return ans
    }
}
```

# 4) 新增：搜索加速索引（Trigram Bloom Block Index）

这是“专业但实现成本可控”的方案：

* 构建：一次扫描，每块 64k chars -> 一个 Bloom bitset（比如 4096 bits）
* 查询：先用 Bloom 过滤候选块，再在候选块里精确 KMP（不会漏结果）
* 索引文件很小：百 MB 小说也就几百 KB～几 MB

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/search/TrigramBloomIndex.kt
package com.example.reader.engines.txt.internal.search

import com.example.reader.engines.txt.internal.open.TxtMeta
import com.example.reader.engines.txt.internal.store.Utf16TextStore
import com.example.reader.engines.txt.internal.util.Hashing
import java.io.File
import java.io.RandomAccessFile
import kotlin.math.min
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.withContext
import kotlin.coroutines.coroutineContext

internal class TrigramBloomIndex private constructor(
    private val file: File,
    val blockChars: Int,
    private val bitsetBits: Int,
    val lengthChars: Long,
    val sampleHash: String
) {

    companion object {
        private const val MAGIC = "TBX1"
        private const val VERSION = 1

        fun openIfValid(file: File, meta: TxtMeta): TrigramBloomIndex? {
            if (!file.exists()) return null
            return try {
                RandomAccessFile(file, "r").use { raf ->
                    val magic = ByteArray(4).also { raf.readFully(it) }.toString(Charsets.US_ASCII)
                    if (magic != MAGIC) return null
                    val ver = raf.readInt()
                    if (ver != VERSION) return null
                    val blockChars = raf.readInt()
                    val bitsetBits = raf.readInt()
                    val lengthChars = raf.readLong()
                    val sampleHash = raf.readStringUtf8()
                    if (lengthChars != meta.lengthChars) return null
                    if (sampleHash != meta.sampleHash) return null
                    TrigramBloomIndex(file, blockChars, bitsetBits, lengthChars, sampleHash)
                }
            } catch (_: Throwable) {
                null
            }
        }

        suspend fun buildIfNeeded(
            file: File,
            lockFile: File,
            store: Utf16TextStore,
            meta: TxtMeta,
            ioDispatcher: CoroutineDispatcher,
            blockChars: Int = 64_000,
            bitsetBits: Int = 4096
        ) = withContext(ioDispatcher) {
            if (openIfValid(file, meta) != null) return@withContext

            lockFile.parentFile?.mkdirs()
            RandomAccessFile(lockFile, "rw").channel.use { ch ->
                ch.lock().use {
                    if (openIfValid(file, meta) != null) return@withContext

                    val tmp = File(file.parentFile, "${file.name}.tmp").apply { if (exists()) delete() }
                    RandomAccessFile(tmp, "rw").use { raf ->
                        raf.setLength(0)
                        raf.write(MAGIC.toByteArray(Charsets.US_ASCII))
                        raf.writeInt(VERSION)
                        raf.writeInt(blockChars)
                        raf.writeInt(bitsetBits)
                        raf.writeLong(meta.lengthChars)
                        raf.writeStringUtf8(meta.sampleHash)

                        val bytesPerBlock = bitsetBits / 8
                        val totalBlocks = ((meta.lengthChars + blockChars - 1) / blockChars).toInt()

                        val bitset = ByteArray(bytesPerBlock)

                        for (bi in 0 until totalBlocks) {
                            coroutineContext.ensureActive()

                            java.util.Arrays.fill(bitset, 0.toByte())
                            val start = bi.toLong() * blockChars.toLong()
                            val len = min(blockChars.toLong(), meta.lengthChars - start).toInt()
                            val chars = store.readChars(start, len)
                            bloomAddAllTrigrams(chars, bitsetBits, bitset)

                            raf.write(bitset)
                        }
                    }

                    if (file.exists()) file.delete()
                    tmp.renameTo(file)
                }
            }
        }

        private fun bloomAddAllTrigrams(chars: CharArray, bits: Int, bitset: ByteArray) {
            if (chars.size < 3) return
            for (i in 0 until chars.size - 2) {
                val c1 = chars[i].code
                val c2 = chars[i + 1].code
                val c3 = chars[i + 2].code

                val h = mix64(c1, c2, c3)
                val (h1, h2) = splitHashes(h)

                setBit(bitset, bits, h1)
                setBit(bitset, bits, h2)
            }
        }

        private fun setBit(bitset: ByteArray, bits: Int, hash: Int) {
            val idx = (hash ushr 1).mod(bits) // keep non-negative
            val byteIdx = idx ushr 3
            val bit = 1 shl (idx and 7)
            bitset[byteIdx] = (bitset[byteIdx].toInt() or bit).toByte()
        }

        private fun mix64(c1: Int, c2: Int, c3: Int): Long {
            // 48-bit payload -> hash64
            var x = (c1.toLong() shl 32) xor (c2.toLong() shl 16) xor c3.toLong()
            // finalizer
            x = x xor (x ushr 33)
            x *= -0xae502812aa7333L
            x = x xor (x ushr 33)
            x *= -0x3b314601e57a13adL
            x = x xor (x ushr 33)
            return x
        }

        private fun splitHashes(h: Long): Pair<Int, Int> {
            val h1 = (h and 0xFFFFFFFFL).toInt()
            val h2 = (h ushr 32).toInt()
            return h1 to h2
        }
    }

    fun bytesPerBlock(): Int = bitsetBits / 8

    fun blocksCount(): Int = ((lengthChars + blockChars - 1) / blockChars).toInt()

    fun blockRange(i: Int): LongRange {
        val start = i.toLong() * blockChars.toLong()
        val endExclusive = (start + blockChars.toLong()).coerceAtMost(lengthChars)
        return start until endExclusive
    }

    fun mayContainAll(fileRaf: RandomAccessFile, blockIndex: Int, trigrams: IntArray): Boolean {
        val bytesPerBlock = bytesPerBlock()
        val headerSize = headerSizeBytes()
        val pos = headerSize.toLong() + blockIndex.toLong() * bytesPerBlock.toLong()
        fileRaf.seek(pos)
        val bitset = ByteArray(bytesPerBlock)
        fileRaf.readFully(bitset)

        for (h in trigrams) {
            val idx = (h ushr 1).mod(bitsetBits)
            val byteIdx = idx ushr 3
            val mask = 1 shl (idx and 7)
            if (((bitset[byteIdx].toInt() and mask) == 0)) return false
        }
        return true
    }

    fun buildQueryTrigramHashes(query: String): IntArray {
        if (query.length < 3) return intArrayOf()
        val hashes = LinkedHashSet<Int>()
        val chars = query.toCharArray()
        for (i in 0 until chars.size - 2) {
            val h = mix64(chars[i].code, chars[i + 1].code, chars[i + 2].code)
            val (h1, h2) = splitHashes(h)
            hashes.add(h1)
            hashes.add(h2)
        }
        return hashes.toIntArray()
    }

    private fun headerSizeBytes(): Int {
        // magic(4) + ver(4) + blockChars(4) + bitsetBits(4) + length(8) + sampleHashLen(2)+hash
        val hashBytes = sampleHash.toByteArray(Charsets.UTF_8)
        return 4 + 4 + 4 + 4 + 8 + 2 + hashBytes.size
    }
}

private fun RandomAccessFile.readStringUtf8(): String =
    com.example.reader.engines.txt.internal.util.readStringUtf8(this)
```

## 4.1 替换 SearchProvider 为“专业版”（不重复旧实现）

新建一个 provider（避免你旧 `TxtSearchProvider.kt` 大改）：

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/provider/TxtSearchProviderPro.kt
package com.example.reader.engines.txt.internal.provider

import com.example.reader.api.provider.SearchHit
import com.example.reader.api.provider.SearchOptions
import com.example.reader.api.provider.SearchProvider
import com.example.reader.engines.txt.internal.open.TxtBookFiles
import com.example.reader.engines.txt.internal.open.TxtMeta
import com.example.reader.engines.txt.internal.search.TrigramBloomIndex
import com.example.reader.engines.txt.internal.store.Utf16TextStore
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import kotlin.math.min
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.channels.Channel
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.channelFlow
import kotlinx.coroutines.launch
import kotlinx.coroutines.withContext
import kotlin.coroutines.coroutineContext
import java.io.RandomAccessFile

internal class TxtSearchProviderPro(
    private val files: TxtBookFiles,
    private val store: Utf16TextStore,
    private val meta: TxtMeta,
    private val ioDispatcher: CoroutineDispatcher
) : SearchProvider {

    override fun search(query: String, options: SearchOptions): Flow<SearchHit> = channelFlow {
        val q = query
        if (q.isBlank()) return@channelFlow

        val startOffset = options.startFrom
            ?.takeIf { it.scheme == LocatorSchemes.TXT_OFFSET }
            ?.value
            ?.toLongOrNull()
            ?.coerceIn(0L, store.lengthChars)
            ?: 0L

        // 尝试加载/构建 bloom index（构建不阻塞本次搜索，后台并行）
        val bloomFile = files.bloomIdx
        val bloomLock = files.bloomLock
        val bloom = TrigramBloomIndex.openIfValid(bloomFile, meta)

        if (bloom == null && q.length >= 3 && meta.lengthChars >= 1_000_000L) {
            launch(ioDispatcher) {
                TrigramBloomIndex.buildIfNeeded(bloomFile, bloomLock, store, meta, ioDispatcher)
            }
        }

        // 选择策略：能用 bloom -> 候选块扫描；否则全量 streaming scan
        if (bloom != null && q.length >= 3) {
            fastSearchWithBloom(bloom, q, options, startOffset)
        } else {
            streamingScan(q, options, startOffset)
        }
    }

    private suspend fun kotlinx.coroutines.channels.ProducerScope<SearchHit>.fastSearchWithBloom(
        bloom: TrigramBloomIndex,
        q: String,
        options: SearchOptions,
        startOffset: Long
    ) = withContext(ioDispatcher) {
        val pattern = q.toCharArray()
        val matcher = KmpMatcher(pattern, caseSensitive = options.caseSensitive)
        val trig = bloom.buildQueryTrigramHashes(if (options.caseSensitive) q else q.lowercase())
        val blocks = bloom.blocksCount()

        RandomAccessFile(files.bloomIdx, "r").use { raf ->
            val candidates = ArrayList<Int>(64)
            val startBlock = (startOffset / bloom.blockChars).toInt().coerceAtLeast(0)

            for (bi in startBlock until blocks) {
                coroutineContext.ensureActive()
                if (bloom.mayContainAll(raf, bi, trig)) {
                    candidates.add(bi)
                    // 防止边界漏：把邻居也加进来
                    if (bi > 0) candidates.add(bi - 1)
                    if (bi + 1 < blocks) candidates.add(bi + 1)
                }
            }

            candidates.sort()
            val uniq = candidates.distinct()

            var hits = 0
            var lastEmittedStart = -1L

            for (bi in uniq) {
                coroutineContext.ensureActive()
                val range = bloom.blockRange(bi)
                val scanStart = (range.first - (pattern.size + 4).toLong()).coerceAtLeast(startOffset)
                val scanEndExclusive = (range.last + (pattern.size + 4).toLong()).coerceAtMost(store.lengthChars)

                val len = (scanEndExclusive - scanStart).toInt().coerceAtLeast(0)
                if (len <= 0) continue

                val chunk = store.readChars(scanStart, len)

                matcher.findAll(chunk) { matchIndex ->
                    coroutineContext.ensureActive()

                    val matchStartGlobal = scanStart + matchIndex.toLong()
                    val matchEndGlobal = matchStartGlobal + pattern.size.toLong()

                    if (matchStartGlobal < startOffset) return@findAll true
                    if (matchStartGlobal == lastEmittedStart) return@findAll true

                    if (options.wholeWord && !isWholeWord(chunk, matchIndex, pattern.size)) {
                        return@findAll true
                    }

                    lastEmittedStart = matchStartGlobal

                    trySend(
                        SearchHit(
                            range = LocatorRange(
                                start = Locator(LocatorSchemes.TXT_OFFSET, matchStartGlobal.toString()),
                                end = Locator(LocatorSchemes.TXT_OFFSET, matchEndGlobal.toString())
                            ),
                            excerpt = buildExcerpt(matchStartGlobal, pattern.size),
                            sectionTitle = null
                        )
                    )

                    hits++
                    hits < options.maxHits
                }
            }
        }
    }

    private suspend fun kotlinx.coroutines.channels.ProducerScope<SearchHit>.streamingScan(
        q: String,
        options: SearchOptions,
        startOffset: Long
    ) = withContext(ioDispatcher) {
        val pattern = q.toCharArray()
        val matcher = KmpMatcher(pattern, caseSensitive = options.caseSensitive)

        val chunkChars = 64_000
        val overlap = (pattern.size - 1).coerceAtLeast(0)

        var carry = CharArray(0)
        var pos = startOffset
        var hits = 0

        while (pos < store.lengthChars && hits < options.maxHits) {
            coroutineContext.ensureActive()

            val toRead = min(chunkChars.toLong(), store.lengthChars - pos).toInt()
            val chunk = store.readChars(pos, toRead)

            val combined = CharArray(carry.size + chunk.size)
            if (carry.isNotEmpty()) System.arraycopy(carry, 0, combined, 0, carry.size)
            if (chunk.isNotEmpty()) System.arraycopy(chunk, 0, combined, carry.size, chunk.size)

            val combinedStartGlobal = pos - carry.size.toLong()

            matcher.findAll(combined) { matchIndex ->
                coroutineContext.ensureActive()

                val matchStartGlobal = combinedStartGlobal + matchIndex.toLong()
                val matchEndGlobal = matchStartGlobal + pattern.size.toLong()

                if (matchStartGlobal < startOffset) return@findAll true
                if (matchIndex + pattern.size <= carry.size) return@findAll true

                if (options.wholeWord && !isWholeWord(combined, matchIndex, pattern.size)) {
                    return@findAll true
                }

                trySend(
                    SearchHit(
                        range = LocatorRange(
                            start = Locator(LocatorSchemes.TXT_OFFSET, matchStartGlobal.toString()),
                            end = Locator(LocatorSchemes.TXT_OFFSET, matchEndGlobal.toString())
                        ),
                        excerpt = buildExcerpt(matchStartGlobal, pattern.size),
                        sectionTitle = null
                    )
                )

                hits++
                hits < options.maxHits
            }

            carry = if (overlap > 0) {
                val keep = min(overlap, combined.size)
                combined.copyOfRange(combined.size - keep, combined.size)
            } else CharArray(0)

            pos += toRead.toLong()
        }
    }

    private fun isWholeWord(hay: CharArray, start: Int, len: Int): Boolean {
        val before = if (start - 1 >= 0) hay[start - 1] else null
        val after = if (start + len < hay.size) hay[start + len] else null
        val beforeOk = before == null || !Character.isLetterOrDigit(before)
        val afterOk = after == null || !Character.isLetterOrDigit(after)
        return beforeOk && afterOk
    }

    private fun buildExcerpt(matchStart: Long, patternLen: Int): String {
        val before = 64
        val after = 96
        val center = matchStart + patternLen / 2
        return store.readAround(center, before, after)
            .replace('\n', ' ')
            .trim()
    }
}
```

# 5) 新增：RenderPage.links（URL/Email）+ capabilities.links=true

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/link/LinkDetector.kt
package com.example.reader.engines.txt.internal.link

import com.example.reader.model.DocumentLink
import com.example.reader.model.LinkTarget

internal object LinkDetector {

    // 粗略但够用：避免过度复杂导致误报/性能差
    private val url = Regex("""(?i)\b(https?://[^\s<>"']+|www\.[^\s<>"']+)""")
    private val email = Regex("""\b[A-Za-z0-9._%+\-]+@[A-Za-z0-9.\-]+\.[A-Za-z]{2,}\b""")

    private val trimTail = setOf('.', ',', '，', '。', '!', '?', '！', '？', ')', ']', '}', '>', '"', '\'')

    fun detect(text: CharSequence, max: Int = 20): List<DocumentLink> {
        val s = text.toString()
        val out = ArrayList<DocumentLink>(8)
        val seen = HashSet<String>()

        fun add(raw: String, isEmail: Boolean) {
            var v = raw
            while (v.isNotEmpty() && trimTail.contains(v.last())) v = v.dropLast(1)
            if (v.isBlank()) return

            val normalized = if (isEmail) "mailto:$v" else normalizeUrl(v)
            if (!seen.add(normalized)) return

            out.add(
                DocumentLink(
                    target = LinkTarget.External(normalized),
                    title = v,
                    bounds = null
                )
            )
        }

        for (m in url.findAll(s)) {
            if (out.size >= max) break
            add(m.value, isEmail = false)
        }
        for (m in email.findAll(s)) {
            if (out.size >= max) break
            add(m.value, isEmail = true)
        }

        return out
    }

    private fun normalizeUrl(v: String): String {
        val t = v.trim()
        return if (t.startsWith("www.", ignoreCase = true)) "http://$t" else t
    }
}
```

# 7) 更新分页渲染：用 SoftBreakIndex 替换每页启发式（核心专业化）

你现在 `TxtPaginator` 用 `SoftBreakProcessor` 每页判断。专业版改成：

* 若 `SoftBreakIndex` 可用：按索引替换该窗口范围内的 `\n` 为 `' '`（软换行），并对非软换行加段距 span
* 若不可用：保持原逻辑（或只做段距）

**你不需要重写 TxtPaginator 整个文件**，只要把 `processed` 生成那段换成索引应用。

给你一段可直接替换 `TxtPaginator.pageAt()` 内“processed 生成”的片段：

```kotlin
// 你在 TxtPaginator 里新增字段：private val softIndex: SoftBreakIndex?
// 并在构造时注入（document/session open 时打开 softbreak.idx）

val paragraphSpacingPx = (config.paragraphSpacingDp * constraints.density).toInt().coerceAtLeast(0)
val ssb = android.text.SpannableStringBuilder(String(raw))

if (softIndex != null && meta.hardWrapLikely) {
    softIndex.forEachNewlineInRange(start, start + raw.size.toLong()) { off, isSoft ->
        val local = (off - start).toInt()
        if (local !in 0 until ssb.length) return@forEachNewlineInRange
        if (isSoft) {
            ssb.setCharAt(local, ' ')
        } else if (paragraphSpacingPx > 0) {
            val next = if (local + 1 < ssb.length) ssb[local + 1] else null
            if (next != '\n') {
                ssb.setSpan(
                    com.example.reader.engines.txt.internal.render.ParagraphSpacingSpan(paragraphSpacingPx),
                    local,
                    local + 1,
                    android.text.Spanned.SPAN_EXCLUSIVE_EXCLUSIVE
                )
            }
        }
    }
} else {
    // 非 hardwrap：只做段距
    if (paragraphSpacingPx > 0) {
        for (i in 0 until ssb.length) {
            if (ssb[i] == '\n') {
                val next = if (i + 1 < ssb.length) ssb[i + 1] else null
                if (next != '\n') {
                    ssb.setSpan(
                        com.example.reader.engines.txt.internal.render.ParagraphSpacingSpan(paragraphSpacingPx),
                        i, i + 1,
                        android.text.Spanned.SPAN_EXCLUSIVE_EXCLUSIVE
                    )
                }
            }
        }
    }
}

val processed: CharSequence = ssb
```

# 8) 强化编码探测（多候选评分）

你现在的编码探测是：BOM -> UTF8合法 -> fallback GB18030。专业版建议：

* BOM 仍最高优先
* 若 UTF‑8 严格解码通过：直接 UTF‑8
* 否则：对一批候选（GB18030/GBK/Big5/Shift_JIS/Windows‑1252 等）做 sample 解码评分，选“替换字符少 + 控制字符少 + 可读字符多”的最优

我不重复你整个 `EncodingDetector.kt`，给你新增一个评分选择函数，你把它接进 detect() 的 fallback 分支即可：

```kotlin
// File: engines/txt/.../internal/encoding/EncodingScorer.kt
package com.example.reader.engines.txt.internal.encoding

import com.example.reader.source.DocumentSource
import java.nio.charset.Charset
import java.nio.charset.CodingErrorAction

internal object EncodingScorer {

    suspend fun pickBest(source: DocumentSource, sampleBytes: Int = 256 * 1024): Charset {
        val candidates = listOf(
            Charsets.UTF_8,
            Charset.forName("GB18030"),
            Charset.forName("GBK"),
            Charset.forName("Big5"),
            Charset.forName("Shift_JIS"),
            Charset.forName("windows-1252")
        )

        val bytes = source.openInputStream().use { input ->
            val buf = ByteArray(sampleBytes)
            val n = input.read(buf)
            if (n <= 0) ByteArray(0) else buf.copyOf(n)
        }

        fun score(cs: Charset): Int {
            val dec = cs.newDecoder()
                .onMalformedInput(CodingErrorAction.REPLACE)
                .onUnmappableCharacter(CodingErrorAction.REPLACE)

            val s = dec.decode(java.nio.ByteBuffer.wrap(bytes)).toString()
            var repl = 0
            var ctrl = 0
            var printable = 0

            for (c in s) {
                when {
                    c == '\uFFFD' -> repl++
                    c == '\n' || c == '\t' || c == '\r' -> printable++
                    c < ' ' -> ctrl++
                    else -> printable++
                }
            }
            // 分数越大越好：强惩罚 replacement，次惩罚控制字符
            return printable * 2 - repl * 50 - ctrl * 10
        }

        return candidates.maxBy { score(it) }
    }
}
```

在 `EncodingDetector.detect()` 里，当 UTF‑8 不合法时，把 fallback 改为：

```kotlin
val best = EncodingScorer.pickBest(source)
ReaderResult.Ok(EncodingResult(best, confidence = 0.65, reason = "scored_candidates"))
```
