```kotlin
// ✅ 修改签名
fun maybePersist(key: RenderKey, index: PageStartsIndex, newAdds: Int) {
    if (!config.persistPagination) return
    if (newAdds <= 0) return
    val bucket = map[key] ?: return
    bucket.newStartsSinceLastWrite += newAdds

    if (bucket.newStartsSinceLastWrite >= config.paginationWriteEveryNewStarts) {
        bucket.newStartsSinceLastWrite = 0
        saveToDisk(key, index)
    }
}
```

> 其它方法不变（getOrCreate/flush/saveToDisk/loadFromDisk）。

### 4.5 新增 `schedulePrefetch()`（放在类内任意位置）

```kotlin
private fun schedulePrefetch(n: Int) {
    val count = n.coerceIn(0, 6)
    if (count <= 0) return

    // 取消上一轮，避免堆积
    prefetchJob?.cancel()
    prefetchJob = scope.launch {
        runCatching { prefetchNeighbors(count) }
    }
}
```

### 4.6 修改 `close()`：取消后台 scope（并保持你已有 flush）

```kotlin
override fun close() {
    // ✅ 先停后台任务
    prefetchJob?.cancel()
    scope.cancel()

    // 你已有的 flush 保持
    renderKey?.let { key ->
        val start = currentSlice?.startChar ?: currentStartChar
        lastPositionStore.save(key, start, force = true)
        paginationStore.flush(key, pageStarts)
    }
}
```
