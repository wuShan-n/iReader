下面给你一套**可直接落地**、严格实现你给定 `core/reader/api` 契约的 **TXT 引擎完整代码（MVP + 可扩展）**。

实现特点：

* ✅ `ReaderEngine.open()`：读取 `DocumentSource`，编码探测（BOM/UTF-8/GB18030…），文本规范化
* ✅ `ReaderDocument`：提供 `metadata()` + `createSession()`
* ✅ `ReaderSession`：提供 `controller + outline/search/text/annotations`（资源 provider 为 null）
* ✅ `ReaderController`：支持 `setLayoutConstraints / setConfig / render / next / prev / goTo / goToProgress / prefetch / invalidate`
* ✅ 分页：使用 `StaticLayout`（兼容 API 21），一次 chunk 生成多页 pageStarts
* ✅ Locator：`scheme=txt.offset`，value=全局字符偏移（Int 字符串）
* ✅ Search：Flow 流式 `indexOf`，支持 maxHits/startFrom/caseSensitive/wholeWord（简单边界）
* ✅ Outline：可选章节识别（中文“第X章”、英文 Chapter）
* ✅ Annotations：**文档级内存存储**（跨 session 共享），支持 decorationsFor(range)

> ⚠️ 说明：`paragraphSpacingDp` / `hyphenation` 在 MVP 中未参与排版（否则会引入复杂 Spannable/段落样式系统，容易造成偏移不稳定）。你后续如果要严格支持段间距，我建议做“段落 token 化 + 布局器”那条路线。

```kotlin
package com.example.reader.txt

import com.example.reader.api.engine.ReaderDocument
import com.example.reader.api.engine.ReaderEngine
import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.open.OpenOptions
import com.example.reader.model.BookFormat
import com.example.reader.source.DocumentSource
import com.example.reader.txt.internal.doc.TxtDocument
import com.example.reader.txt.internal.open.TxtOpener
import com.example.reader.txt.internal.util.toReaderError
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

class TxtEngine(
    private val ioDispatcher: CoroutineDispatcher = Dispatchers.IO
) : ReaderEngine {

    override val supportedFormats: Set<BookFormat> = setOf(BookFormat.TXT)

    override suspend fun open(
        source: DocumentSource,
        options: OpenOptions
    ): ReaderResult<ReaderDocument> = withContext(ioDispatcher) {
        try {
            // 可防御：如果 runtime 误选了 format，也尽量给出合理错误
            if (options.hintFormat != null && options.hintFormat != BookFormat.TXT) {
                return@withContext ReaderResult.Err(
                    ReaderError.UnsupportedFormat(detected = options.hintFormat.name)
                )
            }

            val opener = TxtOpener()
            val opened = opener.open(source, options)

            when (opened) {
                is ReaderResult.Err -> opened
                is ReaderResult.Ok -> {
                    val document = TxtDocument(
                        source = source,
                        openOptions = options,
                        content = opened.value
                    )
                    ReaderResult.Ok(document)
                }
            }
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }
    }
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/txt/internal/open/EncodingDetector.kt
package com.example.reader.txt.internal.open

import java.nio.ByteBuffer
import java.nio.charset.Charset
import java.nio.charset.CodingErrorAction

internal class EncodingDetector(
    private val fallbackCandidates: List<String> = listOf(
        "UTF-8",
        "GB18030",
        "GBK",
        "UTF-16LE",
        "UTF-16BE",
        "ISO-8859-1"
    )
) {

    fun detect(bytes: ByteArray, hintName: String?): Charset {
        // 1) 用户指定（OpenOptions.textEncoding）
        if (!hintName.isNullOrBlank()) {
            runCatching { Charset.forName(hintName.trim()) }.getOrNull()?.let { return it }
        }

        // 2) BOM
        detectBom(bytes)?.let { return it }

        // 3) 尝试候选（优先 UTF-8）
        for (name in fallbackCandidates) {
            val cs = runCatching { Charset.forName(name) }.getOrNull() ?: continue
            if (canDecodeStrict(cs, bytes)) return cs
        }

        // 4) 兜底 UTF-8（尽量别返回 null）
        return Charsets.UTF_8
    }

    private fun detectBom(bytes: ByteArray): Charset? {
        if (bytes.size >= 3 &&
            bytes[0] == 0xEF.toByte() &&
            bytes[1] == 0xBB.toByte() &&
            bytes[2] == 0xBF.toByte()
        ) return Charsets.UTF_8

        if (bytes.size >= 2 &&
            bytes[0] == 0xFF.toByte() &&
            bytes[1] == 0xFE.toByte()
        ) return Charsets.UTF_16LE

        if (bytes.size >= 2 &&
            bytes[0] == 0xFE.toByte() &&
            bytes[1] == 0xFF.toByte()
        ) return Charsets.UTF_16BE

        return null
    }

    private fun canDecodeStrict(charset: Charset, bytes: ByteArray): Boolean {
        return runCatching {
            val decoder = charset.newDecoder()
                .onMalformedInput(CodingErrorAction.REPORT)
                .onUnmappableCharacter(CodingErrorAction.REPORT)
            decoder.decode(ByteBuffer.wrap(bytes))
            true
        }.getOrElse { false }
    }
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/txt/internal/open/TxtOpener.kt
package com.example.reader.txt.internal.open

import com.example.reader.api.error.ReaderResult
import com.example.reader.api.open.OpenOptions
import com.example.reader.source.DocumentSource
import com.example.reader.txt.internal.util.toReaderError
import java.io.ByteArrayOutputStream
import java.io.InputStream
import java.nio.charset.Charset
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

internal data class TxtContent(
    val text: String,
    val charset: Charset,
    val byteSize: Int
)

internal class TxtOpener(
    private val detector: EncodingDetector = EncodingDetector()
) {

    suspend fun open(
        source: DocumentSource,
        options: OpenOptions
    ): ReaderResult<TxtContent> = withContext(Dispatchers.IO) {
        try {
            val bytes = source.openInputStream().use { it.readAllBytesCompat() }
            val charset = detector.detect(bytes, options.textEncoding)

            val rawText = decode(bytes, charset)
            val normalized = normalize(rawText)

            ReaderResult.Ok(
                TxtContent(
                    text = normalized,
                    charset = charset,
                    byteSize = bytes.size
                )
            )
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }
    }

    private fun decode(bytes: ByteArray, charset: Charset): String {
        // Kotlin/Java 会处理 BOM->\uFEFF 的情况（部分情况下仍可能残留），后面 normalize 再清理
        return String(bytes, charset)
    }

    private fun normalize(input: String): String {
        var s = input

        // 去掉开头 BOM 字符（有些 decode 后会保留）
        if (s.isNotEmpty() && s[0] == '\uFEFF') {
            s = s.substring(1)
        }

        // 换行统一：\r\n 和 \r -> \n
        s = s.replace("\r\n", "\n").replace('\r', '\n')

        // tab -> 4 spaces（避免 layout 时 tab 宽度不可控）
        s = s.replace("\t", "    ")

        // 移除一些常见控制字符（保留 \n）
        s = buildString(s.length) {
            for (ch in s) {
                if (ch == '\n') {
                    append(ch)
                } else {
                    val code = ch.code
                    if (code in 0x00..0x08 || code in 0x0B..0x1F) {
                        // skip
                    } else {
                        append(ch)
                    }
                }
            }
        }

        return s
    }
}

private fun InputStream.readAllBytesCompat(): ByteArray {
    // Android 旧版本没有 InputStream.readAllBytes()
    val buffer = ByteArray(16 * 1024)
    val out = ByteArrayOutputStream()
    while (true) {
        val read = read(buffer)
        if (read <= 0) break
        out.write(buffer, 0, read)
    }
    return out.toByteArray()
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/txt/internal/locator/TxtLocatorMapper.kt
package com.example.reader.txt.internal.locator

import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import kotlin.math.roundToInt

internal class TxtLocatorMapper(
    private val textLength: Int
) {

    fun locatorForOffset(offset: Int): Locator {
        val clamped = offset.coerceIn(0, textLength)
        val prog = if (textLength == 0) 0.0 else clamped.toDouble() / textLength.toDouble()
        return Locator(
            scheme = LocatorSchemes.TXT_OFFSET,
            value = clamped.toString(),
            extras = mapOf(
                "progression" to prog.toString()
            )
        )
    }

    fun rangeForOffsets(startOffset: Int, endOffsetExclusive: Int): LocatorRange {
        val s = startOffset.coerceIn(0, textLength)
        val e = endOffsetExclusive.coerceIn(0, textLength)
        return LocatorRange(
            start = locatorForOffset(minOf(s, e)),
            end = locatorForOffset(maxOf(s, e))
        )
    }

    fun offsetOf(locator: Locator): Int? {
        if (locator.scheme != LocatorSchemes.TXT_OFFSET) return null
        return locator.value.toIntOrNull()
    }

    fun offsetOfOrClamp(locator: Locator, fallback: Int = 0): Int {
        val off = offsetOf(locator) ?: fallback
        return off.coerceIn(0, textLength)
    }

    fun offsetFromProgress(percent: Double): Int {
        if (textLength <= 0) return 0
        val p = percent.coerceIn(0.0, 1.0)
        return (p * textLength).roundToInt().coerceIn(0, textLength)
    }
}
```

```kotlin
package com.example.reader.txt.internal.paginate

internal data class PageMap(
    val pageStarts: IntArray,
    val configHash: Long,
    val textLength: Int
) {
    val pageCount: Int get() = pageStarts.size

    fun startOffsetOf(pageIndex: Int): Int =
        pageStarts[pageIndex.coerceIn(0, pageStarts.lastIndex)]

    fun endOffsetOf(pageIndex: Int): Int {
        val i = pageIndex.coerceIn(0, pageStarts.lastIndex)
        return if (i + 1 <= pageStarts.lastIndex) pageStarts[i + 1] else textLength
    }

    fun pageIndexForOffset(offset: Int): Int {
        val off = offset.coerceIn(0, textLength)
        // largest i such that pageStarts[i] <= off
        var lo = 0
        var hi = pageStarts.lastIndex
        var ans = 0
        while (lo <= hi) {
            val mid = (lo + hi) ushr 1
            val v = pageStarts[mid]
            if (v <= off) {
                ans = mid
                lo = mid + 1
            } else {
                hi = mid - 1
            }
        }
        return ans
    }
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/txt/internal/paginate/TxtPaginator.kt
package com.example.reader.txt.internal.paginate

import android.os.Build
import android.text.Layout
import android.text.StaticLayout
import android.text.TextPaint

internal data class TxtLayoutSpec(
    val contentWidthPx: Int,
    val contentHeightPx: Int,
    val paint: TextPaint,
    val lineSpacingAddPx: Float,
    val lineSpacingMult: Float
)

internal class TxtPaginator(
    private val baseChunkChars: Int = 50_000,
    private val maxChunkChars: Int = 300_000
) {

    fun buildPageMap(
        fullText: String,
        spec: TxtLayoutSpec,
        configHash: Long
    ): PageMap {
        val n = fullText.length
        if (n <= 0) {
            return PageMap(intArrayOf(0), configHash = configHash, textLength = 0)
        }

        val starts = ArrayList<Int>(1024)
        starts.add(0)

        var start = 0
        var guard = 0

        while (true) {
            guard++
            if (guard > 200_000) {
                // 防御性退出：避免极端情况下死循环
                break
            }

            val (chunkStarts, reachedEnd) = paginateFrom(fullText, start, spec)

            // chunkStarts[0] == start
            for (i in 1 until chunkStarts.size) {
                val s = chunkStarts[i]
                if (s > starts.last()) starts.add(s)
            }

            if (reachedEnd) break

            val nextStart = starts.last()
            if (nextStart <= start) {
                // 理论上不会发生；兜底前进 1 char
                start = (start + 1).coerceAtMost(n)
                if (start >= n) break
            } else {
                start = nextStart
            }
        }

        return PageMap(
            pageStarts = starts.toIntArray(),
            configHash = configHash,
            textLength = n
        )
    }

    private fun paginateFrom(
        fullText: String,
        startOffset: Int,
        spec: TxtLayoutSpec
    ): Pair<IntArray, Boolean> {
        val n = fullText.length
        if (startOffset >= n) return intArrayOf(n) to true

        var chunkSize = baseChunkChars.coerceAtMost(n - startOffset)

        while (true) {
            val end = (startOffset + chunkSize).coerceAtMost(n)
            val chunk = fullText.subSequence(startOffset, end)

            val layout = createLayout(chunk, spec)

            val starts = splitLayoutToPageStarts(layout, startOffset, spec.contentHeightPx)

            val reachedEnd = (end >= n) && (starts.size <= 1)
            val hasNewPage = starts.size > 1

            if (hasNewPage || end >= n) {
                return starts to reachedEnd
            }

            // 说明：chunk 太小，居然一页装得下（非常罕见，但要避免死循环）
            val next = (chunkSize * 2).coerceAtMost(maxChunkChars).coerceAtMost(n - startOffset)
            if (next <= chunkSize) {
                // 已经无法增长了，只能当作“剩余装得下”处理（会在 end>=n 时返回）
                return starts to true
            }
            chunkSize = next
        }
    }

    private fun splitLayoutToPageStarts(
        layout: StaticLayout,
        globalStartOffset: Int,
        pageHeightPx: Int
    ): IntArray {
        val list = ArrayList<Int>(16)
        list.add(globalStartOffset)

        var pageStartLine = 0
        val lineCount = layout.lineCount

        for (line in 0 until lineCount) {
            val top = layout.getLineTop(pageStartLine)
            val bottom = layout.getLineBottom(line)
            if (bottom - top > pageHeightPx) {
                val newStartInChunk = layout.getLineStart(line)
                val newGlobal = globalStartOffset + newStartInChunk
                if (newGlobal > list.last()) {
                    list.add(newGlobal)
                    pageStartLine = line
                } else {
                    // 防御：避免 newStart == oldStart 导致死循环
                    val forced = (list.last() + 1)
                    if (forced <= globalStartOffset + layout.text.length) {
                        list.add(forced)
                        pageStartLine = line
                    }
                }
            }
        }

        return list.toIntArray()
    }

    private fun createLayout(text: CharSequence, spec: TxtLayoutSpec): StaticLayout {
        val width = spec.contentWidthPx.coerceAtLeast(1)

        return if (Build.VERSION.SDK_INT >= 23) {
            StaticLayout.Builder.obtain(text, 0, text.length, spec.paint, width)
                .setAlignment(Layout.Alignment.ALIGN_NORMAL)
                .setIncludePad(false)
                .setLineSpacing(spec.lineSpacingAddPx, spec.lineSpacingMult)
                .build()
        } else {
            @Suppress("DEPRECATION")
            StaticLayout(
                text,
                spec.paint,
                width,
                Layout.Alignment.ALIGN_NORMAL,
                spec.lineSpacingMult,
                spec.lineSpacingAddPx,
                false
            )
        }
    }
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/txt/internal/paginate/TxtTextMapping.kt
package com.example.reader.txt.internal.paginate

import com.example.reader.api.render.TextMapping
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes

internal class TxtTextMapping(
    private val pageStartOffset: Int,
    private val pageEndOffsetExclusive: Int
) : TextMapping {

    override fun locatorAt(charIndex: Int): Locator {
        val global = (pageStartOffset + charIndex)
            .coerceIn(pageStartOffset, pageEndOffsetExclusive)
        return Locator(LocatorSchemes.TXT_OFFSET, global.toString())
    }

    override fun rangeFor(startChar: Int, endChar: Int): LocatorRange {
        val s = startChar.coerceAtLeast(0)
        val e = endChar.coerceAtLeast(0)
        return LocatorRange(
            start = locatorAt(minOf(s, e)),
            end = locatorAt(maxOf(s, e))
        )
    }

    override fun charRangeFor(range: LocatorRange): IntRange? {
        val s = range.start.takeIf { it.scheme == LocatorSchemes.TXT_OFFSET }?.value?.toIntOrNull() ?: return null
        val e = range.end.takeIf { it.scheme == LocatorSchemes.TXT_OFFSET }?.value?.toIntOrNull() ?: return null

        // 要求完全落在当前页
        if (s < pageStartOffset || e > pageEndOffsetExclusive) return null

        val localStart = (s - pageStartOffset).coerceAtLeast(0)
        val localEndExclusive = (e - pageStartOffset).coerceAtLeast(localStart)

        // IntRange 是 inclusive，所以用 until
        return localStart until localEndExclusive
    }
}
```

```kotlin
import java.util.Locale

internal class TxtDocument(
    val source: DocumentSource,
    override val openOptions: OpenOptions,
    val content: TxtContent
) : ReaderDocument {

    override val id: DocumentId = DocumentId("txt:${source.uri}")

    override val format: BookFormat = BookFormat.TXT

    override val capabilities: DocumentCapabilities = DocumentCapabilities(
        reflowable = true,
        fixedLayout = false,
        outline = true,          // 我们提供启发式目录
        search = true,
        textExtraction = true,
        annotations = true,      // 内存 annotations（后续可替换为持久化桥接）
        links = false
    )

    private val annotationStore = TxtAnnotationStore()

    @Volatile
    private var metadataCache: DocumentMetadata? = null

    @Volatile
    private var outlineCache: List<com.example.reader.model.OutlineNode>? = null

    override suspend fun metadata(): ReaderResult<DocumentMetadata> {
        val cached = metadataCache
        if (cached != null) return ReaderResult.Ok(cached)

        val title = source.displayName
            ?.trim()
            ?.removeSuffix(".txt")
            ?.removeSuffix(".TXT")
            ?.takeIf { it.isNotBlank() }

        val meta = DocumentMetadata(
            title = title ?: "Untitled",
            author = null,
            language = guessLanguageFromCharsetName(content.charset.name()),
            identifier = null,
            extra = mapOf(
                "uri" to source.uri.toString(),
                "displayName" to (source.displayName ?: ""),
                "mimeType" to (source.mimeType ?: ""),
                "charset" to content.charset.name(),
                "byteSize" to content.byteSize.toString()
            )
        )

        metadataCache = meta
        return ReaderResult.Ok(meta)
    }

    override suspend fun createSession(
        initialLocator: Locator?,
        initialConfig: RenderConfig
    ): ReaderResult<ReaderSession> {
        // TXT 只支持 ReflowText
        val config = (initialConfig as? RenderConfig.ReflowText) ?: RenderConfig.ReflowText()

        return ReaderResult.Ok(
            TxtSession(
                document = this,
                initialLocator = initialLocator,
                initialConfig = config,
                annotationStore = annotationStore,
                outlineProvider = TxtOutlineProvider(
                    text = content.text,
                    cached = { outlineCache },
                    cache = { outlineCache = it }
                )
            )
        )
    }

    override fun close() {
        // 当前实现：全文 String 在内存，文档关闭时不需要释放 native 资源
        // 后续如果做大文件模式/索引缓存，可以在这里清理
    }

    private fun guessLanguageFromCharsetName(name: String): String? {
        val n = name.lowercase(Locale.US)
        return when {
            n.contains("gb") -> "zh"
            n.contains("big5") -> "zh"
            n.contains("utf") -> null
            else -> null
        }
    }
}
```

```kotlin
internal class TxtController(
    private val documentId: DocumentId,
    private val fullText: String,
    initialLocator: Locator?,
    initialConfig: RenderConfig.ReflowText,
    private val locatorMapper: TxtLocatorMapper,
    private val annotationStore: TxtAnnotationStore
) : ReaderController {

    private val scope = CoroutineScope(SupervisorJob() + Dispatchers.Default)

    private val mutex = Mutex()

    private var constraints: LayoutConstraints? = null
    private var config: RenderConfig.ReflowText = initialConfig

    private var currentOffset: Int = initialLocator?.let { locatorMapper.offsetOfOrClamp(it, 0) } ?: 0

    private var pageMap: PageMap? = null
    private val paginator = TxtPaginator()

    private val pageCache = com.example.reader.txt.internal.cache.PageLruCache<String, RenderPage>(maxSize = 12)

    private val _events = MutableSharedFlow<ReaderEvent>(
        extraBufferCapacity = 32
    )
    override val events: Flow<ReaderEvent> = _events.asSharedFlow()

    private val _state = MutableStateFlow(
        RenderState(
            locator = locatorMapper.locatorForOffset(currentOffset),
            progression = Progression(percent = 0.0),
            nav = NavigationAvailability(canGoPrev = false, canGoNext = fullText.isNotEmpty()),
            titleInView = null,
            config = config
        )
    )
    override val state: StateFlow<RenderState> = _state

    override suspend fun setLayoutConstraints(constraints: LayoutConstraints): ReaderResult<Unit> =
        mutex.withLock {
            this.constraints = constraints
            pageMap = null
            pageCache.clear()
            ReaderResult.Ok(Unit)
        }

    override suspend fun setConfig(config: RenderConfig): ReaderResult<Unit> =
        mutex.withLock {
            val reflow = config as? RenderConfig.ReflowText
                ?: return@withLock ReaderResult.Err(
                    ReaderError.Internal("TXT engine supports RenderConfig.ReflowText only")
                )
            this.config = reflow
            pageMap = null
            pageCache.clear()
            _state.value = _state.value.copy(config = reflow)
            ReaderResult.Ok(Unit)
        }

    override suspend fun render(policy: RenderPolicy): ReaderResult<RenderPage> =
        renderAtCurrent(policy = policy, emitPageChanged = false)

    override suspend fun next(policy: RenderPolicy): ReaderResult<RenderPage> =
        moveBy(deltaPages = +1, policy = policy)

    override suspend fun prev(policy: RenderPolicy): ReaderResult<RenderPage> =
        moveBy(deltaPages = -1, policy = policy)

    override suspend fun goTo(locator: Locator, policy: RenderPolicy): ReaderResult<RenderPage> {
        return try {
            mutex.withLock {
                currentOffset = locatorMapper.offsetOfOrClamp(locator, currentOffset)
            }
            renderAtCurrent(policy = policy, emitPageChanged = true)
        } catch (t: Throwable) {
            val err = t.toReaderError()
            _events.tryEmit(ReaderEvent.Error(err))
            ReaderResult.Err(err)
        }
    }

    override suspend fun goToProgress(percent: Double, policy: RenderPolicy): ReaderResult<RenderPage> {
        return try {
            val off = locatorMapper.offsetFromProgress(percent)
            mutex.withLock { currentOffset = off }
            renderAtCurrent(policy = policy, emitPageChanged = true)
        } catch (t: Throwable) {
            val err = t.toReaderError()
            _events.tryEmit(ReaderEvent.Error(err))
            ReaderResult.Err(err)
        }
    }

    override suspend fun prefetchNeighbors(count: Int): ReaderResult<Unit> {
        return try {
            val map = ensurePageMap()
            val idx = map.pageIndexForOffset(currentOffset)
            val from = (idx - count).coerceAtLeast(0)
            val to = (idx + count).coerceAtMost(map.pageCount - 1)

            for (i in from..to) {
                // 当前页也可以预热缓存（不影响）
                renderPageAt(map, i, allowCache = true)
            }

            ReaderResult.Ok(Unit)
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }
    }

    override suspend fun invalidate(reason: InvalidateReason): ReaderResult<Unit> =
        mutex.withLock {
            // TXT 内容不太会变，但 config/layout 会变
            pageMap = null
            pageCache.clear()
            ReaderResult.Ok(Unit)
        }

    override fun close() {
        runCatching { scope.cancel() }
        runCatching { pageCache.clear() }
    }

    // -----------------------------
    // internal helpers
    // -----------------------------

    private suspend fun moveBy(deltaPages: Int, policy: RenderPolicy): ReaderResult<RenderPage> {
        return try {
            val map = ensurePageMap()
            val idx = map.pageIndexForOffset(currentOffset)
            val nextIdx = (idx + deltaPages).coerceIn(0, map.pageCount - 1)
            mutex.withLock { currentOffset = map.startOffsetOf(nextIdx) }
            renderAtCurrent(policy = policy, emitPageChanged = true)
        } catch (t: Throwable) {
            val err = t.toReaderError()
            _events.tryEmit(ReaderEvent.Error(err))
            ReaderResult.Err(err)
        }
    }

    private suspend fun renderAtCurrent(
        policy: RenderPolicy,
        emitPageChanged: Boolean
    ): ReaderResult<RenderPage> {
        return try {
            val map = ensurePageMap()
            val idx = map.pageIndexForOffset(currentOffset)
            val pageStart = map.startOffsetOf(idx)

            // 将 currentOffset 对齐到页起点（更符合“页”为单位）
            mutex.withLock { currentOffset = pageStart }

            val pageResult = renderPageAt(map, idx, allowCache = policy.allowCache)

            // 更新 state
            val newState = buildState(map, idx)
            _state.value = newState

            if (emitPageChanged) {
                _events.tryEmit(ReaderEvent.PageChanged(newState.locator))
            }

            // 发送 Rendered 事件
            _events.tryEmit(ReaderEvent.Rendered(pageResult.id, pageResult.metrics))

            // 预取邻页
            if (policy.prefetchNeighbors > 0) {
                prefetchNeighbors(policy.prefetchNeighbors)
            }

            ReaderResult.Ok(pageResult)
        } catch (t: Throwable) {
            val err = t.toReaderError()
            _events.tryEmit(ReaderEvent.Error(err))
            ReaderResult.Err(err)
        }
    }

    private fun buildState(map: PageMap, pageIndex: Int): RenderState {
        val start = map.startOffsetOf(pageIndex)
        val percent = if (map.textLength == 0) 0.0 else start.toDouble() / map.textLength.toDouble()

        return RenderState(
            locator = locatorMapper.locatorForOffset(start),
            progression = Progression(
                percent = percent.coerceIn(0.0, 1.0),
                label = "${pageIndex + 1}/${map.pageCount}",
                current = pageIndex + 1,
                total = map.pageCount
            ),
            nav = NavigationAvailability(
                canGoPrev = pageIndex > 0,
                canGoNext = pageIndex < map.pageCount - 1
            ),
            titleInView = null,
            config = config
        )
    }

    private suspend fun ensurePageMap(): PageMap {
        val existing = pageMap
        val spec = buildLayoutSpec()
        val hash = configHash(spec, config)

        if (existing != null && existing.configHash == hash && existing.textLength == fullText.length) {
            return existing
        }

        return mutex.withLock {
            val again = pageMap
            if (again != null && again.configHash == hash && again.textLength == fullText.length) {
                return@withLock again
            }

            val built = withContext(Dispatchers.Default) {
                paginator.buildPageMap(
                    fullText = fullText,
                    spec = spec,
                    configHash = hash
                )
            }
            pageMap = built
            built
        }
    }

    private fun buildLayoutSpec(): TxtLayoutSpec {
        val c = constraints ?: throw ReaderError.Internal("LayoutConstraints not set for TXT controller")

        val density = c.density
        val fontScale = c.fontScale

        val paddingPx = (config.pagePaddingDp * density).roundToInt().coerceAtLeast(0)
        val contentWidth = (c.viewportWidthPx - paddingPx * 2).coerceAtLeast(1)
        val contentHeight = (c.viewportHeightPx - paddingPx * 2).coerceAtLeast(1)

        val paint = TextPaint(TextPaint.ANTI_ALIAS_FLAG).apply {
            val textSizePx = config.fontSizeSp * density * fontScale
            textSize = textSizePx

            // 可选：字体族
            if (!config.fontFamilyName.isNullOrBlank()) {
                runCatching { typeface = Typeface.create(config.fontFamilyName, Typeface.NORMAL) }
            }
        }

        // MVP：段间距不单独处理；lineHeightMult 用 StaticLayout line spacing
        return TxtLayoutSpec(
            contentWidthPx = contentWidth,
            contentHeightPx = contentHeight,
            paint = paint,
            lineSpacingAddPx = 0f,
            lineSpacingMult = config.lineHeightMult.coerceAtLeast(1.0f)
        )
    }

    private fun configHash(spec: TxtLayoutSpec, config: RenderConfig.ReflowText): Long {
        var h = 17
        h = 31 * h + spec.contentWidthPx
        h = 31 * h + spec.contentHeightPx
        h = 31 * h + spec.paint.textSize.hashCode()
        h = 31 * h + config.lineHeightMult.hashCode()
        h = 31 * h + (config.fontFamilyName?.hashCode() ?: 0)
        h = 31 * h + config.pagePaddingDp.hashCode()
        h = 31 * h + config.fontSizeSp.hashCode()
        return h.toLong()
    }

    private suspend fun renderPageAt(
        map: PageMap,
        pageIndex: Int,
        allowCache: Boolean
    ): RenderPage {
        val idx = pageIndex.coerceIn(0, map.pageCount - 1)
        val start = map.startOffsetOf(idx)
        val end = map.endOffsetOf(idx)

        val pageId = "txt:${documentId.value}:${map.configHash}:$idx"

        if (allowCache) {
            pageCache.get(pageId)?.let { cached ->
                return cached.copy(
                    metrics = RenderMetrics(renderTimeMs = 0L, cacheHit = true)
                )
            }
        }

        val ms = measureTimeMillis {
            // no-op，计时用
        }

        val renderStart = System.currentTimeMillis()

        val pageText = fullText.subSequence(start, end)
        val mapping = TxtTextMapping(start, end)

        // Decorations：按当前页范围过滤（reflow range）
        val pageRange = locatorMapper.rangeForOffsets(start, end)
        val decorations = annotationStore.decorationsForRange(pageRange)

        val page = RenderPage(
            id = PageId(pageId),
            locator = locatorMapper.locatorForOffset(start),
            content = com.example.reader.api.render.RenderContent.Text(
                text = pageText,
                mapping = mapping
            ),
            links = emptyList(),
            decorations = decorations,
            metrics = RenderMetrics(
                renderTimeMs = (System.currentTimeMillis() - renderStart).coerceAtLeast(0L),
                cacheHit = false
            )
        )

        if (allowCache) {
            pageCache.put(pageId, page)
        }

        return page
    }
}
```

```kotlin
package com.example.reader.txt.internal.provider

import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.TextProvider
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import com.example.reader.txt.internal.locator.TxtLocatorMapper

internal class TxtTextProvider(
    private val text: String,
    private val locatorMapper: TxtLocatorMapper
) : TextProvider {

    override suspend fun getText(range: LocatorRange): ReaderResult<String> {
        val s = parseOffset(range.start) ?: return ReaderResult.Err(ReaderError.Internal("Invalid range.start for TXT"))
        val e = parseOffset(range.end) ?: return ReaderResult.Err(ReaderError.Internal("Invalid range.end for TXT"))

        val start = minOf(s, e).coerceIn(0, text.length)
        val end = maxOf(s, e).coerceIn(0, text.length)

        return ReaderResult.Ok(text.substring(start, end))
    }

    override suspend fun getTextAround(locator: Locator, maxChars: Int): ReaderResult<String> {
        val off = locatorMapper.offsetOfOrClamp(locator, 0)
        if (text.isEmpty()) return ReaderResult.Ok("")

        val half = (maxChars.coerceAtLeast(16) / 2)
        val start = (off - half).coerceAtLeast(0)
        val end = (start + maxChars).coerceAtMost(text.length)
        return ReaderResult.Ok(text.substring(start, end))
    }

    private fun parseOffset(locator: Locator): Int? {
        if (locator.scheme != LocatorSchemes.TXT_OFFSET) return null
        return locator.value.toIntOrNull()
    }
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/txt/internal/provider/TxtSearchProvider.kt
package com.example.reader.txt.internal.provider

import com.example.reader.api.provider.SearchHit
import com.example.reader.api.provider.SearchOptions
import com.example.reader.api.provider.SearchProvider
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.txt.internal.locator.TxtLocatorMapper
import kotlinx.coroutines.currentCoroutineContext
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.flow

internal class TxtSearchProvider(
    private val text: String,
    private val locatorMapper: TxtLocatorMapper
) : SearchProvider {

    override fun search(query: String, options: SearchOptions): Flow<SearchHit> = flow {
        if (query.isBlank() || text.isEmpty()) return@flow

        val ignoreCase = !options.caseSensitive
        val q = query

        val startOffset = options.startFrom?.let { locatorMapper.offsetOfOrClamp(it, 0) } ?: 0
        val maxHits = options.maxHits.coerceAtLeast(1)

        var from = startOffset.coerceIn(0, text.length)
        var hits = 0

        val ctx = currentCoroutineContext()

        while (from < text.length) {
            ctx.ensureActive()

            val idx = text.indexOf(q, startIndex = from, ignoreCase = ignoreCase)
            if (idx < 0) break

            val end = idx + q.length

            if (!options.wholeWord || isWholeWordHit(text, idx, end)) {
                val range = LocatorRange(
                    start = locatorMapper.locatorForOffset(idx),
                    end = locatorMapper.locatorForOffset(end)
                )

                val excerpt = buildExcerpt(text, idx, end)

                emit(
                    SearchHit(
                        range = range,
                        excerpt = excerpt,
                        sectionTitle = null
                    )
                )

                hits++
                if (hits >= maxHits) break
            }

            // 防止卡住：至少前进 1
            from = (idx + 1).coerceAtMost(text.length)
        }
    }

    private fun buildExcerpt(text: String, start: Int, end: Int): String {
        val before = 40
        val after = 40
        val s = (start - before).coerceAtLeast(0)
        val e = (end + after).coerceAtMost(text.length)
        return text.substring(s, e)
            .replace('\n', ' ')
            .trim()
    }

    private fun isWholeWordHit(text: String, start: Int, end: Int): Boolean {
        fun isWordChar(ch: Char): Boolean = ch.isLetterOrDigit() || ch == '_'

        val leftOk = if (start <= 0) true else !isWordChar(text[start - 1])
        val rightOk = if (end >= text.length) true else !isWordChar(text[end])
        return leftOk && rightOk
    }
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/txt/internal/provider/TxtOutlineProvider.kt
package com.example.reader.txt.internal.provider

import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.OutlineProvider
import com.example.reader.model.OutlineNode
import com.example.reader.txt.internal.locator.TxtLocatorMapper
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

internal class TxtOutlineProvider(
    private val text: String,
    private val cached: () -> List<OutlineNode>?,
    private val cache: (List<OutlineNode>) -> Unit
) : OutlineProvider {

    // 常见章节识别（可按产品扩展）
    private val cnChapter = Regex("""^\s*第[一二三四五六七八九十百千万0-9]+[章回节].*$""")
    private val enChapter = Regex("""^\s*(chapter|CHAPTER)\s+\d+.*$""")

    override suspend fun getOutline(): ReaderResult<List<OutlineNode>> = withContext(Dispatchers.Default) {
        cached()?.let { return@withContext ReaderResult.Ok(it) }

        val nodes = ArrayList<OutlineNode>(128)
        val mapper = TxtLocatorMapper(text.length)

        // 扫描行：不 split，避免内存峰值
        var lineStart = 0
        var i = 0
        val n = text.length

        fun onLine(lineEndExclusive: Int) {
            val line = text.substring(lineStart, lineEndExclusive).trim()
            if (line.isEmpty()) return

            if (cnChapter.matches(line) || enChapter.matches(line)) {
                nodes.add(
                    OutlineNode(
                        title = line.take(80),
                        locator = mapper.locatorForOffset(lineStart)
                    )
                )
            }
        }

        while (i < n) {
            val ch = text[i]
            if (ch == '\n') {
                onLine(i)
                lineStart = i + 1
            }
            i++
            if (nodes.size >= 500) break
        }
        if (lineStart < n && nodes.size < 500) {
            onLine(n)
        }

        cache(nodes)
        ReaderResult.Ok(nodes)
    }
}
```

```kotlin
package com.example.reader.txt.internal.annotations

import com.example.reader.api.annotation.Decoration
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.annotation.Annotation
import com.example.reader.model.annotation.AnnotationAnchor
import com.example.reader.model.annotation.AnnotationDraft

import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock
import java.util.UUID

internal class TxtAnnotationStore {

    private val mutex = Mutex()
    private val _all = MutableStateFlow<List<Annotation>>(emptyList())
    val all: StateFlow<List<Annotation>> = _all

    suspend fun listAll(): List<Annotation> = _all.value

    suspend fun create(draft: AnnotationDraft): Annotation {
        val now = System.currentTimeMillis()
        val ann = Annotation(
            id = AnnotationId(UUID.randomUUID().toString()),
            type = draft.type,
            anchor = draft.anchor,
            content = draft.content,
            style = draft.style,
            createdAtEpochMs = now,
            updatedAtEpochMs = now,
            extra = draft.extra
        )
        mutex.withLock {
            _all.value = _all.value + ann
        }
        return ann
    }

    suspend fun update(annotation: Annotation) {
        mutex.withLock {
            _all.value = _all.value.map { if (it.id == annotation.id) annotation else it }
        }
    }

    suspend fun delete(id: AnnotationId) {
        mutex.withLock {
            _all.value = _all.value.filterNot { it.id == id }
        }
    }

    suspend fun queryByRange(range: LocatorRange): List<Annotation> {
        val (qs, qe) = parseOffsets(range) ?: return emptyList()
        val a = minOf(qs, qe)
        val b = maxOf(qs, qe)

        return _all.value.filter { ann ->
            val r = (ann.anchor as? AnnotationAnchor.ReflowRange)?.range ?: return@filter false
            val (s, e) = parseOffsets(r) ?: return@filter false
            val x = minOf(s, e)
            val y = maxOf(s, e)
            rangesIntersect(a, b, x, y)
        }
    }

    fun decorationsForRange(range: LocatorRange): List<Decoration> {
        val (qs, qe) = parseOffsets(range) ?: return emptyList()
        val a = minOf(qs, qe)
        val b = maxOf(qs, qe)

        return _all.value.mapNotNull { ann ->
            if (ann.type.name == "BOOKMARK") return@mapNotNull null

            val r = (ann.anchor as? AnnotationAnchor.ReflowRange)?.range ?: return@mapNotNull null
            val (s, e) = parseOffsets(r) ?: return@mapNotNull null
            val x = minOf(s, e)
            val y = maxOf(s, e)

            if (!rangesIntersect(a, b, x, y)) return@mapNotNull null

            Decoration.Reflow(range = r, style = ann.style)
        }
    }

    private fun parseOffsets(range: LocatorRange): Pair<Int, Int>? {
        val s = range.start.takeIf { it.scheme == LocatorSchemes.TXT_OFFSET }?.value?.toIntOrNull() ?: return null
        val e = range.end.takeIf { it.scheme == LocatorSchemes.TXT_OFFSET }?.value?.toIntOrNull() ?: return null
        return s to e
    }

    private fun rangesIntersect(aStart: Int, aEnd: Int, bStart: Int, bEnd: Int): Boolean {
        // treat as [start, end) 半开区间更合理，但这里做近似：只要有重叠
        return aStart < bEnd && bStart < aEnd
    }
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/txt/internal/cache/PageLruCache.kt
package com.example.reader.txt.internal.cache

internal class PageLruCache<K, V>(private val maxSize: Int) {

    private val map = object : LinkedHashMap<K, V>(maxSize, 0.75f, true) {
        override fun removeEldestEntry(eldest: MutableMap.MutableEntry<K, V>?): Boolean {
            return size > maxSize
        }
    }

    fun get(key: K): V? = synchronized(map) { map[key] }

    fun put(key: K, value: V) {
        synchronized(map) { map[key] = value }
    }

    fun clear() {
        synchronized(map) { map.clear() }
    }
}
```

## 12) 错误映射（引擎内）：toReaderError.kt

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/txt/internal/util/ReaderErrors.kt
package com.example.reader.txt.internal.util

import com.example.reader.api.error.ReaderError
import java.io.FileNotFoundException
import java.io.IOException
import kotlinx.coroutines.CancellationException

internal fun Throwable.toReaderError(): ReaderError =
    when (this) {
        is ReaderError -> this
        is CancellationException -> ReaderError.Cancelled(cause = this)
        is FileNotFoundException -> ReaderError.NotFound(cause = this)
        is SecurityException -> ReaderError.PermissionDenied(cause = this)
        is IOException -> ReaderError.Io(cause = this)
        else -> ReaderError.Internal(message = message, cause = this)
    }
```
