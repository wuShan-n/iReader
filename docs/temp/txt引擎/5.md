在 `setLayoutConstraints()` 里，`ensureRenderBucketLocked()` 之后加恢复逻辑（只恢复一次/每个 key 一次）：

```kotlin
override suspend fun setLayoutConstraints(constraints: LayoutConstraints): ReaderResult<Unit> =
    mutex.withLock {
        this.constraints = constraints
        cache.clear()
        currentSlice = null

        totalCharsCache = -1
        val total = totalCharsLocked()
        currentStartChar = if (total == 0) 0 else currentStartChar.coerceIn(0, total - 1)

        ensureRenderBucketLocked()
        val key = renderKey

        // ✅ 若业务没显式传 initialLocator，则从 last-position 恢复
        if (!explicitInitial && key != null && restoredForKey != key) {
            restoredForKey = key
            val restored = lastPositionStore.load(key) // suspend，内部 IO
            if (restored != null && total > 0) {
                currentStartChar = restored.coerceIn(0, total - 1)
                // 确保分页桶包含起点
                pageStarts.seedIfEmpty(0, currentStartChar)
                pageStarts.addStart(currentStartChar)
            }
        }

        ReaderResult.Ok(Unit)
    }
```

在 `renderLocked()` 里，更新 state 后（或者 slice 已经确定后）加节流保存：

```kotlin
private data class TitleInfo(val title: String, val level: Int)

private fun extractTitleWithLevel(line: String): TitleInfo? {
    if (line.isBlank()) return null

    // Markdown：# ~ ######
    val md = Regex("""^\s*(#{1,6})\s+(.+)$""").matchEntire(line)
    if (md != null) return TitleInfo(md.groupValues[2].trim(), md.groupValues[1].length)

    // 中文：第X章/节/回/卷/部/篇（章/卷/部/篇=1，节=2，其它默认1）
    chineseChapter.matchEntire(line)?.let { m ->
        val head = m.groupValues[1].trim()
        val tail = m.groupValues.getOrNull(2)?.trim().orEmpty()
        val title = if (tail.isNotEmpty()) "$head $tail" else head
        val level = when {
            head.contains("节") -> 2
            else -> 1
        }
        return TitleInfo(title, level)
    }

    // 英文 CHAPTER -> level 1
    if (enChapter.matches(line)) return TitleInfo(line.trim(), 1)

    // 数字标题：1. / 1.2 / 1.2.3 -> level=段数
    val num = Regex("""^\s*(\d+(?:\.\d+)*)\s*[\.\、\)]\s+(.+)$""").matchEntire(line)
    if (num != null) {
        val seg = num.groupValues[1]
        val level = seg.count { it == '.' } + 1
        return TitleInfo(line.trim(), level.coerceIn(1, 6))
    }

    return null
}
```

```kotlin
import java.security.MessageDigest

internal object KeyHash {
    fun stableName(key: RenderKey): String {
        val s = key.toString()
        val md = MessageDigest.getInstance("SHA-256").digest(s.toByteArray(Charsets.UTF_8))
        return buildString(20) {
            for (i in 0 until 10) {
                val b = md[i].toInt() and 0xFF
                append("0123456789abcdef"[b ushr 4])
                append("0123456789abcdef"[b and 0x0F])
            }
        }
    }
}
```

### 2) 新增 `TxtLastPositionStore.kt`

路径：`engines/txt/.../internal/paging/TxtLastPositionStore.kt`

```kotlin
package com.example.reader.engines.txt.internal.paging

import com.example.reader.engines.txt.TxtEngineConfig
import java.io.File
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

internal class TxtLastPositionStore(
    private val config: TxtEngineConfig,
    private val docNamespace: String,
    private val charsetName: String,
    private val ioDispatcher: CoroutineDispatcher = Dispatchers.IO
) {
    val minIntervalMs: Long = config.lastPositionMinIntervalMs

    suspend fun load(key: RenderKey): Int? = withContext(ioDispatcher) {
        if (!config.persistLastPosition) return@withContext null
        val f = fileFor(key) ?: return@withContext null
        if (!f.exists()) return@withContext null
        runCatching { decodeVarInt(f.readBytes()) }.getOrNull()
    }

    fun save(key: RenderKey, startChar: Int, force: Boolean = false) {
        if (!config.persistLastPosition) return
        val f = fileFor(key) ?: return
        runCatching {
            f.parentFile?.mkdirs()
            val tmp = File(f.parentFile, f.name + ".tmp")
            tmp.writeBytes(encodeVarInt(startChar.coerceAtLeast(0)))
            if (f.exists()) f.delete()
            tmp.renameTo(f)
        }
    }

    private fun fileFor(key: RenderKey): File? {
        val base = config.cacheDir ?: return null
        val dir = File(base, "reader-txt/lastpos/${docNamespace.hashCode()}_${charsetName.hashCode()}")
        val name = KeyHash.stableName(key) + ".pos"
        return File(dir, name)
    }

    private fun encodeVarInt(value: Int): ByteArray {
        var v = value
        val out = ArrayList<Byte>(5)
        while (true) {
            val b = v and 0x7F
            v = v ushr 7
            if (v == 0) {
                out.add(b.toByte())
                break
            } else {
                out.add((b or 0x80).toByte())
            }
        }
        return out.toByteArray()
    }

    private fun decodeVarInt(bytes: ByteArray): Int {
        var result = 0
        var shift = 0
        var i = 0
        while (i < bytes.size) {
            val b = bytes[i].toInt() and 0xFF
            result = result or ((b and 0x7F) shl shift)
            i++
            if ((b and 0x80) == 0) break
            shift += 7
        }
        return result
    }
}
```

### 3) 新增 `TxtOutlineTreeBuilder.kt`

路径：`engines/txt/.../internal/provider/TxtOutlineTreeBuilder.kt`

```kotlin
package com.example.reader.engines.txt.internal.provider

import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.OutlineNode
import java.util.ArrayDeque

internal class TxtOutlineTreeBuilder(
    private val asTree: Boolean
) {
    private data class MutableNode(
        val level: Int,
        val title: String,
        val startChar: Int,
        val children: MutableList<MutableNode> = mutableListOf()
    )

    private val roots = mutableListOf<MutableNode>()
    private val stack = ArrayDeque<MutableNode>()

    fun add(level: Int, title: String, startChar: Int) {
        val lvl = level.coerceIn(1, 6)
        val node = MutableNode(lvl, title, startChar)

        if (!asTree) {
            roots.add(node)
            return
        }

        while (stack.isNotEmpty() && stack.last().level >= lvl) {
            stack.removeLast()
        }

        if (stack.isEmpty()) {
            roots.add(node)
        } else {
            stack.last().children.add(node)
        }
        stack.addLast(node)
    }

    fun build(): List<OutlineNode> =
        roots.map { it.toOutlineNode() }

    private fun MutableNode.toOutlineNode(): OutlineNode =
        OutlineNode(
            title = title,
            locator = Locator(LocatorSchemes.TXT_OFFSET, startChar.toString()),
            children = children.map { it.toOutlineNode() }
        )
}
```

## 你现在的行为变化（对 UI 非侵入）

* 如果业务没传 `initialLocator`：TXT 会优先恢复“上次阅读位置”（同 RenderKey 下精准恢复到页起点）
* 如果业务传了 `initialLocator`：以业务为准，不会被恢复逻辑覆盖
* 目录现在是树形（如果 `outlineAsTree=true`），UI 直接递归渲染 `OutlineNode.children` 即可
