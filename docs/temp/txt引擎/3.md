本轮优化目标：

* ✅ **分页结果跨 Session 复用（同一 Document 内）**：同一本书同一套布局/字体参数，二次打开/返回阅读页不再从头“重新摸索”页起点。
* ✅ **加入 TXT 目录（OutlineProvider）**：按常见章节/标题规则扫描生成目录，支持跳转。
* ✅ **分页索引按 RenderKey 分桶**：`constraints + config + charset + docId` 不同就分开缓存，避免错页。
* ✅ **PageStartsIndex 增强**：支持 empty 检查与初始化 seed，便于复用。

> 说明：这里的“持久化”是 **Document 生命周期内复用**（同进程）。如果你要跨进程持久化到文件，需要 `EngineEnvironment.cacheDir` 之类的注入口（你目前 TXT engine 还没这个环境接口，所以我先给不破坏结构的版本）。

private fun ensureRenderBucketLocked() {
    val c = constraints ?: return
    val key = RenderKey.of(
        docId = documentId.value,
        charset = store.charset.name(),
        constraints = c,
        config = config
    )
    if (key != renderKey) {
        renderKey = key
        pageStarts = paginationStore.getOrCreate(key)
        // ✅ seed：确保至少有 0 与当前起点（避免空表）
        pageStarts.seedIfEmpty(0, currentStartChar)
    } else {
        pageStarts.seedIfEmpty(0, currentStartChar)
    }
}

### 3.3 修改 `setLayoutConstraints()`：切桶 + seed + clamp

override suspend fun setLayoutConstraints(constraints: LayoutConstraints): ReaderResult<Unit> =
    mutex.withLock {
        this.constraints = constraints
        cache.clear()
        currentSlice = null

        totalCharsCache = -1
        val total = totalCharsLocked()
        currentStartChar = if (total == 0) 0 else currentStartChar.coerceIn(0, total - 1)

        ensureRenderBucketLocked()
        ReaderResult.Ok(Unit)
    }

override suspend fun setConfig(config: RenderConfig): ReaderResult<Unit> =
    mutex.withLock {
        val reflow = when (config) {
            is RenderConfig.ReflowText -> config
            else -> RenderConfig.ReflowText()
        }
        this.config = reflow
        cache.clear()
        currentSlice = null
        _state.value = _state.value.copy(config = reflow)

        ensureRenderBucketLocked()
        ReaderResult.Ok(Unit)
    }

// engines/txt/.../internal/paging/PageStartsIndex.kt

internal class PageStartsIndex {
    private val starts = IntArrayList()

    fun clear() = starts.clear()

    fun isEmpty(): Boolean = starts.size == 0

    fun seedIfEmpty(vararg values: Int) {
        if (!isEmpty()) return
        for (v in values) addStart(v)
    }

    fun addStart(v: Int) {
        val value = v.coerceAtLeast(0)
        starts.addSortedUnique(value)
    }

    // 其余不变...
}

package com.example.reader.engines.txt.internal.paging

import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.RenderConfig

internal data class RenderKey(
    val docId: String,
    val charset: String,
    val viewportW: Int,
    val viewportH: Int,
    val densityBits: Int,
    val fontScaleBits: Int,
    val fontSizeBits: Int,
    val lineHeightBits: Int,
    val paragraphBits: Int,
    val paddingBits: Int,
    val hyphenation: Boolean,
    val fontFamily: String?
) {
    companion object {
        fun placeholder(docId: String, charset: String): RenderKey =
            RenderKey(
                docId, charset,
                viewportW = -1, viewportH = -1,
                densityBits = 0, fontScaleBits = 0,
                fontSizeBits = 0, lineHeightBits = 0,
                paragraphBits = 0, paddingBits = 0,
                hyphenation = false,
                fontFamily = null
            )

        fun of(
            docId: String,
            charset: String,
            constraints: LayoutConstraints,
            config: RenderConfig.ReflowText
        ): RenderKey =
            RenderKey(
                docId = docId,
                charset = charset,
                viewportW = constraints.viewportWidthPx,
                viewportH = constraints.viewportHeightPx,
                densityBits = constraints.density.toBits(),
                fontScaleBits = constraints.fontScale.toBits(),
                fontSizeBits = config.fontSizeSp.toBits(),
                lineHeightBits = config.lineHeightMult.toBits(),
                paragraphBits = config.paragraphSpacingDp.toBits(),
                paddingBits = config.pagePaddingDp.toBits(),
                hyphenation = config.hyphenation,
                fontFamily = config.fontFamilyName
            )
    }
}

## 3) 新增 `TxtOutlineProvider.kt`

放到：`engines/txt/.../internal/provider/TxtOutlineProvider.kt`

```kotlin id="w3c4o5"
package com.example.reader.engines.txt.internal.provider

import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.OutlineProvider
import com.example.reader.engines.txt.internal.storage.TxtTextStore
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.OutlineNode
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext

/**
 * 简单但实用的 TXT 目录：
 * - 中文：第X章/节/回
 * - 英文：CHAPTER/Chapter N
 * - 数字标题：1. / 1、 / 1) ...
 * - Markdown：# / ## ...
 *
 * 产出扁平目录（children 为空），足够支撑跳转。
 */
internal class TxtOutlineProvider(
    private val store: TxtTextStore,
    private val ioDispatcher: CoroutineDispatcher
) : OutlineProvider {

    private val maxNodes = 500
    private val chunkChars = 64 * 1024

    // 常见标题规则（尽量宽松）
    private val chineseChapter = Regex("""^\s*(第.{1,12}[章节回卷部篇])\s*(.*)$""")
    private val enChapter = Regex("""^\s*(CHAPTER|Chapter)\s+\d+.*$""")
    private val numbered = Regex("""^\s*\d{1,4}\s*[\.\、\)]\s+.+$""")
    private val markdown = Regex("""^\s*#{1,6}\s+.+$""")

    override suspend fun getOutline(): ReaderResult<List<OutlineNode>> = withContext(ioDispatcher) {
        try {
            val total = store.totalChars().coerceAtLeast(0)
            if (total == 0) return@withContext ReaderResult.Ok(emptyList())

            val nodes = ArrayList<OutlineNode>(minOf(64, maxNodes))
            var offset = 0
            var carry = ""

            while (offset < total && nodes.size < maxNodes) {
                val text = store.readChars(offset, (chunkChars).coerceAtMost(total - offset))
                if (text.isEmpty()) break

                val merged = carry + text
                val lines = merged.split('\n')

                // 最后一行可能被截断，留到下次
                carry = if (merged.lastOrNull() == '\n') "" else lines.lastOrNull().orEmpty()

                var localPos = 0
                val lineCount = if (carry.isEmpty()) lines.size else (lines.size - 1)
                for (i in 0 until lineCount) {
                    val line = lines[i].trimEnd('\r')
                    val lineStartChar = offset - carry.length + localPos

                    val title = extractTitle(line)
                    if (title != null) {
                        nodes.add(
                            OutlineNode(
                                title = title,
                                locator = Locator(LocatorSchemes.TXT_OFFSET, lineStartChar.coerceAtLeast(0).toString()),
                                children = emptyList()
                            )
                        )
                        if (nodes.size >= maxNodes) break
                    }

                    // +1 for '\n'
                    localPos += (lines[i].length + 1)
                }

                offset += text.length
            }

            ReaderResult.Ok(nodes)
        } catch (t: Throwable) {
            ReaderResult.Ok(emptyList()) // 目录失败不影响阅读（你也可以改成 Err）
        }
    }

    private fun extractTitle(line: String): String? {
        if (line.isBlank()) return null

        chineseChapter.matchEntire(line)?.let { m ->
            val head = m.groupValues[1].trim()
            val tail = m.groupValues.getOrNull(2)?.trim().orEmpty()
            return if (tail.isNotEmpty()) "$head $tail" else head
        }
        if (enChapter.matches(line)) return line.trim()
        if (numbered.matches(line)) return line.trim()
        if (markdown.matches(line)) return line.trim().trimStart('#').trim()

        return null
    }
}
```

# 三、你现在得到的收益

* 同一本 TXT 在同一进程内：**打开 Session / 返回阅读页 / 切回书架再进来**，分页起点会复用，`prev()`、`goToProgress()` 更快更稳。
* 目录（OutlineProvider）立刻可用：你只要在 UI 展示 outline 列表并 `controller.goTo(node.locator)` 即可跳转。
