下面给你一套**可编译、可落地**的 `engines/txt` 引擎实现（按你现有 `core/reader/api` 契约来写），包含：

* ✅ 打开与缓存（规范化为 UTF-16LE `content.u16`，带 meta 指纹复用）
* ✅ 编码探测（BOM + UTF‑8 合法性探测 + fallback）
* ✅ `Utf16TextStore` 随机访问（FileChannel positional read）
* ✅ StaticLayout 分页（按需分页 + LRU 页缓存）
* ✅ “硬换行消除”（动态 SoftBreak 处理，长度不变，定位稳定）
* ✅ `ReaderController`（stateflow + events + next/prev/goTo/goToProgress/prefetch/invalidate）
* ✅ SearchProvider（流式 KMP 扫描，支持 startFrom、wholeWord、maxHits、可取消）
* ✅ OutlineProvider（章节启发式识别，缓存到 json）
* ✅ TextProvider（复制/摘录）

> 你只要把下面文件放进 `engines/txt/src/main/kotlin/...` 对应路径，然后在 `app` 里把 `TxtEngine` 注入 `EngineRegistry` 就能跑起来。

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/encoding/EncodingDetector.kt
package com.example.reader.engines.txt.internal.encoding

import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.engines.txt.internal.util.toReaderError
import com.example.reader.source.DocumentSource
import java.io.BufferedInputStream
import java.nio.charset.Charset
import java.nio.charset.CodingErrorAction
import org.json.JSONObject

internal data class EncodingResult(
    val charset: Charset,
    val confidence: Double,
    val reason: String
) {
    fun toJson(): JSONObject = JSONObject().apply {
        put("charset", charset.name())
        put("confidence", confidence)
        put("reason", reason)
    }
}

internal class EncodingDetector {

    suspend fun detect(
        source: DocumentSource,
        explicit: String?
    ): ReaderResult<EncodingResult> {
        return try {
            if (!explicit.isNullOrBlank()) {
                val cs = Charset.forName(explicit.trim())
                return ReaderResult.Ok(EncodingResult(cs, confidence = 1.0, reason = "explicit"))
            }

            // BOM first
            val bom = sniffBom(source)
            if (bom != null) return ReaderResult.Ok(bom)

            // Try UTF-8 validity on a sample
            val utf8Ok = looksLikeUtf8(source, sampleBytes = 256 * 1024)
            if (utf8Ok) {
                return ReaderResult.Ok(EncodingResult(Charsets.UTF_8, confidence = 0.85, reason = "utf8_valid_sample"))
            }

            // Fallback: GB18030 is a pragmatic choice for CN novels.
            val gb = Charset.forName("GB18030")
            ReaderResult.Ok(EncodingResult(gb, confidence = 0.55, reason = "fallback_gb18030"))
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }
    }

    private suspend fun sniffBom(source: DocumentSource): EncodingResult? {
        source.openInputStream().use { raw ->
            val input = BufferedInputStream(raw)
            input.mark(4)
            val b0 = input.read()
            val b1 = input.read()
            val b2 = input.read()
            val b3 = input.read()
            input.reset()

            // UTF-8 BOM: EF BB BF
            if (b0 == 0xEF && b1 == 0xBB && b2 == 0xBF) {
                return EncodingResult(Charsets.UTF_8, 1.0, "bom_utf8")
            }
            // UTF-16 LE: FF FE
            if (b0 == 0xFF && b1 == 0xFE) {
                return EncodingResult(Charset.forName("UTF-16LE"), 1.0, "bom_utf16le")
            }
            // UTF-16 BE: FE FF
            if (b0 == 0xFE && b1 == 0xFF) {
                return EncodingResult(Charset.forName("UTF-16BE"), 1.0, "bom_utf16be")
            }
            // UTF-32 BOM (optional)
            if (b0 == 0x00 && b1 == 0x00 && b2 == 0xFE && b3 == 0xFF) {
                return EncodingResult(Charset.forName("UTF-32BE"), 1.0, "bom_utf32be")
            }
            if (b0 == 0xFF && b1 == 0xFE && b2 == 0x00 && b3 == 0x00) {
                return EncodingResult(Charset.forName("UTF-32LE"), 1.0, "bom_utf32le")
            }
            return null
        }
    }

    private suspend fun looksLikeUtf8(source: DocumentSource, sampleBytes: Int): Boolean {
        val decoder = Charsets.UTF_8.newDecoder()
            .onMalformedInput(CodingErrorAction.REPORT)
            .onUnmappableCharacter(CodingErrorAction.REPORT)

        return try {
            source.openInputStream().use { input ->
                val buf = ByteArray(sampleBytes)
                val n = input.read(buf)
                if (n <= 0) return false
                decoder.decode(java.nio.ByteBuffer.wrap(buf, 0, n))
                true
            }
        } catch (_: Throwable) {
            false
        }
    }
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/open/TxtOpener.kt
package com.example.reader.engines.txt.internal.open

import com.example.reader.api.error.ReaderResult
import com.example.reader.api.open.OpenOptions
import com.example.reader.engines.txt.internal.encoding.EncodingDetector
import com.example.reader.engines.txt.internal.util.toReaderError
import com.example.reader.model.DocumentId
import com.example.reader.source.DocumentSource
import java.io.BufferedInputStream
import java.io.File
import java.io.RandomAccessFile
import java.security.MessageDigest
import kotlin.math.max
import kotlin.math.min
import kotlin.math.sqrt
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.withContext
import org.json.JSONObject

internal class TxtOpener(
    private val cacheRoot: File,
    private val ioDispatcher: CoroutineDispatcher
) {
    private val encodingDetector = EncodingDetector()

    // bump when you change normalization/schema and want rebuild
    private val schemaVersion = 1

    suspend fun open(source: DocumentSource, options: OpenOptions): ReaderResult<TxtOpenResult> =
        withContext(ioDispatcher) {
            try {
                cacheRoot.mkdirs()

                val sampleHash = computeSampleHash(source, limitBytes = 64 * 1024)
                val docId = computeDocumentId(source, sampleHash)
                val files = buildFiles(docId)

                files.bookDir.mkdirs()

                RandomAccessFile(files.lockFile, "rw").channel.use { ch ->
                    ch.lock().use {
                        val charsetResult = encodingDetector.detect(source, options.textEncoding)
                        val enc = when (charsetResult) {
                            is ReaderResult.Err -> return@withContext charsetResult
                            is ReaderResult.Ok -> charsetResult.value
                        }

                        val cached = tryLoadCached(files, source, sampleHash, enc.charset.name())
                        if (cached != null) {
                            return@withContext ReaderResult.Ok(
                                TxtOpenResult(
                                    documentId = docId,
                                    files = files,
                                    meta = cached
                                )
                            )
                        }

                        // build fresh
                        val meta = buildCache(files, source, sampleHash, enc.charset.name())

                        return@withContext ReaderResult.Ok(
                            TxtOpenResult(
                                documentId = docId,
                                files = files,
                                meta = meta
                            )
                        )
                    }
                }
            } catch (t: Throwable) {
                ReaderResult.Err(t.toReaderError())
            }
        }

    private fun buildFiles(docId: DocumentId): TxtBookFiles {
        val bookDir = File(cacheRoot, docId.value)
        return TxtBookFiles(
            bookDir = bookDir,
            lockFile = File(bookDir, "build.lock"),
            contentU16 = File(bookDir, "content.u16"),
            metaJson = File(bookDir, "meta.json"),
            outlineJson = File(bookDir, "outline.json")
        )
    }

    private fun tryLoadCached(
        files: TxtBookFiles,
        source: DocumentSource,
        sampleHash: String,
        expectedCharset: String
    ): TxtMeta? {
        if (!files.metaJson.exists()) return null
        if (!files.contentU16.exists()) return null

        val obj = JSONObject(files.metaJson.readText())
        val meta = TxtMeta.fromJson(obj)

        if (meta.version != schemaVersion) return null
        if (meta.sourceUri != source.uri.toString()) return null
        if ((meta.sizeBytes ?: -1L) != (source.sizeBytes ?: -1L)) return null
        if (meta.sampleHash != sampleHash) return null
        if (!meta.originalCharset.equals(expectedCharset, ignoreCase = true)) return null

        // quick sanity
        if (meta.lengthChars <= 0L && files.contentU16.length() > 0L) return null

        return meta
    }

    private suspend fun buildCache(
        files: TxtBookFiles,
        source: DocumentSource,
        sampleHash: String,
        charsetName: String
    ): TxtMeta {
        val charset = java.nio.charset.Charset.forName(charsetName)

        val tmp = File(files.bookDir, "content.u16.tmp")
        if (tmp.exists()) tmp.delete()

        val stats = LineStatsCollector(sampleCapacity = 5000)

        // streaming normalize -> UTF-16LE
        val writer = Utf16LeFileWriter(tmp)

        var pendingCR = false
        var lineLen = 0
        var lastNonSpace: Char? = null

        source.openInputStream().use { raw ->
            val input = BufferedInputStream(raw)
            val reader = java.io.InputStreamReader(input, charset)

            val buf = CharArray(16_384)
            while (true) {
                coroutineContext.ensureActive()

                val n = reader.read(buf)
                if (n < 0) break

                var i = 0
                while (i < n) {
                    val c = buf[i]

                    if (pendingCR) {
                        if (c == '\n') {
                            // CRLF -> single '\n'
                            writer.writeChar('\n')
                            stats.onLineFinished(lineLen, lastNonSpace)
                            lineLen = 0
                            lastNonSpace = null
                            pendingCR = false
                            i++
                            continue
                        } else {
                            // lone CR -> '\n'
                            writer.writeChar('\n')
                            stats.onLineFinished(lineLen, lastNonSpace)
                            lineLen = 0
                            lastNonSpace = null
                            pendingCR = false
                            // continue processing current c normally
                        }
                    }

                    when (c) {
                        '\r' -> {
                            pendingCR = true
                        }

                        '\n' -> {
                            writer.writeChar('\n')
                            stats.onLineFinished(lineLen, lastNonSpace)
                            lineLen = 0
                            lastNonSpace = null
                        }

                        else -> {
                            val out = normalizeChar(c)
                            writer.writeChar(out)
                            lineLen++
                            if (!out.isWhitespace()) lastNonSpace = out
                        }
                    }

                    i++
                }
            }
        }

        if (pendingCR) {
            writer.writeChar('\n')
            stats.onLineFinished(lineLen, lastNonSpace)
            lineLen = 0
            lastNonSpace = null
        } else {
            // if file doesn't end with newline, still count last line if any char existed
            stats.onLineFinishedIfHasData(lineLen, lastNonSpace)
        }

        val lengthChars = writer.totalCharsWritten
        writer.close()

        // swap
        if (files.contentU16.exists()) files.contentU16.delete()
        tmp.renameTo(files.contentU16)

        val snapshot = stats.snapshot()
        val meta = TxtMeta(
            version = schemaVersion,
            sourceUri = source.uri.toString(),
            displayName = source.displayName,
            sizeBytes = source.sizeBytes,
            sampleHash = sampleHash,
            originalCharset = charset.name(),
            lengthChars = lengthChars,

            hardWrapLikely = snapshot.hardWrapLikely,
            typicalLineLength = snapshot.medianLineLength,
            meanLineLength = snapshot.mean,
            stdLineLength = snapshot.std,
            endPunctRatio = snapshot.endPunctRatio,
            blankLineRatio = snapshot.blankRatio,

            createdAtEpochMs = System.currentTimeMillis()
        )

        files.metaJson.writeText(meta.toJson().toString())
        return meta
    }

    private fun normalizeChar(c: Char): Char {
        // 保持长度不变：控制字符替换为空格（避免 offset 漂移）
        if (c == '\u0000') return ' '
        if (c < ' ' && c != '\t') return ' '
        return c
    }

    private suspend fun computeSampleHash(source: DocumentSource, limitBytes: Int): String {
        val md = MessageDigest.getInstance("SHA-256")
        source.openInputStream().use { input ->
            val buf = ByteArray(8 * 1024)
            var remaining = limitBytes
            while (remaining > 0) {
                coroutineContext.ensureActive()
                val n = input.read(buf, 0, min(buf.size, remaining))
                if (n <= 0) break
                md.update(buf, 0, n)
                remaining -= n
            }
        }
        return md.digest().toHex()
    }

    private fun computeDocumentId(source: DocumentSource, sampleHash: String): DocumentId {
        val md = MessageDigest.getInstance("SHA-256")
        val base = buildString {
            append(source.uri.toString())
            append('|')
            append(source.sizeBytes ?: -1L)
            append('|')
            append(source.displayName ?: "")
            append('|')
            append(sampleHash)
        }
        md.update(base.toByteArray(Charsets.UTF_8))
        // shorten for path
        val hex = md.digest().toHex().take(32)
        return DocumentId(hex)
    }

    private fun ByteArray.toHex(): String =
        joinToString("") { b -> ((b.toInt() and 0xFF) + 0x100).toString(16).substring(1) }

    // ---------------- stats collector ----------------

    private data class StatsSnapshot(
        val medianLineLength: Int,
        val mean: Double,
        val std: Double,
        val endPunctRatio: Double,
        val blankRatio: Double,
        val hardWrapLikely: Boolean
    )

    private class LineStatsCollector(
        private val sampleCapacity: Int
    ) {
        private var totalLines: Long = 0
        private var blankLines: Long = 0
        private var endPunctLines: Long = 0

        // Welford
        private var mean: Double = 0.0
        private var m2: Double = 0.0

        // reservoir
        private val sample = IntArray(sampleCapacity)
        private var sampleSize = 0

        private val endPunctSet = setOf('。', '！', '？', '.', '!', '?', '…')

        fun onLineFinished(length: Int, lastNonSpace: Char?) {
            totalLines++
            if (length == 0) blankLines++

            // mean/std
            val x = length.toDouble()
            val delta = x - mean
            mean += delta / totalLines
            val delta2 = x - mean
            m2 += delta * delta2

            // end punctuation ratio
            if (lastNonSpace != null && endPunctSet.contains(lastNonSpace)) endPunctLines++

            // reservoir sample
            if (sampleCapacity <= 0) return
            if (sampleSize < sampleCapacity) {
                sample[sampleSize++] = length
            } else {
                // replace with probability
                val j = (kotlin.random.Random.nextLong(totalLines)).toInt()
                if (j < sampleCapacity) sample[j] = length
            }
        }

        fun onLineFinishedIfHasData(length: Int, lastNonSpace: Char?) {
            // for last line without newline: if there was any data, count it as a line
            if (length > 0 || lastNonSpace != null) onLineFinished(length, lastNonSpace)
        }

        fun snapshot(): StatsSnapshot {
            val variance = if (totalLines > 1) (m2 / (totalLines - 1)) else 0.0
            val std = sqrt(max(0.0, variance))
            val meanSafe = if (mean.isFinite() && mean > 0) mean else 1.0

            val median = if (sampleSize > 0) {
                val arr = sample.copyOf(sampleSize)
                arr.sort()
                arr[arr.size / 2]
            } else 0

            val endPunctRatio = if (totalLines > 0) endPunctLines.toDouble() / totalLines else 0.0
            val blankRatio = if (totalLines > 0) blankLines.toDouble() / totalLines else 0.0
            val coeffVar = std / meanSafe

            val hardWrapLikely =
                median in 30..120 &&
                    blankRatio < 0.15 &&
                    coeffVar < 0.55 &&
                    endPunctRatio < 0.55

            return StatsSnapshot(
                medianLineLength = median,
                mean = mean,
                std = std,
                endPunctRatio = endPunctRatio,
                blankRatio = blankRatio,
                hardWrapLikely = hardWrapLikely
            )
        }
    }
}
```

```kotlin
package com.example.reader.engines.txt.internal.open

import java.io.Closeable
import java.io.File
import java.io.FileOutputStream
import java.nio.ByteBuffer
import java.nio.ByteOrder
import java.nio.channels.FileChannel

internal class Utf16LeFileWriter(
    file: File,
    bufferChars: Int = 16_384
) : Closeable {

    private val channel: FileChannel = FileOutputStream(file).channel
    private val buf: ByteBuffer = ByteBuffer
        .allocateDirect(bufferChars * 2)
        .order(ByteOrder.LITTLE_ENDIAN)

    var totalCharsWritten: Long = 0
        private set

    fun writeChar(c: Char) {
        if (buf.remaining() < 2) flush()
        buf.putChar(c)
        totalCharsWritten++
    }

    private fun flush() {
        buf.flip()
        while (buf.hasRemaining()) channel.write(buf)
        buf.clear()
    }

    override fun close() {
        flush()
        channel.force(true)
        channel.close()
    }
}
```

## 4) 随机访问文本存储：Utf16TextStore

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/store/Utf16TextStore.kt
package com.example.reader.engines.txt.internal.store

import java.io.Closeable
import java.io.File
import java.io.FileInputStream
import java.nio.ByteBuffer
import java.nio.ByteOrder
import java.nio.channels.FileChannel
import kotlin.math.min

internal class Utf16TextStore private constructor(
    private val channel: FileChannel,
    val lengthChars: Long
) : Closeable {

    companion object {
        fun open(file: File): Utf16TextStore {
            val ch = FileInputStream(file).channel
            val lenChars = ch.size() / 2L
            return Utf16TextStore(ch, lenChars)
        }
    }

    fun readChars(startChar: Long, count: Int): CharArray {
        if (count <= 0) return CharArray(0)
        if (startChar < 0) return readChars(0, count)
        if (startChar >= lengthChars) return CharArray(0)

        val actual = min(count.toLong(), (lengthChars - startChar)).toInt()
        val bytes = ByteBuffer.allocate(actual * 2).order(ByteOrder.LITTLE_ENDIAN)

        var readTotal = 0
        while (readTotal < actual * 2) {
            val n = channel.read(bytes, startChar * 2L + readTotal)
            if (n <= 0) break
            readTotal += n
        }

        bytes.flip()
        bytes.order(ByteOrder.LITTLE_ENDIAN)
        val charBuf = bytes.asCharBuffer()
        val out = CharArray(charBuf.remaining())
        charBuf.get(out)
        return out
    }

    fun readString(startChar: Long, count: Int): String {
        val chars = readChars(startChar, count)
        return if (chars.isEmpty()) "" else String(chars)
    }

    fun readAround(centerChar: Long, before: Int, after: Int): String {
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/render/SoftBreakProcessor.kt
package com.example.reader.engines.txt.internal.render

import android.text.SpannableStringBuilder
import android.text.Spanned
import kotlin.math.roundToInt

internal class SoftBreakProcessor(
    private val hardWrapLikely: Boolean,
    private val typicalLineLen: Int
) {
    private val strongEndPunct = setOf('。', '！', '？', '.', '!', '?', '…', ':', '：', ';', '；')

    fun process(
        raw: CharArray,
        paragraphSpacingPx: Int
    ): CharSequence {
        if (raw.isEmpty()) return ""

        // build spannable with same length
        val ssb = SpannableStringBuilder(String(raw))
        if (!hardWrapLikely) {
            // 仅段落间距
            if (paragraphSpacingPx > 0) applyParagraphSpacing(ssb, paragraphSpacingPx)
            return ssb
        }

        // iterate lines with a sliding window (line0 + line1)
        var line0Start = 0
        var newline0 = indexOfNewline(ssb, line0Start)
        if (newline0 < 0) {
            // single-line file
            return ssb
        }
        var line0 = computeLineInfo(ssb, line0Start, newline0)

        var line1Start = newline0 + 1
        var newline1 = indexOfNewline(ssb, line1Start)
        var line1End = if (newline1 >= 0) newline1 else ssb.length
        var line1 = computeLineInfo(ssb, line1Start, line1End)

        while (true) {
            val nlIndex = newline0
            if (nlIndex >= ssb.length) break

            val nextChar = if (nlIndex + 1 < ssb.length) ssb[nlIndex + 1] else null

            val consecutiveNewlines = (nextChar == '\n')
            val soft =
                !consecutiveNewlines &&
                    isSoftBreak(line0, line1)

            if (soft) {
                // keep length: replace '\n' -> ' '
                ssb.setCharAt(nlIndex, ' ')
            } else {
                // paragraph spacing only when we keep real newline and it's not an empty line separator
                if (paragraphSpacingPx > 0 && !consecutiveNewlines) {
                    ssb.setSpan(
                        ParagraphSpacingSpan(paragraphSpacingPx),
                        nlIndex,
                        nlIndex + 1,
                        Spanned.SPAN_EXCLUSIVE_EXCLUSIVE
                    )
                }
            }

            // move window forward: line0 <- line1
            if (newline1 < 0) break

            line0Start = line1Start
            newline0 = newline1
            line0 = line1

            line1Start = newline0 + 1
            newline1 = indexOfNewline(ssb, line1Start)
            line1End = if (newline1 >= 0) newline1 else ssb.length
            line1 = computeLineInfo(ssb, line1Start, line1End)

            if (line0Start >= ssb.length) break
        }

        return ssb
    }

    private fun applyParagraphSpacing(ssb: SpannableStringBuilder, paragraphSpacingPx: Int) {
        for (i in 0 until ssb.length) {
            if (ssb[i] == '\n') {
                val next = if (i + 1 < ssb.length) ssb[i + 1] else null
                if (next != '\n') {
                    ssb.setSpan(
                        ParagraphSpacingSpan(paragraphSpacingPx),
                        i,
                        i + 1,
                        Spanned.SPAN_EXCLUSIVE_EXCLUSIVE
                    )
                }
            }
        }
    }

    private data class LineInfo(
        val len: Int,
        val firstNonSpace: Char?,
        val lastNonSpace: Char?
    )

    private fun computeLineInfo(text: CharSequence, start: Int, end: Int): LineInfo {
        val len = (end - start).coerceAtLeast(0)
        var first: Char? = null
        var last: Char? = null

        var i = start
        while (i < end) {
            val c = text[i]
            if (!c.isWhitespace()) {
                first = c
                break
            }
            i++
        }

        var j = end - 1
        while (j >= start) {
            val c = text[j]
            if (!c.isWhitespace()) {
                last = c
                break
            }
            j--
        }

        return LineInfo(len = len, firstNonSpace = first, lastNonSpace = last)
    }

    private fun isSoftBreak(line0: LineInfo, line1: LineInfo): Boolean {
        if (line0.len <= 0 || line1.len <= 0) return false

        val typical = typicalLineLen.coerceAtLeast(40)

        // 过短的行更像自然段/标题，不合并
        if (line0.len < (typical * 0.60).roundToInt()) return false
        if (line1.len < (typical * 0.30).roundToInt()) return false

        // 段末强标点：不要合并
        val last = line0.lastNonSpace
        if (last != null && strongEndPunct.contains(last)) return false

        // 下一行缩进/项目符号：不要合并
        val first = line1.firstNonSpace
        if (first == null) return false
        if (first == '-' || first == '*' || first == '•') return false

        return true
    }

    private fun indexOfNewline(text: CharSequence, from: Int): Int {
        for (i in from until text.length) if (text[i] == '\n') return i
        return -1
    }
}
```

```kotlin
package com.example.reader.engines.txt.internal.pagination

import android.text.StaticLayout
import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.RenderConfig
import com.example.reader.api.render.RenderPolicy
import com.example.reader.engines.txt.internal.layout.StaticLayoutMeasurer
import com.example.reader.engines.txt.internal.layout.TextPaintFactory
import com.example.reader.engines.txt.internal.open.TxtMeta
import com.example.reader.engines.txt.internal.render.SoftBreakProcessor
import com.example.reader.engines.txt.internal.store.Utf16TextStore
import kotlin.math.max
import kotlin.math.min
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext

internal class TxtPaginator(
    private val store: Utf16TextStore,
    private val meta: TxtMeta,
    private val ioDispatcher: CoroutineDispatcher,
    private val defaultDispatcher: CoroutineDispatcher
) {

    private val softBreakProcessor = SoftBreakProcessor(
        hardWrapLikely = meta.hardWrapLikely,
        typicalLineLen = meta.typicalLineLength
    )

    suspend fun pageAt(
        startOffset: Long,
        config: RenderConfig.ReflowText,
        constraints: LayoutConstraints,
        policy: RenderPolicy
    ): PageSlice = withContext(defaultDispatcher) {
        val length = store.lengthChars
        if (length <= 0L) return@withContext PageSlice(0, 0, "")

        val start = startOffset.coerceIn(0L, max(0L, length - 1))
        val paint = TextPaintFactory.create(config, constraints)

        val paddingPx = (config.pagePaddingDp * constraints.density).toInt().coerceAtLeast(0)
        val contentWidth = (constraints.viewportWidthPx - 2 * paddingPx).coerceAtLeast(1)
        val contentHeight = (constraints.viewportHeightPx - 2 * paddingPx).coerceAtLeast(1)

        val windowChars = estimateWindowChars(paint.textSize, contentWidth, contentHeight)
        val toRead = min(windowChars.toLong(), length - start).toInt().coerceAtLeast(0)
        val raw = withContext(ioDispatcher) { store.readChars(start, toRead) }

        val paragraphSpacingPx = (config.paragraphSpacingDp * constraints.density).toInt().coerceAtLeast(0)
        val processed: CharSequence = softBreakProcessor.process(raw, paragraphSpacingPx)

        val layout: StaticLayout = StaticLayoutMeasurer.build(
            text = processed,
            paint = paint,
            widthPx = contentWidth,
            lineHeightMult = config.lineHeightMult,
            hyphenation = config.hyphenation,
            quality = policy.quality
        )

        val lastLine = findLastLineFits(layout, contentHeight)
        val endInWindow = if (lastLine >= 0) {
            layout.getLineEnd(lastLine).coerceIn(1, processed.length.coerceAtLeast(1))
        } else {
            processed.length.coerceAtLeast(1)
        }

        val end = (start + endInWindow.toLong()).coerceAtMost(length)
        val pageText = processed.subSequence(0, endInWindow)

        PageSlice(start = start, end = end, text = pageText)
    }

    private fun findLastLineFits(layout: StaticLayout, heightPx: Int): Int {
        val lineCount = layout.lineCount
        var last = -1
        for (i in 0 until lineCount) {
            if (layout.getLineBottom(i) <= heightPx) last = i else break
        }
        return last
    }

    private fun estimateWindowChars(textSizePx: Float, contentWidth: Int, contentHeight: Int): Int {
        // rough estimate: assume avg char width ~ 0.55 * textSize
        val avgCharWidth = max(6f, textSizePx * 0.55f)
        val charsPerLine = (contentWidth / avgCharWidth).toInt().coerceAtLeast(10)
        val lineHeight = max(8f, textSizePx * 1.35f)
        val linesPerPage = (contentHeight / lineHeight).toInt().coerceAtLeast(5)

        val approx = charsPerLine * linesPerPage
        return (approx * 3).coerceIn(8_000, 200_000)
    }
}
```

```kotlin
internal class TxtController(
    private val documentKey: String,
    private val store: Utf16TextStore,
    private val meta: TxtMeta,
    initialOffset: Long,
    initialConfig: RenderConfig.ReflowText,
    private val maxPageCache: Int,
    private val ioDispatcher: CoroutineDispatcher,
    private val defaultDispatcher: CoroutineDispatcher
) : ReaderController, Closeable {

    private val scope = CoroutineScope(SupervisorJob() + defaultDispatcher)
    private val mutex = Mutex()

    private var constraints: LayoutConstraints? = null
    private var config: RenderConfig.ReflowText = initialConfig

    private var currentStart: Long = initialOffset.coerceAtLeast(0L)
    private var currentEnd: Long = initialOffset.coerceAtLeast(0L)

    private val paginator = TxtPaginator(store, meta, ioDispatcher, defaultDispatcher)

    private val pageCache = object : LruCache<Long, PageSlice>(maxPageCache.coerceAtLeast(1)) {
        override fun sizeOf(key: Long, value: PageSlice): Int = 1
    }

    private val _events = MutableSharedFlow<ReaderEvent>(
        extraBufferCapacity = 32
    )
    override val events: Flow<ReaderEvent> = _events.asSharedFlow()

    private val _state = MutableStateFlow(
        RenderState(
            locator = Locator(LocatorSchemes.TXT_OFFSET, currentStart.toString()),
            progression = Progression(percent = 0.0, label = "0%"),
            nav = NavigationAvailability(canGoPrev = false, canGoNext = meta.lengthChars > 0),
            titleInView = null,
            config = initialConfig
        )
    )
    override val state: StateFlow<RenderState> = _state

    override suspend fun setLayoutConstraints(constraints: LayoutConstraints): ReaderResult<Unit> =
        mutex.withLock {
            this.constraints = constraints
            clearCacheLocked()
            updateStateLocked(endKnown = false)
            ReaderResult.Ok(Unit)
        }

    override suspend fun setConfig(config: RenderConfig): ReaderResult<Unit> =
        mutex.withLock {
            val reflow = config as? RenderConfig.ReflowText
                ?: return@withLock ReaderResult.Err(
                    ReaderError.Internal("TXT engine only supports RenderConfig.ReflowText")
                )

            this.config = reflow
            clearCacheLocked()
            updateStateLocked(endKnown = false)
            ReaderResult.Ok(Unit)
        }

    override suspend fun render(policy: RenderPolicy): ReaderResult<RenderPage> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Err(
                ReaderError.Internal("LayoutConstraints not set")
            )

            val start = currentStart
            val (slice, metrics) = getOrComputePageLocked(start, c, policy)

            currentStart = slice.start
            currentEnd = slice.end

            val page = buildRenderPage(slice, metrics)
            updateStateLocked(endKnown = true)

            _events.tryEmit(ReaderEvent.Rendered(page.id, page.metrics))
            ReaderResult.Ok(page)
        }

    override suspend fun next(policy: RenderPolicy): ReaderResult<RenderPage> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Err(
                ReaderError.Internal("LayoutConstraints not set")
            )

            ensureCurrentEndLocked(c, policy)

            if (currentEnd >= store.lengthChars) {
                return@withLock ReaderResult.Err(ReaderError.Internal("Already at end"))
            }

            val (slice, metrics) = getOrComputePageLocked(currentEnd, c, policy)
            currentStart = slice.start
            currentEnd = slice.end

            val page = buildRenderPage(slice, metrics)
            updateStateLocked(endKnown = true)
            _events.tryEmit(ReaderEvent.PageChanged(page.locator))
            _events.tryEmit(ReaderEvent.Rendered(page.id, page.metrics))
            ReaderResult.Ok(page)
        }

    override suspend fun prev(policy: RenderPolicy): ReaderResult<RenderPage> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Err(
                ReaderError.Internal("LayoutConstraints not set")
            )

            if (currentStart <= 0L) {
                return@withLock ReaderResult.Err(ReaderError.Internal("Already at beginning"))
            }

            val prevStart = findPrevStartLocked(currentStart, c, policy)
            val (slice, metrics) = getOrComputePageLocked(prevStart, c, policy)

            currentStart = slice.start
            currentEnd = slice.end

            val page = buildRenderPage(slice, metrics)
            updateStateLocked(endKnown = true)
            _events.tryEmit(ReaderEvent.PageChanged(page.locator))
            _events.tryEmit(ReaderEvent.Rendered(page.id, page.metrics))
            ReaderResult.Ok(page)
        }

    override suspend fun goTo(locator: Locator, policy: RenderPolicy): ReaderResult<RenderPage> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Err(
                ReaderError.Internal("LayoutConstraints not set")
            )

            val offset = parseOffset(locator)
                ?: return@withLock ReaderResult.Err(ReaderError.Internal("Invalid locator for TXT: $locator"))

            val start = offset.coerceIn(0L, store.lengthChars)
            val (slice, metrics) = getOrComputePageLocked(start, c, policy)
            currentStart = slice.start
            currentEnd = slice.end

            val page = buildRenderPage(slice, metrics)
            updateStateLocked(endKnown = true)
            _events.tryEmit(ReaderEvent.PageChanged(page.locator))
            _events.tryEmit(ReaderEvent.Rendered(page.id, page.metrics))
            ReaderResult.Ok(page)
        }

    override suspend fun goToProgress(percent: Double, policy: RenderPolicy): ReaderResult<RenderPage> {
        val p = percent.coerceIn(0.0, 1.0)
        val offset = (store.lengthChars * p).toLong()
        return goTo(Locator(LocatorSchemes.TXT_OFFSET, offset.toString()), policy)
    }

    override suspend fun prefetchNeighbors(count: Int): ReaderResult<Unit> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Err(
                ReaderError.Internal("LayoutConstraints not set")
            )
            if (count <= 0) return@withLock ReaderResult.Ok(Unit)

            // ensure we know currentEnd so next prefetch starts correctly
            ensureCurrentEndLocked(c, RenderPolicy.Default)

            val starts = mutableListOf<Long>()
            var s = currentEnd
            repeat(count) {
                if (s >= store.lengthChars) return@repeat
                starts.add(s)
                // best-effort chain: compute next end (not cached yet) sequentially
                val slice = paginator.pageAt(s, config, c, RenderPolicy.Default)
                s = slice.end
            }

            // fire and forget (cache warm)
            starts.forEach { start ->
                scopeLaunchPrefetch(start, c)
            }

            ReaderResult.Ok(Unit)
        }

    override suspend fun invalidate(reason: InvalidateReason): ReaderResult<Unit> =
        mutex.withLock {
            clearCacheLocked()
            updateStateLocked(endKnown = false)
            ReaderResult.Ok(Unit)
        }

    override fun close() {
        pageCache.evictAll()
        scope.cancel()
    }

    // ---------------- internal helpers ----------------

    private fun scopeLaunchPrefetch(start: Long, constraints: LayoutConstraints) {
        scope.runCatching {
            kotlinx.coroutines.launch {
                mutex.withLock {
                    if (pageCache.get(start) != null) return@withLock
                }
                val slice = paginator.pageAt(start, config, constraints, RenderPolicy.Default)
                mutex.withLock { pageCache.put(slice.start, slice) }
            }
        }
    }

    private suspend fun ensureCurrentEndLocked(constraints: LayoutConstraints, policy: RenderPolicy) {
        if (currentEnd > currentStart) return
        val (slice, _) = getOrComputePageLocked(currentStart, constraints, policy)
        currentStart = slice.start
        currentEnd = slice.end
    }

    private suspend fun getOrComputePageLocked(
        start: Long,
        constraints: LayoutConstraints,
        policy: RenderPolicy
    ): Pair<PageSlice, RenderMetrics?> {
        val cacheAllowed = policy.allowCache
        if (cacheAllowed) {
            val cached = pageCache.get(start)
            if (cached != null) {
                return cached to RenderMetrics(renderTimeMs = 0, cacheHit = true)
            }
        }

        val t0 = System.nanoTime()
        val slice = paginator.pageAt(start, config, constraints, policy)
        val dtMs = ((System.nanoTime() - t0) / 1_000_000L).coerceAtLeast(0L)

        if (cacheAllowed) pageCache.put(slice.start, slice)
        return slice to RenderMetrics(renderTimeMs = dtMs, cacheHit = false)
    }

    private fun buildRenderPage(slice: PageSlice, metrics: RenderMetrics?): RenderPage {
        val mapping = TxtTextMapping(slice.start, slice.end)
        val loc = Locator(LocatorSchemes.TXT_OFFSET, slice.start.toString())

        val id = PageId("$documentKey:${slice.start}:${UUID.randomUUID()}")
        return RenderPage(
            id = id,
            locator = loc,
            content = com.example.reader.api.render.RenderContent.Text(
                text = slice.text,
                mapping = mapping
            ),
            links = emptyList(),
            decorations = emptyList(),
            metrics = metrics
        )
    }

    private fun updateStateLocked(endKnown: Boolean) {
        val length = store.lengthChars.coerceAtLeast(1L)
        val percent = (currentStart.toDouble() / length.toDouble()).coerceIn(0.0, 1.0)
        val label = "${(percent * 100).roundToInt()}%"

        val canPrev = currentStart > 0L
        val canNext = if (endKnown) currentEnd < store.lengthChars else currentStart < store.lengthChars

        _state.value = _state.value.copy(
            locator = Locator(LocatorSchemes.TXT_OFFSET, currentStart.toString()),
            progression = Progression(percent = percent, label = label),
            nav = NavigationAvailability(canGoPrev = canPrev, canGoNext = canNext),
            config = config
        )
    }

    private fun clearCacheLocked() {
        pageCache.evictAll()
        currentEnd = currentStart
    }

    private fun parseOffset(locator: Locator): Long? {
        if (locator.scheme != LocatorSchemes.TXT_OFFSET) return null
        return locator.value.toLongOrNull()
    }

    private suspend fun findPrevStartLocked(
        currentStart: Long,
        constraints: LayoutConstraints,
        policy: RenderPolicy
    ): Long {
        // heuristic: step back roughly 2 pages then paginate forward to reach currentStart
        val length = store.lengthChars.coerceAtLeast(1L)
        val pct = currentStart.toDouble() / length.toDouble()
        val approxCharsPerPage = (meta.typicalLineLength.coerceAtLeast(40) * 30).toLong() // rough
        val anchor = (currentStart - approxCharsPerPage * 2).coerceAtLeast(0L)

        var s = anchor
        var lastStart = anchor
        while (s < currentStart) {
            val (slice, _) = getOrComputePageLocked(s, constraints, policy)
            lastStart = slice.start
            s = slice.end
            if (s >= currentStart) break
        }
        return lastStart.coerceAtLeast(0L)
    }
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/provider/KmpMatcher.kt
package com.example.reader.engines.txt.internal.provider

import kotlin.math.max

internal class KmpMatcher(
    private val pattern: CharArray,
    private val caseSensitive: Boolean
) {
    private val lps: IntArray = buildLps(pattern)

    fun findAll(hay: CharArray, onMatch: (start: Int) -> Boolean) {
        if (pattern.isEmpty() || hay.isEmpty()) return

        var i = 0
        var j = 0
        while (i < hay.size) {
            val hc = norm(hay[i])
            val pc = norm(pattern[j])
            if (hc == pc) {
                i++; j++
                if (j == pattern.size) {
                    val start = i - j
                    val cont = onMatch(start)
                    if (!cont) return
                    j = lps[j - 1]
                }
            } else {
                if (j != 0) {
                    j = lps[j - 1]
                } else {
                    i++
                }
            }
        }
    }

    private fun norm(c: Char): Char {
        return if (caseSensitive) c else c.lowercaseChar()
    }

    private fun buildLps(p: CharArray): IntArray {
        val lps = IntArray(p.size)
        var len = 0
        var i = 1
        while (i < p.size) {
            if (p[i] == p[len]) {
                len++
                lps[i] = len
                i++
            } else {
                if (len != 0) {
                    len = lps[len - 1]
                } else {
                    lps[i] = 0
                    i++
                }
            }
        }
        return lps
    }
}
```

```kotlin
// File: engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/provider/TxtSearchProvider.kt
package com.example.reader.engines.txt.internal.provider

import com.example.reader.api.provider.SearchHit
import com.example.reader.api.provider.SearchOptions
import com.example.reader.api.provider.SearchProvider
import com.example.reader.engines.txt.internal.open.TxtMeta
import com.example.reader.engines.txt.internal.store.Utf16TextStore
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import kotlin.math.min
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.flow
import kotlinx.coroutines.flow.flowOn

internal class TxtSearchProvider(
    private val store: Utf16TextStore,
    private val meta: TxtMeta,
    private val ioDispatcher: CoroutineDispatcher
) : SearchProvider {

    override fun search(query: String, options: SearchOptions): Flow<SearchHit> = flow {
        val q = query
        if (q.isBlank()) return@flow

        val startOffset = options.startFrom
            ?.takeIf { it.scheme == LocatorSchemes.TXT_OFFSET }
            ?.value
            ?.toLongOrNull()
            ?.coerceIn(0L, store.lengthChars)
            ?: 0L

        val pattern = q.toCharArray()
        val matcher = KmpMatcher(pattern, caseSensitive = options.caseSensitive)

        val chunkChars = 64_000
        val overlap = (pattern.size - 1).coerceAtLeast(0)

        var carry = CharArray(0)
        var pos = startOffset
        var hits = 0

        while (pos < store.lengthChars) {
            coroutineContext.ensureActive()

            val toRead = min(chunkChars.toLong(), store.lengthChars - pos).toInt()
            val chunk = store.readChars(pos, toRead)

            val combined = CharArray(carry.size + chunk.size)
            if (carry.isNotEmpty()) System.arraycopy(carry, 0, combined, 0, carry.size)
            if (chunk.isNotEmpty()) System.arraycopy(chunk, 0, combined, carry.size, chunk.size)

            val combinedStartGlobal = pos - carry.size.toLong()

            matcher.findAll(combined) { matchIndex ->
                coroutineContext.ensureActive()

                val matchStartGlobal = combinedStartGlobal + matchIndex.toLong()
                val matchEndGlobal = matchStartGlobal + pattern.size.toLong()

                // skip matches before requested start
                if (matchStartGlobal < startOffset) return@findAll true

                // avoid duplicates: matches fully inside carry were already emitted
                if (matchIndex + pattern.size <= carry.size) return@findAll true

                if (options.wholeWord && !isWholeWord(combined, matchIndex, pattern.size)) {
                    return@findAll true
                }

                val range = LocatorRange(
                    start = Locator(LocatorSchemes.TXT_OFFSET, matchStartGlobal.toString()),
                    end = Locator(LocatorSchemes.TXT_OFFSET, matchEndGlobal.toString())
                )

                val excerpt = buildExcerpt(matchStartGlobal, pattern.size)

                emit(
                    SearchHit(
                        range = range,
                        excerpt = excerpt,
                        sectionTitle = null
                    )
                )

                hits++
                hits < options.maxHits
            }

            if (overlap > 0) {
                val keep = min(overlap, combined.size)
                carry = combined.copyOfRange(combined.size - keep, combined.size)
            } else {
                carry = CharArray(0)
            }

            pos += toRead.toLong()
        }
    }.flowOn(ioDispatcher)

    private fun isWholeWord(hay: CharArray, start: Int, len: Int): Boolean {
        val before = if (start - 1 >= 0) hay[start - 1] else null
        val after = if (start + len < hay.size) hay[start + len] else null
        val beforeOk = before == null || !Character.isLetterOrDigit(before)
        val afterOk = after == null || !Character.isLetterOrDigit(after)
        return beforeOk && afterOk
    }

    private fun buildExcerpt(matchStart: Long, patternLen: Int): String {
        val before = 64
        val after = 96
        val center = matchStart + patternLen / 2
        return store.readAround(center, before, after)
            .replace('\n', ' ')
            .trim()
    }
}
```

```kotlin
                val maxChars = 200_000
                val len = (end - start).toInt().coerceAtLeast(0)
                val capped = min(len, maxChars)

                val text = store.readString(start, capped)
                ReaderResult.Ok(text)
            } catch (t: Throwable) {
                ReaderResult.Err(com.example.reader.engines.txt.internal.util.run { t.toReaderError() })
            }
        }

    override suspend fun getTextAround(locator: Locator, maxChars: Int): ReaderResult<String> =
        withContext(ioDispatcher) {
            try {
                val center = parseOffset(locator) ?: return@withContext ReaderResult.Err(
                    ReaderError.Internal("Invalid TXT locator: $locator")
                )
                val half = (maxChars / 2).coerceAtLeast(32)
                val text = store.readAround(center, half, half)
                ReaderResult.Ok(text)
            } catch (t: Throwable) {
                ReaderResult.Err(com.example.reader.engines.txt.internal.util.run { t.toReaderError() })
            }
        }

    private fun parseOffset(locator: Locator): Long? {
        if (locator.scheme != LocatorSchemes.TXT_OFFSET) return null
        return locator.value.toLongOrNull()
```

```kotlin

package com.example.reader.engines.txt.internal.provider

import com.example.reader.api.error.ReaderResult

import com.example.reader.engines.txt.internal.open.TxtBookFiles

import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.OutlineNode
import kotlin.math.min
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.withContext
import org.json.JSONArray
import org.json.JSONObject

internal class TxtOutlineProvider(
    private val files: TxtBookFiles,
    private val store: Utf16TextStore,
    private val meta: TxtMeta,
    private val ioDispatcher: CoroutineDispatcher
) : OutlineProvider {

    override suspend fun getOutline(): ReaderResult<List<OutlineNode>> =
        withContext(ioDispatcher) {
            try {
                val cached = loadCacheIfValid()
                if (cached != null) return@withContext ReaderResult.Ok(cached)

                val outline = scanChapters()
                saveCache(outline)
                ReaderResult.Ok(outline)
            } catch (t: Throwable) {
                ReaderResult.Err(com.example.reader.engines.txt.internal.util.run { t.toReaderError() })
            }
        }

    private fun loadCacheIfValid(): List<OutlineNode>? {
        if (!files.outlineJson.exists()) return null
        val obj = JSONObject(files.outlineJson.readText())
        val version = obj.optInt("version", -1)
        val sampleHash = obj.optString("sampleHash", "")
        if (version != 1) return null
        if (sampleHash != meta.sampleHash) return null

        val arr = obj.getJSONArray("items")
        val out = ArrayList<OutlineNode>(arr.length())
        for (i in 0 until arr.length()) {
            val it = arr.getJSONObject(i)
            val title = it.getString("title")
            val offset = it.getLong("offset")
            out.add(
                OutlineNode(
                    title = title,
                    locator = Locator(LocatorSchemes.TXT_OFFSET, offset.toString()),
                    children = emptyList()
                )
            )
        }
        return out
    }

    private fun saveCache(outline: List<OutlineNode>) {
        val arr = JSONArray()
        outline.forEach {
            val offset = it.locator.value.toLongOrNull() ?: 0L
            arr.put(JSONObject().apply {
                put("title", it.title)
                put("offset", offset)
            })
        }
        val obj = JSONObject().apply {
            put("version", 1)
            put("sampleHash", meta.sampleHash)
            put("items", arr)
        }
        files.outlineJson.writeText(obj.toString())
    }

    private suspend fun scanChapters(): List<OutlineNode> {
        val detector = ChapterDetector()

        val chunkChars = 64_000
        var pos = 0L
        var carry = ""

        val out = ArrayList<OutlineNode>(64)
        val seenTitles = HashSet<String>()

        while (pos < store.lengthChars) {
            coroutineContext.ensureActive()

            val toRead = min(chunkChars.toLong(), store.lengthChars - pos).toInt()
            val chunk = store.readString(pos, toRead)
            val combined = carry + chunk

            var lineStartInCombined = 0
            while (true) {
                val nl = combined.indexOf('\n', startIndex = lineStartInCombined)
                if (nl < 0) break

                val line = combined.substring(lineStartInCombined, nl)
                val lineTrim = line.trim()

                val lineStartGlobal = (pos - carry.length.toLong()) + lineStartInCombined.toLong()

                if (detector.isChapterTitle(lineTrim)) {
                    if (seenTitles.add(lineTrim)) {
                        out.add(
                            OutlineNode(
                                title = lineTrim,
                                locator = Locator(LocatorSchemes.TXT_OFFSET, lineStartGlobal.toString())
                            )
                        )
                        // 防止误报过多
                        if (out.size >= 300) return out
                    }
                }

                lineStartInCombined = nl + 1
            }

            carry = combined.substring(lineStartInCombined)
            pos += toRead.toLong()
        }

        return out
    }
}
```

```kotlin
package com.example.reader.engines.txt.internal.provider

internal class ChapterDetector {

    private val cn = Regex("""^第[零一二三四五六七八九十百千万0-9]{1,9}[章节回卷部篇].{0,30}$""")
    private val en = Regex("""^(Chapter|CHAPTER)\s+\d+.*$""")
    private val pro = Regex("""^(Prologue|Epilogue|PROLOGUE|EPILOGUE)$""")

    fun isChapterTitle(line: String): Boolean {
        if (line.isBlank()) return false
        if (line.length > 40) return false

        if (cn.matches(line)) return true
        if (en.matches(line)) return true
        if (pro.matches(line)) return true

        return false
    }
}
```
