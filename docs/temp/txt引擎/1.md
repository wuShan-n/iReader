下面给你一套**可直接放进 `engines/txt` 模块编译运行**的 TXT 引擎实现（匹配你现有的 `core/model`、`core/files`、`core/reader/api`、`core/reader/runtime` 契约）。
实现内容：**打开/编码探测、分页渲染（StaticLayout）、定位（txt.offset=charOffset）、next/prev/goTo/goToProgress、TextProvider、SearchProvider、（可用的内存版 AnnotationProvider）**。

> 说明：这版 TextStore 采用 **整本读入内存**（多数 txt 书籍 < 5-10MB 非常稳）。如果你要支持超大 txt（几十/上百 MB），我可以在下一步把 TextStore 升级成“稀疏索引 + 准随机访问”版本。

```kotlin
class TxtEngine(
    private val ioDispatcher: CoroutineDispatcher = Dispatchers.IO
) : ReaderEngine {

    override val supportedFormats: Set<BookFormat> = setOf(BookFormat.TXT)

    override suspend fun open(
        source: DocumentSource,
        options: OpenOptions
    ): ReaderResult<ReaderDocument> = withContext(ioDispatcher) {

            val charset = TxtCharsetDetector.detect(source, options.textEncoding, ioDispatcher)
            ReaderResult.Ok(

        }

internal fun Throwable.toReaderError(): ReaderError =
    when (this) {
        is ReaderError -> this
        is java.io.FileNotFoundException -> ReaderError.NotFound(cause = this)
        is SecurityException -> ReaderError.PermissionDenied(cause = this)
        is java.util.concurrent.CancellationException -> ReaderError.Cancelled(cause = this)
        is kotlinx.coroutines.CancellationException -> ReaderError.Cancelled(cause = this)
        is java.io.IOException -> ReaderError.Io(cause = this)
        else -> ReaderError.Internal(message = message, cause = this)
    }
```

```kotlin
package com.example.reader.engines.txt.internal.open

import com.example.reader.source.DocumentSource
import java.nio.charset.Charset
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext

internal object TxtCharsetDetector {

    /**
     * 优先级：
     * 1) options.textEncoding（用户指定）
     * 2) BOM（UTF-8 / UTF-16LE / UTF-16BE）
     * 3) 默认 UTF-8
     */
    suspend fun detect(
        source: DocumentSource,
        overrideName: String?,
        ioDispatcher: CoroutineDispatcher
    ): Charset = withContext(ioDispatcher) {
        overrideName?.let { name ->
            return@withContext Charset.forName(name)
        }

        // BOM sniff
        val header = source.openInputStream().use { input ->
            val buf = ByteArray(4)
            val read = input.read(buf)
            if (read <= 0) ByteArray(0) else buf.copyOf(read)
        }

        // UTF-8 BOM: EF BB BF
        if (header.size >= 3 &&
            header[0] == 0xEF.toByte() &&
            header[1] == 0xBB.toByte() &&
            header[2] == 0xBF.toByte()
        ) return@withContext Charsets.UTF_8

        // UTF-16 LE BOM: FF FE
        if (header.size >= 2 &&
            header[0] == 0xFF.toByte() &&
            header[1] == 0xFE.toByte()
        ) return@withContext Charsets.UTF_16LE

        // UTF-16 BE BOM: FE FF
        if (header.size >= 2 &&
            header[0] == 0xFE.toByte() &&
            header[1] == 0xFF.toByte()
        ) return@withContext Charsets.UTF_16BE

        // 默认
        Charsets.UTF_8
    }
}
```

```kotlin
internal class TxtDocument(
    private val source: DocumentSource,
    override val openOptions: OpenOptions,
    private val charset: Charset,
    private val ioDispatcher: CoroutineDispatcher
) : ReaderDocument {

    override val id: DocumentId = DocumentId("txt:${stableId(source)}")
    override val format: BookFormat = BookFormat.TXT

    override val capabilities: DocumentCapabilities =
        DocumentCapabilities(
            reflowable = true,
            fixedLayout = false,
            outline = false,
            search = true,
            textExtraction = true,
            annotations = true, // 本实现提供 session 内存版标注
            links = false
        )

        InMemoryTxtTextStore(
            source = source,
            charset = charset,
            ioDispatcher = ioDispatcher
        )
    }

    override suspend fun metadata(): ReaderResult<DocumentMetadata> =
        withContext(ioDispatcher) {
            ReaderResult.Ok(
                DocumentMetadata(
                    title = source.displayName ?: "Untitled",
                    author = null,
                    language = null,
                    identifier = null,
                    extra = buildMap {
                        put("charset", charset.name())
                        source.sizeBytes?.let { put("sizeBytes", it.toString()) }
                        source.mimeType?.let { put("mimeType", it) }
                    }
                )
            )
        }

    override suspend fun createSession(
        initialLocator: Locator?,
        initialConfig: RenderConfig
    ): ReaderResult<ReaderSession> = withContext(ioDispatcher) {
        val locator = normalizeLocator(initialLocator)
        val config = normalizeConfig(initialConfig)
        TxtSession.create(
            documentId = id,
            store = textStore,
            initialLocator = locator,
            initialConfig = config,
            ioDispatcher = ioDispatcher
        )
    }

    override fun close() {
        runCatching { textStore.close() }
    }

    private fun normalizeLocator(locator: Locator?): Locator {
        if (locator == null) return Locator(LocatorSchemes.TXT_OFFSET, "0")
        if (locator.scheme != LocatorSchemes.TXT_OFFSET) return Locator(LocatorSchemes.TXT_OFFSET, "0")
        val v = locator.value.toIntOrNull() ?: 0
        return Locator(LocatorSchemes.TXT_OFFSET, v.coerceAtLeast(0).toString(), locator.extras)
    }

    private fun normalizeConfig(config: RenderConfig): RenderConfig.ReflowText =
        when (config) {
            is RenderConfig.ReflowText -> config
            else -> RenderConfig.ReflowText()
        }

    private fun stableId(source: DocumentSource): String {
        val base = buildString {
            append(source.uri.toString())
            append("|")
            append(source.displayName ?: "")
            append("|")
            append(source.sizeBytes?.toString() ?: "")
        }
        // 简单稳定 hash（避免暴露 uri 全文给外部）
        return base.hashCode().toString()
    }
}
```

## 4) `engines/txt/src/main/kotlin/com/example/reader/engines/txt/internal/storage/InMemoryTxtTextStore.kt`

```kotlin
package com.example.reader.engines.txt.internal.storage

import com.example.reader.source.DocumentSource
import java.io.Closeable
import java.nio.charset.Charset
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext

internal interface TxtTextStore : Closeable {
    val charset: Charset
    suspend fun totalChars(): Int
    suspend fun readRange(startChar: Int, endCharExclusive: Int): String
    suspend fun readChars(startChar: Int, maxChars: Int): String
    suspend fun readAround(charOffset: Int, maxChars: Int): String
}

internal class InMemoryTxtTextStore(
    private val source: DocumentSource,
    override val charset: Charset,
    private val ioDispatcher: CoroutineDispatcher,
    private val maxBytes: Long = 20L * 1024L * 1024L // 20MB：够用且安全
) : TxtTextStore {

    @Volatile
    private var cached: String? = null

    private suspend fun ensureLoaded(): String = withContext(ioDispatcher) {
        cached?.let { return@withContext it }

        // 可用 sizeBytes 做防御
        val size = source.sizeBytes
        if (size != null && size > maxBytes) {
            // 你可以把这里改成“切换到 IndexedTextStore”
            throw IllegalStateException("TXT too large for in-memory store: $size bytes")
        }

        val text = source.openInputStream().use { input ->
            val bytes = input.readBytes()
            if (bytes.size.toLong() > maxBytes) {
                throw IllegalStateException("TXT too large for in-memory store: ${bytes.size} bytes")
            }
            // 去 BOM（如果存在）
            decodeRemovingBom(bytes, charset)
        }
        cached = text
        text
    }

    override suspend fun totalChars(): Int = ensureLoaded().length

    override suspend fun readRange(startChar: Int, endCharExclusive: Int): String {
        val s = ensureLoaded()
        val start = startChar.coerceIn(0, s.length)
        val end = endCharExclusive.coerceIn(start, s.length)
        return s.substring(start, end)
    }

    override suspend fun readChars(startChar: Int, maxChars: Int): String {
        val s = ensureLoaded()
        val start = startChar.coerceIn(0, s.length)
        val end = (start + maxChars).coerceAtMost(s.length)
        return s.substring(start, end)
    }

    override suspend fun readAround(charOffset: Int, maxChars: Int): String {
        val s = ensureLoaded()
        val half = (maxChars / 2).coerceAtLeast(1)
        val start = (charOffset - half).coerceIn(0, s.length)
        val end = (start + maxChars).coerceAtMost(s.length)
        return s.substring(start, end)
    }

    override fun close() {
        cached = null
    }

    private fun decodeRemovingBom(bytes: ByteArray, charset: Charset): String {
        // 如果 detector 选了 UTF-8/UTF-16 并且 bytes 带 BOM，这里再去一次更稳
        if (bytes.size >= 3 &&
            bytes[0] == 0xEF.toByte() && bytes[1] == 0xBB.toByte() && bytes[2] == 0xBF.toByte()
        ) {
            return bytes.copyOfRange(3, bytes.size).toString(charset)
        }
        if (bytes.size >= 2 &&
            ((bytes[0] == 0xFF.toByte() && bytes[1] == 0xFE.toByte()) ||
             (bytes[0] == 0xFE.toByte() && bytes[1] == 0xFF.toByte()))
        ) {
            return bytes.copyOfRange(2, bytes.size).toString(charset)
        }
        return bytes.toString(charset)
    }
}
```

```kotlin
package com.example.reader.engines.txt.internal.paging

import android.text.Layout
import android.text.StaticLayout
import android.text.TextPaint
import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.RenderConfig
import com.example.reader.engines.txt.internal.storage.TxtTextStore
import kotlin.math.max
import kotlin.math.min
import kotlin.math.roundToInt

internal data class PageSlice(
    val startChar: Int,
    val endChar: Int,
    val text: String
)

internal class TxtPager(
    private val store: TxtTextStore
) {

    suspend fun pageAt(
        startChar: Int,
        constraints: LayoutConstraints,
        config: RenderConfig.ReflowText
    ): PageSlice {
        val total = store.totalChars()
        val start = startChar.coerceIn(0, max(0, total))
        if (total == 0) return PageSlice(0, 0, "")

        val paddingPx = dpToPx(config.pagePaddingDp, constraints.density)
        val contentWidth = max(1, constraints.viewportWidthPx - paddingPx * 2)
        val contentHeight = max(1, constraints.viewportHeightPx - paddingPx * 2)

        // 读一个足够大的窗口供 StaticLayout 测量
        val candidateMax = max(2_000, estimateCharsPerPage(constraints, config) * 3)
        val text = store.readChars(start, min(candidateMax, total - start))

        if (text.isEmpty()) return PageSlice(start, start, "")

        val paint = TextPaint(TextPaint.ANTI_ALIAS_FLAG).apply {
            textSize = spToPx(config.fontSizeSp, constraints.density, constraints.fontScale)
        }

        val layout = StaticLayout.Builder
            .obtain(text, 0, text.length, paint, contentWidth)
            .setAlignment(Layout.Alignment.ALIGN_NORMAL)
            .setIncludePad(false)
            .setLineSpacing(0f, config.lineHeightMult)
            .apply {
                if (config.hyphenation) {
                    setHyphenationFrequency(Layout.HYPHENATION_FREQUENCY_NORMAL)
                }
            }
            .build()

        val y = (contentHeight - 1).coerceAtLeast(0)
        val lastLine = layout.getLineForVertical(y).coerceIn(0, max(0, layout.lineCount - 1))
        var endInText = layout.getLineEnd(lastLine).coerceIn(0, text.length)

        // 防御：避免 0 长死循环
        if (endInText <= 0 && text.isNotEmpty()) endInText = 1

        val endChar = (start + endInText).coerceIn(start, total)
        val pageText = text.substring(0, endInText)

        return PageSlice(start, endChar, pageText)
    }

    fun estimateCharsPerPage(
        constraints: LayoutConstraints,
        config: RenderConfig.ReflowText
    ): Int {
        val paddingPx = dpToPx(config.pagePaddingDp, constraints.density)
        val contentWidth = max(1, constraints.viewportWidthPx - paddingPx * 2)
        val contentHeight = max(1, constraints.viewportHeightPx - paddingPx * 2)

        val paint = TextPaint(TextPaint.ANTI_ALIAS_FLAG).apply {
            textSize = spToPx(config.fontSizeSp, constraints.density, constraints.fontScale)
        }

        val fm = paint.fontMetrics
        val lineHeightPx = max(1f, (fm.descent - fm.ascent) * config.lineHeightMult)
        val lines = max(1, (contentHeight / lineHeightPx).toInt())

        // 用一个代表性字符估算平均宽度（中文/英文都还行）
        val avgCharW = max(1f, paint.measureText("中"))
        val charsPerLine = max(5, (contentWidth / avgCharW).toInt())

        return (charsPerLine * lines).coerceIn(300, 50_000)
    }

    private fun dpToPx(dp: Float, density: Float): Int =
        (dp * density).roundToInt()

    private fun spToPx(sp: Float, density: Float, fontScale: Float): Float =
        sp * density * fontScale
}
```

```kotlin
package com.example.reader.engines.txt.internal.paging

internal class PaginationCache(
    private val maxPages: Int = 24
) {
    private val lru = object : LinkedHashMap<Int, PageSlice>(maxPages, 0.75f, true) {
        override fun removeEldestEntry(eldest: MutableMap.MutableEntry<Int, PageSlice>?): Boolean {
            return size > maxPages
        }
    }

    fun get(startChar: Int): PageSlice? = synchronized(lru) { lru[startChar] }

    fun put(slice: PageSlice) {
        synchronized(lru) { lru[slice.startChar] = slice }
    }

    fun clear() {
        synchronized(lru) { lru.clear() }
    }
}
```

```kotlin
package com.example.reader.engines.txt.internal.controller

import com.example.reader.api.render.TextMapping
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes

internal class TxtTextMapping(
    private val pageStartChar: Int
) : TextMapping {

    override fun locatorAt(charIndex: Int): Locator =
        Locator(LocatorSchemes.TXT_OFFSET, (pageStartChar + charIndex).toString())

    override fun rangeFor(startChar: Int, endChar: Int): LocatorRange =
        LocatorRange(locatorAt(startChar), locatorAt(endChar))

    override fun charRangeFor(range: LocatorRange): IntRange? {
        if (range.start.scheme != LocatorSchemes.TXT_OFFSET) return null
        val s = range.start.value.toIntOrNull() ?: return null
        val e = range.end.value.toIntOrNull() ?: return null
        val a = (s - pageStartChar).coerceAtLeast(0)
        val b = (e - pageStartChar).coerceAtLeast(a)
        return a until b
    }
}
```

```kotlin
package com.example.reader.engines.txt.internal.controller

import android.os.SystemClock
import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.AnnotationQuery
import com.example.reader.api.provider.AnnotationProvider
import com.example.reader.api.render.InvalidateReason
import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.NavigationAvailability
import com.example.reader.api.render.PageId
import com.example.reader.api.render.ReaderController
import com.example.reader.api.render.ReaderEvent
import com.example.reader.api.render.RenderConfig
import com.example.reader.api.render.RenderContent
import com.example.reader.api.render.RenderMetrics
import com.example.reader.api.render.RenderPage
import com.example.reader.api.render.RenderPolicy
import com.example.reader.api.render.RenderState
import com.example.reader.engines.txt.internal.paging.PageSlice
import com.example.reader.engines.txt.internal.paging.PaginationCache
import com.example.reader.engines.txt.internal.paging.TxtPager
import com.example.reader.engines.txt.internal.storage.TxtTextStore
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.Progression
import kotlin.math.max
import kotlin.math.min
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.MutableSharedFlow
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asSharedFlow
import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock
import kotlinx.coroutines.withContext

internal class TxtController(
    private val store: TxtTextStore,
    private val pager: TxtPager,
    private val ioDispatcher: CoroutineDispatcher,
    private val annotations: AnnotationProvider?
) : ReaderController {

    private val mutex = Mutex()

    private var constraints: LayoutConstraints? = null
    private var config: RenderConfig.ReflowText = RenderConfig.ReflowText()

    private var currentStartChar: Int = 0
    private var currentSlice: PageSlice? = null

    private val historyStarts = ArrayDeque<Int>() // 用于 prev（最稳）
    private val cache = PaginationCache(maxPages = 36)

    private val _state = MutableStateFlow(
        RenderState(
            locator = Locator(LocatorSchemes.TXT_OFFSET, "0"),
            progression = Progression(percent = 0.0, label = "0%"),
            nav = NavigationAvailability(canGoPrev = false, canGoNext = false),
            titleInView = null,
            config = config
        )
    )
    override val state: StateFlow<RenderState> = _state

    private val _events = MutableSharedFlow<ReaderEvent>(extraBufferCapacity = 64)
    override val events: Flow<ReaderEvent> = _events.asSharedFlow()

    override suspend fun setLayoutConstraints(constraints: LayoutConstraints): ReaderResult<Unit> =
        mutex.withLock {
            this.constraints = constraints
            cache.clear()
            currentSlice = null
            ReaderResult.Ok(Unit)
        }

    override suspend fun setConfig(config: RenderConfig): ReaderResult<Unit> =
        mutex.withLock {
            val reflow = when (config) {
                is RenderConfig.ReflowText -> config
                else -> RenderConfig.ReflowText()
            }
            this.config = reflow
            cache.clear()
            currentSlice = null
            _state.value = _state.value.copy(config = reflow)
            ReaderResult.Ok(Unit)
        }

    override suspend fun render(policy: RenderPolicy): ReaderResult<RenderPage> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Err(
                ReaderError.Internal("LayoutConstraints not set")
            )
            return@withLock renderLocked(c, policy, keepHistory = true)
        }

    override suspend fun next(policy: RenderPolicy): ReaderResult<RenderPage> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Err(
                ReaderError.Internal("LayoutConstraints not set")
            )

            val slice = ensureSliceLocked(c, policy)
                ?: return@withLock ReaderResult.Err(ReaderError.Internal("Failed to render current page"))

            val total = store.totalChars()
            if (slice.endChar >= total) {
                return@withLock ReaderResult.Err(ReaderError.Internal("Already at end"))
            }

            historyStarts.addLast(currentStartChar)
            currentStartChar = slice.endChar
            currentSlice = null
            return@withLock renderLocked(c, policy, keepHistory = false)
        }

    override suspend fun prev(policy: RenderPolicy): ReaderResult<RenderPage> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Err(
                ReaderError.Internal("LayoutConstraints not set")
            )

            val prevStart = if (historyStarts.isNotEmpty()) {
                historyStarts.removeLast()
            } else {
                // 无历史时：用启发式反推上一页
                computePrevStartLocked(c, policy)
            }

            if (prevStart == currentStartChar) {
                return@withLock ReaderResult.Err(ReaderError.Internal("Already at beginning"))
            }

            currentStartChar = prevStart
            currentSlice = null
            return@withLock renderLocked(c, policy, keepHistory = false)
        }

    override suspend fun goTo(locator: Locator, policy: RenderPolicy): ReaderResult<RenderPage> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Err(
                ReaderError.Internal("LayoutConstraints not set")
            )

            val target = parseTxtOffset(locator) ?: 0
            val start = findReasonablePageStartLocked(target, c)
            historyStarts.clear()
            currentStartChar = start
            currentSlice = null
            return@withLock renderLocked(c, policy, keepHistory = false)
        }

    override suspend fun goToProgress(percent: Double, policy: RenderPolicy): ReaderResult<RenderPage> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Err(
                ReaderError.Internal("LayoutConstraints not set")
            )
            val total = store.totalChars().coerceAtLeast(0)
            if (total == 0) return@withLock renderLocked(c, policy, keepHistory = false)

            val p = percent.coerceIn(0.0, 1.0)
            val target = (p * total).toInt().coerceIn(0, max(0, total - 1))
            val start = findReasonablePageStartLocked(target, c)

            historyStarts.clear()
            currentStartChar = start
            currentSlice = null
            return@withLock renderLocked(c, policy, keepHistory = false)
        }

    override suspend fun prefetchNeighbors(count: Int): ReaderResult<Unit> =
        mutex.withLock {
            val c = constraints ?: return@withLock ReaderResult.Ok(Unit)
            val slice = ensureSliceLocked(c, RenderPolicy.Default) ?: return@withLock ReaderResult.Ok(Unit)

            val total = store.totalChars()
            // next
            if (slice.endChar < total) {
                prefetchAtLocked(slice.endChar, c)
            }
            // prev：用历史或启发式
            val prevStart = if (historyStarts.isNotEmpty()) historyStarts.last() else computePrevStartLocked(c, RenderPolicy.Default)
            if (prevStart >= 0 && prevStart < slice.startChar) {
                prefetchAtLocked(prevStart, c)
            }

            ReaderResult.Ok(Unit)
        }

    override suspend fun invalidate(reason: InvalidateReason): ReaderResult<Unit> =
        mutex.withLock {
            cache.clear()
            currentSlice = null
            if (reason != InvalidateReason.LAYOUT_CHANGED) {
                // 布局变化通常会由 setLayoutConstraints 触发清空历史
                // 这里保留历史也行；先不动
            }
            ReaderResult.Ok(Unit)
        }

    override fun close() {
        // no-op（document 关闭时会释放 store）
    }

    // ---------------- internal ----------------

    private suspend fun renderLocked(
        constraints: LayoutConstraints,
        policy: RenderPolicy,
        keepHistory: Boolean
    ): ReaderResult<RenderPage> = withContext(ioDispatcher) {
        try {
            val startMs = SystemClock.elapsedRealtime()
            val (slice, cacheHit) = getOrBuildSliceLocked(currentStartChar, constraints, policy)

            currentSlice = slice

            val mapping = TxtTextMapping(slice.startChar)
            val locator = Locator(LocatorSchemes.TXT_OFFSET, slice.startChar.toString())
            val total = store.totalChars().coerceAtLeast(0)
            val pct = if (total == 0) 0.0 else slice.startChar.toDouble() / total.toDouble()

            val nav = NavigationAvailability(
                canGoPrev = slice.startChar > 0,
                canGoNext = slice.endChar < total
            )

            val progression = Progression(
                percent = pct.coerceIn(0.0, 1.0),
                label = "${(pct * 100).toInt()}%"
            )

            // decorations（可选）
            val decorations = runCatching {
                annotations?.decorationsFor(
                    AnnotationQuery(
                        range = LocatorRange(
                            start = Locator(LocatorSchemes.TXT_OFFSET, slice.startChar.toString()),
                            end = Locator(LocatorSchemes.TXT_OFFSET, slice.endChar.toString())
                        )
                    )
                )?.let { res ->
                    when (res) {
                        is ReaderResult.Ok -> res.value
                        is ReaderResult.Err -> emptyList()
                    }
                } ?: emptyList()
            }.getOrElse { emptyList() }

            val pageId = PageId("txt:${slice.startChar}")
            val metrics = RenderMetrics(
                renderTimeMs = (SystemClock.elapsedRealtime() - startMs),
                cacheHit = cacheHit
            )

            val page = RenderPage(
                id = pageId,
                locator = locator,
                content = RenderContent.Text(
                    text = slice.text,
                    mapping = mapping
                ),
                links = emptyList(),
                decorations = decorations,
                metrics = metrics
            )

            _state.value = RenderState(
                locator = locator,
                progression = progression,
                nav = nav,
                titleInView = null,
                config = config
            )

            _events.tryEmit(ReaderEvent.PageChanged(locator))
            _events.tryEmit(ReaderEvent.Rendered(pageId, metrics))

            ReaderResult.Ok(page)
        } catch (t: Throwable) {
            _events.tryEmit(ReaderEvent.Error(t))
            ReaderResult.Err(ReaderError.Internal(message = t.message, cause = t))
        }
    }

    private suspend fun ensureSliceLocked(
        constraints: LayoutConstraints,
        policy: RenderPolicy
    ): PageSlice? {
        return currentSlice ?: getOrBuildSliceLocked(currentStartChar, constraints, policy).first
    }

    private suspend fun getOrBuildSliceLocked(
        startChar: Int,
        constraints: LayoutConstraints,
        policy: RenderPolicy
    ): Pair<PageSlice, Boolean> {
        cache.get(startChar)?.let { return it to true }

        val slice = pager.pageAt(startChar, constraints, config)
        cache.put(slice)
        return slice to false
    }

    private suspend fun prefetchAtLocked(startChar: Int, constraints: LayoutConstraints) {
        if (cache.get(startChar) != null) return
        val slice = pager.pageAt(startChar, constraints, config)
        cache.put(slice)
    }

    private fun parseTxtOffset(locator: Locator): Int? {
        if (locator.scheme != LocatorSchemes.TXT_OFFSET) return null
        return locator.value.toIntOrNull()
    }

    private suspend fun findReasonablePageStartLocked(
        targetChar: Int,
        constraints: LayoutConstraints
    ): Int {
        val total = store.totalChars().coerceAtLeast(0)
        if (total == 0) return 0
        val t = targetChar.coerceIn(0, max(0, total - 1))

        val estimate = pager.estimateCharsPerPage(constraints, config)
        var start = (t - estimate).coerceAtLeast(0)

        // 向前推进几次直到包含 target
        repeat(12) {
            val slice = pager.pageAt(start, constraints, config)
            cache.put(slice)
            if (t < slice.endChar) return slice.startChar
            if (slice.endChar >= total) return slice.startChar
            start = slice.endChar
        }
        return start.coerceIn(0, max(0, total - 1))
    }

    private suspend fun computePrevStartLocked(
        constraints: LayoutConstraints,
        policy: RenderPolicy
    ): Int {
        val total = store.totalChars().coerceAtLeast(0)
        if (total == 0) return 0
        if (currentStartChar <= 0) return 0

        val estimate = pager.estimateCharsPerPage(constraints, config)
        var probe = (currentStartChar - estimate * 2).coerceAtLeast(0)

        var lastStart = 0
        repeat(12) {
            val slice = pager.pageAt(probe, constraints, config)
            cache.put(slice)

            if (slice.endChar >= currentStartChar) {
                // lastStart 是上一页起点
                return lastStart.coerceAtMost(currentStartChar)
            }
            lastStart = slice.startChar
            probe = slice.endChar
        }
        return max(0, currentStartChar - estimate)
    }
}
```

```kotlin
        val safe = maxChars.coerceIn(32, 10_000)
        return ReaderResult.Ok(store.readAround(off.coerceAtLeast(0), safe))
    }

    private fun parse(locator: Locator): Int? {
        if (locator.scheme != LocatorSchemes.TXT_OFFSET) return null
        return locator.value.toIntOrNull()
    }
```

```kotlin
package com.example.reader.engines.txt.internal.provider

import com.example.reader.api.provider.SearchHit
import com.example.reader.api.provider.SearchOptions
import com.example.reader.api.provider.SearchProvider
import com.example.reader.engines.txt.internal.storage.TxtTextStore
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.flow
import kotlinx.coroutines.withContext

internal class TxtSearchProvider(
    private val store: TxtTextStore,
    private val ioDispatcher: CoroutineDispatcher = Dispatchers.IO
) : SearchProvider {

    override fun search(query: String, options: SearchOptions): Flow<SearchHit> = flow {
        if (query.isBlank()) return@flow

        val q = if (options.caseSensitive) query else query.lowercase()
        val chunkSize = 64 * 1024
        val overlap = (q.length + 64).coerceAtMost(8_192)

        val total = withContext(ioDispatcher) { store.totalChars() }
        var cursor = options.startFrom?.let { parse(it) } ?: 0
        cursor = cursor.coerceIn(0, total)

        var emitted = 0

        while (cursor < total && emitted < options.maxHits) {
            val read = withContext(ioDispatcher) {
                store.readChars(cursor, (chunkSize + overlap).coerceAtMost(total - cursor))
            }
            if (read.isEmpty()) break

            val hay = if (options.caseSensitive) read else read.lowercase()

            var idx = 0
            while (idx >= 0 && emitted < options.maxHits) {
                idx = hay.indexOf(q, startIndex = idx)
                if (idx < 0) break

                // wholeWord：简单边界判定
                if (options.wholeWord && !isWholeWord(hay, idx, q.length)) {
                    idx += q.length
                    continue
                }

                val start = cursor + idx
                val end = start + q.length

                val excerpt = buildExcerpt(read, idx, q.length)

                emit(
                    SearchHit(
                        range = LocatorRange(
                            start = Locator(LocatorSchemes.TXT_OFFSET, start.toString()),
                            end = Locator(LocatorSchemes.TXT_OFFSET, end.toString())
                        ),
                        excerpt = excerpt,
                        sectionTitle = null
                    )
                )
                emitted++
                idx += q.length
            }

            // 下一段，保留 overlap 防止跨边界漏命中
            cursor += chunkSize
        }
    }

    private fun parse(locator: Locator): Int {
        if (locator.scheme != LocatorSchemes.TXT_OFFSET) return 0
        return locator.value.toIntOrNull() ?: 0
    }

    private fun isWholeWord(text: String, start: Int, len: Int): Boolean {
        val before = start - 1
        val after = start + len
        val leftOk = before < 0 || !isWordChar(text[before])
        val rightOk = after >= text.length || !isWordChar(text[after])
        return leftOk && rightOk
    }

    private fun isWordChar(c: Char): Boolean =
        c.isLetterOrDigit() || c == '_'

    private fun buildExcerpt(src: String, hitStart: Int, hitLen: Int): String {
        val left = (hitStart - 40).coerceAtLeast(0)
        val right = (hitStart + hitLen + 60).coerceAtMost(src.length)
        val prefix = if (left > 0) "…" else ""
        val suffix = if (right < src.length) "…" else ""
        return prefix + src.substring(left, right).replace('\n', ' ') + suffix
    }
}
```

```kotlin
package com.example.reader.engines.txt.internal.provider

import com.example.reader.api.annotation.Decoration
import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.AnnotationProvider
import com.example.reader.api.provider.AnnotationQuery
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.annotation.Annotation
import com.example.reader.model.annotation.AnnotationAnchor
import com.example.reader.model.annotation.AnnotationDraft
import com.example.reader.model.annotation.AnnotationId
import java.util.UUID
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.asStateFlow

internal class InMemoryAnnotationProvider : AnnotationProvider {

    private val state = MutableStateFlow<List<Annotation>>(emptyList())

    override fun observeAll(): Flow<List<Annotation>> = state.asStateFlow()

    override suspend fun listAll(): ReaderResult<List<Annotation>> =
        ReaderResult.Ok(state.value)

    override suspend fun query(query: AnnotationQuery): ReaderResult<List<Annotation>> {
        val all = state.value
        val range = query.range
        if (range != null) {
            val qs = parse(range.start) ?: 0
            val qe = parse(range.end) ?: 0
            val a = minOf(qs, qe)
            val b = maxOf(qs, qe)
            val filtered = all.filter { ann ->
                val r = ann.anchor as? AnnotationAnchor.ReflowRange ?: return@filter false
                val as0 = parse(r.range.start) ?: return@filter false
                val ae0 = parse(r.range.end) ?: return@filter false
                overlaps(a, b, minOf(as0, ae0), maxOf(as0, ae0))
            }
            return ReaderResult.Ok(filtered)
        }
        return ReaderResult.Ok(all)
    }

    override suspend fun create(draft: AnnotationDraft): ReaderResult<Annotation> {
        val anchor = draft.anchor as? AnnotationAnchor.ReflowRange
            ?: return ReaderResult.Err(ReaderError.Internal("TXT supports only ReflowRange annotations"))
        if (!isTxt(anchor.range.start) || !isTxt(anchor.range.end)) {
            return ReaderResult.Err(ReaderError.Internal("Invalid txt.offset anchor"))
        }

        val now = System.currentTimeMillis()
        val ann = Annotation(
            id = AnnotationId(UUID.randomUUID().toString()),
            type = draft.type,
            anchor = draft.anchor,
            content = draft.content,
            style = draft.style,
            createdAtEpochMs = now,
            updatedAtEpochMs = now,
            extra = draft.extra
        )
        state.value = state.value + ann
        return ReaderResult.Ok(ann)
    }

    override suspend fun update(annotation: Annotation): ReaderResult<Unit> {
        state.value = state.value.map { if (it.id == annotation.id) annotation.copy(updatedAtEpochMs = System.currentTimeMillis()) else it }
        return ReaderResult.Ok(Unit)
    }

    override suspend fun delete(id: AnnotationId): ReaderResult<Unit> {
        state.value = state.value.filterNot { it.id == id }
        return ReaderResult.Ok(Unit)
    }

    override suspend fun decorationsFor(query: AnnotationQuery): ReaderResult<List<Decoration>> {
        val anns = query(query).let {
            when (it) {
                is ReaderResult.Ok -> it.value
                is ReaderResult.Err -> return ReaderResult.Err(it.error)
            }
        }
        val decorations = anns.mapNotNull { ann ->
            val r = ann.anchor as? AnnotationAnchor.ReflowRange ?: return@mapNotNull null
            Decoration.Reflow(range = r.range, style = ann.style)
        }
        return ReaderResult.Ok(decorations)
    }

    private fun isTxt(locator: Locator): Boolean = locator.scheme == LocatorSchemes.TXT_OFFSET

    private fun parse(locator: Locator): Int? {
        if (locator.scheme != LocatorSchemes.TXT_OFFSET) return null
        return locator.value.toIntOrNull()
    }

    private fun overlaps(a0: Int, a1: Int, b0: Int, b1: Int): Boolean {
        // [a0,a1) 与 [b0,b1) 是否相交
        return a0 < b1 && b0 < a1
    }
}
```

```kotlin
import com.example.reader.model.SessionId
import java.util.UUID
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext

internal class TxtSession private constructor(
    override val id: SessionId,
    override val controller: ReaderController,
    override val outline: OutlineProvider?,
    override val search: SearchProvider?,
    override val text: TextProvider?,
    override val annotations: AnnotationProvider?,
    override val resources: ResourceProvider?
) : ReaderSession {

    override fun close() {
        runCatching { controller.close() }
    }

    companion object {
        suspend fun create(
            documentId: DocumentId,
            store: TxtTextStore,
            initialLocator: Locator,
            initialConfig: com.example.reader.api.render.RenderConfig.ReflowText,
            ioDispatcher: CoroutineDispatcher
        ): ReaderResult<ReaderSession> = withContext(ioDispatcher) {
            try {
                val pager = TxtPager(store)
                val annotationProvider = InMemoryAnnotationProvider()

                val controller = TxtController(
                    store = store,
                    pager = pager,
                    ioDispatcher = ioDispatcher,
                    annotations = annotationProvider
                ).apply {
                    // 先设置 config
                    setConfig(initialConfig)
                    // 初始位置（locator 只有 offset）
                    val off = if (initialLocator.scheme == LocatorSchemes.TXT_OFFSET)
                        initialLocator.value.toIntOrNull() ?: 0
                    else 0
                    // controller 需要 layout 后才能 render；这里只设置内部起点
                    // 通过 goTo 可在 UI 约束设置后调用；这里我们直接写入起点逻辑：
                    // TxtController 当前实现起点为 0，外层 feature 可在拿到 handle 后 goTo(initialLocator)
                    // 为了更“开箱即用”，我们这里尝试 goTo（若没约束会返回 Err，不影响）
                    runCatching { goTo(initialLocator) }
                }

                val session = TxtSession(
                    id = SessionId(UUID.randomUUID().toString()),
                    controller = controller,
                    outline = null,
                    search = TxtSearchProvider(store, ioDispatcher),
                    text = TxtTextProvider(store),
                    annotations = annotationProvider,
                    resources = null
                )

                ReaderResult.Err(com.example.reader.engines.txt.toReaderError(t))
            }

            com.example.reader.engines.txt.run { t.toReaderError() }
    }
```
