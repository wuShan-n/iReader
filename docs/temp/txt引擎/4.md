```kotlin
fun addSortedUnique(value: Int): Boolean {
    if (_size == 0) {
        add(value)
        return true
    }
    val idx = binarySearch(value)
    if (idx >= 0) return false // already exists
    val ins = -idx - 1
    ensureCapacity(_size + 1)
    if (ins < _size) {
        System.arraycopy(arr, ins, arr, ins + 1, _size - ins)
    }
    arr[ins] = value
    _size++
    return true
}

fun toIntArrayCopy(): IntArray = arr.copyOf(_size)
```

## 2) 新增 `TxtOutlineCache.kt`

路径：`engines/txt/.../internal/provider/TxtOutlineCache.kt`

```kotlin
package com.example.reader.engines.txt.internal.provider

import com.example.reader.engines.txt.TxtEngineConfig
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.OutlineNode
import java.io.File

internal class TxtOutlineCache(
    private val config: TxtEngineConfig,
    private val docNamespace: String,
    private val charsetName: String
) {
    @Volatile private var memory: List<OutlineNode>? = null

    fun getMemory(): List<OutlineNode>? = memory
    fun setMemory(nodes: List<OutlineNode>) { memory = nodes }

    fun loadFromDisk(): List<OutlineNode>? {
        if (!config.persistOutline) return null
        val f = file() ?: return null
        if (!f.exists()) return null

        return runCatching {
            val lines = f.readLines(Charsets.UTF_8)
            lines.mapNotNull { line ->
                val tab = line.indexOf('\t')
                if (tab <= 0) return@mapNotNull null
                val off = line.substring(0, tab).toIntOrNull() ?: return@mapNotNull null
                val title = line.substring(tab + 1)
                OutlineNode(
                    title = title,
                    locator = Locator(LocatorSchemes.TXT_OFFSET, off.toString()),
                    children = emptyList()
                )
            }
        }.getOrNull()
    }

    fun saveToDisk(nodes: List<OutlineNode>) {
        if (!config.persistOutline) return
        val f = file() ?: return
        runCatching {
            f.parentFile?.mkdirs()
            val tmp = File(f.parentFile, f.name + ".tmp")
            tmp.writeText(
                buildString {
                    for (n in nodes) {
                        val off = n.locator.value
                        append(off).append('\t').append(n.title.replace('\n', ' ')).append('\n')
                    }
                },
                Charsets.UTF_8
            )
            if (f.exists()) f.delete()
            tmp.renameTo(f)
        }
    }

    private fun file(): File? {
        val base = config.cacheDir ?: return null
        val dir = File(base, "reader-txt/outline")
        val safe = "${docNamespace.hashCode()}_${charsetName.hashCode()}.txt"
        return File(dir, safe)
    }
}
```

## 3) 新增 `TxtPaginationStore.kt`（替换你原来的同名文件）

> 这份是**带磁盘 backing** 的完整实现。用它替换原文件即可。

路径：`engines/txt/.../internal/paging/TxtPaginationStore.kt`

```kotlin
package com.example.reader.engines.txt.internal.paging

import com.example.reader.engines.txt.TxtEngineConfig
import java.io.File
import java.security.MessageDigest
import java.util.concurrent.ConcurrentHashMap

internal class TxtPaginationStore(
    private val config: TxtEngineConfig,
    private val docNamespace: String,
    private val charsetName: String
) {
    private data class Bucket(
        val index: PageStartsIndex,
        var newStartsSinceLastWrite: Int = 0
    )

    private val map = ConcurrentHashMap<RenderKey, Bucket>()

    fun getOrCreate(key: RenderKey): PageStartsIndex {
        return map.getOrPut(key) {
            val idx = PageStartsIndex()
            idx.seedIfEmpty(0)

            // 尝试从磁盘加载
            if (config.persistPagination) {
                loadFromDisk(key)?.let { loaded ->
                    idx.mergeFrom(loaded)
                }
            }
            Bucket(idx)
        }.index
    }

    fun maybePersist(key: RenderKey, index: PageStartsIndex) {
        if (!config.persistPagination) return
        val bucket = map[key] ?: return
        bucket.newStartsSinceLastWrite++

        if (bucket.newStartsSinceLastWrite >= config.paginationWriteEveryNewStarts) {
            bucket.newStartsSinceLastWrite = 0
            saveToDisk(key, index)
        }
    }

    fun flush(key: RenderKey, index: PageStartsIndex) {
        if (!config.persistPagination) return
        saveToDisk(key, index)
        map[key]?.newStartsSinceLastWrite = 0
    }

    private fun loadFromDisk(key: RenderKey): IntArray? {
        val f = fileFor(key) ?: return null
        if (!f.exists()) return null
        return runCatching { PageStartsCodec.decode(f.readBytes()) }.getOrNull()
    }

    private fun saveToDisk(key: RenderKey, index: PageStartsIndex) {
        val f = fileFor(key) ?: return
        runCatching {
            f.parentFile?.mkdirs()
            val tmp = File(f.parentFile, f.name + ".tmp")
            val bytes = PageStartsCodec.encode(index.snapshot())
            tmp.writeBytes(bytes)
            if (f.exists()) f.delete()
            tmp.renameTo(f)
        }
    }

    private fun fileFor(key: RenderKey): File? {
        val base = config.cacheDir ?: return null
        val dir = File(base, "reader-txt/pagination/${docNamespace.hashCode()}_${charsetName.hashCode()}")
        val name = stableName(key) + ".bin"
        return File(dir, name)
    }

    private fun stableName(key: RenderKey): String {
        val s = key.toString()
        val md = MessageDigest.getInstance("SHA-256").digest(s.toByteArray(Charsets.UTF_8))
        // 取前 10 bytes -> 20 hex 字符，足够且文件名短
        return buildString(20) {
            for (i in 0 until 10) {
                val b = md[i].toInt() and 0xFF
                append("0123456789abcdef"[b ushr 4])
                append("0123456789abcdef"[b and 0x0F])
            }
        }
    }
}
```

## 4) 新增 `PageStartsCodec.kt`（delta + varint 压缩）

路径：`engines/txt/.../internal/paging/PageStartsCodec.kt`

```kotlin
package com.example.reader.engines.txt.internal.paging

import java.io.ByteArrayOutputStream

internal object PageStartsCodec {

    fun encode(sortedUnique: IntArray): ByteArray {
        val out = ByteArrayOutputStream(sortedUnique.size * 2)
        var prev = 0
        for (v in sortedUnique) {
            val delta = (v - prev).coerceAtLeast(0)
            writeVarInt(out, delta)
            prev = v
        }
        return out.toByteArray()
    }

    fun decode(bytes: ByteArray): IntArray {
        val list = ArrayList<Int>(256)
        var i = 0
        var prev = 0
        while (i < bytes.size) {
            val (value, next) = readVarInt(bytes, i)
            i = next
            prev += value
            list.add(prev)
        }
        return list.toIntArray()
    }

    private fun writeVarInt(out: ByteArrayOutputStream, value: Int) {
        var v = value
        while (true) {
            val b = v and 0x7F
            v = v ushr 7
            if (v == 0) {
                out.write(b)
                return
            } else {
                out.write(b or 0x80)
            }
        }
    }

    private fun readVarInt(bytes: ByteArray, start: Int): Pair<Int, Int> {
        var result = 0
        var shift = 0
        var i = start
        while (i < bytes.size) {
            val b = bytes[i].toInt() and 0xFF
            result = result or ((b and 0x7F) shl shift)
            i++
            if ((b and 0x80) == 0) break
            shift += 7
        }
        return result to i
    }
}
```
