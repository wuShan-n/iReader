## A) 分层缓存（DRAFT 小缓存 + FINAL 大缓存）

### 1) 新增：`TieredTileCache.kt`

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/cache/TieredTileCache.kt
package com.example.reader.pdf.internal.cache

import android.graphics.Bitmap
import com.example.reader.api.render.RenderPolicy

internal class TieredTileCache(
    private val draftCache: TileCache,
    private val finalCache: TileCache
) {
    fun get(key: String, quality: RenderPolicy.Quality): Bitmap? =
        when (quality) {
            RenderPolicy.Quality.DRAFT -> draftCache.get(key)
            RenderPolicy.Quality.FINAL -> finalCache.get(key)
        }

    fun put(key: String, bitmap: Bitmap, quality: RenderPolicy.Quality) {
        when (quality) {
            RenderPolicy.Quality.DRAFT -> draftCache.put(key, bitmap)
            RenderPolicy.Quality.FINAL -> finalCache.put(key, bitmap)
        }
    }

    fun evictAll() {
        draftCache.evictAll()
        finalCache.evictAll()
    }

    fun entryCount(quality: RenderPolicy.Quality): Int =
        when (quality) {
            RenderPolicy.Quality.DRAFT -> draftCache.entryCount()
            RenderPolicy.Quality.FINAL -> finalCache.entryCount()
        }
}
```

// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/cache/TieredCacheSizing.kt
package com.example.reader.pdf.internal.cache

internal data class TieredCacheBudget(
    val draftBytes: Int,
    val finalBytes: Int
)

internal fun defaultPdfTieredCacheBudget(
    totalBytes: Int = defaultPdfTileCacheBytes(),
    finalRatio: Float = 0.75f
): TieredCacheBudget {
    val ratio = finalRatio.coerceIn(0.5f, 0.9f)
    val finalB = (totalBytes * ratio).toInt().coerceAtLeast(8 * 1024 * 1024)
    val draftB = (totalBytes - finalB).coerceAtLeast(4 * 1024 * 1024)
    return TieredCacheBudget(draftBytes = draftB, finalBytes = finalB)
}

## B) TileProvider 接入分层缓存 + in-flight 去重（适配你上一轮）

### 3) 修改：`PdfTileProvider.kt`（缓存从 `TileCache` 换成 `TieredTileCache`）

> 只贴需要改的内容：构造参数/缓存读写点/evictAll。

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/render/PdfTileProvider.kt
package com.example.reader.pdf.internal.render

import android.graphics.Bitmap
import android.graphics.Rect
import com.example.reader.api.render.RenderPolicy
import com.example.reader.api.render.TileProvider
import com.example.reader.api.render.TileRequest
import com.example.reader.pdf.internal.backend.PdfBackend
import com.example.reader.pdf.internal.backend.PdfPageSize
import com.example.reader.pdf.internal.backend.PdfRenderMode
import com.example.reader.pdf.internal.cache.TieredTileCache
import com.example.reader.pdf.internal.cache.TileInflight
import java.io.Closeable
import java.util.concurrent.atomic.AtomicBoolean
import kotlin.math.max
import kotlinx.coroutines.CancellationException
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.withContext
import kotlin.coroutines.coroutineContext

internal class PdfTileProvider(
    private val backend: PdfBackend,
    private val pageIndex: Int,
    private val basePage: PdfPageSize,
    private val baseScale: Float,
    private val rotationDegrees: Int,
    private val allowCache: Boolean,
    private val cacheDraft: Boolean,            // 是否缓存 DRAFT（分层缓存下仍然有意义）
    private val cache: TieredTileCache,         // ✅ 修改：分层缓存
    private val inflight: TileInflight,
    private val ioDispatcher: CoroutineDispatcher,
    private val isValid: () -> Boolean
) : TileProvider, Closeable {

    private val closed = AtomicBoolean(false)

    override suspend fun renderTile(request: TileRequest): Bitmap = withContext(ioDispatcher) {
        coroutineContext.ensureActive()
        if (closed.get() || !isValid()) throw CancellationException("stale tile provider")

        val scale = max(0.05f, baseScale * request.scale)

        val canCacheThis = allowCache && (request.quality == RenderPolicy.Quality.FINAL || cacheDraft)
        val key = if (canCacheThis) buildKey(request, scale) else ""

        if (canCacheThis) {
            cache.get(key, request.quality)?.let { return@withContext it }
            return@withContext inflight.getOrAwait(key) {
                cache.get(key, request.quality) ?: renderInternal(request, scale, key, canCacheThis)
            }
        }

        renderInternal(request, scale, key = "", canCacheThis = false)
    }

    private suspend fun renderInternal(
        request: TileRequest,
        scale: Float,
        key: String,
        canCacheThis: Boolean
    ): Bitmap {
        coroutineContext.ensureActive()

        val bmpConfig = when (request.quality) {
            RenderPolicy.Quality.DRAFT -> Bitmap.Config.RGB_565
            RenderPolicy.Quality.FINAL -> Bitmap.Config.ARGB_8888
        }

        val w = request.widthPx.coerceAtLeast(1)
        val h = request.heightPx.coerceAtLeast(1)
        val bytesPerPixel = if (bmpConfig == Bitmap.Config.RGB_565) 2 else 4
        val estBytes = 1L * w * h * bytesPerPixel

        val max = Runtime.getRuntime().maxMemory().coerceAtLeast(128L * 1024 * 1024)
        val hardLimit = max / 10L

        if (estBytes > hardLimit) {
            if (allowCache) cache.evictAll()
            return Bitmap.createBitmap(1, 1, Bitmap.Config.ARGB_8888)
        }

        try {
            val bmp = Bitmap.createBitmap(w, h, bmpConfig)
            val clip = Rect(0, 0, bmp.width, bmp.height)

            val matrix = buildPdfTileMatrix(
                basePage = basePage,
                scale = scale,
                rotationDegrees = rotationDegrees,
                leftPx = request.leftPx,
                topPx = request.topPx
            )

            val mode = when (request.quality) {
                RenderPolicy.Quality.DRAFT -> PdfRenderMode.DRAFT
                RenderPolicy.Quality.FINAL -> PdfRenderMode.FINAL
            }

            backend.render(pageIndex, bmp, clip, matrix, mode)

            if (canCacheThis && bmp.width > 1 && bmp.height > 1) {
                cache.put(key, bmp, request.quality) // ✅ 修改：写入分层缓存
            }
            return bmp
        } catch (ce: CancellationException) {
            throw ce
        } catch (oom: OutOfMemoryError) {
            if (allowCache) cache.evictAll()
            return Bitmap.createBitmap(1, 1, Bitmap.Config.ARGB_8888)
        }
    }

    override fun close() {
        closed.set(true)
    }

    // buildKey(...) 保持不变
}
```

## C) 当前页首屏 FINAL 自动预热（停住后补清晰）

核心做法：Controller 内建一个 `CoroutineScope`，在每次 `render()` 成功后 **debounce** 触发 `preheatCurrentPageFinal()`：

* 用户连续滑动/缩放时：前一个 job 会 cancel，不会浪费
* 用户停住一小会：后台预热首屏 FINAL tile（写入 FINAL cache），UI 下一次请求会直接命中

> 只贴需要新增/替换的片段。

    draftCache = com.example.reader.pdf.internal.cache.TileCache(tierBudget.draftBytes),
    finalCache = com.example.reader.pdf.internal.cache.TileCache(tierBudget.finalBytes)

#### 4.4 新增：`scheduleFinalPreheatIfIdle()` + `preheatCurrentPageFinal()`

```kotlin
// File: engines/pdf/.../PdfController.kt 末尾附近新增

private fun scheduleFinalPreheatIfIdle() {
    val c = constraints ?: return
    val gen = generation.get()

    finalPreheatJob?.cancel()
    finalPreheatJob = scope.launch {
        delay(finalPreheatDebounceMs)
        if (!isActive) return@launch
        // 若期间发生翻页/缩放/旋转/布局改变，则放弃（代际不同）
        if (generation.get() != gen) return@launch
        preheatCurrentPageFinal(c, gen)
    }
}

private suspend fun preheatCurrentPageFinal(
    c: LayoutConstraints,
    gen: Long
) {
    // 只预热首屏（viewport 尺寸）；UI 常见就是这样请求 tile
    val base = backend.pageSize(pageIndex)
    val transform = computeTransform(base, config, c)

    val w = c.viewportWidthPx.coerceAtMost(transform.pageWidthPx).coerceAtLeast(1)
    val h = c.viewportHeightPx.coerceAtMost(transform.pageHeightPx).coerceAtLeast(1)

    val provider = PdfTileProvider(
        backend = backend,
        pageIndex = pageIndex,
        basePage = base,
        baseScale = transform.scale,
        rotationDegrees = config.rotationDegrees,
        allowCache = true,
        cacheDraft = false,     // FINAL 预热只进 final tier
        cache = cache,
        inflight = inflight,    // 与前台共享，避免重复渲染
        ioDispatcher = ioDispatcher,
        isValid = { generation.get() == gen }
    )

    try {
        provider.renderTile(
            TileRequest(
                leftPx = 0,
                topPx = 0,
                widthPx = w,
                heightPx = h,
                scale = 1.0f,
                quality = RenderPolicy.Quality.FINAL
            )
        )
    } finally {
        provider.close()
    }
}
```

## 这两项优化的“体感提升”

* **快速滑动/缩放**：DRAFT 命中更高（小缓存），首屏出现更快
* **停住后清晰化**：你不需要 UI 改逻辑，Controller 会自动把当前页首屏 FINAL 预热进 FINAL cache
* **CPU/IO 更省**：in-flight 去重 + debounce，避免重复渲染
