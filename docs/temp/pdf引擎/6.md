下面是 **“PdfRenderer 负责渲染 + Pdfium 负责 outline/links/text/search”** 的**新增/修改代码**（不重复你已有文件）。
实现策略：**渲染仍走你现有 PlatformPdfBackend(PdfRenderer)**；在 `PdfDocumentImpl` 打开时同时（best-effort）打开 Pdfium 文档，Session 里注入 `OutlineProvider/TextProvider/SearchProvider`，并且 `PdfController.render()` 填充 `RenderPage.links`。

## 1.1 新增 `PdfiumDocument.kt`

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/pdfium/PdfiumDocument.kt
package com.example.reader.pdf.internal.pdfium

import android.content.Context
import android.graphics.RectF
import android.os.ParcelFileDescriptor
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.OutlineNode
import com.example.reader.model.Locator
import com.example.reader.model.NormalizedRect
import com.example.reader.model.DocumentLink
import com.example.reader.model.LinkTarget
import com.shockwave.pdfium.PdfDocument
import com.shockwave.pdfium.PdfiumCore
import java.io.Closeable
import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock
import kotlin.coroutines.coroutineContext
import kotlinx.coroutines.ensureActive

internal class PdfiumDocument private constructor(
    private val context: Context,
    private val core: PdfiumCore,
    private val doc: PdfDocument,
    private val pfd: ParcelFileDescriptor,
    private val textApi: PdfiumTextApi?
) : Closeable {

    private val mutex = Mutex()

    val pageCount: Int get() = core.getPageCount(doc)

    val supportsText: Boolean get() = textApi != null

    companion object {
        fun open(
            context: Context,
            pfd: ParcelFileDescriptor,
            password: String?
        ): PdfiumDocument {
            val core = PdfiumCore(context)
            val pdfDoc = if (password.isNullOrEmpty()) {
                core.newDocument(pfd)
            } else {
                core.newDocument(pfd, password)
            }
            val textApi = PdfiumTextApi.tryCreate(core)
            return PdfiumDocument(context.applicationContext, core, pdfDoc, pfd, textApi)
        }
    }

    suspend fun outline(): List<OutlineNode> = mutex.withLock {
        val toc = core.getTableOfContents(doc) ?: emptyList()
        toc.map { it.toOutlineNode() }
    }

    suspend fun pageLinks(pageIndex: Int, rotationDegrees: Int): List<DocumentLink> = mutex.withLock {
        coroutineContext.ensureActive()
        core.openPage(doc, pageIndex)

        val pageW = core.getPageWidthPoint(doc, pageIndex).toFloat().coerceAtLeast(1f)
        val pageH = core.getPageHeightPoint(doc, pageIndex).toFloat().coerceAtLeast(1f)

        val links = core.getPageLinks(doc, pageIndex) ?: emptyList()
        links.mapNotNull { link ->
            val bounds: RectF? = link.bounds
            val rects = bounds?.let {
                val n = NormalizedRect(
                    left = (it.left / pageW).coerceIn(0f, 1f),
                    top = (it.top / pageH).coerceIn(0f, 1f),
                    right = (it.right / pageW).coerceIn(0f, 1f),
                    bottom = (it.bottom / pageH).coerceIn(0f, 1f)
                )
                listOf(rotateNormalizedRect(n, rotationDegrees))
            }

            val target: LinkTarget? = when {
                link.uri != null -> LinkTarget.External(link.uri)
                link.destPageIdx >= 0 -> LinkTarget.Internal(
                    Locator(LocatorSchemes.PDF_PAGE, link.destPageIdx.toString())
                )
                else -> null
            }

            target?.let { DocumentLink(target = it, title = null, bounds = rects) }
        }
    }

    suspend fun extractPageText(pageIndex: Int): String = mutex.withLock {
        coroutineContext.ensureActive()
        val api = textApi ?: throw UnsupportedOperationException("Pdfium text API not available")
        core.openPage(doc, pageIndex)
        api.extractPageText(core, doc, pageIndex)
    }

    override fun close() {
        runCatching { core.closeDocument(doc) }
        runCatching { pfd.close() }
    }

    private fun PdfDocument.Bookmark.toOutlineNode(): OutlineNode {
        val page = pageIdx.coerceAtLeast(0)
        val children = (children ?: emptyList()).map { it.toOutlineNode() }
        return OutlineNode(
            title = title ?: "",
            locator = Locator(LocatorSchemes.PDF_PAGE, page.toString()),
            children = children
        )
    }
}

/** 把“未旋转页面坐标系”的 normalized rect 转到“旋转后显示坐标系” */
internal fun rotateNormalizedRect(r: NormalizedRect, rotationDegrees: Int): NormalizedRect {
    val rot = ((rotationDegrees % 360) + 360) % 360
    return when (rot) {
        0 -> r
        90 -> NormalizedRect(
            left = (1f - r.bottom),
            top = r.left,
            right = (1f - r.top),
            bottom = r.right
        )
        180 -> NormalizedRect(
            left = (1f - r.right),
            top = (1f - r.bottom),
            right = (1f - r.left),
            bottom = (1f - r.top)
        )
        270 -> NormalizedRect(
            left = r.top,
            top = (1f - r.right),
            right = r.bottom,
            bottom = (1f - r.left)
        )
        else -> r
    }
}
```

## 1.2 新增 `PdfiumTextApi.kt`（反射探测，避免库版本差异导致直接编译/运行崩）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/pdfium/PdfiumTextApi.kt
package com.example.reader.pdf.internal.pdfium

import com.shockwave.pdfium.PdfDocument
import com.shockwave.pdfium.PdfiumCore
import java.lang.reflect.Method

/**
 * PdfiumAndroid 的“文本 API”在不同版本/分支里名字可能不同。
 * 这里用反射做 best-effort：存在就开启 text/search；不存在就让能力为 false。
 */
internal class PdfiumTextApi private constructor(
    private val openTextPage: Method,
    private val closeTextPage: Method,
    private val textCountChars: Method,
    private val textGetText: Method
) {
    companion object {
        fun tryCreate(core: PdfiumCore): PdfiumTextApi? {
            val cls = core.javaClass

            val open = cls.methods.firstOrNull { it.name == "openTextPage" && it.parameterTypes.size == 2 }
                ?: return null
            val close = cls.methods.firstOrNull { it.name == "closeTextPage" && it.parameterTypes.size == 1 }
                ?: return null

            // 常见：textPageCountChars(long) / textPageGetText(long, start, len)
            val count = cls.methods.firstOrNull { it.name == "textPageCountChars" && it.parameterTypes.size == 1 }
                ?: return null
            val getText = cls.methods.firstOrNull { it.name == "textPageGetText" && it.parameterTypes.size == 3 }
                ?: return null

            return PdfiumTextApi(open, close, count, getText)
        }
    }

    fun extractPageText(core: PdfiumCore, doc: PdfDocument, pageIndex: Int): String {
        val textPagePtr = openTextPage.invoke(core, doc, pageIndex) as Long
        try {
            val count = (textCountChars.invoke(core, textPagePtr) as Int).coerceAtLeast(0)
            if (count == 0) return ""
            // 取全量文本（你也可以限制最大长度）
            val result = textGetText.invoke(core, textPagePtr, 0, count) as? String
            return result ?: ""
        } finally {
            closeTextPage.invoke(core, textPagePtr)
        }
    }
}
```

## 2.2 新增 `PdfTextProvider.kt`

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/provider/PdfTextProvider.kt
package com.example.reader.pdf.internal.provider

import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.TextProvider
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import com.example.reader.pdf.internal.pdfium.PdfiumDocument
import com.example.reader.pdf.internal.util.toReaderError

internal class PdfTextProvider(
    private val pdfium: PdfiumDocument
) : TextProvider {

    override suspend fun getText(range: LocatorRange): ReaderResult<String> =
        try {
            val s = range.start
            val e = range.end
            if (s.scheme != LocatorSchemes.PDF_PAGE || e.scheme != LocatorSchemes.PDF_PAGE) {
                return ReaderResult.Ok("")
            }
            val sp = s.value.toIntOrNull() ?: 0
            val ep = e.value.toIntOrNull() ?: sp
            if (sp != ep) {
                // 简化实现：跨页先不拼接（可按需要拼）
                return ReaderResult.Ok(pdfium.extractPageText(sp))
            }
            ReaderResult.Ok(pdfium.extractPageText(sp))
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }

    override suspend fun getTextAround(locator: Locator, maxChars: Int): ReaderResult<String> =
        try {
            if (locator.scheme != LocatorSchemes.PDF_PAGE) return ReaderResult.Ok("")
            val page = locator.value.toIntOrNull() ?: 0
            val text = pdfium.extractPageText(page)
            ReaderResult.Ok(text.take(maxChars.coerceAtLeast(0)))
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }
}
```

## 2.3 新增 `PdfSearchProvider.kt`（按页扫描的 baseline，可逐步升级为“带 rects 高亮”）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/provider/PdfSearchProvider.kt
package com.example.reader.pdf.internal.provider

import com.example.reader.api.provider.SearchHit
import com.example.reader.api.provider.SearchOptions
import com.example.reader.api.provider.SearchProvider
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import com.example.reader.pdf.internal.pdfium.PdfiumDocument
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.flow
import kotlinx.coroutines.withContext
import kotlin.coroutines.coroutineContext

internal class PdfSearchProvider(
    private val pdfium: PdfiumDocument
) : SearchProvider {

    override fun search(query: String, options: SearchOptions): Flow<SearchHit> = flow {
        val q = query.takeIf { it.isNotBlank() } ?: return@flow

        val startPage = options.startFrom
            ?.takeIf { it.scheme == LocatorSchemes.PDF_PAGE }
            ?.value?.toIntOrNull()
            ?.coerceAtLeast(0)
            ?: 0

        val maxHits = options.maxHits.coerceAtLeast(1)
        var hits = 0

        for (page in startPage until pdfium.pageCount) {
            coroutineContext.ensureActive()

            val text = withContext(Dispatchers.IO) { pdfium.extractPageText(page) }
            if (text.isEmpty()) continue

            val hay = if (options.caseSensitive) text else text.lowercase()
            val needle = if (options.caseSensitive) q else q.lowercase()

            var idx = 0
            while (true) {
                coroutineContext.ensureActive()
                val found = hay.indexOf(needle, startIndex = idx)
                if (found < 0) break

                // excerpt：命中前后各取一些
                val s = (found - 24).coerceAtLeast(0)
                val e = (found + needle.length + 24).coerceAtMost(text.length)
                val excerpt = text.substring(s, e).replace('\n', ' ')

                val loc = Locator(LocatorSchemes.PDF_PAGE, page.toString())
                emit(
                    SearchHit(
                        range = LocatorRange(start = loc, end = loc),
                        excerpt = excerpt,
                        sectionTitle = null
                    )
                )

                hits++
                if (hits >= maxHits) return@flow
                idx = found + needle.length
            }
        }
    }
}
```

# 5) 修改：`PdfDocumentImpl` 同时打开 Pdfium（best-effort）并设置 capabilities

关键点：

* PdfRenderer 渲染仍用 `opened.pfd`
* Pdfium 用 **dup 出来的 pfd**（互不影响）
* Pdfium 文本 API 若不可用，则 `textExtraction/search=false`

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/open/PdfDocumentImpl.kt
package com.example.reader.pdf.internal.open

import android.content.Context
import android.os.ParcelFileDescriptor
import com.example.reader.api.engine.ReaderDocument
import com.example.reader.api.engine.ReaderSession
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.open.OpenOptions
import com.example.reader.api.render.RenderConfig
import com.example.reader.model.BookFormat
import com.example.reader.model.DocumentCapabilities
import com.example.reader.model.DocumentId
import com.example.reader.model.DocumentMetadata
import com.example.reader.model.Locator
import com.example.reader.pdf.internal.backend.PlatformPdfBackend
import com.example.reader.pdf.internal.pdfium.PdfiumDocument
import com.example.reader.pdf.internal.session.PdfSessionImpl
import com.example.reader.pdf.internal.util.toReaderError
import com.example.reader.source.DocumentSource
import java.io.Closeable
import java.util.UUID
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext

internal class PdfDocumentImpl(
    private val appContext: Context, // ✅ 新增：用于 PdfiumCore(context)
    private val source: DocumentSource,
    override val openOptions: OpenOptions,
    private val opened: OpenedPdf,
    private val ioDispatcher: CoroutineDispatcher
) : ReaderDocument, Closeable {

    private val backend = PlatformPdfBackend(opened.pfd)

    // ✅ 新增：pdfium best-effort
    private val pdfium: PdfiumDocument? = runCatching {
        val dup = ParcelFileDescriptor.dup(opened.pfd.fileDescriptor)
        PdfiumDocument.open(appContext, dup, openOptions.password)
    }.getOrNull()

    override val id: DocumentId =
        DocumentId(UUID.nameUUIDFromBytes(source.uri.toString().toByteArray()).toString())

    override val format: BookFormat = BookFormat.PDF

    override val capabilities: DocumentCapabilities = DocumentCapabilities(
        reflowable = false,
        fixedLayout = true,
        outline = pdfium != null,
        links = pdfium != null,
        textExtraction = (pdfium?.supportsText == true),
        search = (pdfium?.supportsText == true),
        annotations = true
    )

    override suspend fun metadata(): ReaderResult<DocumentMetadata> = withContext(ioDispatcher) {
        try {
            ReaderResult.Ok(
                DocumentMetadata(
                    title = source.displayName,
                    extra = buildMap {
                        source.mimeType?.let { put("mimeType", it) }
                        source.sizeBytes?.let { put("sizeBytes", it.toString()) }
                        put("pageCount", backend.pageCount.toString())
                        put("pdfium", (pdfium != null).toString())
                        put("pdfiumText", (pdfium?.supportsText == true).toString())
                    }
                )
            )
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }
    }

    override suspend fun createSession(
        initialLocator: Locator?,
        initialConfig: RenderConfig
    ): ReaderResult<ReaderSession> = withContext(ioDispatcher) {
        try {
            val config = (initialConfig as? RenderConfig.FixedPage) ?: RenderConfig.FixedPage()
            val startPage = when {
                initialLocator == null -> 0
                initialLocator.scheme == LocatorSchemes.PDF_PAGE -> initialLocator.value.toIntOrNull() ?: 0
                else -> 0
            }.coerceIn(0, (backend.pageCount - 1).coerceAtLeast(0))

            ReaderResult.Ok(
                PdfSessionImpl(
                    backend = backend,
                    startPageIndex = startPage,
                    initialConfig = config,
                    capabilities = capabilities,
                    ioDispatcher = ioDispatcher,
                    pdfium = pdfium // ✅ 新增
                )
            )
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }
    }

    override fun close() {
        runCatching { pdfium?.close() }     // ✅ 新增
        runCatching { backend.close() }
        runCatching { opened.pfd.close() }
        opened.tempFile?.let { runCatching { it.delete() } }
    }
}
```

> 上面引用了 `LocatorSchemes.PDF_PAGE`，你需要确保 `PdfDocumentImpl.kt` 顶部 import 了 `com.example.reader.model.LocatorSchemes`（新增 import）。

# 7) 当前实现说明（你现在得到什么）

完成以上改动后：

* ✅ 渲染仍走 PdfRenderer + Tiles（你现有体验不会回退）
* ✅ `RenderPage.links` 会输出来自 Pdfium 的热区链接（含 bounds + internal/external target）
* ✅ `Session.outline` 有目录树（Pdfium TOC）
* ✅ `Session.text` / `Session.search` **如果 Pdfium text API 可用**则开启；不可用则自动降级为 null（capabilities 也会反映）
