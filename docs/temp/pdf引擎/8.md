# 1) 新增：Search 高亮存储与通知（跨 SearchProvider ↔ Controller）

## 1.1 新增 `PdfHighlightStore.kt`

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/provider/PdfHighlightStore.kt
package com.example.reader.pdf.internal.provider

import com.example.reader.model.NormalizedRect
import java.util.concurrent.ConcurrentHashMap
import java.util.concurrent.atomic.AtomicLong
import kotlinx.coroutines.flow.MutableSharedFlow
import kotlinx.coroutines.flow.SharedFlow

internal class PdfHighlightStore(
    private val maxRectsPerPage: Int = 128
) {
    private val gen = AtomicLong(0L)
    private val map = ConcurrentHashMap<Int, List<NormalizedRect>>()

    private val _updates = MutableSharedFlow<Int>(extraBufferCapacity = 64)
    val updates: SharedFlow<Int> = _updates

    /** 开启一次新的搜索（清空旧高亮） */
    fun newSearch(): Long {
        val g = gen.incrementAndGet()
        map.clear()
        _updates.tryEmit(ALL_PAGES)
        return g
    }

    fun add(pageIndex: Int, rects: List<NormalizedRect>, g: Long) {
        if (g != gen.get() || rects.isEmpty()) return

        // 合并并限流
        val merged = (map[pageIndex].orEmpty() + rects)
            .distinctBy { "${it.left},${it.top},${it.right},${it.bottom}" }
            .take(maxRectsPerPage)

        map[pageIndex] = merged
        _updates.tryEmit(pageIndex)
    }

    fun rectsFor(pageIndex: Int): List<NormalizedRect> = map[pageIndex].orEmpty()

    companion object {
        const val ALL_PAGES = -1
    }
}
```

# 2) 修改：`PdfSearchProvider.kt`（写入高亮 store + 继续在 extras 写 rects）

```kotlin
// File: engines/pdf/.../PdfController.kt
// ✅ 新增：在类内部某处（比如字段后）加一个 init block

init {
    val store = highlightStore ?: return
    scope.launch {
        store.updates.collect { page ->
            if (page == PdfHighlightStore.ALL_PAGES || page == pageIndex) {
                // 用 PageChanged 作为“当前页内容变化”的信号，提示 UI 重新 render
                _events.tryEmit(ReaderEvent.PageChanged(com.example.reader.pdf.internal.util.pdfPageLocator(pageIndex)))
            }
        }
    }
```

# 5) 优化 Links：在 PdfiumDocument 内做“按页+旋转”缓存，Session 直接用

你现在 linksResolver 是在 session 每次 openPage + mapRectToDevice。把它提升到 `PdfiumDocument` 里缓存起来。

## 5.1 修改：`PdfiumDocument.kt` 新增 `pageLinksNormalized(...)`

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/pdfium/PdfiumDocument.kt
// ✅ 新增：缓存 key=(pageIndex<<2)+rotateIndex，value=List<PdfLinkRaw>

import android.graphics.RectF
import com.example.reader.pdf.internal.cache.IntLruCache
import com.example.reader.pdf.internal.render.PdfLinkRaw
import com.example.reader.model.LinkTarget
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes

private val linkNormCache = IntLruCache<List<PdfLinkRaw>>(maxEntries = 128)

suspend fun pageLinksNormalized(pageIndex: Int, rotationDegrees: Int): List<PdfLinkRaw> {
    val rotate = pdfiumRotateIndex(rotationDegrees)
    val key = (pageIndex shl 2) + rotate
    linkNormCache.get(key)?.let { return it }

    val base = 10_000
    val out = mutableListOf<PdfLinkRaw>()
    withPage(pageIndex) { page ->
        val links = page.getPageLinks()
        for (l in links) {
            val target = when {
                l.uri != null -> LinkTarget.External(l.uri)
                l.destPageIdx != null -> LinkTarget.Internal(
                    Locator(LocatorSchemes.PDF_PAGE, l.destPageIdx.toString())
                )
                else -> null
            } ?: continue

            val device = page.mapRectToDevice(
                startX = 0, startY = 0,
                sizeX = base, sizeY = base,
                rotate = rotate,
                coords = l.bounds
            )
            val n = rectToNormalized(device, base)
            out += PdfLinkRaw(
                target = target,
                bounds = RectF(n.left, n.top, n.right, n.bottom) // normalized 0..1
            )
        }
    }

    linkNormCache.put(key, out)
    return out
}
```

> 另外，在你已有的 `evictAllCaches()` 里把 `linkNormCache.evictAll()` 也加上（如果你有这个方法）。

# 这一轮优化带来的效果

* ✅ 搜索时：命中会自动在当前页 `RenderPage.decorations` 里变成高亮（不需要 feature 自己解析 rects）
* ✅ 搜索命中更新会触发 `PageChanged`，更容易让 UI 触发一次重新 render
* ✅ 链接热区：从“每次 render 都解析”变成“按页+旋转缓存”，滚动/缩放更省
