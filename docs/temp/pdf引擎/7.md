```kotlin
package com.example.reader.pdf.internal.cache

import java.util.LinkedHashMap

internal class IntLruCache<V>(
    private val maxEntries: Int
) {
    private val map = object : LinkedHashMap<Int, V>(maxEntries, 0.75f, true) {
        override fun removeEldestEntry(eldest: MutableMap.MutableEntry<Int, V>?): Boolean =
            size > maxEntries
    }

    @Synchronized fun get(key: Int): V? = map[key]
    @Synchronized fun put(key: Int, value: V) { map[key] = value }
    @Synchronized fun evictAll() = map.clear()
}
```

## 2) 新增：Pdfium 封装（io.legere 版本）+ outline/links/text 缓存

> 这个替换你之前的 `PdfiumDocument.kt`（如果同名，直接覆盖即可）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/pdfium/PdfiumDocument.kt
package com.example.reader.pdf.internal.pdfium

import android.graphics.RectF
import android.os.ParcelFileDescriptor
import com.example.reader.model.DocumentLink
import com.example.reader.model.LinkTarget
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.NormalizedRect
import com.example.reader.model.OutlineNode
import com.example.reader.pdf.internal.cache.IntLruCache
import com.example.reader.pdf.internal.util.useSafely
import io.legere.pdfiumandroid.api.Bookmark
import io.legere.pdfiumandroid.api.Link
import io.legere.pdfiumandroid.suspend.PdfDocumentKt
import io.legere.pdfiumandroid.suspend.PdfPageKt
import io.legere.pdfiumandroid.suspend.PdfTextPageKt
import io.legere.pdfiumandroid.suspend.PdfiumCoreKt
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.Dispatchers
import java.io.Closeable

internal class PdfiumDocument private constructor(
    private val core: PdfiumCoreKt,
    private val doc: PdfDocumentKt,
    val pageCount: Int
) : Closeable {

    // ✅ 缓存：目录/链接/文本
    private var outlineCache: List<OutlineNode>? = null
    private val linksCache = IntLruCache<List<CachedLink>>(maxEntries = 64)
    private val textCache = IntLruCache<String>(maxEntries = 16)

    val supportsText: Boolean = true // API 有 textPage；具体页面可能 0 chars（扫描件）

    companion object {
        suspend fun open(
            pfd: ParcelFileDescriptor,
            dispatcher: CoroutineDispatcher = Dispatchers.IO,
            password: String?
        ): PdfiumDocument {
            val core = PdfiumCoreKt(dispatcher)
            val doc = core.newDocument(pfd, password)
            val pageCount = doc.getPageCount()
            return PdfiumDocument(core, doc, pageCount)
        }
    }

    suspend fun outline(): List<OutlineNode> {
        outlineCache?.let { return it }
        val toc = doc.getTableOfContents()
        val mapped = toc.map { it.toOutlineNode() }
        outlineCache = mapped
        return mapped
    }

    suspend fun pageLinks(pageIndex: Int, rotationDegrees: Int): List<DocumentLink> {
        val cached = linksCache.get(pageIndex)
        val raw = cached ?: loadLinksUncached(pageIndex).also { linksCache.put(pageIndex, it) }
        return raw.mapNotNull { it.toDocumentLink(rotationDegrees) }
    }

    suspend fun pageText(pageIndex: Int): String {
        textCache.get(pageIndex)?.let { return it }
        val text = loadTextUncached(pageIndex)
        textCache.put(pageIndex, text)
        return text
    }

    fun evictAllCaches() {
        outlineCache = null
        linksCache.evictAll()
        textCache.evictAll()
    }

    override fun close() {
        runCatching { doc.close() }
        // core 本身无需 close（doc close 会释放 native 资源 + 关闭 pfd dup）
    }

    // ---------------- internal helpers ----------------

    private suspend fun loadLinksUncached(pageIndex: Int): List<CachedLink> {
        val page = requireNotNull(doc.openPage(pageIndex)) { "openPage($pageIndex) failed" }
        return page.useSafely { p ->
            val links: List<Link> = p.getPageLinks()
            links.mapNotNull { link ->
                val target = when {
                    link.uri != null -> LinkTarget.External(link.uri)
                    link.destPageIdx != null -> LinkTarget.Internal(
                        Locator(LocatorSchemes.PDF_PAGE, link.destPageIdx.toString())
                    )
                    else -> null
                } ?: return@mapNotNull null

                // link.bounds 是 PDF 坐标；不要直接 /widthPoint —— 用 mapRectToDevice 做归一化更稳
                CachedLink(
                    target = target,
                    bounds = link.bounds
                )
            }
        }
    }

    private suspend fun loadTextUncached(pageIndex: Int): String {
        val page = requireNotNull(doc.openPage(pageIndex)) { "openPage($pageIndex) failed" }
        return page.useSafely { p ->
            val textPage = p.openTextPage()
            textPage.useSafely { tp: PdfTextPageKt ->
                val count = tp.textPageCountChars().coerceAtLeast(0)
                if (count == 0) return@useSafely ""
                tp.textPageGetText(0, count) ?: ""
            }
        }
    }

    internal suspend fun withPage(
        pageIndex: Int,
        block: suspend (PdfPageKt) -> Unit
    ) {
        val page = requireNotNull(doc.openPage(pageIndex)) { "openPage($pageIndex) failed" }
        page.useSafely { p -> block(p) }
    }

    internal suspend fun withTextPage(
        pageIndex: Int,
        block: suspend (PdfPageKt, PdfTextPageKt) -> Unit
    ) {
        val page = requireNotNull(doc.openPage(pageIndex)) { "openPage($pageIndex) failed" }
        page.useSafely { p ->
            val textPage = p.openTextPage()
            textPage.useSafely { tp ->
                block(p, tp)
            }
        }
    }

    private fun Bookmark.toOutlineNode(): OutlineNode {
        val page = pageIdx.toInt().coerceAtLeast(0)
        return OutlineNode(
            title = title ?: "",
            locator = Locator(LocatorSchemes.PDF_PAGE, page.toString()),
            children = children.map { it.toOutlineNode() }
        )
    }

    private data class CachedLink(
        val target: LinkTarget,
        val bounds: RectF
    ) {
        suspend fun toDocumentLink(rotationDegrees: Int): DocumentLink? {
            // bounds -> normalized rect（含旋转），交给调用方传 rotate
            // 这里先返回“未归一化”的，归一化在 provider/Controller 用 page.mapRectToDevice 做
            return DocumentLink(target = target, bounds = null)
        }
    }
}

/** rotationDegrees -> pdfium rotate index (0,1,2,3) */
internal fun pdfiumRotateIndex(rotationDegrees: Int): Int {
    val r = ((rotationDegrees % 360) + 360) % 360
    return when (r) {
        90 -> 1
        180 -> 2
        270 -> 3
        else -> 0
    }
}

internal fun rectToNormalized(device: android.graphics.Rect, base: Int): NormalizedRect {
    val l = (device.left.toFloat() / base).coerceIn(0f, 1f)
    val t = (device.top.toFloat() / base).coerceIn(0f, 1f)
    val r = (device.right.toFloat() / base).coerceIn(0f, 1f)
    val b = (device.bottom.toFloat() / base).coerceIn(0f, 1f)
    return NormalizedRect(
        left = minOf(l, r),
        top = minOf(t, b),
        right = maxOf(l, r),
        bottom = maxOf(t, b)
    )
}
```

## 5) 新增：`PdfSearchProvider.kt`（用 PDFium find + 输出命中 rects）

> 这个是本轮最关键的“优化点”：
> ✅ 由“全页扫描文本”升级为 **PDFium find**（更快）
> ✅ 每个命中附带 **rects**（用于 UI 高亮）写入 `LocatorRange.extras["pdf.rects"]`

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/provider/PdfSearchProvider.kt
package com.example.reader.pdf.internal.provider

import android.graphics.RectF
import com.example.reader.api.provider.SearchHit
import com.example.reader.api.provider.SearchOptions
import com.example.reader.api.provider.SearchProvider
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.NormalizedRect
import com.example.reader.pdf.internal.pdfium.PdfiumDocument
import com.example.reader.pdf.internal.pdfium.pdfiumRotateIndex
import com.example.reader.pdf.internal.pdfium.rectToNormalized
import io.legere.pdfiumandroid.api.FindFlags
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.flow
import kotlin.coroutines.coroutineContext

internal class PdfSearchProvider(
    private val pdfium: PdfiumDocument,
    private val rotationDegreesProvider: () -> Int
) : SearchProvider {

    override fun search(query: String, options: SearchOptions): Flow<SearchHit> = flow {
        val q = query.trim()
        if (q.isEmpty()) return@flow

        val flags = buildSet {
            if (options.caseSensitive) add(FindFlags.MatchCase)
            if (options.wholeWord) add(FindFlags.MatchWholeWord)
        }

        val startPage = options.startFrom
            ?.takeIf { it.scheme == LocatorSchemes.PDF_PAGE }
            ?.value?.toIntOrNull()
            ?.coerceAtLeast(0)
            ?: 0

        val maxHits = options.maxHits.coerceAtLeast(1)
        var emitted = 0

        for (pageIndex in startPage until pdfium.pageCount) {
            coroutineContext.ensureActive()
            if (emitted >= maxHits) return@flow

            val rotation = rotationDegreesProvider()
            val rotate = pdfiumRotateIndex(rotation)
            val base = 10_000 // 归一化基准：mapRectToDevice 的 sizeX/sizeY

            pdfium.withTextPage(pageIndex) { page, textPage ->
                val find = textPage.findStart(q, flags, 0) ?: return@withTextPage
                find.useSafely { fr ->
                    while (fr.findNext()) {
                        coroutineContext.ensureActive()
                        val startIdx = fr.getSchResultIndex()
                        if (startIdx < 0) continue

                        // 命中 rects（PDF 坐标 -> mapRectToDevice -> normalized）
                        val rects: List<NormalizedRect> = buildList {
                            val n = textPage.textPageCountRects(startIdx, q.length).coerceAtLeast(0)
                            for (i in 0 until n) {
                                val r: RectF = textPage.textPageGetRect(i) ?: continue
                                val device = page.mapRectToDevice(
                                    startX = 0, startY = 0,
                                    sizeX = base, sizeY = base,
                                    rotate = rotate,
                                    coords = r
                                )
                                add(rectToNormalized(device, base))
                            }
                        }

                        val excerpt = buildExcerpt(
                            text = pdfium.pageText(pageIndex),
                            start = startIdx,
                            len = q.length
                        )

                        val loc = Locator(LocatorSchemes.PDF_PAGE, pageIndex.toString())
                        val extras = if (rects.isNotEmpty()) {
                            mapOf(PdfRectCodec.EXTRA_RECTS to PdfRectCodec.encode(rects))
                        } else {
                            emptyMap()
                        }

                        emit(
                            SearchHit(
                                range = LocatorRange(start = loc, end = loc, extras = extras),
                                excerpt = excerpt,
                                sectionTitle = null
                            )
                        )

                        emitted++
                        if (emitted >= maxHits) return@useSafely
                    }
                }
            }
        }
    }

    private fun buildExcerpt(text: String, start: Int, len: Int): String {
        if (text.isEmpty()) return ""
        val s = (start - 24).coerceAtLeast(0)
        val e = (start + len + 24).coerceAtMost(text.length)
        return text.substring(s, e).replace('\n', ' ')
    }
}
```

## 6) 修改：`PdfController.kt`（把 linksResolver 真正接入 pdfium + mapRectToDevice 归一化）

> 你之前 controller 里 `links = emptyList()`，现在改为：

* 从 `linksResolver(pageIndex, rotationDegrees)` 得到 pdfium `Link` 列表或 `RectF` 列表
* 用 `page.mapRectToDevice(... rotate ...)` 归一化为 `NormalizedRect`
* 输出 `RenderPage.links`

由于你的 `PdfController` 不直接持有 PdfiumDocument（避免强耦合），这里我建议把 `linksResolver` 的签名升级为：
`suspend (pageIndex:Int, rotationDegrees:Int) -> List<PdfLinkRaw>`
