package com.example.reader.pdf.internal.open

import android.content.Context
import android.os.ParcelFileDescriptor
import com.example.reader.api.open.OpenOptions
import com.example.reader.source.DocumentSource
import java.io.File
import java.io.FileOutputStream
import java.util.UUID
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext

internal data class OpenedPdf(
    val pfd: ParcelFileDescriptor,
    /** 如果为了获得可 seek 的 FD 而落盘，会保存 tempFile；close 时需要清理 */
    val tempFile: File? = null
)

internal class PdfOpener(
    private val context: Context,
    private val ioDispatcher: CoroutineDispatcher
) {
    suspend fun open(source: DocumentSource, options: OpenOptions): OpenedPdf =
        withContext(ioDispatcher) {
            // 1) 优先直接拿 PFD（很多 SAF provider 可行）
            val direct = source.openFileDescriptor("r")
            if (direct != null) return@withContext OpenedPdf(pfd = direct, tempFile = null)

            // 2) 兜底：把 InputStream copy 到 cache，再以 File PFD 打开（确保 seekable）
            val dir = File(context.cacheDir, "reader_pdf").apply { mkdirs() }
            val file = File(dir, "tmp_${UUID.randomUUID()}.pdf")

            source.openInputStream().use { input ->
                FileOutputStream(file).use { out ->
                    val buf = ByteArray(DEFAULT_BUF_SIZE)
                    while (true) {
                        val n = input.read(buf)
                        if (n <= 0) break
                        out.write(buf, 0, n)
                    }
                    out.flush()
                }
            }

            val pfd = ParcelFileDescriptor.open(file, ParcelFileDescriptor.MODE_READ_ONLY)
            OpenedPdf(pfd = pfd, tempFile = file)
        }
}

// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/backend/PlatformPdfBackend.kt
package com.example.reader.pdf.internal.backend

import android.graphics.Bitmap
import android.graphics.Matrix
import android.graphics.Rect
import android.graphics.pdf.PdfRenderer
import android.os.ParcelFileDescriptor
import java.io.Closeable
import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock

/**
 * 使用 android.graphics.pdf.PdfRenderer
 * 注意：文档要求同一时刻只能打开一个 Page，所以这里用 Mutex 串行 render。
 */
internal class PlatformPdfBackend(
    pfd: ParcelFileDescriptor
) : PdfBackend, Closeable {

    private val renderer = PdfRenderer(pfd)
    private val mutex = Mutex()

    override val pageCount: Int get() = renderer.pageCount

    override fun pageSize(pageIndex: Int, rotationDegrees: Int): PdfPageSize {
        renderer.openPage(pageIndex).use { page ->
            val w = page.width
            val h = page.height
            return if (rotationDegrees % 180 == 0) PdfPageSize(w, h) else PdfPageSize(h, w)
        }
    }

    override fun render(
        pageIndex: Int,
        bitmap: Bitmap,
        clip: Rect,
        matrix: Matrix,
        mode: PdfRenderMode
    ) {
        // PdfRenderer.Page 需要严格 open/render/close
        // 且同一时刻只能有一个 Page 打开 -> mutex
        // 这里是阻塞 API，调用方一般在 Dispatchers.IO
        runBlockingMutex {
            renderer.openPage(pageIndex).use { page ->
                val renderMode = when (mode) {
                    PdfRenderMode.DRAFT -> PdfRenderer.Page.RENDER_MODE_FOR_DISPLAY
                    PdfRenderMode.FINAL -> PdfRenderer.Page.RENDER_MODE_FOR_PRINT
                }
                page.render(bitmap, clip, matrix, renderMode)
            }
        }
    }

    private fun runBlockingMutex(block: () -> Unit) {
        // 由于 PdfRenderer.render 本身是同步阻塞，直接阻塞锁即可
        // （调用方应放在 IO dispatcher）
        kotlinx.coroutines.runBlocking {
            mutex.withLock { block() }
        }
    }

    override fun close() {
        renderer.close()
    }
}

package com.example.reader.pdf.internal.render

import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.RenderConfig
import com.example.reader.pdf.internal.backend.PdfPageSize
import kotlin.math.max
import kotlin.math.min

internal data class PageTransform(
    val pageWidthPx: Int,
    val pageHeightPx: Int,
    val scale: Float // page原始单位 -> 目标像素
)

internal fun computeTransform(
    page: PdfPageSize,
    config: RenderConfig.FixedPage,
    constraints: LayoutConstraints
): PageTransform {
    val viewportW = constraints.viewportWidthPx.toFloat().coerceAtLeast(1f)
    val viewportH = constraints.viewportHeightPx.toFloat().coerceAtLeast(1f)

    val pageW = page.width.toFloat().coerceAtLeast(1f)
    val pageH = page.height.toFloat().coerceAtLeast(1f)

    val fitScale = when (config.fitMode) {
        RenderConfig.FitMode.FIT_WIDTH -> viewportW / pageW
        RenderConfig.FitMode.FIT_HEIGHT -> viewportH / pageH
        RenderConfig.FitMode.FIT_PAGE -> min(viewportW / pageW, viewportH / pageH)
        RenderConfig.FitMode.FREE -> 1f
    }

    val scale = max(0.05f, fitScale * config.zoom)

    val scaledW = (pageW * scale).toInt().coerceAtLeast(1)
    val scaledH = (pageH * scale).toInt().coerceAtLeast(1)

    return PageTransform(
        pageWidthPx = scaledW,
        pageHeightPx = scaledH,
        scale = scale
    )
}

// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/render/PdfTileProvider.kt
package com.example.reader.pdf.internal.render

import android.graphics.Bitmap
import android.graphics.Matrix
import android.graphics.Rect
import com.example.reader.api.render.RenderPolicy
import com.example.reader.api.render.TileProvider
import com.example.reader.api.render.TileRequest
import com.example.reader.pdf.internal.backend.PdfBackend
import com.example.reader.pdf.internal.backend.PdfRenderMode
import com.example.reader.pdf.internal.cache.TileCache
import java.io.Closeable
import java.util.concurrent.atomic.AtomicBoolean
import kotlin.math.max
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext

internal class PdfTileProvider(
    private val backend: PdfBackend,
    private val pageIndex: Int,
    private val baseScale: Float,
    private val rotationDegrees: Int,
    private val tileCache: TileCache,
    private val ioDispatcher: CoroutineDispatcher
) : TileProvider, Closeable {

    private val closed = AtomicBoolean(false)

    override suspend fun renderTile(request: TileRequest): Bitmap = withContext(ioDispatcher) {
        check(!closed.get()) { "TileProvider closed" }

        val scale = max(0.05f, baseScale * request.scale)
        val key = buildKey(request, scale)

        tileCache.get(key)?.let { cached ->
            // 直接返回缓存（注意：Bitmap 共享需 UI 不修改）
            return@withContext cached
        }

        val bmp = Bitmap.createBitmap(
            request.widthPx.coerceAtLeast(1),
            request.heightPx.coerceAtLeast(1),
            Bitmap.Config.ARGB_8888
        )

        val clip = Rect(0, 0, bmp.width, bmp.height)

        // matrix: 先 scale，再平移，把 page 上 (left,top) 区域挪到 tile 原点
        val matrix = Matrix().apply {
            setScale(scale, scale)
            postTranslate(-request.leftPx.toFloat(), -request.topPx.toFloat())
            // rotation：这里留钩子；如果你要支持 90/180/270，建议在 backend 层做坐标系统一
            // 或在这里 pre/postRotate 围绕原点/页面中心做转换
        }

        val mode = when (request.quality) {
            RenderPolicy.Quality.DRAFT -> PdfRenderMode.DRAFT
            RenderPolicy.Quality.FINAL -> PdfRenderMode.FINAL
        }

        backend.render(
            pageIndex = pageIndex,
            bitmap = bmp,
            clip = clip,
            matrix = matrix,
            mode = mode
        )

        tileCache.put(key, bmp)
        bmp
    }

    private fun buildKey(req: TileRequest, scale: Float): String {
        // scale 量化，减少碎片（例如 1.234 -> 1.25）
        val bucket = ((scale * 100f).toInt() / 5) * 5 // 0.05 步进（你也可改 0.1/0.25）
        return "p=$pageIndex|l=${req.leftPx}|t=${req.topPx}|w=${req.widthPx}|h=${req.heightPx}|s=$bucket|r=$rotationDegrees|q=${req.quality}"
    }

    override fun close() {
        closed.set(true)
    }
}

        DocumentId(UUID.nameUUIDFromBytes(source.uri.toString().toByteArray()).toString())

    override val capabilities: DocumentCapabilities = DocumentCapabilities(
        reflowable = false,
        fixedLayout = true,

        search = false,          // 同上
        textExtraction = false,  // 同上
        annotations = true,      // FixedRects 可先做
        links = false            // 后续接 Pdfium / API35+ 再开启
    )

    override suspend fun metadata(): ReaderResult<DocumentMetadata> = withContext(ioDispatcher) {
        try {
            ReaderResult.Ok(
                DocumentMetadata(
                    title = source.displayName,
                    extra = buildMap {
                        source.mimeType?.let { put("mimeType", it) }
                        source.sizeBytes?.let { put("sizeBytes", it.toString()) }
                        put("pageCount", backend.pageCount.toString())
                    }
                )
            )
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }
    }

    override suspend fun createSession(
        initialLocator: Locator?,
        initialConfig: RenderConfig
    ): ReaderResult<ReaderSession> = withContext(ioDispatcher) {
        try {
            val config = when (initialConfig) {
                is RenderConfig.FixedPage -> initialConfig
                else -> RenderConfig.FixedPage() // 防御：PDF 强制 fixed
            }

            val startPage = when {
                initialLocator == null -> 0
                initialLocator.scheme == "pdf.page" -> initialLocator.value.toIntOrNull() ?: 0
                else -> 0
            }.coerceIn(0, (backend.pageCount - 1).coerceAtLeast(0))

            ReaderResult.Ok(
                PdfSessionImpl(
                    documentId = id,
                    backend = backend,
                    startPageIndex = startPage,
                    initialConfig = config,
                    capabilities = capabilities,
                    ioDispatcher = ioDispatcher
                )
            )
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }
    }

    override fun close() {
        runCatching { backend.close() }
        runCatching { opened.pfd.close() }
        opened.tempFile?.let { runCatching { it.delete() } }
    }
}

package com.example.reader.pdf.internal.render

import com.example.reader.api.error.ReaderResult
import com.example.reader.api.render.InvalidateReason
import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.NavigationAvailability
import com.example.reader.api.render.PageId
import com.example.reader.api.render.ReaderController
import com.example.reader.api.render.ReaderEvent
import com.example.reader.api.render.RenderConfig
import com.example.reader.api.render.RenderContent
import com.example.reader.api.render.RenderMetrics
import com.example.reader.api.render.RenderPage
import com.example.reader.api.render.RenderPolicy
import com.example.reader.api.render.RenderState
import com.example.reader.model.Progression
import com.example.reader.pdf.internal.backend.PdfBackend
import com.example.reader.pdf.internal.cache.TileCache
import com.example.reader.pdf.internal.util.pdfPageLocator
import com.example.reader.pdf.internal.util.requirePdfPageIndex
import com.example.reader.pdf.internal.util.toReaderError
import kotlin.math.floor
import kotlin.math.max
import kotlin.system.measureTimeMillis
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.channels.BufferOverflow
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.MutableSharedFlow
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.withContext

internal class PdfController(
    private val backend: PdfBackend,
    startPageIndex: Int,
    initialConfig: RenderConfig.FixedPage,
    private val pageCount: Int,
    private val ioDispatcher: CoroutineDispatcher
) : ReaderController {

    private val _events = MutableSharedFlow<ReaderEvent>(
        replay = 0,
        extraBufferCapacity = 16,
        onBufferOverflow = BufferOverflow.DROP_OLDEST
    )
    override val events: Flow<ReaderEvent> = _events

    private var constraints: LayoutConstraints? = null
    private var config: RenderConfig.FixedPage = initialConfig
    private var pageIndex: Int = startPageIndex.coerceIn(0, (pageCount - 1).coerceAtLeast(0))

    // 经验值：给 16~32MB tile cache；你可改成按设备内存比例计算
    private val tileCache = TileCache(maxBytes = 24 * 1024 * 1024)

    private val _state = MutableStateFlow(buildState())
    override val state: StateFlow<RenderState> = _state

    override suspend fun setLayoutConstraints(constraints: LayoutConstraints): ReaderResult<Unit> =
        withContext(ioDispatcher) {
            try {
                this@PdfController.constraints = constraints
                _state.value = buildState()
                ReaderResult.Ok(Unit)
            } catch (t: Throwable) {
                ReaderResult.Err(t.toReaderError())
            }
        }

    override suspend fun setConfig(config: RenderConfig): ReaderResult<Unit> =
        withContext(ioDispatcher) {
            try {
                val fixed = config as? RenderConfig.FixedPage
                    ?: return@withContext ReaderResult.Err(
                        com.example.reader.api.error.ReaderError.Internal("PDF requires RenderConfig.FixedPage")
                    )
                this@PdfController.config = fixed
                tileCache.evictAll()
                _state.value = buildState()
                ReaderResult.Ok(Unit)
            } catch (t: Throwable) {
                ReaderResult.Err(t.toReaderError())
            }
        }

    override suspend fun render(policy: RenderPolicy): ReaderResult<RenderPage> =
        withContext(ioDispatcher) {
            try {
                val c = constraints ?: return@withContext ReaderResult.Err(
                    com.example.reader.api.error.ReaderError.Internal("LayoutConstraints not set")
                )

                val basePageSize = backend.pageSize(pageIndex, rotationDegrees = config.rotationDegrees)
                val transform = computeTransform(basePageSize, config, c)

                val cacheHitBox = booleanArrayOf(false)
                val metrics: RenderMetrics?
                val page: RenderPage

                val timeMs = measureTimeMillis {
                    val provider = PdfTileProvider(
                        backend = backend,
                        pageIndex = pageIndex,
                        baseScale = transform.scale,
                        rotationDegrees = config.rotationDegrees,
                        tileCache = tileCache,
                        ioDispatcher = ioDispatcher
                    )

                    // 这里返回 Tiles：UI 会按视口请求 tile
                    page = RenderPage(
                        id = PageId("pdf:${pageIndex}:${transform.scale}:${config.rotationDegrees}:${config.fitMode}:${config.zoom}"),
                        locator = pdfPageLocator(pageIndex),
                        content = RenderContent.Tiles(
                            pageWidthPx = transform.pageWidthPx,
                            pageHeightPx = transform.pageHeightPx,
                            tileProvider = provider
                        ),
                        links = emptyList(),
                        decorations = emptyList(),
                        metrics = null
                    )
                }

                metrics = RenderMetrics(renderTimeMs = timeMs, cacheHit = cacheHitBox[0])

                // 更新 state（locator/config/progression/nav）
                _state.value = buildState()
                _events.tryEmit(ReaderEvent.Rendered(page.id, metrics))

                ReaderResult.Ok(page.copy(metrics = metrics))
            } catch (t: Throwable) {
                _events.tryEmit(ReaderEvent.Error(t))
                ReaderResult.Err(t.toReaderError())
            }
        }

    override suspend fun next(policy: RenderPolicy): ReaderResult<RenderPage> {
        if (pageIndex >= pageCount - 1) return ReaderResult.Err(
            com.example.reader.api.error.ReaderError.Internal("No next page")
        )
        pageIndex++
        _events.tryEmit(ReaderEvent.PageChanged(pdfPageLocator(pageIndex)))
        return render(policy)
    }

    override suspend fun prev(policy: RenderPolicy): ReaderResult<RenderPage> {
        if (pageIndex <= 0) return ReaderResult.Err(
            com.example.reader.api.error.ReaderError.Internal("No previous page")
        )
        pageIndex--
        _events.tryEmit(ReaderEvent.PageChanged(pdfPageLocator(pageIndex)))
        return render(policy)
    }

    override suspend fun goTo(locator: com.example.reader.model.Locator, policy: RenderPolicy): ReaderResult<RenderPage> {
        return try {
            val idx = locator.requirePdfPageIndex().coerceIn(0, (pageCount - 1).coerceAtLeast(0))
            pageIndex = idx
            _events.tryEmit(ReaderEvent.PageChanged(pdfPageLocator(pageIndex)))
            render(policy)
        } catch (t: Throwable) {
            ReaderResult.Err(t.toReaderError())
        }
    }

    override suspend fun goToProgress(percent: Double, policy: RenderPolicy): ReaderResult<RenderPage> {
        val p = percent.coerceIn(0.0, 1.0)
        val idx = floor(p * max(1, pageCount - 1)).toInt().coerceIn(0, (pageCount - 1).coerceAtLeast(0))
        pageIndex = idx
        _events.tryEmit(ReaderEvent.PageChanged(pdfPageLocator(pageIndex)))
        return render(policy)
    }

    override suspend fun prefetchNeighbors(count: Int): ReaderResult<Unit> = withContext(ioDispatcher) {
        // 最小实现：先不做（不影响功能），后续你可以在这里：
        // - 为相邻页预热低清 tile（DRAFT）
        // - 或者预先计算 pageSize/links/outline 等元信息
        ReaderResult.Ok(Unit)
    }

    override suspend fun invalidate(reason: InvalidateReason): ReaderResult<Unit> = withContext(ioDispatcher) {
        when (reason) {
            InvalidateReason.CONTENT_CHANGED,
            InvalidateReason.CONFIG_CHANGED,
            InvalidateReason.LAYOUT_CHANGED -> tileCache.evictAll()
        }
        ReaderResult.Ok(Unit)
    }

    private fun buildState(): RenderState {
        val idx = pageIndex.coerceIn(0, (pageCount - 1).coerceAtLeast(0))
        val percent = if (pageCount <= 1) 0.0 else idx.toDouble() / (pageCount - 1).toDouble()
        return RenderState(
            locator = pdfPageLocator(idx),
            progression = Progression(
                percent = percent,
                label = "${idx + 1}/$pageCount",
                current = idx + 1,
                total = pageCount
            ),
            nav = NavigationAvailability(
                canGoPrev = idx > 0,
                canGoNext = idx < pageCount - 1
            ),
            titleInView = null,
            config = config
        )
    }

    override fun close() {
        tileCache.evictAll()
    }
}
