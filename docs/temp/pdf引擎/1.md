下面给你一套**可直接落地**的 `engines/pdf` 实现（基于 **PDFium**，支持：**渲染（Tiles）/目录（Outline）/文本抽取/搜索/链接热区/标注覆盖（可插拔存储）**），完全对齐你给的 `core/reader/api` 契约与 `runtime` 编排方式。

实现选型我用的是 **PdfiumAndroidKt（io.legere.*）**：它提供 coroutine/suspend API，并且明确强调“打开 document/page/textPage 很贵，要缓存”，以及 **ARGB_8888 比 RGB_565 快**（RGB_565 会做额外缓冲与转换）——所以我在渲染里默认用 ARGB_8888 + LRU page cache + LRU tile bitmap cache。 ([GitHub][1])

```kotlin

import android.util.Log
import com.example.reader.api.engine.ReaderDocument
import com.example.reader.api.engine.ReaderEngine
import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.open.OpenOptions
import com.example.reader.engines.pdf.internal.PdfDocumentImpl
import com.example.reader.engines.pdf.internal.PdfEngineConfig
import com.example.reader.engines.pdf.internal.annotation.InMemoryPdfAnnotationStore
import com.example.reader.engines.pdf.internal.annotation.PdfAnnotationStore
import com.example.reader.engines.pdf.internal.util.sha1Hex
import com.example.reader.model.BookFormat
import com.example.reader.model.DocumentCapabilities
import com.example.reader.model.DocumentId
import com.example.reader.source.DocumentSource
import io.legere.pdfiumandroid.api.Config
import io.legere.pdfiumandroid.api.LoggerInterface
import io.legere.pdfiumandroid.suspend.PdfDocumentKt
import io.legere.pdfiumandroid.suspend.PdfiumCoreKt
import java.io.ByteArrayOutputStream
import java.io.IOException
import java.util.Locale
import kotlinx.coroutines.withContext

/**
 * ✅ PDF 引擎：实现 core/reader/api 的 ReaderEngine
 *
 * 特性：
 * - Tiles 渲染（支持大页、缩放、预取）
 * - Outline（目录）
 * - Text extraction（复制/摘录）
 * - Search（按页扫描，可取消）
 * - Links（热区 -> DocumentLink）
 * - Annotations（引擎只负责“覆盖/定位”，存储可插拔）
 */
class PdfEngine(
    private val engineConfig: PdfEngineConfig = PdfEngineConfig(),
    private val annotationStore: PdfAnnotationStore = InMemoryPdfAnnotationStore()
) : ReaderEngine {

    override val supportedFormats: Set<BookFormat> = setOf(BookFormat.PDF)

    private val pdfium: PdfiumCoreKt = PdfiumCoreKt(
        dispatcher = engineConfig.ioDispatcher,
        config = Config().copy(
            logger = object : LoggerInterface {
                override fun d(tag: String, message: String?) {
                    Log.d("Pdfium/$tag", message ?: "")
                }
                override fun e(tag: String, t: Throwable?, message: String?) {
                    Log.e("Pdfium/$tag", message ?: "", t)
                }
            }
        )
    )

    override suspend fun open(
        source: DocumentSource,
        options: OpenOptions
    ): ReaderResult<ReaderDocument> = withContext(engineConfig.ioDispatcher) {
        try {
            // 1) 优先 PFD（随机访问、内存友好）
            val pfd: ParcelFileDescriptor? = source.openFileDescriptor("r")
            val (pdfDoc, closeablePfd) = if (pfd != null) {
                val doc = pdfium.newDocument(pfd, options.password)
                doc to pfd
            } else {
                // 2) 兜底：读入内存（不建议大文件）
                val bytes = readAllBytesWithLimit(source, engineConfig.maxPdfBytesInMemoryFallback)
                val doc = pdfium.newDocument(bytes, options.password)
                doc to null
            }

            val id = buildDocumentId(source)
            val caps = DocumentCapabilities(
                reflowable = false,
                fixedLayout = true,
                outline = engineConfig.enableOutline,
                search = engineConfig.enableSearch && engineConfig.enableTextExtraction,
                textExtraction = engineConfig.enableTextExtraction,
                annotations = engineConfig.enableAnnotations,
                links = engineConfig.enableLinks
            )

            ReaderResult.Ok(
                PdfDocumentImpl(
                    id = id,
                    source = source,
                    openOptions = options,
                    capabilities = caps,
                    pdfDocument = pdfDoc,
                    pfdToClose = closeablePfd,
                    engineConfig = engineConfig,
                    annotationStore = annotationStore
                )
            )
        } catch (t: Throwable) {
            ReaderResult.Err(t.toPdfReaderError())
        }
    }

    private fun buildDocumentId(source: DocumentSource): DocumentId {
        val raw = buildString {
            append(source.uri.toString())
            append("|")
            append(source.displayName ?: "")
            append("|")
            append(source.sizeBytes ?: -1L)
            append("|")
            append(source.mimeType ?: "")
        }
        return DocumentId("pdf:${sha1Hex(raw)}")
    }

    private suspend fun readAllBytesWithLimit(source: DocumentSource, limitBytes: Long): ByteArray {
        val sizeHint = source.sizeBytes
        if (sizeHint != null && sizeHint > limitBytes) {
            throw IOException("PDF too large for in-memory fallback: $sizeHint > $limitBytes")
        }
        return source.openInputStream().use { input ->
            val out = ByteArrayOutputStream()
            val buf = ByteArray(DEFAULT_BUFFER_SIZE)
            var total = 0L
            while (true) {
                val read = input.read(buf)
                if (read <= 0) break
                total += read
                if (total > limitBytes) {
                    throw IOException("PDF too large for in-memory fallback: $total > $limitBytes")
                }
                out.write(buf, 0, read)
            }
            out.toByteArray()
        }
    }
}

private fun Throwable.toPdfReaderError(): ReaderError {
    val msg = (message ?: "").lowercase(Locale.US)
    return when {
        this is ReaderError -> this
        this is SecurityException -> ReaderError.PermissionDenied(cause = this)
        this is java.io.FileNotFoundException -> ReaderError.NotFound(cause = this)
        msg.contains("password") || msg.contains("encrypted") -> ReaderError.InvalidPassword(cause = this)
        this is IOException -> ReaderError.Io(cause = this)
        else -> ReaderError.Internal(cause = this)
    }
}
```

```kotlin
    val enableTextExtraction: Boolean = true,
    val enableSearch: Boolean = true,
    val enableOutline: Boolean = true,
    val enableLinks: Boolean = true,
    val enableAnnotations: Boolean = true,

    /** 当拿不到 PFD 时，读入内存的最大允许字节数（避免 OOM） */
    val maxPdfBytesInMemoryFallback: Long = 32L * 1024 * 1024
) {
    companion object {
        fun defaultBitmapCacheBytes(): Int {
            val maxMem = Runtime.getRuntime().maxMemory()
            val target = maxMem / 8
            val capped = min(target, 256L * 1024 * 1024)
            val floored = max(capped, 32L * 1024 * 1024)
            return floored.toInt()
        }
    }
}
```

```kotlin
import com.example.reader.source.DocumentSource
import io.legere.pdfiumandroid.suspend.PdfDocumentKt
import java.util.UUID
import java.util.concurrent.atomic.AtomicBoolean
import kotlinx.coroutines.withContext

internal class PdfDocumentImpl(
    override val id: DocumentId,
    private val source: DocumentSource,
    override val openOptions: OpenOptions,
    override val capabilities: DocumentCapabilities,
    private val pdfDocument: PdfDocumentKt,
    private val pfdToClose: ParcelFileDescriptor?,
    private val engineConfig: PdfEngineConfig,
    private val annotationStore: PdfAnnotationStore
) : ReaderDocument {

    override val format: BookFormat = BookFormat.PDF

    private val closed = AtomicBoolean(false)

    override suspend fun metadata(): ReaderResult<DocumentMetadata> =
        withContext(engineConfig.ioDispatcher) {
            try {
                val meta = pdfDocument.getDocumentMeta()
                ReaderResult.Ok(PdfMetaMapper.toDocumentMetadata(meta))
            } catch (t: Throwable) {
                ReaderResult.Err(t.toPdfReaderError())
            }
        }

    override suspend fun createSession(
        initialLocator: Locator?,
        initialConfig: RenderConfig
    ): ReaderResult<ReaderSession> = withContext(engineConfig.ioDispatcher) {
        try {
            if (closed.get()) return@withContext ReaderResult.Err(ReaderError.Internal("Document closed"))

            val pageCount = pdfDocument.getPageCount()

            val startPage = initialLocator.toPageIndexOrNull()?.coerceIn(0, (pageCount - 1).coerceAtLeast(0)) ?: 0
            val fixedConfig = initialConfig.asFixedPage()

            val pageCache = PdfPageCache(
                pdfDocument = pdfDocument,
                ioDispatcher = engineConfig.ioDispatcher,
                maxPages = engineConfig.pageCacheSize,
                enableTextPage = engineConfig.enableTextExtraction
            )

            val controller = PdfReaderController(
                documentId = id,
                pageCount = pageCount,
                pageCache = pageCache,
                engineConfig = engineConfig,
                annotationStore = annotationStore,
                initialPageIndex = startPage,
                initialConfig = fixedConfig
            )

            val outline = if (capabilities.outline) PdfOutlineProviderImpl(pdfDocument, engineConfig) else null
            val text = if (capabilities.textExtraction) PdfTextProviderImpl(id, pageCache, engineConfig) else null
            val search = if (capabilities.search && text != null) PdfSearchProviderImpl(pageCount, text, engineConfig) else null
            val annotations = if (capabilities.annotations) annotationStore.asProvider(id) else null

            ReaderResult.Ok(
                PdfReaderSession(
                    id = com.example.reader.model.SessionId(UUID.randomUUID().toString()),
                    controller = controller,
                    outline = outline,
                    search = search,
                    text = text,
                    annotations = annotations
                )
            )
        } catch (t: Throwable) {
            ReaderResult.Err(t.toPdfReaderError())
        }
    }

    override fun close() {
        if (!closed.compareAndSet(false, true)) return

        runCatching { pdfDocument.close() }
        runCatching { pfdToClose?.close() }
    }

    private fun Locator?.toPageIndexOrNull(): Int? {
        if (this == null) return null
        if (scheme != LocatorSchemes.PDF_PAGE) return null
        return value.toIntOrNull()
    }
}

private fun RenderConfig.asFixedPage(): RenderConfig.FixedPage =

        is RenderConfig.FixedPage -> this
        is RenderConfig.ReflowText -> RenderConfig.FixedPage() // 防御：PDF 不支持 Reflow
```

# 6) Page + TextPage LRU 缓存（专业级）

> PdfiumAndroidKt 明确说 opening doc/page/textPage 很贵，所以我们做 **LRU + refCount**，避免“用着用着被 evict 关掉”。 ([GitHub][1])

## `engines/pdf/src/main/java/com/example/reader/engines/pdf/internal/cache/PdfPageCache.kt`

```kotlin
package com.example.reader.engines.pdf.internal.cache

import com.example.reader.engines.pdf.internal.util.toPdfReaderError
import io.legere.pdfiumandroid.suspend.PdfDocumentKt
import io.legere.pdfiumandroid.suspend.PdfPageKt
import io.legere.pdfiumandroid.suspend.PdfTextPageKt
import java.io.Closeable
import java.util.LinkedHashMap
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext
import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock

internal class PdfPageCache(
    private val pdfDocument: PdfDocumentKt,
    private val ioDispatcher: CoroutineDispatcher,
    private val maxPages: Int,
    private val enableTextPage: Boolean
) : Closeable {

    internal data class Holder(
        val page: PdfPageKt,
        val textPage: PdfTextPageKt?
    )

    private data class Entry(
        val holder: Holder,
        var refCount: Int = 0
    )

    private val mutex = Mutex()

    // accessOrder=true => LRU
    private val lru: LinkedHashMap<Int, Entry> = LinkedHashMap(maxPages, 0.75f, true)

    suspend fun <T> withPage(pageIndex: Int, block: suspend (Holder) -> T): T {
        val entry = acquire(pageIndex)
        try {
            return block(entry.holder)
        } finally {
            release(pageIndex)
        }
    }

    suspend fun prefetch(pageIndex: Int) {
        val entry = acquire(pageIndex)
        release(pageIndex)
        // 不做更多事：仅确保打开并进入 LRU
        // 若需要预渲染 tile，可在 controller 里扩展
    }

    private suspend fun acquire(pageIndex: Int): Entry = mutex.withLock {
        val existing = lru[pageIndex]
        if (existing != null) {
            existing.refCount++
            return existing
        }

        val holder = openHolder(pageIndex)
        val entry = Entry(holder = holder, refCount = 1)
        lru[pageIndex] = entry

        trimToSizeLocked()
        return entry
    }

    private suspend fun release(pageIndex: Int) = mutex.withLock {
        val entry = lru[pageIndex] ?: return
        entry.refCount = (entry.refCount - 1).coerceAtLeast(0)
        trimToSizeLocked()
    }

    private fun trimToSizeLocked() {
        if (lru.size <= maxPages) return

        val iterator = lru.entries.iterator()
        while (lru.size > maxPages && iterator.hasNext()) {
            val (pageIndex, entry) = iterator.next()
            if (entry.refCount == 0) {
                iterator.remove()
                closeHolder(entry.holder)
            } else {
                // 这个 entry 还在用，跳过
            }
        }
    }

    private suspend fun openHolder(pageIndex: Int): Holder = withContext(ioDispatcher) {
        val page = pdfDocument.openPage(pageIndex)
            ?: throw IllegalStateException("Cannot open PDF page: $pageIndex")

        val textPage = if (enableTextPage) {
            runCatching { page.openTextPage() }.getOrNull()
        } else {
            null
        }

        Holder(page = page, textPage = textPage)
    }

    private fun closeHolder(holder: Holder) {
        runCatching { holder.textPage?.close() }
        runCatching { holder.page.close() }
    }

    override fun close() {
        runCatching {
            val entries = runBlockingSnapshot()
            entries.forEach { closeHolder(it.holder) }
        }
    }

    private fun runBlockingSnapshot(): List<Entry> {
        // 关闭时允许“尽力而为”
        return try {
            val list = mutableListOf<Entry>()
            lru.values.forEach { list.add(it) }
            lru.clear()
            list
        } catch (_: Throwable) {
            emptyList()
        }
    }
}
```

> 注意：这里 `close()` 的实现是“尽力而为”。如果你希望严格协程语义，可以把 `close()` 改成 `suspend fun closeSuspend()` 并让 controller/session 在协程里关闭。

```kotlin
package com.example.reader.engines.pdf.internal.cache

import android.graphics.Bitmap
import android.util.LruCache

internal class PdfBitmapCache(maxBytes: Int) {

    private val cache = object : LruCache<String, Bitmap>(maxBytes / 1024) {
        override fun sizeOf(key: String, value: Bitmap): Int {
            val bytes = runCatching { value.allocationByteCount }.getOrElse { value.byteCount }
            return bytes / 1024
        }
    }

    fun get(key: String): Bitmap? = cache.get(key)

    fun put(key: String, bitmap: Bitmap) {
        cache.put(key, bitmap)
    }

    fun evictAll() {
        cache.evictAll()
    }
}
```

# 8) Render 数学与 Tiles 渲染

## `engines/pdf/src/main/java/com/example/reader/engines/pdf/internal/render/PdfRenderMath.kt`

```kotlin
package com.example.reader.engines.pdf.internal.render

import android.graphics.Matrix
import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.RenderConfig
import io.legere.pdfiumandroid.api.Size
import io.legere.pdfiumandroid.suspend.PdfPageKt
import kotlin.math.min
import kotlin.math.roundToInt

internal data class PdfPageRenderMetrics(
    val pageIndex: Int,
    val pageWidthPt: Int,
    val pageHeightPt: Int,
    val pageWidthPx: Int,     // base pixels (scale=1.0)
    val pageHeightPx: Int,
    val baseMatrix: Matrix    // points -> base pixels
)

internal object PdfRenderMath {

    suspend fun compute(
        page: PdfPageKt,
        constraints: LayoutConstraints,
        config: RenderConfig.FixedPage
    ): PdfPageRenderMetrics {
        val screenDpi = (constraints.density * 160f).roundToInt().coerceAtLeast(1)

        val pageWidthPt = page.getPageWidthPoint().coerceAtLeast(1)
        val pageHeightPt = page.getPageHeightPoint().coerceAtLeast(1)

        val sizeAtDpi: Size = page.getPageSize(screenDpi)
        val pageW = sizeAtDpi.width.coerceAtLeast(1)
        val pageH = sizeAtDpi.height.coerceAtLeast(1)

        val fitScale = when (config.fitMode) {
            RenderConfig.FitMode.FIT_WIDTH -> constraints.viewportWidthPx.toFloat() / pageW.toFloat()
            RenderConfig.FitMode.FIT_HEIGHT -> constraints.viewportHeightPx.toFloat() / pageH.toFloat()
            RenderConfig.FitMode.FIT_PAGE -> min(
                constraints.viewportWidthPx.toFloat() / pageW.toFloat(),
                constraints.viewportHeightPx.toFloat() / pageH.toFloat()
            )
            RenderConfig.FitMode.FREE -> 1f
        }

        val baseScale = (fitScale * config.zoom).coerceAtLeast(0.01f)

        val baseWpx = (pageW * baseScale).roundToInt().coerceAtLeast(1)
        val baseHpx = (pageH * baseScale).roundToInt().coerceAtLeast(1)

        // points -> pixels: 先 points->px@dpi，再 * baseScale
        // px@dpi ≈ points * dpi / 72
        val pxPerPt = screenDpi / 72f
        val scale = pxPerPt * baseScale

        val matrix = Matrix().apply {
            postScale(scale, scale)
            // rotationDegrees 可后续扩展：这里先不叠加额外旋转（大部分 PDF 自带 rotation 在 page 内部已处理）
        }

        return PdfPageRenderMetrics(
            pageIndex = page.pageIndex,
            pageWidthPt = pageWidthPt,
            pageHeightPt = pageHeightPt,
            pageWidthPx = baseWpx,
            pageHeightPx = baseHpx,
            baseMatrix = matrix
        )
    }
}
```

## `engines/pdf/src/main/java/com/example/reader/engines/pdf/internal/render/PdfTileProvider.kt`

```kotlin
package com.example.reader.engines.pdf.internal.render

import android.graphics.Bitmap
import android.graphics.Matrix
import android.graphics.RectF
import com.example.reader.api.render.RenderPolicy
import com.example.reader.api.render.TileProvider
import com.example.reader.api.render.TileRequest
import com.example.reader.engines.pdf.internal.PdfEngineConfig
import com.example.reader.engines.pdf.internal.cache.PdfBitmapCache
import com.example.reader.engines.pdf.internal.cache.PdfPageCache
import java.io.Closeable
import java.util.concurrent.atomic.AtomicBoolean
import kotlin.math.roundToInt

internal class PdfTileProvider(
    private val pageIndex: Int,
    private val baseMetrics: PdfPageRenderMetrics,
    private val pageCache: PdfPageCache,
    private val engineConfig: PdfEngineConfig,
    private val bitmapCache: PdfBitmapCache
) : TileProvider, Closeable {

    private val closed = AtomicBoolean(false)

    override suspend fun renderTile(request: TileRequest): Bitmap {
        check(!closed.get()) { "TileProvider closed" }

        val key = buildKey(request)
        bitmapCache.get(key)?.let { cached ->
            return cached
        }

        val scale = request.scale.coerceAtLeast(0.1f)
        val outW = (request.widthPx * scale).roundToInt().coerceAtLeast(1)
        val outH = (request.heightPx * scale).roundToInt().coerceAtLeast(1)

        // PdfiumAndroidKt 推荐 ARGB_8888（RGB_565 会慢，因为要做额外转换） :contentReference[oaicite:3]{index=3}
        val bitmap = Bitmap.createBitmap(outW, outH, Bitmap.Config.ARGB_8888)

        val matrix = Matrix(baseMetrics.baseMatrix).apply {
            // 让输出分辨率提升：device pixels = base pixels * scale
            postScale(scale, scale)
            // 把 tile 的左上角平移到 (0,0)
            postTranslate(-request.leftPx * scale, -request.topPx * scale)
        }

        val clip = RectF(0f, 0f, outW.toFloat(), outH.toFloat())

        // 注意：renderTile 是“局部渲染”，真正重活在这里
        pageCache.withPage(pageIndex) { holder ->
            holder.page.renderPageBitmap(
                bitmap = bitmap,
                matrix = matrix,
                clipRect = clip,
                renderAnnot = engineConfig.renderPdfAnnotations,
                textMask = engineConfig.textMask,
                canvasColor = engineConfig.canvasColor,
                pageBackgroundColor = engineConfig.pageBackgroundColor
            )
        }

        // DRAFT: 你可以在这里做降采样/模糊等策略；现在保持一致
        if (request.quality == RenderPolicy.Quality.DRAFT) {
            // 可选：不缓存 draft
        } else {
            bitmapCache.put(key, bitmap)
        }

        return bitmap
    }

    override fun close() {
        closed.set(true)
    }

    private fun buildKey(r: TileRequest): String =
        buildString {
            append("p=").append(pageIndex)
            append("|l=").append(r.leftPx)
            append("|t=").append(r.topPx)
            append("|w=").append(r.widthPx)
            append("|h=").append(r.heightPx)
            append("|s=").append(r.scale)
            append("|q=").append(r.quality.name)
            // baseMetrics 变了（缩放/fit）也应该区分：用 base page size + matrix 的 scale 做指纹
            append("|bw=").append(baseMetrics.pageWidthPx)
            append("|bh=").append(baseMetrics.pageHeightPx)
        }
}
```

```kotlin
internal class PdfReaderController(
    private val documentId: DocumentId,
    private val pageCount: Int,
    private val pageCache: PdfPageCache,
    private val engineConfig: PdfEngineConfig,
    private val annotationStore: PdfAnnotationStore,
    initialPageIndex: Int,
    initialConfig: RenderConfig.FixedPage
) : ReaderController, Closeable {

    private val mutex = Mutex()

    private var constraints: LayoutConstraints? = null
    private var config: RenderConfig.FixedPage = initialConfig
    private var pageIndex: Int = initialPageIndex.coerceIn(0, (pageCount - 1).coerceAtLeast(0))

    private val bitmapCache = PdfBitmapCache(engineConfig.bitmapCacheMaxBytes)

    private val _state = MutableStateFlow(
        RenderState(
            locator = pageLocator(pageIndex),
            progression = Progression(
                percent = if (pageCount <= 0) 0.0 else (pageIndex + 1).toDouble() / pageCount.toDouble(),
                label = "${pageIndex + 1}/$pageCount",
                current = pageIndex + 1,
                total = pageCount
            ),
            nav = NavigationAvailability(
                canGoPrev = pageIndex > 0,
                canGoNext = pageIndex < pageCount - 1
            ),
            titleInView = null,
            config = config
        )
    )

    private val _events = MutableSharedFlow<ReaderEvent>(extraBufferCapacity = 64)

    override val state: StateFlow<RenderState> = _state.asStateFlow()
    override val events: Flow<ReaderEvent> = _events.asSharedFlow()

    override suspend fun setLayoutConstraints(constraints: LayoutConstraints): ReaderResult<Unit> =
        mutex.withLock {
            this.constraints = constraints
            _state.value = _state.value.copy(config = config)
            ReaderResult.Ok(Unit)
        }

    override suspend fun setConfig(config: RenderConfig): ReaderResult<Unit> =
        mutex.withLock {
            this.config = when (config) {
                is RenderConfig.FixedPage -> config
                is RenderConfig.ReflowText -> RenderConfig.FixedPage() // 防御
            }
            _state.value = _state.value.copy(config = this.config)
            ReaderResult.Ok(Unit)
        }

    override suspend fun render(policy: RenderPolicy): ReaderResult<RenderPage> {
        val snap = mutex.withLock {
            val c = constraints ?: return@withLock Snapshot.error("LayoutConstraints not set")
            Snapshot.ok(
                pageIndex = pageIndex,
                config = config,
                constraints = c
            )
        }

        if (!snap.ok) {
            return ReaderResult.Err(ReaderError.Internal(snap.errorMessage ?: "Invalid state"))
        }

        val result = runCatching {
            var produced: RenderPage? = null
            val ms = measureTimeMillis {
                produced = buildRenderPage(snap.pageIndex, snap.constraints!!, snap.config!!)
            }
            produced!!.copy(metrics = RenderMetrics(renderTimeMs = ms, cacheHit = false))
        }.fold(
            onSuccess = { ReaderResult.Ok(it) },
            onFailure = { ReaderResult.Err(it.toPdfReaderError()) }
        )

        // events
        when (result) {
            is ReaderResult.Ok -> _events.tryEmit(ReaderEvent.Rendered(result.value.id, result.value.metrics))
            is ReaderResult.Err -> _events.tryEmit(ReaderEvent.Error(result.error))
        }

        return result
    }

    override suspend fun next(policy: RenderPolicy): ReaderResult<RenderPage> {
        val moved = mutex.withLock {
            if (pageIndex >= pageCount - 1) return@withLock false
            pageIndex++
            updateStateLocked()
            true
        }
        if (moved) _events.tryEmit(ReaderEvent.PageChanged(pageLocator(pageIndex)))
        return render(policy)
    }

    override suspend fun prev(policy: RenderPolicy): ReaderResult<RenderPage> {
        val moved = mutex.withLock {
            if (pageIndex <= 0) return@withLock false
            pageIndex--
            updateStateLocked()
            true
        }
        if (moved) _events.tryEmit(ReaderEvent.PageChanged(pageLocator(pageIndex)))
        return render(policy)
    }

    override suspend fun goTo(locator: Locator, policy: RenderPolicy): ReaderResult<RenderPage> {
        val target = locator.toPageIndexOrNull()
            ?: return ReaderResult.Err(ReaderError.Internal("Invalid locator for PDF: $locator"))

        val moved = mutex.withLock {
            val newIndex = target.coerceIn(0, (pageCount - 1).coerceAtLeast(0))
            if (newIndex == pageIndex) return@withLock false
            pageIndex = newIndex
            updateStateLocked()
            true
        }
        if (moved) _events.tryEmit(ReaderEvent.PageChanged(pageLocator(pageIndex)))
        return render(policy)
    }

    override suspend fun goToProgress(percent: Double, policy: RenderPolicy): ReaderResult<RenderPage> {
        val p = percent.coerceIn(0.0, 1.0)
        val target = if (pageCount <= 0) 0 else floor(p * pageCount).toInt().coerceIn(0, pageCount - 1)
        return goTo(pageLocator(target), policy)
    }

    override suspend fun prefetchNeighbors(count: Int): ReaderResult<Unit> {
        val current = mutex.withLock { pageIndex }
        val n = count.coerceIn(0, 6)
        val targets = (current - n..current + n).filter { it in 0 until pageCount }
        runCatching {
            targets.forEach { pageCache.prefetch(it) }
        }.onFailure {
            return ReaderResult.Err(it.toPdfReaderError())
        }
        return ReaderResult.Ok(Unit)
    }

    override suspend fun invalidate(reason: InvalidateReason): ReaderResult<Unit> {
        // 这里最常见的是：清 tile cache（比如 zoom/fit 改了）
        bitmapCache.evictAll()
        return ReaderResult.Ok(Unit)
    }

    override fun close() {
        runCatching { pageCache.close() }
        bitmapCache.evictAll()
    }

    private suspend fun buildRenderPage(
        pageIndex: Int,
        constraints: LayoutConstraints,
        config: RenderConfig.FixedPage
    ): RenderPage {
        val locator = pageLocator(pageIndex)

        val metrics = pageCache.withPage(pageIndex) { holder ->
            PdfRenderMath.compute(holder.page, constraints, config)
        }

        val tileProvider = PdfTileProvider(
            pageIndex = pageIndex,
            baseMetrics = metrics,
            pageCache = pageCache,
            engineConfig = engineConfig,
            bitmapCache = bitmapCache
        )

        val links = if (engineConfig.enableLinks) {
            PdfLinkExtractor.extractLinks(pageCache, pageIndex, metrics.pageWidthPt, metrics.pageHeightPt)
        } else {
            emptyList()
        }

        val decorations = if (engineConfig.enableAnnotations) {
            val provider: AnnotationProvider = annotationStore.asProvider(documentId)
            provider.decorationsFor(AnnotationQuery(page = locator)).fold(
                onOk = { it },
                onErr = { emptyList() }
            )
        } else {
            emptyList()
        }

        return RenderPage(
            id = PageId("pdf:$documentId:$pageIndex:${metrics.pageWidthPx}x${metrics.pageHeightPx}:${config.fitMode}:${config.zoom}:${config.rotationDegrees}"),
            locator = locator,
            content = RenderContent.Tiles(
                pageWidthPx = metrics.pageWidthPx,
                pageHeightPx = metrics.pageHeightPx,
                tileProvider = tileProvider
            ),
            links = links,
            decorations = decorations,
            metrics = null
        )
    }

    private fun updateStateLocked() {
        _state.value = _state.value.copy(
            locator = pageLocator(pageIndex),
            progression = Progression(
                percent = if (pageCount <= 0) 0.0 else (pageIndex + 1).toDouble() / pageCount.toDouble(),
                label = "${pageIndex + 1}/$pageCount",
                current = pageIndex + 1,
                total = pageCount
            ),
            nav = NavigationAvailability(
                canGoPrev = pageIndex > 0,
                canGoNext = pageIndex < pageCount - 1
            ),
            config = config
        )
    }

    private fun pageLocator(index: Int): Locator =
        Locator(scheme = LocatorSchemes.PDF_PAGE, value = index.toString())

    private fun Locator.toPageIndexOrNull(): Int? =
        if (scheme == LocatorSchemes.PDF_PAGE) value.toIntOrNull() else null

    private data class Snapshot(
        val ok: Boolean,
        val pageIndex: Int = 0,
        val constraints: LayoutConstraints? = null,
        val config: RenderConfig.FixedPage? = null,
        val errorMessage: String? = null
    ) {
        companion object {
            fun ok(pageIndex: Int, constraints: LayoutConstraints, config: RenderConfig.FixedPage) =
                Snapshot(ok = true, pageIndex = pageIndex, constraints = constraints, config = config)
            fun error(message: String) = Snapshot(ok = false, errorMessage = message)
        }
    }
}

private inline fun <T> com.example.reader.api.error.ReaderResult<T>.fold(
    onOk: (T) -> T,
    onErr: (com.example.reader.api.error.ReaderError) -> T
): T = when (this) {
    is com.example.reader.api.error.ReaderResult.Ok -> onOk(value)
    is com.example.reader.api.error.ReaderResult.Err -> onErr(error)
}
```

```kotlin
import com.example.reader.api.error.ReaderResult
```

```kotlin
import com.example.reader.model.LocatorSchemes

internal class PdfTextProviderImpl(
    private val documentId: DocumentId,
    private val pageCache: PdfPageCache,
    private val config: PdfEngineConfig
) : TextProvider {

    override suspend fun getText(range: LocatorRange): ReaderResult<String> =
        withContext(config.ioDispatcher) {
            try {
                val pageIndex = range.start.toPageIndexOrNull()
                    ?: return@withContext ReaderResult.Err(
                        com.example.reader.api.error.ReaderError.Internal("Invalid range locator: ${range.start}")
                    )

                val text = getPageText(pageIndex).orEmpty()
                val (start, end) = extractCharRange(range)
                if (start != null && end != null && start in 0..text.length && end in 0..text.length && start < end) {
                    ReaderResult.Ok(text.substring(start, end))
                } else {
                    ReaderResult.Ok(text)
                }
            } catch (t: Throwable) {
                ReaderResult.Err(t.toPdfReaderError())
            }
        }

    override suspend fun getTextAround(locator: Locator, maxChars: Int): ReaderResult<String> =
        withContext(config.ioDispatcher) {
            try {
                val pageIndex = locator.toPageIndexOrNull()
                    ?: return@withContext ReaderResult.Err(
                        com.example.reader.api.error.ReaderError.Internal("Invalid locator: $locator")
                    )

                val text = getPageText(pageIndex).orEmpty()
                ReaderResult.Ok(text.take(maxChars.coerceAtLeast(0)))
            } catch (t: Throwable) {
                ReaderResult.Err(t.toPdfReaderError())
            }
        }

    internal suspend fun getPageText(pageIndex: Int): String? =
        pageCache.withPage(pageIndex) { holder ->
            val tp = holder.textPage ?: return@withPage null
            val count = tp.textPageCountChars()
            if (count <= 0) return@withPage ""
            tp.textPageGetText(0, count)
        }

    private fun Locator.toPageIndexOrNull(): Int? =
        if (scheme == LocatorSchemes.PDF_PAGE) value.toIntOrNull() else null

    private fun extractCharRange(range: LocatorRange): Pair<Int?, Int?> {
        val s = range.start.extras["charStart"]?.toIntOrNull()
            ?: range.start.extras["charIndex"]?.toIntOrNull()
        val e = range.end.extras["charEnd"]?.toIntOrNull()
            ?: range.end.extras["charIndex"]?.toIntOrNull()
        return s to e
    }
}
```

## `engines/pdf/src/main/java/com/example/reader/engines/pdf/internal/provider/PdfSearchProviderImpl.kt`

```kotlin
package com.example.reader.engines.pdf.internal.provider

import com.example.reader.api.provider.SearchHit
import com.example.reader.api.provider.SearchOptions
import com.example.reader.api.provider.SearchProvider
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import java.util.Locale
import kotlin.math.max
import kotlinx.coroutines.currentCoroutineContext
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.flow
import kotlinx.coroutines.flow.flowOn

internal class PdfSearchProviderImpl(
    private val pageCount: Int,
    private val textProvider: PdfTextProviderImpl,
    private val config: com.example.reader.engines.pdf.internal.PdfEngineConfig
) : SearchProvider {

    override fun search(query: String, options: SearchOptions): Flow<SearchHit> = flow {
        val q = query.trim()
        if (q.isEmpty()) return@flow

        val ctx = currentCoroutineContext()
        val startPage = options.startFrom?.toPageIndexOrNull()?.coerceIn(0, (pageCount - 1).coerceAtLeast(0)) ?: 0

        val needle = if (options.caseSensitive) q else q.lowercase(Locale.ROOT)

        var hits = 0
        for (pageIndex in startPage until pageCount) {
            ctx.ensureActive()

            val rawText = textProvider.getPageText(pageIndex) ?: continue
            val hay = if (options.caseSensitive) rawText else rawText.lowercase(Locale.ROOT)

            var from = 0
            while (true) {
                ctx.ensureActive()
                val idx = hay.indexOf(needle, startIndex = from)
                if (idx < 0) break

                val end = idx + needle.length
                if (options.wholeWord && !isWholeWord(hay, idx, end)) {
                    from = end
                    continue
                }

                val excerpt = buildExcerpt(rawText, idx, end)
                val range = LocatorRange(
                    start = Locator(
                        scheme = LocatorSchemes.PDF_PAGE,
                        value = pageIndex.toString(),
                        extras = mapOf("charIndex" to idx.toString(), "charStart" to idx.toString())
                    ),
                    end = Locator(
                        scheme = LocatorSchemes.PDF_PAGE,
                        value = pageIndex.toString(),
                        extras = mapOf("charIndex" to end.toString(), "charEnd" to end.toString())
                    )
                )

                emit(
                    SearchHit(
                        range = range,
                        excerpt = excerpt,
                        sectionTitle = null
                    )
                )

                hits++
                if (hits >= options.maxHits) return@flow

                from = end
            }
        }
    }.flowOn(config.ioDispatcher)

    private fun Locator.toPageIndexOrNull(): Int? =
        if (scheme == LocatorSchemes.PDF_PAGE) value.toIntOrNull() else null

    private fun isWholeWord(text: String, start: Int, end: Int): Boolean {
        fun isWordChar(c: Char): Boolean = c.isLetterOrDigit() || c == '_'
        val leftOk = start == 0 || !isWordChar(text[start - 1])
        val rightOk = end >= text.length || !isWordChar(text[end])
        return leftOk && rightOk
    }

    private fun buildExcerpt(rawText: String, start: Int, end: Int): String {
        val left = max(0, start - 40)
        val right = minOf(rawText.length, end + 40)
        return rawText.substring(left, right)
            .replace('\n', ' ')
            .replace('\r', ' ')
            .trim()
    }
}
```

这里我用**反射**方式从 Pdfium 的 Link 对象取字段（避免你因上游 API 字段名变动而编译失败）。如果反射取不到，就降级为 `emptyList()`。

## `engines/pdf/src/main/java/com/example/reader/engines/pdf/internal/link/PdfLinkExtractor.kt`

```kotlin
package com.example.reader.engines.pdf.internal.link

import android.graphics.RectF
import com.example.reader.engines.pdf.internal.cache.PdfPageCache
import com.example.reader.engines.pdf.internal.util.Reflect
import com.example.reader.model.DocumentLink
import com.example.reader.model.LinkTarget
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.NormalizedRect
import kotlin.math.max
import kotlin.math.min

internal object PdfLinkExtractor {

    suspend fun extractLinks(
        pageCache: PdfPageCache,
        pageIndex: Int,
        pageWidthPt: Int,
        pageHeightPt: Int
    ): List<DocumentLink> {
        return pageCache.withPage(pageIndex) { holder ->
            val anyLinks: List<Any> = runCatching { holder.page.getPageLinks() as List<Any> }.getOrElse { emptyList() }
            anyLinks.mapNotNull { mapLink(it, pageWidthPt, pageHeightPt) }
        }
    }

    private fun mapLink(linkObj: Any, pageWpt: Int, pageHpt: Int): DocumentLink? {
        val uri = Reflect.string(linkObj, "uri", "getUri", "url", "getUrl")
        val dest = Reflect.int(linkObj, "destPageIdx", "getDestPageIdx", "pageIdx", "getPageIdx")
        val bounds = Reflect.rectF(linkObj, "bounds", "getBounds", "rect", "getRect", "box", "getBox")

        val target = when {
            uri != null && uri.isNotBlank() -> LinkTarget.External(uri)
            dest != null && dest >= 0 -> LinkTarget.Internal(
                Locator(LocatorSchemes.PDF_PAGE, dest.toString())
            )
            else -> return null
        }

        val normalized = bounds?.let { rect ->
            listOf(rectToNormalized(rect, pageWpt, pageHpt))
        }

        return DocumentLink(
            target = target,
            title = null,
            bounds = normalized
        )
    }

    private fun rectToNormalized(r: RectF, pageWpt: Int, pageHpt: Int): NormalizedRect {
        val w = pageWpt.coerceAtLeast(1).toFloat()
        val h = pageHpt.coerceAtLeast(1).toFloat()

        val left = (r.left / w).coerceIn(0f, 1f)
        val right = (r.right / w).coerceIn(0f, 1f)

        // PDF 坐标往往 y 向上；我们把它转换成 UI 常用 y 向下：top=0, bottom=1
        val y0 = min(r.top, r.bottom)
        val y1 = max(r.top, r.bottom)

        val top = (1f - (y1 / h)).coerceIn(0f, 1f)
        val bottom = (1f - (y0 / h)).coerceIn(0f, 1f)

        return NormalizedRect(
            left = min(left, right),
            top = min(top, bottom),
            right = max(left, right),
            bottom = max(top, bottom)
        )
    }
}
```

```kotlin
package com.example.reader.engines.pdf.internal.util

import android.graphics.RectF

internal object Reflect {

    fun string(obj: Any, vararg names: String): String? =
        names.firstNotNullOfOrNull { n ->
            invokeNoArg(obj, n) as? String ?: getField(obj, n) as? String
        }

    fun int(obj: Any, vararg names: String): Int? =
        names.firstNotNullOfOrNull { n ->
            when (val v = invokeNoArg(obj, n) ?: getField(obj, n)) {
                is Int -> v
                is Long -> v.toInt()
                is Number -> v.toInt()
                is String -> v.toIntOrNull()
                else -> null
            }
        }

    fun long(obj: Any, vararg names: String): Long? =
        names.firstNotNullOfOrNull { n ->
            when (val v = invokeNoArg(obj, n) ?: getField(obj, n)) {
                is Long -> v
                is Int -> v.toLong()
                is Number -> v.toLong()
                is String -> v.toLongOrNull()
                else -> null
            }
        }

    @Suppress("UNCHECKED_CAST")
    fun list(obj: Any, vararg names: String): List<Any>? =
        names.firstNotNullOfOrNull { n ->
            when (val v = invokeNoArg(obj, n) ?: getField(obj, n)) {
                is List<*> -> v.filterNotNull() as List<Any>
                else -> null
            }
        }

    fun rectF(obj: Any, vararg names: String): RectF? =
        names.firstNotNullOfOrNull { n ->
            when (val v = invokeNoArg(obj, n) ?: getField(obj, n)) {
                is RectF -> v
                else -> null
            }
        }

    private fun invokeNoArg(obj: Any, name: String): Any? =
        runCatching {
            val m = obj.javaClass.methods.firstOrNull { it.name == name && it.parameterTypes.isEmpty() }
            m?.isAccessible = true
            m?.invoke(obj)
        }.getOrNull()

    private fun getField(obj: Any, name: String): Any? =
        runCatching {
            val f = obj.javaClass.fields.firstOrNull { it.name == name }
                ?: obj.javaClass.declaredFields.firstOrNull { it.name == name }
            f?.isAccessible = true
            f?.get(obj)
        }.getOrNull()
}
```

```kotlin
package com.example.reader.engines.pdf.internal.annotation

import com.example.reader.api.annotation.Decoration
import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.AnnotationProvider
import com.example.reader.api.provider.AnnotationQuery
import com.example.reader.model.DocumentId
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.annotation.Annotation
import com.example.reader.model.annotation.AnnotationDraft
import com.example.reader.model.annotation.AnnotationId
import com.example.reader.model.annotation.AnnotationAnchor
import java.util.UUID
import java.util.concurrent.ConcurrentHashMap
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.asStateFlow

internal class InMemoryPdfAnnotationStore : PdfAnnotationStore {

    private data class DocState(
        val flow: MutableStateFlow<List<Annotation>> = MutableStateFlow(emptyList())
    )

    private val docs = ConcurrentHashMap<String, DocState>()

    override fun asProvider(documentId: DocumentId): AnnotationProvider {
        val state = docs.getOrPut(documentId.value) { DocState() }
        return InMemoryPdfAnnotationProvider(documentId, state)
    }
}

private class InMemoryPdfAnnotationProvider(
    private val documentId: DocumentId,
    private val state: InMemoryPdfAnnotationStore.DocState
) : AnnotationProvider {

    override fun observeAll(): Flow<List<Annotation>> = state.flow.asStateFlow()

    override suspend fun listAll(): ReaderResult<List<Annotation>> =
        ReaderResult.Ok(state.flow.value)

    override suspend fun query(query: AnnotationQuery): ReaderResult<List<Annotation>> {
        val all = state.flow.value
        val filtered = when {
            query.page != null -> {
                val page = query.page
                all.filter { ann ->
                    when (val a = ann.anchor) {
                        is AnnotationAnchor.FixedRects -> a.page.samePage(page)
                        else -> false
                    }
                }
            }
            query.range != null -> all // PDF fixed 这里不按 range 筛
            else -> all
        }
        return ReaderResult.Ok(filtered)
    }

    override suspend fun create(draft: AnnotationDraft): ReaderResult<Annotation> {
        val now = System.currentTimeMillis()
        val ann = Annotation(
            id = AnnotationId(UUID.randomUUID().toString()),
            type = draft.type,
            anchor = draft.anchor,
            content = draft.content,
            style = draft.style,
            createdAtEpochMs = now,
            updatedAtEpochMs = now,
            extra = draft.extra
        )
        state.flow.value = state.flow.value + ann
        return ReaderResult.Ok(ann)
    }

    override suspend fun update(annotation: Annotation): ReaderResult<Unit> {
        val now = System.currentTimeMillis()
        val updated = annotation.copy(updatedAtEpochMs = now)
        state.flow.value = state.flow.value.map { if (it.id == updated.id) updated else it }
        return ReaderResult.Ok(Unit)
    }

    override suspend fun delete(id: AnnotationId): ReaderResult<Unit> {
        state.flow.value = state.flow.value.filterNot { it.id == id }
        return ReaderResult.Ok(Unit)
    }

    override suspend fun decorationsFor(query: AnnotationQuery): ReaderResult<List<Decoration>> {
        val anns = query(query).getOrNull().orEmpty()
        val decos = anns.mapNotNull { ann ->
            when (val a = ann.anchor) {
                is AnnotationAnchor.FixedRects -> Decoration.Fixed(
                    page = a.page,
                    rects = a.rects,
                    style = ann.style
                )
                is AnnotationAnchor.ReflowRange -> null
            }
        }
        return ReaderResult.Ok(decos)
    }

    private fun AnnotationAnchor.FixedRects.pageSameIndex(page: Locator): Boolean =
        this.page.samePage(page)

    private fun Locator.samePage(other: Locator): Boolean {
        if (scheme != LocatorSchemes.PDF_PAGE || other.scheme != LocatorSchemes.PDF_PAGE) return false
        return value == other.value
    }
}

private fun <T> ReaderResult<T>.getOrNull(): T? =
    when (this) {
        is ReaderResult.Ok -> value
        is ReaderResult.Err -> null
    }
```

```kotlin
import java.security.MessageDigest

internal fun sha1Hex(input: String): String {
    val md = MessageDigest.getInstance("SHA-1")
    val bytes = md.digest(input.toByteArray(Charsets.UTF_8))
    return buildString(bytes.size * 2) {
        for (b in bytes) {
            append(((b.toInt() shr 4) and 0xF).toString(16))
            append((b.toInt() and 0xF).toString(16))
        }
    }
```
