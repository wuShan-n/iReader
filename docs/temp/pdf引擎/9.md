# 1) ✅ 可清空高亮（不改 core API）：SearchProvider 额外实现一个可选控制接口

## 1.1 修改：`PdfHighlightStore.kt`（新增 `clearHighlights()`）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/provider/PdfHighlightStore.kt
package com.example.reader.pdf.internal.provider

import com.example.reader.model.NormalizedRect
import java.util.concurrent.ConcurrentHashMap
import java.util.concurrent.atomic.AtomicLong
import kotlinx.coroutines.flow.MutableSharedFlow
import kotlinx.coroutines.flow.SharedFlow

internal class PdfHighlightStore(
    private val maxRectsPerPage: Int = 128
) {
    private val gen = AtomicLong(0L)
    private val map = ConcurrentHashMap<Int, List<NormalizedRect>>()

    private val _updates = MutableSharedFlow<Int>(extraBufferCapacity = 64)
    val updates: SharedFlow<Int> = _updates

    fun newSearch(): Long {
        val g = gen.incrementAndGet()
        map.clear()
        _updates.tryEmit(ALL_PAGES)
        return g
    }

    /** ✅ 新增：显式清空（给“清除高亮”按钮用） */
    fun clearHighlights() {
        gen.incrementAndGet()
        map.clear()
        _updates.tryEmit(ALL_PAGES)
    }

    fun add(pageIndex: Int, rects: List<NormalizedRect>, g: Long) {
        if (g != gen.get() || rects.isEmpty()) return

        val merged = (map[pageIndex].orEmpty() + rects)
            .distinctBy { "${it.left},${it.top},${it.right},${it.bottom}" }
            .take(maxRectsPerPage)

        map[pageIndex] = merged
        _updates.tryEmit(pageIndex)
    }

    fun rectsFor(pageIndex: Int): List<NormalizedRect> = map[pageIndex].orEmpty()

    companion object {
        const val ALL_PAGES = -1
    }
}
```

## 1.3 修改：`PdfSearchProvider.kt`（实现 HighlightControl；query 为空时清空并结束）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/provider/PdfSearchProvider.kt
package com.example.reader.pdf.internal.provider

import android.graphics.RectF
import com.example.reader.api.provider.SearchHit
import com.example.reader.api.provider.SearchOptions
import com.example.reader.api.provider.SearchProvider
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.NormalizedRect
import com.example.reader.pdf.internal.pdfium.PdfiumDocument
import com.example.reader.pdf.internal.pdfium.pdfiumRotateIndex
import com.example.reader.pdf.internal.pdfium.rectToNormalized
import com.example.reader.pdf.internal.util.useSafely
import io.legere.pdfiumandroid.api.FindFlags
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.flow
import kotlin.coroutines.coroutineContext

internal class PdfSearchProvider(
    private val pdfium: PdfiumDocument,
    private val rotationDegreesProvider: () -> Int,
    private val highlightStore: PdfHighlightStore
) : SearchProvider, HighlightControl {

    override fun clearHighlights() {
        highlightStore.clearHighlights()
    }

    override fun search(query: String, options: SearchOptions): Flow<SearchHit> = flow {
        val q = query.trim()

        // ✅ 允许 “清空高亮” 的另一条路径：search("") 即清空并结束
        if (q.isEmpty()) {
            highlightStore.clearHighlights()
            return@flow
        }

        val searchGen = highlightStore.newSearch()

        val flags = buildSet {
            if (options.caseSensitive) add(FindFlags.MatchCase)
            if (options.wholeWord) add(FindFlags.MatchWholeWord)
        }

        val startPage = options.startFrom
            ?.takeIf { it.scheme == LocatorSchemes.PDF_PAGE }
            ?.value?.toIntOrNull()
            ?.coerceAtLeast(0)
            ?: 0

        val maxHits = options.maxHits.coerceAtLeast(1)
        var emitted = 0

        // ✅ 懒加载章节映射（只算一次）
        val titleResolver: suspend (Int) -> String? = { page -> pdfium.sectionTitleForPage(page) }

        for (pageIndex in startPage until pdfium.pageCount) {
            coroutineContext.ensureActive()
            if (emitted >= maxHits) return@flow

            val rotation = rotationDegreesProvider()
            val rotate = pdfiumRotateIndex(rotation)
            val base = 10_000

            pdfium.withTextPage(pageIndex) { page, textPage ->
                val find = textPage.findStart(q, flags, 0) ?: return@withTextPage
                find.useSafely { fr ->
                    while (fr.findNext()) {
                        coroutineContext.ensureActive()
                        val startIdx = fr.getSchResultIndex()
                        if (startIdx < 0) continue

                        val rects: List<NormalizedRect> = buildList {
                            val n = textPage.textPageCountRects(startIdx, q.length).coerceAtLeast(0)
                            for (i in 0 until n) {
                                val r: RectF = textPage.textPageGetRect(i) ?: continue
                                val device = page.mapRectToDevice(
                                    startX = 0, startY = 0,
                                    sizeX = base, sizeY = base,
                                    rotate = rotate,
                                    coords = r
                                )
                                add(rectToNormalized(device, base))
                            }
                        }

                        // ✅ 写入高亮 store，让 controller 自动 decorations
                        highlightStore.add(pageIndex, rects, searchGen)

                        val excerpt = buildExcerpt(
                            text = pdfium.pageText(pageIndex),
                            start = startIdx,
                            len = q.length
                        )

                        val loc = Locator(LocatorSchemes.PDF_PAGE, pageIndex.toString())
                        val extras = if (rects.isNotEmpty()) {
                            mapOf(PdfRectCodec.EXTRA_RECTS to PdfRectCodec.encode(rects))
                        } else emptyMap()

                        emit(
                            SearchHit(
                                range = LocatorRange(start = loc, end = loc, extras = extras),
                                excerpt = excerpt,
                                sectionTitle = titleResolver(pageIndex) // ✅ 新增：章节标题
                            )
                        )

                        emitted++
                        if (emitted >= maxHits) return@useSafely
                    }
                }
            }
        }
    }

    private fun buildExcerpt(text: String, start: Int, len: Int): String {
        if (text.isEmpty()) return ""
        val s = (start - 24).coerceAtLeast(0)
        val e = (start + len + 24).coerceAtMost(text.length)
        return text.substring(s, e).replace('\n', ' ')
    }
}
```

# 2) ✅ SearchHit.sectionTitle：建立“页 → 章节标题”索引（来自 Outline）

## 2.1 新增：`PdfOutlineIndex.kt`

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/pdfium/PdfOutlineIndex.kt
package com.example.reader.pdf.internal.pdfium

import com.example.reader.model.LocatorSchemes
import com.example.reader.model.OutlineNode
import kotlin.math.max

internal class PdfOutlineIndex private constructor(
    private val starts: List<Pair<Int, String>>
) {
    fun titleForPage(pageIndex: Int): String? {
        if (starts.isEmpty()) return null
        // binary search last start <= pageIndex
        var lo = 0
        var hi = starts.size - 1
        var ans = -1
        while (lo <= hi) {
            val mid = (lo + hi) ushr 1
            val p = starts[mid].first
            if (p <= pageIndex) {
                ans = mid
                lo = mid + 1
            } else hi = mid - 1
        }
        return if (ans >= 0) starts[ans].second else null
    }

    companion object {
        fun build(outline: List<OutlineNode>): PdfOutlineIndex {
            val pairs = mutableListOf<Pair<Int, String>>()
            fun walk(node: OutlineNode) {
                val page = node.locator
                    .takeIf { it.scheme == LocatorSchemes.PDF_PAGE }
                    ?.value?.toIntOrNull()
                val title = node.title.trim()
                if (page != null && title.isNotEmpty()) {
                    pairs += (max(0, page) to title)
                }
                node.children.forEach(::walk)
            }
            outline.forEach(::walk)

            // sort by page asc; for same page, keep the last one (deeper nodes often later)
            val sorted = pairs.sortedWith(compareBy({ it.first }))
            val compact = mutableListOf<Pair<Int, String>>()
            for ((p, t) in sorted) {
                if (compact.isNotEmpty() && compact.last().first == p) {
                    compact[compact.lastIndex] = p to t
                } else {
                    compact += p to t
                }
            }
            return PdfOutlineIndex(compact)
        }
    }
}
```

## 2.2 修改：`PdfiumDocument.kt`（新增 `sectionTitleForPage()` + index 缓存 + cache 失效）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/pdfium/PdfiumDocument.kt
package com.example.reader.pdf.internal.pdfium

import com.example.reader.model.OutlineNode

// ✅ 新增字段：索引缓存
private var outlineIndexCache: PdfOutlineIndex? = null

// ✅ 新增：给 SearchProvider 用的章节标题查询
suspend fun sectionTitleForPage(pageIndex: Int): String? {
    val idx = outlineIndexCache ?: PdfOutlineIndex.build(outline()).also { outlineIndexCache = it }
    return idx.titleForPage(pageIndex)
}

// 如果你已有 evictAllCaches()，加一行：
fun evictAllCaches() {
    outlineCache = null
    outlineIndexCache = null // ✅ 新增
    linksCache.evictAll()
    textCache.evictAll()
    linkNormCache.evictAll() // 如果你上轮加了这个
}
```
