## 2) 修改：`PlatformPdfBackend.kt`（去掉 `runBlocking`，统一用 Mutex + suspend；加 pageSize 缓存）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/backend/PlatformPdfBackend.kt
package com.example.reader.pdf.internal.backend

import android.graphics.Bitmap
import android.graphics.Matrix
import android.graphics.Rect
import android.graphics.pdf.PdfRenderer
import android.os.ParcelFileDescriptor
import com.example.reader.pdf.internal.cache.PageSizeCache
import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock

internal class PlatformPdfBackend(
    pfd: ParcelFileDescriptor
) : PdfBackend {

    private val renderer = PdfRenderer(pfd)

    /** PdfRenderer 限制：同一时间只能打开一个 Page。 */
    private val pageMutex = Mutex()

    /** page size 频繁读取会反复 openPage，做个轻量 cache。 */
    private val sizeCache = PageSizeCache(maxEntries = 64)

    override val pageCount: Int get() = renderer.pageCount

    override suspend fun pageSize(pageIndex: Int): PdfPageSize {
        sizeCache.get(pageIndex)?.let { return it }

        return pageMutex.withLock {
            sizeCache.get(pageIndex)?.let { return@withLock it }
            renderer.openPage(pageIndex).use { page ->
                PdfPageSize(page.width, page.height).also { sizeCache.put(pageIndex, it) }
            }
        }
    }

    override suspend fun render(
        pageIndex: Int,
        bitmap: Bitmap,
        clip: Rect,
        matrix: Matrix,
        mode: PdfRenderMode
    ) {
        pageMutex.withLock {
            renderer.openPage(pageIndex).use { page ->
                val renderMode = when (mode) {
                    PdfRenderMode.DRAFT -> PdfRenderer.Page.RENDER_MODE_FOR_DISPLAY
                    PdfRenderMode.FINAL -> PdfRenderer.Page.RENDER_MODE_FOR_PRINT
                }
                page.render(bitmap, clip, matrix, renderMode)
            }
        }
    }

    override fun close() {
        renderer.close()
    }
}
```

## 3) 新增：`PageSizeCache.kt`（轻量 LRU，避免重复 openPage）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/cache/PageSizeCache.kt
package com.example.reader.pdf.internal.cache

import com.example.reader.pdf.internal.backend.PdfPageSize
import java.util.LinkedHashMap

internal class PageSizeCache(
    private val maxEntries: Int
) {
    private val map = object : LinkedHashMap<Int, PdfPageSize>(maxEntries, 0.75f, true) {
        override fun removeEldestEntry(eldest: MutableMap.MutableEntry<Int, PdfPageSize>?): Boolean {
            return size > maxEntries
        }
    }

    @Synchronized fun get(pageIndex: Int): PdfPageSize? = map[pageIndex]
    @Synchronized fun put(pageIndex: Int, size: PdfPageSize) { map[pageIndex] = size }
    @Synchronized fun clear() = map.clear()
}
```

## 4) 修改：`FitTransform.kt`（把“旋转后的页尺寸”与“原始尺寸”分离）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/render/FitTransform.kt
package com.example.reader.pdf.internal.render

import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.RenderConfig
import com.example.reader.pdf.internal.backend.PdfPageSize
import kotlin.math.max
import kotlin.math.min

internal data class PageTransform(
    /** 旋转后的逻辑页尺寸（用于 UI 坐标系 / TileRequest 范围） */
    val pageWidthPx: Int,
    val pageHeightPx: Int,
    /** 原始 page 坐标 -> 旋转后像素坐标 的 scale */
    val scale: Float
)

internal fun rotatedSize(base: PdfPageSize, rotationDegrees: Int): PdfPageSize {
    val r = ((rotationDegrees % 360) + 360) % 360
    return if (r == 90 || r == 270) PdfPageSize(base.height, base.width) else base
}

internal fun computeTransform(
    basePage: PdfPageSize,
    config: RenderConfig.FixedPage,
    constraints: LayoutConstraints
): PageTransform {
    val rotated = rotatedSize(basePage, config.rotationDegrees)

    val viewportW = constraints.viewportWidthPx.toFloat().coerceAtLeast(1f)
    val viewportH = constraints.viewportHeightPx.toFloat().coerceAtLeast(1f)

    val pageW = rotated.width.toFloat().coerceAtLeast(1f)
    val pageH = rotated.height.toFloat().coerceAtLeast(1f)

    val fitScale = when (config.fitMode) {
        RenderConfig.FitMode.FIT_WIDTH -> viewportW / pageW
        RenderConfig.FitMode.FIT_HEIGHT -> viewportH / pageH
        RenderConfig.FitMode.FIT_PAGE -> min(viewportW / pageW, viewportH / pageH)
        RenderConfig.FitMode.FREE -> 1f
    }

    val scale = max(0.05f, fitScale * config.zoom)

    val scaledW = (pageW * scale).toInt().coerceAtLeast(1)
    val scaledH = (pageH * scale).toInt().coerceAtLeast(1)

    return PageTransform(
        pageWidthPx = scaledW,
        pageHeightPx = scaledH,
        scale = scale
    )
}
```

## 5) 新增：`PdfMatrix.kt`（正确支持 0/90/180/270 旋转 + tile 裁剪）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/render/PdfMatrix.kt
package com.example.reader.pdf.internal.render

import android.graphics.Matrix
import com.example.reader.pdf.internal.backend.PdfPageSize

internal fun buildPdfTileMatrix(
    basePage: PdfPageSize,     // 未旋转原始页尺寸
    scale: Float,
    rotationDegrees: Int,
    leftPx: Int,
    topPx: Int
): Matrix {
    val r = ((rotationDegrees % 360) + 360) % 360
    val w = basePage.width.toFloat() * scale
    val h = basePage.height.toFloat() * scale

    return Matrix().apply {
        // 先把 page 映射到“缩放后的像素空间”
        setScale(scale, scale)

        // 再做旋转（围绕原点），并把内容平移回第一象限
        when (r) {
            0 -> Unit
            90 -> {
                postRotate(90f)
                postTranslate(h, 0f)
            }
            180 -> {
                postRotate(180f)
                postTranslate(w, h)
            }
            270 -> {
                postRotate(270f)
                postTranslate(0f, w)
            }
            else -> {
                // 非 90 倍数不建议支持（会导致 UI 坐标/命中复杂化）
                postRotate(r.toFloat())
            }
        }

        // 最后把 tile 的左上角对齐到 bitmap 原点（实现“只画局部”）
        postTranslate(-leftPx.toFloat(), -topPx.toFloat())
    }
}
```

## 6) 修改：`PdfTileProvider.kt`（支持 allowCache；DRAFT 用 RGB_565；调用 suspend backend.render；支持 rotation）

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/render/PdfTileProvider.kt
package com.example.reader.pdf.internal.render

import android.graphics.Bitmap
import android.graphics.Rect
import com.example.reader.api.render.RenderPolicy
import com.example.reader.api.render.TileProvider
import com.example.reader.api.render.TileRequest
import com.example.reader.pdf.internal.backend.PdfBackend
import com.example.reader.pdf.internal.backend.PdfPageSize
import com.example.reader.pdf.internal.backend.PdfRenderMode
import com.example.reader.pdf.internal.cache.TileCache
import java.io.Closeable
import java.util.concurrent.atomic.AtomicBoolean
import kotlin.math.max
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.withContext

internal class PdfTileProvider(
    private val backend: PdfBackend,
    private val pageIndex: Int,
    private val basePage: PdfPageSize,     // 未旋转原始页尺寸
    private val baseScale: Float,
    private val rotationDegrees: Int,
    private val allowCache: Boolean,
    private val tileCache: TileCache,
    private val ioDispatcher: CoroutineDispatcher
) : TileProvider, Closeable {

    private val closed = AtomicBoolean(false)

    override suspend fun renderTile(request: TileRequest): Bitmap = withContext(ioDispatcher) {
        check(!closed.get()) { "TileProvider closed" }

        val scale = max(0.05f, baseScale * request.scale)
        val key = if (allowCache) buildKey(request, scale) else ""

        if (allowCache) {
            tileCache.get(key)?.let { return@withContext it }
        }

        val config = when (request.quality) {
            RenderPolicy.Quality.DRAFT -> Bitmap.Config.RGB_565
            RenderPolicy.Quality.FINAL -> Bitmap.Config.ARGB_8888
        }

        val bmp = Bitmap.createBitmap(
            request.widthPx.coerceAtLeast(1),
            request.heightPx.coerceAtLeast(1),
            config
        )

        val clip = Rect(0, 0, bmp.width, bmp.height)

        val matrix = buildPdfTileMatrix(
            basePage = basePage,
            scale = scale,
            rotationDegrees = rotationDegrees,
            leftPx = request.leftPx,
            topPx = request.topPx
        )

        val mode = when (request.quality) {
            RenderPolicy.Quality.DRAFT -> PdfRenderMode.DRAFT
            RenderPolicy.Quality.FINAL -> PdfRenderMode.FINAL
        }

        backend.render(
            pageIndex = pageIndex,
            bitmap = bmp,
            clip = clip,
            matrix = matrix,
            mode = mode
        )

        if (allowCache) tileCache.put(key, bmp)
        bmp
    }

    private fun buildKey(req: TileRequest, scale: Float): String {
        // scale 量化，减少碎片（0.05 步进）
        val bucket = ((scale * 100f).toInt() / 5) * 5
        return "p=$pageIndex|l=${req.leftPx}|t=${req.topPx}|w=${req.widthPx}|h=${req.heightPx}|s=$bucket|r=$rotationDegrees|q=${req.quality}"
    }

    override fun close() {
        closed.set(true)
    }
}
```

```kotlin
// 只贴“需要改的段落”，其余保持不变：
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/render/PdfController.kt

// ✅ 修改 render(policy) 内部：
// - backend.pageSize 变为 suspend
// - computeTransform 传 basePage（未旋转）
// - PdfTileProvider 传 basePage + allowCache

override suspend fun render(policy: RenderPolicy): ReaderResult<RenderPage> =
    withContext(ioDispatcher) {
        try {
            val c = constraints ?: return@withContext ReaderResult.Err(
                com.example.reader.api.error.ReaderError.Internal("LayoutConstraints not set")
            )

            val basePageSize = backend.pageSize(pageIndex) // ✅ suspend
            val transform = computeTransform(basePageSize, config, c)

            val page: RenderPage
            val timeMs = measureTimeMillis {
                val provider = PdfTileProvider(
                    backend = backend,
                    pageIndex = pageIndex,
                    basePage = basePageSize,                 // ✅ 新增
                    baseScale = transform.scale,
                    rotationDegrees = config.rotationDegrees,
                    allowCache = policy.allowCache,          // ✅ 新增
                    tileCache = tileCache,
                    ioDispatcher = ioDispatcher
                )

                page = RenderPage(
                    id = PageId("pdf:${pageIndex}:${transform.scale}:${config.rotationDegrees}:${config.fitMode}:${config.zoom}:${policy.allowCache}"),
                    locator = pdfPageLocator(pageIndex),
                    content = RenderContent.Tiles(
                        pageWidthPx = transform.pageWidthPx,
                        pageHeightPx = transform.pageHeightPx,
                        tileProvider = provider
                    ),
                    links = emptyList(),
                    decorations = emptyList(),
                    metrics = null
                )
            }

            val metrics = RenderMetrics(renderTimeMs = timeMs, cacheHit = false)

            _state.value = buildState()
            _events.tryEmit(ReaderEvent.Rendered(page.id, metrics))

            ReaderResult.Ok(page.copy(metrics = metrics))
        } catch (t: Throwable) {
            _events.tryEmit(ReaderEvent.Error(t))
            ReaderResult.Err(t.toReaderError())
        }
    }
```

### 这轮优化带来的实际收益

* ✅ **彻底移除 `runBlocking`**（避免潜在死锁/线程阻塞/卡顿）
* ✅ **符合 PdfRenderer “同一时刻只能打开一个 Page”** 的约束（Mutex 串行化，且 pageSize/render 共用锁）
* ✅ **pageSize LRU 缓存**：避免频繁 openPage（尤其 layout/zoom 改动时）
* ✅ **正确支持旋转 0/90/180/270**（page 尺寸、tile matrix 都一致）
* ✅ **RenderPolicy.allowCache 生效**（你已有 policy 字段，这里真正用起来了）
* ✅ **DRAFT 用 RGB_565** 降内存 + 更快首屏（FINAL 仍用 ARGB_8888）
