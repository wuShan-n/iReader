// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/cache/TileCache.kt
package com.example.reader.pdf.internal.cache

import android.graphics.Bitmap
import android.util.LruCache

internal class TileCache(maxBytes: Int) {

    private val cache = object : LruCache<String, Bitmap>(maxBytes) {
        override fun sizeOf(key: String, value: Bitmap): Int = value.allocationByteCount
    }

    @Synchronized fun get(key: String): Bitmap? = cache.get(key)

    @Synchronized fun put(key: String, bitmap: Bitmap) {
        cache.put(key, bitmap)
    }

    @Synchronized fun evictAll() = cache.evictAll()

    @Synchronized fun entryCount(): Int = cache.snapshot().size
}
```

// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/cache/CacheSizing.kt
package com.example.reader.pdf.internal.cache

internal fun defaultPdfTileCacheBytes(): Int {
    val max = Runtime.getRuntime().maxMemory().coerceAtLeast(128L * 1024 * 1024) // 最低按 128MB 估算
    val target = (max / 20L) // ~5%
    val min = 16L * 1024 * 1024
    val cap = 96L * 1024 * 1024
    return target.coerceIn(min, cap).toInt()
}
```

## 3) 修改：`PdfTileProvider.kt`（新增代际校验 + cancellation 友好 + OOM 降级）

> 只贴**需要改的部分**：构造参数、closed/valid 检查、异常处理。

```kotlin
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/render/PdfTileProvider.kt
package com.example.reader.pdf.internal.render

import android.graphics.Bitmap
import android.graphics.Rect
import com.example.reader.api.render.RenderPolicy
import com.example.reader.api.render.TileProvider
import com.example.reader.api.render.TileRequest
import com.example.reader.pdf.internal.backend.PdfBackend
import com.example.reader.pdf.internal.backend.PdfPageSize
import com.example.reader.pdf.internal.backend.PdfRenderMode
import com.example.reader.pdf.internal.cache.TileCache
import java.io.Closeable
import java.util.concurrent.atomic.AtomicBoolean
import kotlin.math.max
import kotlinx.coroutines.CancellationException
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.withContext
import kotlin.coroutines.coroutineContext

internal class PdfTileProvider(
    private val backend: PdfBackend,
    private val pageIndex: Int,
    private val basePage: PdfPageSize,
    private val baseScale: Float,
    private val rotationDegrees: Int,
    private val allowCache: Boolean,
    private val tileCache: TileCache,
    private val ioDispatcher: CoroutineDispatcher,
    /** ✅ 新增：由 Controller 注入“是否仍然有效” */
    private val isValid: () -> Boolean
) : TileProvider, Closeable {

    private val closed = AtomicBoolean(false)

    override suspend fun renderTile(request: TileRequest): Bitmap = withContext(ioDispatcher) {
        coroutineContext.ensureActive()

        if (closed.get() || !isValid()) {
            throw CancellationException("stale tile provider")
        }

        val scale = max(0.05f, baseScale * request.scale)
        val key = if (allowCache) buildKey(request, scale) else ""

        if (allowCache) tileCache.get(key)?.let { return@withContext it }

        try {
            val bmpConfig = when (request.quality) {
                RenderPolicy.Quality.DRAFT -> Bitmap.Config.RGB_565
                RenderPolicy.Quality.FINAL -> Bitmap.Config.ARGB_8888
            }

            val bmp = Bitmap.createBitmap(
                request.widthPx.coerceAtLeast(1),
                request.heightPx.coerceAtLeast(1),
                bmpConfig
            )

            val clip = Rect(0, 0, bmp.width, bmp.height)

            val matrix = buildPdfTileMatrix(
                basePage = basePage,
                scale = scale,
                rotationDegrees = rotationDegrees,
                leftPx = request.leftPx,
                topPx = request.topPx
            )

            val mode = when (request.quality) {
                RenderPolicy.Quality.DRAFT -> PdfRenderMode.DRAFT
                RenderPolicy.Quality.FINAL -> PdfRenderMode.FINAL
            }

            backend.render(pageIndex, bmp, clip, matrix, mode)

            if (allowCache) tileCache.put(key, bmp)
            bmp
        } catch (oom: OutOfMemoryError) {
            // ✅ 降级：清缓存保命，避免一次 OOM 直接把进程打死
            tileCache.evictAll()
            Bitmap.createBitmap(1, 1, Bitmap.Config.ARGB_8888)
        }
    }

    override fun close() {
        closed.set(true)
    }

    // buildKey(...) 不变
}
```

### 4.3 修改：`render(policy)` 创建 provider 时注入 isValid + 关闭旧 provider + metrics cacheHit 更真实

```kotlin
override suspend fun render(policy: RenderPolicy): ReaderResult<RenderPage> =
    withContext(ioDispatcher) {
        try {
            val c = constraints ?: return@withContext ReaderResult.Err(
                com.example.reader.api.error.ReaderError.Internal("LayoutConstraints not set")
            )

            val gen = generation.get()
            val basePageSize = backend.pageSize(pageIndex)
            val transform = computeTransform(basePageSize, config, c)

            val page: RenderPage
            val timeMs = measureTimeMillis {
                val provider = PdfTileProvider(
                    backend = backend,
                    pageIndex = pageIndex,
                    basePage = basePageSize,
                    baseScale = transform.scale,
                    rotationDegrees = config.rotationDegrees,
                    allowCache = policy.allowCache,
                    tileCache = tileCache,
                    ioDispatcher = ioDispatcher,
                    isValid = { generation.get() == gen } // ✅ 代际校验
                )

                // ✅ 关闭旧 provider（减少旧任务继续画）
                currentProvider?.close()
                currentProvider = provider

                page = RenderPage(
                    id = PageId("pdf:${pageIndex}:${transform.scale}:${config.rotationDegrees}:${config.fitMode}:${config.zoom}:${policy.allowCache}"),
                    locator = pdfPageLocator(pageIndex),
                    content = RenderContent.Tiles(
                        pageWidthPx = transform.pageWidthPx,
                        pageHeightPx = transform.pageHeightPx,
                        tileProvider = provider
                    ),
                    links = emptyList(),
                    decorations = emptyList(),
                    metrics = null
                )
            }

            val metrics = RenderMetrics(
                renderTimeMs = timeMs,
                cacheHit = policy.allowCache && tileCache.entryCount() > 0
            )

            _state.value = buildState()
            _events.tryEmit(ReaderEvent.Rendered(page.id, metrics))
            ReaderResult.Ok(page.copy(metrics = metrics))
        } catch (t: Throwable) {
            _events.tryEmit(ReaderEvent.Error(t))
            ReaderResult.Err(t.toReaderError())
        }
    }
```

### 4.4 新增：`prefetchNeighbors()`（预热邻页首屏 DRAFT tile，限制并发）

```kotlin
override suspend fun prefetchNeighbors(count: Int): ReaderResult<Unit> = withContext(ioDispatcher) {
    try {
        val c = constraints ?: return@withContext ReaderResult.Ok(Unit)
        if (count <= 0 || pageCount <= 0) return@withContext ReaderResult.Ok(Unit)

        val targets = buildList {
            for (i in 1..count) {
                val prev = pageIndex - i
                val next = pageIndex + i
                if (prev >= 0) add(prev)
                if (next < pageCount) add(next)
            }
        }
        if (targets.isEmpty()) return@withContext ReaderResult.Ok(Unit)

        // ✅ 限制并发：避免瞬间开很多 bitmap
        val sem = Semaphore(permits = 2)

        coroutineScope {
            targets.forEach { idx ->
                launch {
                    sem.withPermit {
                        prefetchOnePage(idx, c)
                    }
                }
            }
        }

        ReaderResult.Ok(Unit)
    } catch (t: Throwable) {
        ReaderResult.Err(t.toReaderError())
    }
}

// ✅ 新增：预热单页首屏（0,0, viewportW, viewportH）DRAFT tile
private suspend fun prefetchOnePage(
    targetPageIndex: Int,
    c: LayoutConstraints
) {
    val base = backend.pageSize(targetPageIndex)
    val transform = computeTransform(base, config, c)

    val w = c.viewportWidthPx.coerceAtMost(transform.pageWidthPx).coerceAtLeast(1)
    val h = c.viewportHeightPx.coerceAtMost(transform.pageHeightPx).coerceAtLeast(1)

    val provider = PdfTileProvider(
        backend = backend,
        pageIndex = targetPageIndex,
        basePage = base,
        baseScale = transform.scale,
        rotationDegrees = config.rotationDegrees,
        allowCache = true,
        tileCache = tileCache,
        ioDispatcher = ioDispatcher,
        isValid = { true } // prefetch 独立任务，不受当前代际影响
    )

    try {
        provider.renderTile(
            com.example.reader.api.render.TileRequest(
                leftPx = 0,
                topPx = 0,
                widthPx = w,
                heightPx = h,
                scale = 1.0f,
                quality = RenderPolicy.Quality.DRAFT
            )
        )
    } finally {
        provider.close()
    }
}
```

## 这轮优化你会立刻感受到的变化

* **翻页/缩放/旋转**时，旧页 tile 请求会自然被 `CancellationException` 终止（代际失效），不会继续抢 IO/CPU
* `prefetchNeighbors()` 真正生效：邻页首屏更快出现（DRAFT 先顶上）
* tile cache 不再写死 24MB：在大内存机型上更顺滑，小内存机型更稳
* 并发安全：多 tile 并发请求不会把 LruCache 打爆线程问题
* OOM 场景降级：不至于“一次大页放大渲染就崩”
