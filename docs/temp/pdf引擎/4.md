下面继续优化（**只给修改/新增**），本轮重点解决两个大痛点：

1. **同一 tile 并发请求“打桩”**（重复渲染浪费 CPU/IO）→ 做 *in-flight 去重*
2. **大 tile/高 zoom 容易 OOM** → 做 *内存预估 + 软降级*（清缓存 + 返回占位）

## ✅ 新增：`TileInflight.kt`（并发去重：同 key 只渲染一次，其它 await）

```kotlin id="dk2q89"
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/cache/TileInflight.kt
package com.example.reader.pdf.internal.cache

import android.graphics.Bitmap
import java.util.concurrent.ConcurrentHashMap
import kotlinx.coroutines.CompletableDeferred
import kotlinx.coroutines.Deferred

internal class TileInflight {

    private val map: ConcurrentHashMap<String, Deferred<Bitmap>> = ConcurrentHashMap()

    suspend fun getOrAwait(key: String, block: suspend () -> Bitmap): Bitmap {
        val existing = map[key]
        if (existing != null) return existing.await()

        val deferred = CompletableDeferred<Bitmap>()
        val prev = map.putIfAbsent(key, deferred)
        if (prev != null) return prev.await()

        try {
            val value = block()
            deferred.complete(value)
            return value
        } catch (t: Throwable) {
            deferred.completeExceptionally(t)
            throw t
        } finally {
            map.remove(key, deferred)
        }
    }

    fun clear() = map.clear()
}
```

## ✅ 修改：`PdfTileProvider.kt`（接入 in-flight 去重 + 内存预估 + OOM 降级更稳）

> 只贴需要改的点：构造参数、新逻辑（其余保持你现有版本）。

```kotlin id="b8j5d0"
// File: engines/pdf/src/main/java/com/example/reader/pdf/internal/render/PdfTileProvider.kt
package com.example.reader.pdf.internal.render

import android.graphics.Bitmap
import android.graphics.Rect
import com.example.reader.api.render.RenderPolicy
import com.example.reader.api.render.TileProvider
import com.example.reader.api.render.TileRequest
import com.example.reader.pdf.internal.backend.PdfBackend
import com.example.reader.pdf.internal.backend.PdfPageSize
import com.example.reader.pdf.internal.backend.PdfRenderMode
import com.example.reader.pdf.internal.cache.TileCache
import com.example.reader.pdf.internal.cache.TileInflight
import java.io.Closeable
import java.util.concurrent.atomic.AtomicBoolean
import kotlin.math.max
import kotlinx.coroutines.CancellationException
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.withContext
import kotlin.coroutines.coroutineContext

internal class PdfTileProvider(
    private val backend: PdfBackend,
    private val pageIndex: Int,
    private val basePage: PdfPageSize,
    private val baseScale: Float,
    private val rotationDegrees: Int,
    private val allowCache: Boolean,
    private val cacheDraft: Boolean,            // ✅ 新增：是否缓存 DRAFT
    private val tileCache: TileCache,
    private val inflight: TileInflight,         // ✅ 新增：并发去重
    private val ioDispatcher: CoroutineDispatcher,
    private val isValid: () -> Boolean
) : TileProvider, Closeable {

    private val closed = AtomicBoolean(false)

    override suspend fun renderTile(request: TileRequest): Bitmap = withContext(ioDispatcher) {
        coroutineContext.ensureActive()

        if (closed.get() || !isValid()) throw CancellationException("stale tile provider")

        val scale = max(0.05f, baseScale * request.scale)

        val canCacheThis = allowCache && (request.quality == RenderPolicy.Quality.FINAL || cacheDraft)
        val key = if (canCacheThis) buildKey(request, scale) else ""

        if (canCacheThis) {
            tileCache.get(key)?.let { return@withContext it }
            // ✅ in-flight 去重：同一 key 只渲染一次
            return@withContext inflight.getOrAwait(key) {
                tileCache.get(key) ?: renderInternal(request, scale, canCacheThis, key)
            }
        }

        // 不缓存：直接渲染
        renderInternal(request, scale, canCacheThis = false, key = "")
    }

    private suspend fun renderInternal(
        request: TileRequest,
        scale: Float,
        canCacheThis: Boolean,
        key: String
    ): Bitmap {
        coroutineContext.ensureActive()

        val bmpConfig = when (request.quality) {
            RenderPolicy.Quality.DRAFT -> Bitmap.Config.RGB_565
            RenderPolicy.Quality.FINAL -> Bitmap.Config.ARGB_8888
        }

        // ✅ 内存预估：过大直接软降级，避免 OOM 把进程打死
        val w = request.widthPx.coerceAtLeast(1)
        val h = request.heightPx.coerceAtLeast(1)
        val bytesPerPixel = if (bmpConfig == Bitmap.Config.RGB_565) 2 else 4
        val estBytes = 1L * w * h * bytesPerPixel

        val max = Runtime.getRuntime().maxMemory().coerceAtLeast(128L * 1024 * 1024)
        val hardLimit = max / 10L // 单 tile 最多吃 10% heap（经验值，够稳）

        if (estBytes > hardLimit) {
            if (allowCache) tileCache.evictAll()
            return Bitmap.createBitmap(1, 1, Bitmap.Config.ARGB_8888)
        }

        try {
            val bmp = Bitmap.createBitmap(w, h, bmpConfig)
            val clip = Rect(0, 0, bmp.width, bmp.height)

            val matrix = buildPdfTileMatrix(
                basePage = basePage,
                scale = scale,
                rotationDegrees = rotationDegrees,
                leftPx = request.leftPx,
                topPx = request.topPx
            )

            val mode = when (request.quality) {
                RenderPolicy.Quality.DRAFT -> PdfRenderMode.DRAFT
                RenderPolicy.Quality.FINAL -> PdfRenderMode.FINAL
            }

            backend.render(pageIndex, bmp, clip, matrix, mode)

            if (canCacheThis && bmp.width > 1 && bmp.height > 1) {
                tileCache.put(key, bmp)
            }
            return bmp
        } catch (ce: CancellationException) {
            throw ce
        } catch (oom: OutOfMemoryError) {
            if (allowCache) tileCache.evictAll()
            return Bitmap.createBitmap(1, 1, Bitmap.Config.ARGB_8888)
        }
    }

    override fun close() {
        closed.set(true)
    }

    // buildKey(...) 保持不变
}
```

## 你会立刻看到的效果

* 同一个 tile 被 UI 多次请求（Compose 重组/快速滑动/多指缩放）时：**只渲染一次，其它 await 结果**
* 预取与前台渲染撞车：**不会重复渲染**
* 大 tile / 高 zoom：**提前拦截内存爆炸**，即使撞上 OOM 也能清缓存自救，不轻易崩进程
