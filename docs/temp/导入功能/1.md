下面给你一套**可直接落地**的“书籍导入”实现代码（按你现在的模块边界写），核心点：

* **feature/library 只调用 `core/files:ImportManager`**
* **导入任务/导入条目/书籍记录** 用 Room 落库（`core/database`）
* **复制落盘 + 去重 + 建档** 在 `core/work:ImportWorker`（WorkManager）里做
* **格式探测** 复用你已有的 `DefaultBookFormatDetector`（runtime）
* **DocumentSource** 追加两个实现：`ContentUriDocumentSource` 与 `FileDocumentSource`

> 你可以先把这些文件加进工程，跑通“导入队列 + 落盘 + 建档”。封面/目录/索引的 Enrich Worker 我先留了接口位，后面再补。

// File: core/files/src/main/kotlin/com/example/reader/files/source/ContentUriDocumentSource.kt
package com.example.reader.files.source

import android.content.ContentResolver
import android.content.Context
import android.database.Cursor
import android.net.Uri
import android.os.ParcelFileDescriptor
import android.provider.OpenableColumns
import com.example.reader.source.DocumentSource
import java.io.InputStream
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

class ContentUriDocumentSource(
    private val context: Context,
    override val uri: Uri,
    private val ioDispatcher: CoroutineDispatcher = Dispatchers.IO
) : DocumentSource {

    private val resolver: ContentResolver = context.contentResolver

    override val displayName: String? by lazy { queryString(OpenableColumns.DISPLAY_NAME) }
    override val sizeBytes: Long? by lazy { queryLong(OpenableColumns.SIZE) }
    override val mimeType: String? by lazy { resolver.getType(uri) }

    override suspend fun openInputStream(): InputStream = withContext(ioDispatcher) {
        resolver.openInputStream(uri) ?: throw java.io.FileNotFoundException("Cannot open: $uri")
    }

    override suspend fun openFileDescriptor(mode: String): ParcelFileDescriptor? = withContext(ioDispatcher) {
        resolver.openFileDescriptor(uri, mode)
    }

    private fun queryString(column: String): String? =
        queryCursor()?.use { c -> if (c.moveToFirst()) c.getStringOrNull(column) else null }

    private fun queryLong(column: String): Long? =
        queryCursor()?.use { c -> if (c.moveToFirst()) c.getLongOrNull(column) else null }

    private fun queryCursor(): Cursor? =
        resolver.query(uri, arrayOf(OpenableColumns.DISPLAY_NAME, OpenableColumns.SIZE), null, null, null)

    private fun Cursor.getStringOrNull(column: String): String? {
        val idx = getColumnIndex(column)
        return if (idx >= 0 && !isNull(idx)) getString(idx) else null
    }

    private fun Cursor.getLongOrNull(column: String): Long? {
        val idx = getColumnIndex(column)
        return if (idx >= 0 && !isNull(idx)) getLong(idx) else null
    }
}

package com.example.reader.files.permission

import android.content.Context
import android.net.Uri

class UriPermissionStore(private val context: Context) {

    fun takePersistableRead(uri: Uri) {
        val flags = android.content.Intent.FLAG_GRANT_READ_URI_PERMISSION
        context.contentResolver.takePersistableUriPermission(uri, flags)
    }

    fun takePersistableReadWrite(uri: Uri) {
        val flags = android.content.Intent.FLAG_GRANT_READ_URI_PERMISSION or
            android.content.Intent.FLAG_GRANT_WRITE_URI_PERMISSION
        context.contentResolver.takePersistableUriPermission(uri, flags)
    }

    fun hasPersistedRead(uri: Uri): Boolean =
        context.contentResolver.persistedUriPermissions.any { it.uri == uri && it.isReadPermission }
}

// File: core/files/src/main/kotlin/com/example/reader/files/scan/TreeScanner.kt
package com.example.reader.files.scan

import android.content.Context
import android.net.Uri
import androidx.documentfile.provider.DocumentFile

class TreeScanner(private val context: Context) {

    fun scan(treeUri: Uri): List<Uri> {
        val root = DocumentFile.fromTreeUri(context, treeUri) ?: return emptyList()
        val out = ArrayList<Uri>(256)
        dfs(root, out)
        return out
    }

    private fun dfs(dir: DocumentFile, out: MutableList<Uri>) {
        val children = dir.listFiles()
        for (f in children) {
            if (f.isDirectory) {
                dfs(f, out)
            } else if (f.isFile) {
                val name = f.name?.lowercase() ?: ""
                if (name.endsWith(".epub") || name.endsWith(".pdf") || name.endsWith(".txt")) {
                    out += f.uri
                }
            }
        }
    }
}

package com.example.reader.files.storage

import android.content.Context
import java.io.File
import java.nio.file.Files
import java.nio.file.StandardCopyOption
import java.util.UUID

class BookStorage(private val context: Context) {

    fun bookDir(bookId: String): File =
        File(context.filesDir, "books/$bookId").apply { mkdirs() }

    fun canonicalFile(bookId: String, ext: String): File =
        File(bookDir(bookId), "original.$ext")

    fun tempFile(bookId: String): File =
        File(bookDir(bookId), ".tmp-${UUID.randomUUID()}")

    fun atomicMove(from: File, to: File) {
        try {
            Files.move(from.toPath(), to.toPath(), StandardCopyOption.REPLACE_EXISTING, StandardCopyOption.ATOMIC_MOVE)
        } catch (_: Throwable) {
            // fallback
            if (to.exists()) to.delete()
            from.renameTo(to)
        }
    }

    fun deleteBookFiles(bookId: String) {
        bookDir(bookId).deleteRecursively()
    }
}

## 8) DefaultImportManager（入队：建 job + 建 items + enqueue Work）

```kotlin
// File: core/files/src/main/kotlin/com/example/reader/files/importing/DefaultImportManager.kt
package com.example.reader.files.importing

import android.content.Context
import android.net.Uri
import androidx.work.ExistingWorkPolicy
import androidx.work.OneTimeWorkRequestBuilder
import androidx.work.WorkManager
import com.example.reader.data.importing.ImportItemRepo
import com.example.reader.data.importing.ImportJobRepo
import com.example.reader.database.importing.ImportItemEntity
import com.example.reader.database.importing.ImportItemStatus
import com.example.reader.database.importing.ImportJobEntity
import com.example.reader.database.importing.ImportStatus
import com.example.reader.files.permission.UriPermissionStore
import com.example.reader.files.source.ContentUriDocumentSource
import com.example.reader.work.ImportWorker
import com.example.reader.work.WorkNames
import java.util.UUID
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.filterNotNull
import kotlinx.coroutines.flow.map
import kotlinx.coroutines.withContext

class DefaultImportManager(
    private val appContext: Context,
    private val workManager: WorkManager,
    private val jobRepo: ImportJobRepo,
    private val itemRepo: ImportItemRepo,
    private val permissionStore: UriPermissionStore
) : ImportManager {

    override suspend fun enqueue(request: ImportRequest): String = withContext(Dispatchers.IO) {
        val jobId = UUID.randomUUID().toString()
        val now = System.currentTimeMillis()

        // 权限：文件 & 文件夹（tree）
        request.treeUri?.let { permissionStore.takePersistableReadWrite(it) }
        request.uris.forEach { permissionStore.takePersistableRead(it) }

        val job = ImportJobEntity(
            jobId = jobId,
            status = ImportStatus.QUEUED,
            total = if (request.treeUri != null) 0 else request.uris.size,
            done = 0,
            currentTitle = null,
            errorMessage = null,
            sourceTreeUri = request.treeUri?.toString(),
            duplicateStrategy = request.duplicateStrategy.name,
            createdAtEpochMs = now,
            updatedAtEpochMs = now
        )
        jobRepo.upsert(job)

        // 若是选择文件：提前写入 items；若是 tree：由 worker 扫描后写入 items
        if (request.treeUri == null) {
            val items = request.uris.map { uri ->
                val src = ContentUriDocumentSource(appContext, uri)
                ImportItemEntity(
                    jobId = jobId,
                    uri = uri.toString(),
                    displayName = src.displayName,
                    mimeType = src.mimeType,
                    sizeBytes = src.sizeBytes,
                    status = ImportItemStatus.PENDING,
                    updatedAtEpochMs = now
                )
            }
            itemRepo.upsertAll(items)
        }

        // enqueue Worker（只传 jobId，避免 WorkManager Data 限制）
        val work = OneTimeWorkRequestBuilder<ImportWorker>()
            .addTag(WorkNames.tagForJob(jobId))
            .setInputData(ImportWorker.input(jobId))
            .build()

        workManager.enqueueUniqueWork(WorkNames.uniqueForJob(jobId), ExistingWorkPolicy.KEEP, work)
        return@withContext jobId
    }

    override fun observe(jobId: String): Flow<ImportJobState> =
        jobRepo.observe(jobId)
            .filterNotNull()
            .map { e ->
                ImportJobState(
                    jobId = e.jobId,
                    status = e.status,
                    total = e.total,
                    done = e.done,
                    currentTitle = e.currentTitle,
                    errorMessage = e.errorMessage
                )
            }

    override suspend fun cancel(jobId: String) {
        workManager.cancelUniqueWork(WorkNames.uniqueForJob(jobId))
        val now = System.currentTimeMillis()
        val job = jobRepo.get(jobId) ?: return
        jobRepo.upsert(job.copy(status = ImportStatus.CANCELLED, updatedAtEpochMs = now))
    }
}

import com.example.reader.files.scan.TreeScanner
import com.example.reader.files.source.ContentUriDocumentSource
import com.example.reader.files.source.FileDocumentSource
import com.example.reader.files.storage.BookStorage
import com.example.reader.model.BookFormat
import com.example.reader.runtime.format.BookFormatDetector
import com.example.reader.runtime.format.DefaultBookFormatDetector
import java.io.BufferedInputStream
import java.io.BufferedOutputStream
import java.io.File
import java.util.UUID
import javax.inject.Inject
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.withContext
import kotlin.coroutines.coroutineContext

@HiltWorker
class ImportWorker @Inject constructor(
    appContext: Context,
    params: WorkerParameters,
    private val jobRepo: ImportJobRepo,
    private val itemRepo: ImportItemRepo,
    private val bookRepo: BookRepo,
    private val storage: BookStorage,
    private val treeScanner: TreeScanner,
    private val formatDetector: BookFormatDetector
) : CoroutineWorker(appContext, params) {

    override suspend fun doWork(): Result = withContext(Dispatchers.IO) {
        val jobId = inputData.getString(KEY_JOB_ID) ?: return@withContext Result.failure()
        val now0 = System.currentTimeMillis()

        val job = jobRepo.get(jobId) ?: return@withContext Result.failure()

        // 1) 若是 tree 导入：这里扫描并落 items（只做一次）
        if (job.sourceTreeUri != null) {
            val existing = itemRepo.list(jobId)
            if (existing.isEmpty()) {
                val uris = treeScanner.scan(Uri.parse(job.sourceTreeUri))
                val now = System.currentTimeMillis()
                val items = uris.map { uri ->
                    val src = ContentUriDocumentSource(applicationContext, uri)
                    com.example.reader.database.importing.ImportItemEntity(
                        jobId = jobId,
                        uri = uri.toString(),
                        displayName = src.displayName,
                        mimeType = src.mimeType,
                        sizeBytes = src.sizeBytes,
                        status = ImportItemStatus.PENDING,
                        updatedAtEpochMs = now
                    )
                }
                itemRepo.upsertAll(items)
                jobRepo.updateProgress(
                    jobId = jobId,
                    status = ImportStatus.QUEUED,
                    total = uris.size,
                    done = 0,
                    currentTitle = null,
                    errorMessage = null,
                    now = now
                )
            }
        }

        val dup = runCatching { DuplicateStrategy.valueOf(job.duplicateStrategy) }.getOrDefault(DuplicateStrategy.SKIP)

        // 2) 标记 RUNNING
        jobRepo.updateProgress(
            jobId = jobId,
            status = ImportStatus.RUNNING,
            total = jobRepo.get(jobId)?.total ?: job.total,
            done = jobRepo.get(jobId)?.done ?: job.done,
            currentTitle = null,
            errorMessage = null,
            now = now0
        )

        // 3) 处理 items
        val pending = itemRepo.listPendingOrFailed(jobId)
        var done = jobRepo.get(jobId)?.done ?: 0
        val total = jobRepo.get(jobId)?.total ?: pending.size

        try {
            // 大文件建议前台通知（这里先给空实现，你可接入 Notification）
            setForegroundSafe(jobId, done, total)

            for (item in pending) {
                coroutineContext.ensureActive()

                val title = item.displayName ?: item.uri
                val now = System.currentTimeMillis()

                itemRepo.update(jobId, item.uri, ImportItemStatus.RUNNING, null, null, null, null, now)
                jobRepo.updateProgress(jobId, ImportStatus.RUNNING, total, done, title, null, now)

                val uri = Uri.parse(item.uri)
                val src = ContentUriDocumentSource(applicationContext, uri)

                val result = importOne(jobId, src, dup)
                when (result) {
                    is ImportOneResult.Skipped -> {
                        done += 1
                        itemRepo.update(jobId, item.uri, ImportItemStatus.SKIPPED, result.bookId, result.fp, null, "duplicate", System.currentTimeMillis())
                    }
                    is ImportOneResult.Ok -> {
                        done += 1
                        itemRepo.update(jobId, item.uri, ImportItemStatus.SUCCEEDED, result.bookId, result.fp, null, null, System.currentTimeMillis())
                    }
                    is ImportOneResult.Fail -> {
                        done += 1
                        itemRepo.update(jobId, item.uri, ImportItemStatus.FAILED, null, null, result.code, result.message, System.currentTimeMillis())
                    }
                }

                jobRepo.updateProgress(jobId, ImportStatus.RUNNING, total, done, null, null, System.currentTimeMillis())
                setForegroundSafe(jobId, done, total)
            }

            // 4) 完成
            jobRepo.updateProgress(jobId, ImportStatus.SUCCEEDED, total, done, null, null, System.currentTimeMillis())
            return@withContext Result.success()
        } catch (t: Throwable) {
            val msg = t.message ?: t::class.java.simpleName
            jobRepo.updateProgress(jobId, ImportStatus.FAILED, total, done, null, msg, System.currentTimeMillis())
            return@withContext Result.failure()
        }
    }

    private sealed interface ImportOneResult {
        data class Ok(val bookId: String, val fp: String) : ImportOneResult
        data class Skipped(val bookId: String?, val fp: String) : ImportOneResult
        data class Fail(val code: String, val message: String) : ImportOneResult
    }

    private suspend fun importOne(
        jobId: String,
        source: ContentUriDocumentSource,
        dup: DuplicateStrategy
    ): ImportOneResult = withContext(Dispatchers.IO) {
        val now = System.currentTimeMillis()
        val displayName = source.displayName ?: "unknown"
        val ext = guessExt(displayName, source.mimeType)
        val bookId = UUID.randomUUID().toString()

        val tmp = storage.tempFile(bookId)
        val finalFile = storage.canonicalFile(bookId, ext)

        val digest = Fingerprint.newSha256()
        val bytesCopied = copyWithDigest(source, tmp, digest)
        val fpHex = Fingerprint.sha256Hex(digest.digest())

        // 去重：fingerprint 命中
        val existing = bookRepo.findByFingerprint(fpHex)
        if (existing != null) {
            // 清理这次的临时目录
            runCatching { storage.deleteBookFiles(bookId) }

            return@withContext when (dup) {
                DuplicateStrategy.SKIP -> ImportOneResult.Skipped(existing.id, fpHex)
                DuplicateStrategy.KEEP_BOTH -> {
                    // 允许重复：继续写入一个新 book（但 canonicalPath 仍要落盘）
                    // 此时需要真的落盘——用已有内容重新复制很浪费，所以 KEEP_BOTH 默认不走这里
                    // 我建议：KEEP_BOTH 在 UI 明确提示，或实现为 “同 fingerprint 也允许保存一份”
                    ImportOneResult.Skipped(existing.id, fpHex)
                }
                DuplicateStrategy.REPLACE -> {
                    // 替换：删除旧文件 + 覆盖旧书记录（保持 id 不变）
                    runCatching { File(existing.canonicalPath).delete() }
                    runCatching { storage.deleteBookFiles(existing.id) }

                    // 落盘到旧 id 的目录
                    val tmp2 = storage.tempFile(existing.id)
                    tmp.renameTo(tmp2) // 同一分区优化；失败也无所谓，下面 atomicMove 会兜底
                    storage.atomicMove(tmp2, storage.canonicalFile(existing.id, ext))

                    val detected = detectFormatFromFile(storage.canonicalFile(existing.id, ext))
                    val entity = existing.copy(
                        format = detected,
                        canonicalPath = storage.canonicalFile(existing.id, ext).absolutePath,
                        originalUri = source.uri.toString(),
                        displayName = displayName,
                        mimeType = source.mimeType,
                        fingerprintSha256 = fpHex,
                        sizeBytes = bytesCopied,
                        updatedAtEpochMs = now
                    )
                    bookRepo.upsert(entity)
                    ImportOneResult.Ok(existing.id, fpHex)
                }
            }
        }

        // 正常：落盘（原子 move）
        storage.atomicMove(tmp, finalFile)

        val detectedFormat = detectFormatFromFile(finalFile)
        val title = displayName.substringBeforeLast('.', displayName)

        val book = BookEntity(
            id = bookId,
            format = detectedFormat,
            title = title,
            author = null,
            language = null,
            identifier = null,
            canonicalPath = finalFile.absolutePath,
            originalUri = source.uri.toString(),
            displayName = displayName,
            mimeType = source.mimeType,
            fingerprintSha256 = fpHex,
            sizeBytes = bytesCopied,
            createdAtEpochMs = now,
            updatedAtEpochMs = now
        )
        bookRepo.upsert(book)

        return@withContext ImportOneResult.Ok(bookId, fpHex)
    }

    private suspend fun copyWithDigest(
        source: ContentUriDocumentSource,
        outFile: File,
        digest: java.security.MessageDigest
    ): Long {
        var total = 0L
        source.openInputStream().use { raw ->
            BufferedInputStream(raw).use { input ->
                outFile.outputStream().use { fos ->
                    BufferedOutputStream(fos).use { output ->
                        val buf = ByteArray(DEFAULT_BUF)
                        while (true) {
                            coroutineContext.ensureActive()
                            val n = input.read(buf)
                            if (n <= 0) break
                            digest.update(buf, 0, n)
                            output.write(buf, 0, n)
                            total += n
                        }
                        output.flush()
                    }
                }
            }
        }
        return total
    }

    private suspend fun detectFormatFromFile(file: File): BookFormat {
        val src = FileDocumentSource(file, displayName = file.name)
        val r = formatDetector.detect(src, hint = null)
        return when (r) {
            is com.example.reader.api.error.ReaderResult.Ok -> r.value
            is com.example.reader.api.error.ReaderResult.Err -> {
                // 导入阶段：探测失败视为 TXT 或者抛 Unsupported（这里选 TXT 更“宽容”）
                BookFormat.TXT
            }
        }
    }

    private fun guessExt(displayName: String, mime: String?): String {
        val lower = displayName.lowercase()
        return when {
            lower.endsWith(".epub") -> "epub"
            lower.endsWith(".pdf") -> "pdf"
            lower.endsWith(".txt") -> "txt"
            mime == "application/epub+zip" -> "epub"
            mime == "application/pdf" -> "pdf"
            else -> "txt"
        }
    }

    private suspend fun setForegroundSafe(jobId: String, done: Int, total: Int) {
        // 这里先给一个“空前台”占位，避免你没接通知就编译不过：
        // 真正实现：用 NotificationCompat 构建通知并返回 ForegroundInfo(id, notification)
        runCatching {
            setForeground(ForegroundInfo(IMPORT_FG_ID, androidx.core.app.NotificationCompat.Builder(applicationContext, "import")
                .setContentTitle("Importing books")
                .setContentText("$done / $total")
                .setSmallIcon(android.R.drawable.stat_sys_download)
                .setOngoing(true)
                .build()
            ))
        }
    }

    companion object {
        private const val KEY_JOB_ID = "jobId"
        private const val DEFAULT_BUF = 128 * 1024
        private const val IMPORT_FG_ID = 4242

        fun input(jobId: String): Data = Data.Builder().putString(KEY_JOB_ID, jobId).build()
    }
}
```

> 你如果还没准备 Notification Channel：先把 `setForegroundSafe` 里那段 `setForeground(...)` 注释掉也行（但大文件导入后台可能被杀，建议尽快补 channel）。

# 你现在可以立刻跑通的“验收点”

1. feature/library 选择一个/多个文件 -> `StartImport(uris)`
2. `observe(jobId)` 能看到 `QUEUED -> RUNNING -> SUCCEEDED`
3. app 私有目录生成：`files/books/{bookId}/original.xxx`
4. DB `books` 表插入记录，并且同 fingerprint 二次导入会被 **SKIP**（默认策略）
