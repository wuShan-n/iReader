## ✅ 修改：`core/files/storage/BookStorage.kt`

> 新增 import 临时目录与 `importTempFile()`；并保证 `canonicalFile()` 一定会创建 bookDir。

```kotlin
// File: core/files/src/main/kotlin/com/example/reader/files/storage/BookStorage.kt

package com.example.reader.files.storage

import android.content.Context
import java.io.File
import java.nio.file.Files
import java.nio.file.StandardCopyOption
import java.util.UUID

class BookStorage(private val context: Context) {

    fun bookDir(bookId: String): File =
        File(context.filesDir, "books/$bookId").apply { mkdirs() }

    // ✅ 修改：保证目录存在
    fun canonicalFile(bookId: String, ext: String): File =
        File(bookDir(bookId), "original.$ext")

    // ✅ 新增：统一的导入临时目录（不绑定 bookId）
    private fun importTmpDir(): File =
        File(context.filesDir, "import_tmp").apply { mkdirs() }

    // ✅ 新增：导入临时文件
    fun importTempFile(): File =
        File(importTmpDir(), ".tmp-${UUID.randomUUID()}")

    fun atomicMove(from: File, to: File) {
        try {
            Files.move(from.toPath(), to.toPath(), StandardCopyOption.REPLACE_EXISTING, StandardCopyOption.ATOMIC_MOVE)
        } catch (_: Throwable) {
            if (to.exists()) to.delete()
            from.renameTo(to)
        }
    }

    fun deleteBookFiles(bookId: String) {
        bookDir(bookId).deleteRecursively()
    }

    // ✅ 新增：只删除 canonical（不动目录下的其他缓存文件）
    fun deleteCanonical(bookId: String) {
        val dir = bookDir(bookId)
        dir.listFiles()
            ?.filter { it.isFile && it.name.startsWith("original.") }
            ?.forEach { runCatching { it.delete() } }
    }
}
```

// File: core/work/src/main/kotlin/com/example/reader/work/notification/ImportForeground.kt
package com.example.reader.work.notification

import android.app.NotificationChannel
import android.app.NotificationManager
import android.content.Context
import android.os.Build
import androidx.core.app.NotificationCompat
import androidx.work.ForegroundInfo

object ImportForeground {

    const val CHANNEL_ID = "import"
    private const val CHANNEL_NAME = "Book Import"
    private const val NOTIFICATION_ID = 4242

    fun info(
        context: Context,
        done: Int,
        total: Int,
        currentTitle: String?
    ): ForegroundInfo {
        ensureChannel(context)

        val text = if (total > 0) "$done / $total" else "$done"
        val title = currentTitle?.takeIf { it.isNotBlank() } ?: "Importing books"

        val notification = NotificationCompat.Builder(context, CHANNEL_ID)
            .setSmallIcon(android.R.drawable.stat_sys_download)
            .setContentTitle(title)
            .setContentText(text)
            .setOnlyAlertOnce(true)
            .setOngoing(true)
            .setProgress(total.coerceAtLeast(0), done.coerceAtLeast(0), total <= 0)
            .build()

        return ForegroundInfo(NOTIFICATION_ID, notification)
    }

    private fun ensureChannel(context: Context) {
        if (Build.VERSION.SDK_INT < Build.VERSION_CODES.O) return
        val nm = context.getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager
        if (nm.getNotificationChannel(CHANNEL_ID) != null) return

        val channel = NotificationChannel(
            CHANNEL_ID,
            CHANNEL_NAME,
            NotificationManager.IMPORTANCE_LOW
        ).apply {
            description = "Shows progress while importing books"
            setShowBadge(false)
        }
        nm.createNotificationChannel(channel)
    }
}

import java.io.FileNotFoundException
import java.io.IOException
import java.util.zip.ZipException
import kotlinx.coroutines.CancellationException

internal fun Throwable.toImportError(): Pair<String, String> = when (this) {
    is ReaderError -> code to (message ?: code)
    is CancellationException -> "CANCELLED" to (message ?: "Cancelled")
    is FileNotFoundException -> "NOT_FOUND" to (message ?: "Not found")
    is SecurityException -> "PERMISSION_DENIED" to (message ?: "Permission denied")
    is ZipException -> "CORRUPT_OR_INVALID" to (message ?: "Corrupt or invalid")
    is IOException -> "IO" to (message ?: "I/O error")
    else -> "INTERNAL" to (message ?: this::class.java.simpleName)
}

    val msg = t.message ?: t::class.java.simpleName
    jobRepo.updateProgress(jobId, ImportStatus.FAILED, total, done, null, msg, System.currentTimeMillis())
    return@withContext Result.failure()
}

### 3) 修改：`importOne()`（关键优化版）

> ✅ 改动点：
>
> * 一律先拷到 `storage.importTempFile()`
> * 算出 fp 后再决定 targetId（新书 / 替换旧书 / 保留两份）
> * SKIP：删 temp 返回 Skipped
> * KEEP_BOTH：继续落盘并插入新 BookEntity（fp 可重复）
> * REPLACE：用旧 bookId，先 `deleteCanonical(oldId)` 再覆盖

```kotlin
// ImportWorker 内：替换整个 importOne(...) 实现

private sealed interface ImportOneResult {
    data class Ok(val bookId: String, val fp: String) : ImportOneResult
    data class Skipped(val bookId: String?, val fp: String) : ImportOneResult
    data class Fail(val code: String, val message: String) : ImportOneResult
}

private suspend fun importOne(
    source: com.example.reader.files.source.ContentUriDocumentSource,
    dup: com.example.reader.files.importing.DuplicateStrategy
): ImportOneResult = withContext(Dispatchers.IO) {
    val now = System.currentTimeMillis()
    val displayName = source.displayName ?: "unknown"
    val ext = guessExt(displayName, source.mimeType)

    val temp = storage.importTempFile()
    val digest = com.example.reader.files.hash.Fingerprint.newSha256()

    try {
        val bytesCopied = copyWithDigest(source, temp, digest)
        val fpHex = com.example.reader.files.hash.Fingerprint.sha256Hex(digest.digest())

        val existing = bookRepo.findByFingerprint(fpHex)

        // ✅ 根据策略决定目标 bookId
        val targetId: String = when {
            existing == null -> java.util.UUID.randomUUID().toString()
            dup == com.example.reader.files.importing.DuplicateStrategy.SKIP ->
                return@withContext ImportOneResult.Skipped(existing.id, fpHex).also { runCatching { temp.delete() } }
            dup == com.example.reader.files.importing.DuplicateStrategy.REPLACE ->
                existing.id
            else -> // KEEP_BOTH
                java.util.UUID.randomUUID().toString()
        }

        // ✅ REPLACE：只删 canonical，不删整目录（保留其它缓存/未来可复用）
        if (existing != null && dup == com.example.reader.files.importing.DuplicateStrategy.REPLACE) {
            storage.deleteCanonical(targetId)
        }

        val finalFile = storage.canonicalFile(targetId, ext)
        storage.atomicMove(temp, finalFile)

        val detectedFormat = detectFormatFromFile(finalFile)

        val baseTitle = displayName.substringBeforeLast('.', displayName)
        val title = when {
            existing != null && dup == com.example.reader.files.importing.DuplicateStrategy.REPLACE ->
                existing.title ?: baseTitle
            else -> baseTitle
        }

        val entity = if (existing != null && dup == com.example.reader.files.importing.DuplicateStrategy.REPLACE) {
            existing.copy(
                format = detectedFormat,
                title = title,
                canonicalPath = finalFile.absolutePath,
                originalUri = source.uri.toString(),
                displayName = displayName,
                mimeType = source.mimeType,
                fingerprintSha256 = fpHex,
                sizeBytes = bytesCopied,
                updatedAtEpochMs = now,
                coverPath = null // ✅ 替换后封面可能失效，后续 enrich 重新生成
            )
        } else {
            com.example.reader.database.book.BookEntity(
                id = targetId,
                format = detectedFormat,
                title = title,
                author = null,
                language = null,
                identifier = null,
                canonicalPath = finalFile.absolutePath,
                originalUri = source.uri.toString(),
                displayName = displayName,
                mimeType = source.mimeType,
                fingerprintSha256 = fpHex,
                sizeBytes = bytesCopied,
                createdAtEpochMs = now,
                updatedAtEpochMs = now
            )
        }

        bookRepo.upsert(entity)
        return@withContext ImportOneResult.Ok(targetId, fpHex)
    } catch (t: Throwable) {
        runCatching { temp.delete() }
        val (code, msg) = t.toImportError()
        return@withContext ImportOneResult.Fail(code, msg)
    }
}
```
