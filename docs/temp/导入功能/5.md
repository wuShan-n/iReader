## ✅ 新增：封面解码降采样 `BitmapDecode.kt`

```kotlin
// File: core/work/src/main/kotlin/com/example/reader/work/enrich/BitmapDecode.kt
package com.example.reader.work.enrich

import android.graphics.Bitmap
import android.graphics.BitmapFactory
import kotlin.math.max

object BitmapDecode {

    fun decodeSampled(bytes: ByteArray, reqW: Int, reqH: Int): Bitmap? {
        val bounds = BitmapFactory.Options().apply { inJustDecodeBounds = true }
        BitmapFactory.decodeByteArray(bytes, 0, bytes.size, bounds)
        if (bounds.outWidth <= 0 || bounds.outHeight <= 0) return null

        val sample = calcInSampleSize(bounds.outWidth, bounds.outHeight, reqW, reqH)
        val opts = BitmapFactory.Options().apply {
            inSampleSize = sample
            inPreferredConfig = Bitmap.Config.ARGB_8888
        }
        return BitmapFactory.decodeByteArray(bytes, 0, bytes.size, opts)
    }

    private fun calcInSampleSize(w: Int, h: Int, reqW: Int, reqH: Int): Int {
        var inSampleSize = 1
        if (h > reqH || w > reqW) {
            val halfH = h / 2
            val halfW = w / 2
            while ((halfH / inSampleSize) >= reqH && (halfW / inSampleSize) >= reqW) {
                inSampleSize *= 2
            }
        }
        return max(1, inSampleSize)
    }
}
```

## ✅ 新增：从 EPUB 资源直接提取封面 `CoverExtractor.kt`

> 依赖：引擎在 `DocumentMetadata.extra` 里放一个封面资源路径（任意一个 key 命中即可）。
>
> * 推荐引擎写入：`extra["coverHref"] = "OEBPS/Images/cover.jpg"`（或类似）

```kotlin
// File: core/work/src/main/kotlin/com/example/reader/work/enrich/CoverExtractor.kt
package com.example.reader.work.enrich

import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.ResourceProvider
import com.example.reader.model.DocumentMetadata
import java.io.ByteArrayOutputStream
import java.io.File

object CoverExtractor {

    // 兼容多种 key（引擎任选其一写入）
    private val keys = listOf(
        "coverHref",
        "coverPath",
        "cover",
        "cover_href",
        "cover_path"
    )

    suspend fun tryExtractAndSave(
        metadata: DocumentMetadata?,
        resources: ResourceProvider?,
        outFile: File,
        reqW: Int,
        reqH: Int
    ): Boolean {
        val rp = resources ?: return false
        val md = metadata ?: return false

        val path = keys.firstNotNullOfOrNull { k -> md.extra[k]?.takeIf { it.isNotBlank() } }
            ?: return false

        val streamRes = rp.openResource(path)
        if (streamRes !is ReaderResult.Ok) return false

        val bytes = streamRes.value.use { it.readAllBytesCompat(limitBytes = 10 * 1024 * 1024) } // 10MB 上限
            ?: return false

        val bmp = BitmapDecode.decodeSampled(bytes, reqW, reqH) ?: return false
        val scaled = if (bmp.width == reqW && bmp.height == reqH) bmp
        else android.graphics.Bitmap.createScaledBitmap(bmp, reqW, reqH, true)

        BitmapIO.savePng(outFile, scaled)
        return true
    }

    private fun java.io.InputStream.readAllBytesCompat(limitBytes: Int): ByteArray? {
        val out = ByteArrayOutputStream()
        val buf = ByteArray(32 * 1024)
        var total = 0
        while (true) {
            val n = read(buf)
            if (n <= 0) break
            total += n
            if (total > limitBytes) return null
            out.write(buf, 0, n)
        }
        return out.toByteArray()
    }
}
```

### 2) 替换你当前 “needCover” 分支为下面版本（关键优化）

> 行为：
>
> * **TXT**：直接占位图（不创建 session）
> * **EPUB**：优先 `CoverExtractor` 从资源提取；提取失败 -> 占位图（默认不 render）
> * **PDF**：走 render 生成首屏缩略图（保持原逻辑）

把你当前 cover 部分替换为：

```kotlin
// EnrichWorker.kt 内：在 doc.use { ... } 里，替换 needCover 块

var coverPath: String? = book.coverPath
val coverFile = storage.coverFile(book.id)
val titleForCover = (newTitle ?: book.title ?: book.displayName ?: "Untitled")

if (needCover) {
    when (book.format) {
        com.example.reader.model.BookFormat.TXT -> {
            // ✅ TXT：不渲染，直接占位
            val bmp = CoverRenderer.renderCoverBitmap(
                page = com.example.reader.api.render.RenderPage(
                    id = com.example.reader.api.render.PageId("placeholder"),
                    locator = Locator(LocatorSchemes.TXT_OFFSET, "0"),
                    content = com.example.reader.api.render.RenderContent.Text(text = titleForCover)
                ),
                desiredWidth = thumbW,
                desiredHeight = thumbH,
                titleFallback = titleForCover
            )
            BitmapIO.savePng(coverFile, bmp)
            coverPath = coverFile.absolutePath
        }

        com.example.reader.model.BookFormat.EPUB -> {
            // ✅ EPUB：优先资源提取；失败则占位（默认不 render）

            val config = com.example.reader.runtime.render.RenderDefaults.configFor(doc.capabilities)
            val sessionRes = doc.createSession(initialLocator = initialLocator, initialConfig = config)

            if (sessionRes is ReaderResult.Ok) {
                sessionRes.value.use { session ->
                    // 优先：从 metadata.extra 提供的 coverHref/coverPath 等提取
                    val extracted = CoverExtractor.tryExtractAndSave(
                        metadata = docMetaSnapshot,      // ✅ 看下面的“docMetaSnapshot”新增
                        resources = session.resources,
                        outFile = coverFile,
                        reqW = thumbW,
                        reqH = thumbH
                    )

                    if (extracted) {
                        coverPath = coverFile.absolutePath
                    } else if (renderCoverForEpub) {
                        // 可选：你若想对 EPUB 也 render（默认关）
                        session.controller.setLayoutConstraints(
                            LayoutConstraints(thumbW, thumbH, density, 1.0f)
                        )
                        val pageRes = session.controller.render(
                            RenderPolicy(RenderPolicy.Quality.FINAL, allowCache = false, prefetchNeighbors = 0)

                        if (pageRes is ReaderResult.Ok) {
                            val bmp = CoverRenderer.renderCoverBitmap(
                                page = pageRes.value,
                                desiredWidth = thumbW,
                                desiredHeight = thumbH,
                                titleFallback = titleForCover
                            )
                            BitmapIO.savePng(coverFile, bmp)
                            coverPath = coverFile.absolutePath
                        } else {
                            // fallback 占位
                            BitmapIO.savePng(coverFile, CoverRenderer.renderCoverBitmap(
                                page = com.example.reader.api.render.RenderPage(
                                    id = com.example.reader.api.render.PageId("placeholder"),
                                    locator = Locator(LocatorSchemes.EPUB_CFI, "epubcfi(/)"),
                                    content = com.example.reader.api.render.RenderContent.Text(text = titleForCover)
                                ),
                                desiredWidth = thumbW,
                                desiredHeight = thumbH,
                                titleFallback = titleForCover
                            ))
                            coverPath = coverFile.absolutePath
                        }
                    } else {
                        // 默认：不 render，直接占位
                        val bmp = CoverRenderer.renderCoverBitmap(
                            page = com.example.reader.api.render.RenderPage(
                                id = com.example.reader.api.render.PageId("placeholder"),
                                locator = Locator(LocatorSchemes.EPUB_CFI, "epubcfi(/)"),
                                content = com.example.reader.api.render.RenderContent.Text(text = titleForCover)
                            ),
                            desiredWidth = thumbW,
                            desiredHeight = thumbH,
                            titleFallback = titleForCover
                        )
                        BitmapIO.savePng(coverFile, bmp)
                        coverPath = coverFile.absolutePath
                    }
                }
            } else {
                // session 失败：占位
                val bmp = CoverRenderer.renderCoverBitmap(
                    page = com.example.reader.api.render.RenderPage(
                        id = com.example.reader.api.render.PageId("placeholder"),
                        locator = Locator(LocatorSchemes.EPUB_CFI, "epubcfi(/)"),
                        content = com.example.reader.api.render.RenderContent.Text(text = titleForCover)
                    ),
                    desiredWidth = thumbW,
                    desiredHeight = thumbH,
                    titleFallback = titleForCover
                )
                BitmapIO.savePng(coverFile, bmp)
                coverPath = coverFile.absolutePath
            }
        }

        com.example.reader.model.BookFormat.PDF -> {
            // ✅ PDF：render（保留你原先逻辑）

            val config = com.example.reader.runtime.render.RenderDefaults.configFor(doc.capabilities)
            val sessionRes = doc.createSession(initialLocator = initialLocator, initialConfig = config)

            if (sessionRes is ReaderResult.Ok) {
                sessionRes.value.use { session ->
                    session.controller.setLayoutConstraints(

                            page = pageRes.value,
                            desiredWidth = thumbW,
                            desiredHeight = thumbH,
                            titleFallback = titleForCover
                        )
                        BitmapIO.savePng(coverFile, bmp)
                        coverPath = coverFile.absolutePath
                    }
                }
            }
        }
    }
```
