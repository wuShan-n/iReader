### ✅ 新增 `CoverRenderer.kt`

```kotlin
// File: core/work/src/main/kotlin/com/example/reader/work/enrich/CoverRenderer.kt
package com.example.reader.work.enrich

import android.graphics.Bitmap
import android.graphics.Canvas
import android.graphics.Color
import android.graphics.Paint
import com.example.reader.api.render.RenderContent
import com.example.reader.api.render.RenderPage
import com.example.reader.api.render.TileRequest

object CoverRenderer {

    fun renderCoverBitmap(
        page: RenderPage,
        desiredWidth: Int,
        desiredHeight: Int,
        titleFallback: String
    ): Bitmap {
        return when (val c = page.content) {
            is RenderContent.BitmapPage -> scaleTo(c.bitmap, desiredWidth, desiredHeight)

            is RenderContent.Tiles -> {
                // 尽量渲全页一张 tile，再缩放
                val scale = minScale(
                    desiredWidth.toFloat() / c.pageWidthPx.toFloat(),
                    desiredHeight.toFloat() / c.pageHeightPx.toFloat()
                )

                val bmp = runCatching {
                    // full-page tile
                    c.tileProvider.renderTile(
                        TileRequest(
                            leftPx = 0,
                            topPx = 0,
                            widthPx = c.pageWidthPx,
                            heightPx = c.pageHeightPx,
                            scale = scale
                        )
                    )
                }.getOrNull()

                if (bmp != null) scaleTo(bmp, desiredWidth, desiredHeight)
                else placeholder(desiredWidth, desiredHeight, titleFallback)
            }

            is RenderContent.Text,
            is RenderContent.Html -> placeholder(desiredWidth, desiredHeight, titleFallback)
        }
    }

    private fun minScale(a: Float, b: Float): Float = if (a < b) a else b

    private fun scaleTo(src: Bitmap, w: Int, h: Int): Bitmap {
        if (src.width == w && src.height == h) return src
        return Bitmap.createScaledBitmap(src, w, h, true)
    }

    private fun placeholder(w: Int, h: Int, title: String): Bitmap {
        val bmp = Bitmap.createBitmap(w, h, Bitmap.Config.ARGB_8888)
        val canvas = Canvas(bmp)
        canvas.drawColor(Color.WHITE)

        val paint = Paint(Paint.ANTI_ALIAS_FLAG).apply {
            color = Color.BLACK
            textSize = (w * 0.06f).coerceAtLeast(24f)
        }

        val pad = (w * 0.08f)
        val lines = breakLines(title.ifBlank { "Untitled" }, maxCharsPerLine = 18, maxLines = 6)

        var y = h * 0.25f
        for (line in lines) {
            canvas.drawText(line, pad, y, paint)
            y += paint.textSize * 1.25f
        }
        return bmp
    }

    private fun breakLines(text: String, maxCharsPerLine: Int, maxLines: Int): List<String> {
        val out = ArrayList<String>(maxLines)
        var i = 0
        while (i < text.length && out.size < maxLines) {
            val end = (i + maxCharsPerLine).coerceAtMost(text.length)
            out += text.substring(i, end)
            i = end
        }
        return out
    }
}
```

## 7) 新增：`EnrichWorker.kt`（核心）

> 逻辑：
>
> * 从 `ImportItemRepo.listSucceededBookIds(jobId)` 拿本次导入成功的书
> * 对每本书：若缺 metadata 或 cover 才处理
> * metadata：`runtime.openDocument(fileSource).metadata()`
> * cover：`runtime.openSession(...)-> setLayoutConstraints -> render -> 生成 cover.png`
> * 全程 best-effort：单本失败不会让整个 worker 失败（除非被取消）

```kotlin
// File: core/work/src/main/kotlin/com/example/reader/work/enrich/EnrichWorker.kt
package com.example.reader.work.enrich

import android.content.Context
import androidx.hilt.work.HiltWorker
import androidx.work.CoroutineWorker
import androidx.work.WorkerParameters
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.open.OpenOptions
import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.RenderPolicy
import com.example.reader.data.book.BookRepo
import com.example.reader.data.importing.ImportItemRepo
import com.example.reader.files.source.FileDocumentSource
import com.example.reader.files.storage.BookStorage
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.runtime.ReaderRuntime
import com.example.reader.runtime.render.RenderDefaults
import com.example.reader.work.notification.ImportForeground
import java.io.File
import javax.inject.Inject
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.ensureActive
import kotlinx.coroutines.withContext
import kotlin.coroutines.coroutineContext
import kotlin.math.min

@HiltWorker
class EnrichWorker @Inject constructor(
    appContext: Context,
    params: WorkerParameters,
    private val importItemRepo: ImportItemRepo,
    private val bookRepo: BookRepo,
    private val storage: BookStorage,
    private val runtime: ReaderRuntime
) : CoroutineWorker(appContext, params) {

    override suspend fun doWork(): Result = withContext(Dispatchers.IO) {
        val jobId = EnrichWorkerInput.jobId(inputData) ?: return@withContext Result.failure()
        val bookIds = importItemRepo.listSucceededBookIds(jobId)

        if (bookIds.isEmpty()) return@withContext Result.success()

        val dm = applicationContext.resources.displayMetrics
        val density = dm.density
        val thumbW = min(720, dm.widthPixels.coerceAtLeast(480))
        val thumbH = (thumbW * 4 / 3).coerceAtLeast(720)

        var done = 0
        val total = bookIds.size

        try {
            setForeground(ImportForeground.info(applicationContext, done, total, "Enriching…"))

            for (id in bookIds) {
                coroutineContext.ensureActive()

                val book = bookRepo.getById(id) ?: run {
                    done++
                    setForeground(ImportForeground.info(applicationContext, done, total, "Enriching…"))
                    continue
                }

                val needMeta = book.title.isNullOrBlank() || book.author.isNullOrBlank() ||
                    book.language.isNullOrBlank() || book.identifier.isNullOrBlank()

                val needCover = book.coverPath.isNullOrBlank() || !File(book.coverPath).exists()

                if (!needMeta && !needCover) {
                    done++
                    setForeground(ImportForeground.info(applicationContext, done, total, "Enriching…"))
                    continue
                }

                val file = File(book.canonicalPath)
                if (!file.exists()) {
                    done++
                    setForeground(ImportForeground.info(applicationContext, done, total, "Enriching…"))
                    continue
                }

                val source = FileDocumentSource(file, displayName = book.displayName ?: file.name)

                // ---- metadata ----
                var newTitle: String? = book.title
                var newAuthor: String? = book.author
                var newLang: String? = book.language
                var newIdentifier: String? = book.identifier

                if (needMeta) {
                    val docRes = runtime.openDocument(
                        source = source,
                        options = OpenOptions(hintFormat = book.format)
                    )

                    if (docRes is ReaderResult.Ok) {
                        docRes.value.use { doc ->
                            when (val m = doc.metadata()) {
                                is ReaderResult.Ok -> {
                                    val md = m.value
                                    if (newTitle.isNullOrBlank()) newTitle = md.title
                                    if (newAuthor.isNullOrBlank()) newAuthor = md.author
                                    if (newLang.isNullOrBlank()) newLang = md.language
                                    if (newIdentifier.isNullOrBlank()) newIdentifier = md.identifier
                                }
                                is ReaderResult.Err -> {
                                    // ignore: best-effort
                                }
                            }
                        }
                    }
                }

                // ---- cover ----
                var coverPath: String? = book.coverPath
                if (needCover) {
                    val initialLocator: Locator? =
                        if (book.format.name == "PDF") Locator(LocatorSchemes.PDF_PAGE, "0") else null

                    val sessRes = runtime.openSession(
                        source = source,
                        options = OpenOptions(hintFormat = book.format),
                        initialLocator = initialLocator,
                        initialConfig = null // runtime 默认（PDF/FIX vs Reflow）
                    )

                    if (sessRes is ReaderResult.Ok) {
                        sessRes.value.use { handle ->
                            val constraints = LayoutConstraints(
                                viewportWidthPx = thumbW,
                                viewportHeightPx = thumbH,
                                density = density,
                                fontScale = 1.0f
                            )

                            // 有些引擎需要 constraints 才能 render
                            handle.controller.setLayoutConstraints(constraints)

                            val policy = RenderPolicy(
                                quality = RenderPolicy.Quality.FINAL,
                                allowCache = false,
                                prefetchNeighbors = 0
                            )

                            val pageRes = handle.controller.render(policy)
                            if (pageRes is ReaderResult.Ok) {
                                val coverFile = storage.coverFile(book.id)
                                val titleForCover = (newTitle ?: book.title ?: book.displayName ?: "Untitled")
                                val bmp = CoverRenderer.renderCoverBitmap(
                                    page = pageRes.value,
                                    desiredWidth = thumbW,
                                    desiredHeight = thumbH,
                                    titleFallback = titleForCover
                                )
                                BitmapIO.savePng(coverFile, bmp)
                                coverPath = coverFile.absolutePath
                            }
                        }
                    }
                }

                // ---- persist ----
                val now = System.currentTimeMillis()
                val updated = book.copy(
                    title = newTitle,
                    author = newAuthor,
                    language = newLang,
                    identifier = newIdentifier,
                    coverPath = coverPath,
                    updatedAtEpochMs = now
                )
                bookRepo.upsert(updated)

                done++
                setForeground(ImportForeground.info(applicationContext, done, total, updated.title ?: "Enriching…"))
            }

            return@withContext Result.success()
        } catch (ce: kotlinx.coroutines.CancellationException) {
            throw ce
        } catch (_: Throwable) {
            // enrich 失败通常不该阻塞主流程；这里返回 success，避免无限重试
            return@withContext Result.success()
        }
    }
}
```
