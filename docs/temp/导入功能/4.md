下面只给**阶段 B（Enrich）进一步优化**需要的**修改/新增代码**（不重复你已有的代码与我上次已给过且未变的部分）。

本轮优化点：

* **同一本书只 open 一次**：`openDocument → metadata → createSession → render cover`（避免 `openSession` 再开一次文档）
* **前台通知/进度更新节流**：避免每本都 `setForeground` 造成抖动与开销
* **Enrich 入队加约束**：默认在“电量不低”下跑，减少后台被杀/体验更稳
* **单本书失败不影响整体**：仍是 best-effort，但更细粒度

## 1) ✅ 新增：进度节流器 `ProgressThrottle.kt`

```kotlin
// File: core/work/src/main/kotlin/com/example/reader/work/enrich/ProgressThrottle.kt
package com.example.reader.work.enrich

class ProgressThrottle(
    private val minIntervalMs: Long = 500L
) {
    private var lastUpdateMs: Long = 0L

    fun shouldUpdate(nowMs: Long = System.currentTimeMillis()): Boolean {
        if (nowMs - lastUpdateMs < minIntervalMs) return false
        lastUpdateMs = nowMs
        return true
    }

    fun force(nowMs: Long = System.currentTimeMillis()) {
        lastUpdateMs = nowMs
    }
}
```

## 2) ✅ 修改：`EnrichWorker.kt`（只贴需要替换/新增的部分）

> 关键变化：**不再调用 `runtime.openSession()`**，改为：
>
> * `runtime.openDocument()` 打开一次
> * `doc.metadata()` 取元数据
> * `doc.createSession(initialLocator, config)` 渲染封面
> * 最后统一 close（doc.use + handle/use）

### 2.1 替换 `doWork()` 中对每本书的处理核心逻辑（建议直接替换你 for-loop 内的主体）

```kotlin
// File: core/work/src/main/kotlin/com/example/reader/work/enrich/EnrichWorker.kt
// 位置：for (id in bookIds) { ... } 内部主体替换为下面逻辑

val throttle = ProgressThrottle(minIntervalMs = 500L)
// ... 在 for 循环之前初始化一次即可

for (id in bookIds) {
    coroutineContext.ensureActive()

    val book = bookRepo.getById(id) ?: run {
        done++
        if (throttle.shouldUpdate()) {
            setForeground(ImportForeground.info(applicationContext, done, total, "Enriching…"))
        }
        continue
    }

    val needMeta = book.title.isNullOrBlank() || book.author.isNullOrBlank() ||
        book.language.isNullOrBlank() || book.identifier.isNullOrBlank()

    val needCover = book.coverPath.isNullOrBlank() || !File(book.coverPath).exists()

    if (!needMeta && !needCover) {
        done++
        if (throttle.shouldUpdate()) {
            setForeground(ImportForeground.info(applicationContext, done, total, book.title ?: "Enriching…"))
        }
        continue
    }

    val file = File(book.canonicalPath)
    if (!file.exists()) {
        done++
        if (throttle.shouldUpdate()) {
            setForeground(ImportForeground.info(applicationContext, done, total, book.title ?: "Enriching…"))
        }
        continue
    }

    val source = FileDocumentSource(file, displayName = book.displayName ?: file.name)

    // ---- 单本书 best-effort：失败不影响其它 ----
    runCatching {
        val docRes = runtime.openDocument(
            source = source,
            options = OpenOptions(hintFormat = book.format)
        )

        if (docRes !is ReaderResult.Ok) return@runCatching

        docRes.value.use { doc ->
            // ---- metadata ----
            var newTitle: String? = book.title
            var newAuthor: String? = book.author
            var newLang: String? = book.language
            var newIdentifier: String? = book.identifier

            if (needMeta) {
                when (val m = doc.metadata()) {
                    is ReaderResult.Ok -> {
                        val md = m.value
                        if (newTitle.isNullOrBlank()) newTitle = md.title
                        if (newAuthor.isNullOrBlank()) newAuthor = md.author
                        if (newLang.isNullOrBlank()) newLang = md.language
                        if (newIdentifier.isNullOrBlank()) newIdentifier = md.identifier
                    }
                    is ReaderResult.Err -> Unit // ignore
                }
            }

            // ---- cover ----
            var coverPath: String? = book.coverPath
            if (needCover) {
                val initialLocator: Locator? =
                    if (book.format.name == "PDF") Locator(LocatorSchemes.PDF_PAGE, "0") else null

                val config = RenderDefaults.configFor(doc.capabilities) // ✅ 不再走 runtime.openSession 的默认
                val sessionRes = doc.createSession(initialLocator = initialLocator, initialConfig = config)

                if (sessionRes is ReaderResult.Ok) {
                    sessionRes.value.use { session ->
                        val constraints = LayoutConstraints(
                            viewportWidthPx = thumbW,
                            viewportHeightPx = thumbH,
                            density = density,
                            fontScale = 1.0f
                        )
                        session.controller.setLayoutConstraints(constraints)

                        val policy = RenderPolicy(
                            quality = RenderPolicy.Quality.FINAL,
                            allowCache = false,
                            prefetchNeighbors = 0
                        )

                        val pageRes = session.controller.render(policy)
                        if (pageRes is ReaderResult.Ok) {
                            val coverFile = storage.coverFile(book.id)
                            val titleForCover = (newTitle ?: book.title ?: book.displayName ?: "Untitled")
                            val bmp = CoverRenderer.renderCoverBitmap(
                                page = pageRes.value,
                                desiredWidth = thumbW,
                                desiredHeight = thumbH,
                                titleFallback = titleForCover
                            )
                            BitmapIO.savePng(coverFile, bmp)
                            coverPath = coverFile.absolutePath
                        }
                    }
                }
            }

            // ---- persist（只有发生变化才写库，减少写入） ----
            val now = System.currentTimeMillis()
            val changed =
                newTitle != book.title ||
                newAuthor != book.author ||
                newLang != book.language ||
                newIdentifier != book.identifier ||
                coverPath != book.coverPath

            if (changed) {
                val updated = book.copy(
                    title = newTitle,
                    author = newAuthor,
                    language = newLang,
                    identifier = newIdentifier,
                    coverPath = coverPath,
                    updatedAtEpochMs = now
                )
                bookRepo.upsert(updated)
            }
        }
    }

    done++
    if (throttle.shouldUpdate()) {
        setForeground(ImportForeground.info(applicationContext, done, total, book.title ?: "Enriching…"))
    }
}

// 循环结束强制更新一次
throttle.force()
setForeground(ImportForeground.info(applicationContext, done, total, "Enrich complete"))
```

> 你会注意到：现在只会 `openDocument()` 一次，**metadata 与 cover 都从同一个 document 衍生**，整体开销明显下降。

## 3) ✅ 修改：Import 完成后 enqueue Enrich 时加约束（更稳）

> 只改你在 `ImportWorker` 成功后 enqueue enrich 的那段：增加 `Constraints`（电量不低；如果你想更保守可以加“充电中/空闲”）

```kotlin
// File: core/work/src/main/kotlin/com/example/reader/work/ImportWorker.kt
// 位置：你 enqueue EnrichWorker 的代码块里，新增 constraints 并 setConstraints()

val constraints = androidx.work.Constraints.Builder()
    .setRequiresBatteryNotLow(true)
    .build()

val enrich = androidx.work.OneTimeWorkRequestBuilder<com.example.reader.work.enrich.EnrichWorker>()
    .addTag(com.example.reader.work.WorkNames.tagEnrichForJob(jobId))
    .setInputData(com.example.reader.work.enrich.EnrichWorkerInput.data(jobId))
    .setConstraints(constraints)
    .build()
```
