## ✅ 新增：EPUB Zip/OPF 解析与封面提取（不依赖引擎）

### `core/work/.../enrich/epub/EpubZipEnricher.kt`

```kotlin
package com.example.reader.work.enrich.epub

import android.util.Xml
import com.example.reader.model.DocumentMetadata
import com.example.reader.work.enrich.BitmapDecode
import com.example.reader.work.enrich.BitmapIO
import java.io.File
import java.io.InputStream
import java.util.zip.ZipFile

object EpubZipEnricher {

    data class Parsed(
        val metadata: DocumentMetadata,
        val coverPathInZip: String?
    )

    fun parse(file: File): Parsed? {
        ZipFile(file).use { zip ->
            val opfPath = findOpfPath(zip) ?: return null
            val opfEntry = zip.getEntry(opfPath) ?: return null
            zip.getInputStream(opfEntry).use { opf ->
                return parseOpf(opfPath, opf)
            }
        }
    }

    fun tryExtractCoverToPng(
        file: File,
        coverPathInZip: String,
        outFile: File,
        reqW: Int,
        reqH: Int
    ): Boolean {
        ZipFile(file).use { zip ->
            val entry = zip.getEntry(coverPathInZip) ?: return false
            val bytes = zip.getInputStream(entry).use { it.readAllBytesCapped(12 * 1024 * 1024) } ?: return false
            val bmp = BitmapDecode.decodeSampled(bytes, reqW, reqH) ?: return false
            val scaled = if (bmp.width == reqW && bmp.height == reqH) bmp
            else android.graphics.Bitmap.createScaledBitmap(bmp, reqW, reqH, true)
            BitmapIO.savePng(outFile, scaled)
            return true
        }
    }

    // ---------- internals ----------

    private fun findOpfPath(zip: ZipFile): String? {
        val container = zip.getEntry("META-INF/container.xml") ?: return null
        zip.getInputStream(container).use { input ->
            val parser = Xml.newPullParser()
            parser.setInput(input, null)

            var event = parser.eventType
            while (event != org.xmlpull.v1.XmlPullParser.END_DOCUMENT) {
                if (event == org.xmlpull.v1.XmlPullParser.START_TAG && parser.name == "rootfile") {
                    val fullPath = parser.getAttributeValue(null, "full-path")
                        ?: parser.getAttributeValue("", "full-path")
                    if (!fullPath.isNullOrBlank()) return normalizeZipPath(fullPath)
                }
                event = parser.next()
            }
        }
        return null
    }

    private fun parseOpf(opfPath: String, opfStream: InputStream): Parsed? {
        val baseDir = opfPath.substringBeforeLast('/', "")
        val parser = Xml.newPullParser()
        parser.setInput(opfStream, null)

        var title: String? = null
        var creator: String? = null
        var language: String? = null
        var identifier: String? = null

        var coverIdFromMeta: String? = null
        var coverHrefFromProperties: String? = null
        val manifestIdToHref = HashMap<String, String>(64)

        var event = parser.eventType
        while (event != org.xmlpull.v1.XmlPullParser.END_DOCUMENT) {
            if (event == org.xmlpull.v1.XmlPullParser.START_TAG) {
                val name = parser.name
                when {
                    name.endsWith("title", ignoreCase = true) && title == null && isDcTag(parser) ->
                        title = parser.nextTextOrNull()

                    name.endsWith("creator", ignoreCase = true) && creator == null && isDcTag(parser) ->
                        creator = parser.nextTextOrNull()

                    name.endsWith("language", ignoreCase = true) && language == null && isDcTag(parser) ->
                        language = parser.nextTextOrNull()

                    name.endsWith("identifier", ignoreCase = true) && identifier == null && isDcTag(parser) ->
                        identifier = parser.nextTextOrNull()

                    name.equals("meta", ignoreCase = true) -> {
                        val metaName = parser.getAttributeValue(null, "name")
                        val metaContent = parser.getAttributeValue(null, "content")
                        if (metaName.equals("cover", ignoreCase = true) && !metaContent.isNullOrBlank()) {
                            coverIdFromMeta = metaContent
                        }
                    }

                    name.equals("item", ignoreCase = true) -> {
                        val id = parser.getAttributeValue(null, "id") ?: ""
                        val href = parser.getAttributeValue(null, "href")
                        val props = parser.getAttributeValue(null, "properties") ?: ""

                        if (!id.isBlank() && !href.isNullOrBlank()) {
                            manifestIdToHref[id] = href
                        }
                        // OPF3: properties="cover-image"
                        if (coverHrefFromProperties == null && props.split(' ').any { it.equals("cover-image", true) }) {
                            if (!href.isNullOrBlank()) coverHrefFromProperties = href
                        }
                    }
                }
            }
            event = parser.next()
        }

        val coverHref = when {
            coverHrefFromProperties != null -> coverHrefFromProperties
            coverIdFromMeta != null -> manifestIdToHref[coverIdFromMeta!!]
            else -> null
        }

        val coverPath = coverHref?.let { normalizeZipPath(resolveZipPath(baseDir, it)) }

        val md = DocumentMetadata(
            title = title?.trim().takeUnless { it.isNullOrBlank() },
            author = creator?.trim().takeUnless { it.isNullOrBlank() },
            language = language?.trim().takeUnless { it.isNullOrBlank() },
            identifier = identifier?.trim().takeUnless { it.isNullOrBlank() },
            extra = buildMap {
                if (!coverPath.isNullOrBlank()) put("coverPath", coverPath!!)
            }
        )
        return Parsed(md, coverPath)
    }

    private fun isDcTag(parser: org.xmlpull.v1.XmlPullParser): Boolean {
        val prefix = parser.prefix ?: ""
        val ns = parser.namespace ?: ""
        return prefix.equals("dc", true) || ns.contains("purl.org/dc", true)
    }

    private fun org.xmlpull.v1.XmlPullParser.nextTextOrNull(): String? =
        runCatching { nextText() }.getOrNull()

    private fun resolveZipPath(baseDir: String, href: String): String {
        if (baseDir.isBlank()) return href
        return "$baseDir/$href"
    }

    private fun normalizeZipPath(path: String): String {
        val parts = path.replace('\\', '/').split('/').filter { it.isNotEmpty() }
        val stack = ArrayDeque<String>(parts.size)
        for (p in parts) {
            when (p) {
                "." -> Unit
                ".." -> if (stack.isNotEmpty()) stack.removeLast()
                else -> stack.addLast(p)
            }
        }
        return stack.joinToString("/")
    }

    private fun InputStream.readAllBytesCapped(limitBytes: Int): ByteArray? {
        val out = java.io.ByteArrayOutputStream()
        val buf = ByteArray(32 * 1024)
        var total = 0
        while (true) {
            val n = read(buf)
            if (n <= 0) break
            total += n
            if (total > limitBytes) return null
            out.write(buf, 0, n)
        }
        return out.toByteArray()
    }
}
```

### 3) **替换**你当前 per-book 处理逻辑中对 EPUB 的分支（核心）

把“EPUB 的 metadata/cover 仍通过 openDocument/createSession”那段替换为下面逻辑：

```kotlin
// EnrichWorker.kt：在处理每本书时，分支到 EPUB 时用下面逻辑（替换原 EPUB 路径）

if (book.format == com.example.reader.model.BookFormat.EPUB) {
    val parsed = EpubZipEnricher.parse(file)

    var newTitle: String? = book.title
    var newAuthor: String? = book.author
    var newLang: String? = book.language
    var newIdentifier: String? = book.identifier

    if (needMeta && parsed != null) {
        val md = parsed.metadata
        if (newTitle.isNullOrBlank()) newTitle = md.title
        if (newAuthor.isNullOrBlank()) newAuthor = md.author
        if (newLang.isNullOrBlank()) newLang = md.language
        if (newIdentifier.isNullOrBlank()) newIdentifier = md.identifier
    }

    var coverPath: String? = book.coverPath
    if (needCover) {
        val coverFile = storage.coverFile(book.id)
        val titleForCover = (newTitle ?: book.title ?: book.displayName ?: "Untitled")

        val extracted = parsed?.coverPathInZip?.let { pathInZip ->
            runCatching {
                EpubZipEnricher.tryExtractCoverToPng(
                    file = file,
                    coverPathInZip = pathInZip,
                    outFile = coverFile,
                    reqW = thumbW,
                    reqH = thumbH
                )
            }.getOrDefault(false)
        } ?: false

        if (!extracted) {
            val bmp = CoverRenderer.placeholderBitmap(thumbW, thumbH, titleForCover)
            BitmapIO.savePng(coverFile, bmp)
        }
        coverPath = coverFile.absolutePath
    }

    val now = System.currentTimeMillis()
    val changed =
        newTitle != book.title ||
        newAuthor != book.author ||
        newLang != book.language ||
        newIdentifier != book.identifier ||
        coverPath != book.coverPath

    if (changed) {
        bookRepo.upsert(
            book.copy(
                title = newTitle,
                author = newAuthor,
                language = newLang,
                identifier = newIdentifier,
                coverPath = coverPath,
                updatedAtEpochMs = now
            )
        )
    }
}
```
