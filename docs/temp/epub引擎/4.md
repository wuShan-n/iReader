> 核心思路（不改你的 API）：
>
> * `EpubController` 增加 **spineIndex + pageIndex** 状态，`next/prev` 先章内翻页，越界再跨章。
> * `RenderPage` 使用 `LocatorSchemes.REFLOW_PAGE`（`"$spine:$page"`）作为稳定分页定位。
> * HTML 使用 **CSS columns** + 注入 JS `scrollLeft = pageIndex * viewportWidth` 来实现章内分页显示。
> * 链接点击在 WebView 内拦截：内部链接转换为 `reader://goto?scheme=...&value=...`，由 feature 统一拦截后调用 `controller.goTo(locator)`（只依赖 core api）。
> * 同时提供 `RenderPage.links`（bounds 为空）方便非 WebView 场景或辅助功能。

# ✅ 新增 1：章内分页估算器（不依赖 WebView 回传）

### `engines/epub/.../internal/pagination/ReflowPaginator.kt`（新增）

```kotlin
package com.example.reader.engines.epub.internal.pagination

import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.RenderConfig
import com.example.reader.engines.epub.internal.cache.SimpleLruCache
import com.example.reader.engines.epub.internal.open.EpubContainer
import com.example.reader.engines.epub.internal.text.XhtmlTextExtractor
import java.io.File
import kotlin.math.ceil
import kotlin.math.max

internal class ReflowPaginator(
    private val container: EpubContainer
) {
    // 缓存每章文本长度（避免每次都解析）
    private val lengthCache = SimpleLruCache<Int, Int>(maxSize = 64)

    fun pageCount(spineIndex: Int, constraints: LayoutConstraints?, config: RenderConfig.ReflowText): Int {
        if (constraints == null) return 1

        val textLen = lengthCache.getOrPut(spineIndex) {
            val path = container.spinePath(spineIndex)
            val file = File(container.rootDir, path)
            runCatching { XhtmlTextExtractor.extract(file).length }
                .getOrElse { runCatching { file.readText().length }.getOrDefault(0) }
        }.coerceAtLeast(0)

        val charsPerPage = estimateCharsPerPage(textLen, constraints, config).coerceAtLeast(300)
        val pages = ceil(textLen.toDouble() / charsPerPage.toDouble()).toInt().coerceAtLeast(1)

        // 防御：极端文件避免 pageCount 爆炸
        return pages.coerceAtMost(5000)
    }

    private fun estimateCharsPerPage(textLen: Int, c: LayoutConstraints, cfg: RenderConfig.ReflowText): Int {
        val fontPx = cfg.fontSizeSp * c.density * c.fontScale
        val linePx = max(1f, fontPx * cfg.lineHeightMult)
        val paddingPx = cfg.pagePaddingDp * c.density

        val usableW = (c.viewportWidthPx - (paddingPx * 2)).coerceAtLeast(200f)
        val usableH = (c.viewportHeightPx - (paddingPx * 2)).coerceAtLeast(200f)

        val linesPerPage = (usableH / linePx).toInt().coerceAtLeast(8)

        // 粗略判断 CJK，字符宽≈fontPx；拉丁≈0.55*fontPx
        val sampleIsCjk = guessCjk(textLen)
        val charWidth = if (sampleIsCjk) max(6f, fontPx * 0.95f) else max(6f, fontPx * 0.55f)

        val charsPerLine = (usableW / charWidth).toInt().coerceAtLeast(12)
        val raw = (charsPerLine * linesPerPage * 0.9).toInt()

        return raw.coerceAtLeast(400)
    }

    private fun guessCjk(textLen: Int): Boolean {
        // 这里只能用长度做极简推断（更精确需要采样文本内容）
        // 对中文/日文书籍一般章节文本更短且更密，这里让策略稍偏向 CJK
        return textLen in 1..20000
    }
}
```

# ✅ 新增 2：链接提取（输出 RenderPage.links，bounds 为空）

### `engines/epub/.../internal/link/LinkExtractor.kt`（新增）

```kotlin
package com.example.reader.engines.epub.internal.link

import android.util.Xml
import com.example.reader.engines.epub.internal.cache.SimpleLruCache
import com.example.reader.engines.epub.internal.open.EpubContainer
import com.example.reader.engines.epub.internal.parser.PathResolver
import com.example.reader.model.DocumentLink
import com.example.reader.model.LinkTarget
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import java.io.File
import java.io.FileInputStream
import org.xmlpull.v1.XmlPullParser

internal class LinkExtractor(
    private val container: EpubContainer
) {
    private val cache = SimpleLruCache<Int, List<DocumentLink>>(maxSize = 48)

    fun linksForSpine(spineIndex: Int): List<DocumentLink> {
        return cache.getOrPut(spineIndex) {
            val spineRel = container.spinePath(spineIndex)
            val file = File(container.rootDir, spineRel)
            runCatching { extractFromXhtml(file, spineRel) }.getOrElse { emptyList() }
        }
    }

    private fun extractFromXhtml(file: File, baseRelPath: String): List<DocumentLink> {
        val out = mutableListOf<DocumentLink>()

        FileInputStream(file).use { input ->
            val p = Xml.newPullParser()
            p.setInput(input, "UTF-8")

            var event = p.eventType
            while (event != XmlPullParser.END_DOCUMENT) {
                if (event == XmlPullParser.START_TAG) {
                    val name = p.name.substringAfter(':')
                    if (name.equals("a", ignoreCase = true)) {
                        val href = p.getAttributeValue(null, "href") ?: ""
                        if (href.isNotBlank()) {
                            val title = readInnerTextIfAny(p)?.trim().takeIf { !it.isNullOrBlank() }

                            val link = toDocumentLink(baseRelPath, href, title)
                            if (link != null) out += link
                        }
                    }
                }
                event = p.next()
            }
        }
        return out
    }

    private fun toDocumentLink(baseRelPath: String, hrefRaw: String, title: String?): DocumentLink? {
        val href = hrefRaw.trim()
        if (href.startsWith("javascript:", ignoreCase = true)) return null

        val isAbsoluteScheme = Regex("^[a-zA-Z][a-zA-Z0-9+.-]*:").containsMatchIn(href)
        val isMailOrTel = href.startsWith("mailto:", true) || href.startsWith("tel:", true)

        return if (isAbsoluteScheme && !isMailOrTel) {
            // 外链（http/https 等）
            DocumentLink(
                target = LinkTarget.External(url = href),
                title = title
            )
        } else {
            // 内链：相对路径或 #fragment
            val resolved = PathResolver.resolveFrom(baseRelPath, href)
            val frag = PathResolver.fragmentOf(href)
            val locator = Locator(
                scheme = LocatorSchemes.EPUB_CFI,
                value = "href:$resolved#${frag}".trimEnd('#')
            )
            DocumentLink(
                target = LinkTarget.Internal(locator),
                title = title
            )
        }
    }

    private fun readInnerTextIfAny(p: XmlPullParser): String? {
        // <a> 内可能有嵌套标签，这里尽量读到第一个 TEXT；不强求完美
        var event = p.next()
        if (event == XmlPullParser.TEXT) return p.text
        return null
    }
}
```

### `engines/epub/.../internal/render/HtmlComposer.kt`（修改：函数签名与实现要点）

```kotlin
// 新增 import
import com.example.reader.api.render.LayoutConstraints
import com.example.reader.engines.epub.internal.cache.SimpleLruCache

// 新增：模板缓存（不把 pageIndex/anchor 放入 key，避免爆炸）
private data class TplKey(
    val spineIndex: Int,
    val cfgHash: Int,
    val vw: Int,
    val vh: Int
)
private val templateCache = SimpleLruCache<TplKey, String>(maxSize = 24)

fun compose(
    container: EpubContainer,
    spineIndex: Int,
    config: RenderConfig.ReflowText,
    constraints: LayoutConstraints?,
    pageIndex: Int,
    baseRelPath: String,
    anchorId: String?
): String {
    // 未设置 layout 时退化成“滚动模式”HTML
    val vw = constraints?.viewportWidthPx ?: 0
    val vh = constraints?.viewportHeightPx ?: 0
    val key = TplKey(spineIndex, config.hashCode(), vw, vh)

    val tpl = templateCache.getOrPut(key) {
        val path = container.spinePath(spineIndex)
        val file = File(container.rootDir, path)
        val raw = file.readTextSafely()
        val body = extractBodyInner(raw) ?: raw

        val css = buildPagedCss(config, constraints)

        val baseHref = container.spineUri(spineIndex).toString()

        """
        <!doctype html>
        <html>
          <head>
            <meta charset="utf-8"/>
            <meta name="viewport" content="width=device-width, initial-scale=1.0"/>
            <base href="$baseHref"/>
            <style>$css</style>
          </head>
          <body>
            <div class="reader-content">$body</div>
            <script>
              (function(){
                const VW = __VW__;
                const PAGE = __PAGE__;
                const BASE_REL = "__BASE_REL__";
                const ANCHOR = "__ANCHOR__";

                function normalizePath(path){
                  const parts = [];
                  path.split('/').forEach(seg=>{
                    if(!seg || seg === '.') return;
                    if(seg === '..'){ if(parts.length) parts.pop(); return; }
                    parts.push(seg);
                  });
                  return parts.join('/');
                }

                function resolveFrom(baseRel, href){
                  const raw = (href || '').split('#')[0] || '';
                  const baseDir = baseRel.includes('/') ? baseRel.substring(0, baseRel.lastIndexOf('/')) : '';
                  const joined = raw ? (baseDir ? (baseDir + '/' + raw) : raw) : baseRel;
                  return normalizePath(joined);
                }

                function isAbsoluteScheme(h){
                  return /^[a-zA-Z][a-zA-Z0-9+.-]*:/.test(h);
                }

                function goToPage(){
                  if(!VW || VW <= 0) return;
                  const scroller = document.scrollingElement || document.documentElement;
                  scroller.scrollLeft = Math.max(0, PAGE) * VW;
                }

                function goToAnchor(){
                  if(!ANCHOR) return;
                  const el = document.getElementById(ANCHOR) || document.querySelector('[name="'+ANCHOR+'"]');
                  if(el && el.scrollIntoView) el.scrollIntoView();
                }

                document.addEventListener('click', function(e){
                  const a = e.target && e.target.closest ? e.target.closest('a[href]') : null;
                  if(!a) return;
                  const href = a.getAttribute('href') || '';
                  if(!href) return;

                  // 外链：http/https 等
                  if(isAbsoluteScheme(href) && !href.startsWith('mailto:') && !href.startsWith('tel:')){
                    e.preventDefault();
                    window.location.href = 'reader://external?url=' + encodeURIComponent(href);
                    return;
                  }

                  // 内链：#frag 或相对路径
                  e.preventDefault();
                  const frag = href.includes('#') ? href.split('#')[1] : '';
                  const resolved = resolveFrom(BASE_REL, href);
                  const value = 'href:' + resolved + (frag ? ('#' + frag) : '');
                  window.location.href = 'reader://goto?scheme=epub.cfi&value=' + encodeURIComponent(value);
                }, true);

                window.addEventListener('load', function(){
                  goToPage();
                  // anchor 优先（如果有 anchor，会覆盖 page 的视觉位置）
                  goToAnchor();
                });
              })();
            </script>
          </body>
        </html>
        """.trimIndent()
    }

    // 注意：这里做轻量替换，避免每页都重新读盘
    return tpl
        .replace("__VW__", vw.toString())
        .replace("__PAGE__", pageIndex.toString())
        .replace("__BASE_REL__", escapeJs(baseRelPath))
        .replace("__ANCHOR__", escapeJs(anchorId ?: ""))
}

private fun buildPagedCss(cfg: RenderConfig.ReflowText, c: LayoutConstraints?): String {
    val density = c?.density ?: 1f
    val vw = c?.viewportWidthPx ?: 0
    val vh = c?.viewportHeightPx ?: 0

    val fontPx = cfg.fontSizeSp * density * (c?.fontScale ?: 1f)
    val paddingPx = cfg.pagePaddingDp * density

    // 有 layout 才启用 columns 分页
    val columns = if (vw > 0 && vh > 0) """
      html, body { height: ${vh}px; width: ${vw}px; overflow: hidden; }
      .reader-content{
        height: ${vh}px;
        -webkit-column-width: ${vw}px;
        column-width: ${vw}px;
        -webkit-column-gap: 0px;
        column-gap: 0px;
        column-fill: auto;
      }
    """.trimIndent() else ""

    return """
      body{
        margin:0;
        padding:${paddingPx}px;
        font-size:${fontPx}px;
        line-height:${cfg.lineHeightMult};
        word-break: break-word;
        overflow-wrap: anywhere;
      }
      img{ max-width:100%; height:auto; }
      a{ color: inherit; }
      $columns
    """.trimIndent()
}

private fun escapeJs(s: String): String =
    s.replace("\\", "\\\\").replace("\"", "\\\"").replace("\n", "\\n")
```

override suspend fun next(policy: RenderPolicy): ReaderResult<RenderPage> {
    val pc = currentPageCount()
    if (pageIndex < pc - 1) {
        pageIndex++
    } else if (spineIndex < container.spineCount - 1) {
        spineIndex++
        pageIndex = 0
    }
    pendingAnchorId = null
    _state.value = buildState()
    _events.tryEmit(ReaderEvent.PageChanged(_state.value.locator))
    return renderCurrent(policy, emittedAsNavigation = true)
}

override suspend fun prev(policy: RenderPolicy): ReaderResult<RenderPage> {
    if (pageIndex > 0) {
        pageIndex--
    } else if (spineIndex > 0) {
        spineIndex--
        pageIndex = (currentPageCount() - 1).coerceAtLeast(0)
    }
    pendingAnchorId = null
    _state.value = buildState()
    _events.tryEmit(ReaderEvent.PageChanged(_state.value.locator))
    return renderCurrent(policy, emittedAsNavigation = true)

#### 6) 修改 goTo(locator)：支持 reflow.page 与 href:#frag

```kotlin
override suspend fun goTo(locator: Locator, policy: RenderPolicy): ReaderResult<RenderPage> {
    when (locator.scheme) {
        LocatorSchemes.REFLOW_PAGE -> {
            // value 形如 "12:3"
            val (s, p) = parseReflowPage(locator.value)
                ?: return ReaderResult.Err(ReaderError.CorruptOrInvalid("Invalid reflow.page: ${locator.value}"))
            spineIndex = s.coerceIn(0, container.spineCount - 1)
            val pc = currentPageCount()
            pageIndex = p.coerceIn(0, pc - 1)
            pendingAnchorId = null
        }

        LocatorSchemes.EPUB_CFI -> {
            // 支持：spine:idx / href:path#frag
            val v = locator.value
            val targetSpine = when {
                v.startsWith("spine:") -> v.removePrefix("spine:").toIntOrNull()
                v.startsWith("href:") -> {
                    val href = v.removePrefix("href:").substringBefore('#')
                    container.spineIndexOfHref(href)
                }
                else -> null
            } ?: return ReaderResult.Err(ReaderError.CorruptOrInvalid("Invalid epub locator: $locator"))

            spineIndex = targetSpine.coerceIn(0, container.spineCount - 1)
            pageIndex = 0

            val frag = v.substringAfter('#', "")
            pendingAnchorId = frag.takeIf { it.isNotBlank() }
        }

        else -> return ReaderResult.Err(ReaderError.CorruptOrInvalid("Unsupported locator scheme: ${locator.scheme}"))
    }

    _state.value = buildState()
    _events.tryEmit(ReaderEvent.PageChanged(_state.value.locator))
    return renderCurrent(policy, emittedAsNavigation = true)
}

private fun parseReflowPage(value: String): Pair<Int, Int>? {
    val parts = value.split(':')
    if (parts.size != 2) return null
    val s = parts[0].toIntOrNull() ?: return null
    val p = parts[1].toIntOrNull() ?: return null
    return s to p
}
```

private fun buildState(): RenderState {
    val chapterTotal = container.spineCount.coerceAtLeast(1)
    val chapterCur = spineIndex.coerceIn(0, chapterTotal - 1) + 1

    val pc = currentPageCount().coerceAtLeast(1)
    val pageCur = pageIndex.coerceIn(0, pc - 1) + 1

    val percent = ((spineIndex.toDouble() + (pageCur - 1).toDouble() / pc.toDouble())
        / (chapterTotal.toDouble())).coerceIn(0.0, 1.0)

    return RenderState(
        locator = Locator(LocatorSchemes.REFLOW_PAGE, "$spineIndex:$pageIndex"),
        progression = Progression(
            percent = percent,
            label = "Chapter $chapterCur/$chapterTotal · Page $pageCur/$pc",
            current = pageCur,
            total = pc
        ),
        nav = NavigationAvailability(
            canGoPrev = !(spineIndex == 0 && pageIndex == 0),
            canGoNext = !(spineIndex == chapterTotal - 1 && pageIndex == pc - 1)
        ),
        titleInView = container.titleForSpine(spineIndex),
        config = config
    )
}

# ✅ 新增 3（feature 侧最小对接）：拦截 reader://goto 进行跳转

> 这是“链接热区/跳转”闭环的关键（只依赖 core api，不依赖 engines）。
> 你把它用到 WebViewClient 的 `shouldOverrideUrlLoading` 里即可。

### `feature/reader/.../web/ReaderWebViewLinkRouter.kt`（新增）

```kotlin
package com.example.reader.feature.reader.web

import android.net.Uri
import com.example.reader.api.render.ReaderController
import com.example.reader.model.Locator
import kotlinx.coroutines.CoroutineScope
import kotlinx.coroutines.launch

object ReaderWebViewLinkRouter {

    fun tryHandle(url: String, controller: ReaderController, scope: CoroutineScope): Boolean {
        val uri = runCatching { Uri.parse(url) }.getOrNull() ?: return false
        if (uri.scheme != "reader") return false

        return when (uri.host) {
            "goto" -> {
                val scheme = uri.getQueryParameter("scheme") ?: return true
                val value = uri.getQueryParameter("value") ?: return true
                scope.launch { controller.goTo(Locator(scheme, value)) }
                true
            }
            "external" -> {
                // 由你决定：用 Intent.ACTION_VIEW 打开 uri.getQueryParameter("url")
                true
            }
            else -> true
        }
    }
}
```
