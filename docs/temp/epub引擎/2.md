## 新增 1：并发打开同一本书时的解压锁

### `engines/epub/src/main/java/.../internal/open/EpubLocks.kt`（新增）

```kotlin
package com.example.reader.engines.epub.internal.open

import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock
import java.util.concurrent.ConcurrentHashMap

internal object EpubLocks {
    private val locks = ConcurrentHashMap<String, Mutex>()

    suspend fun <T> withDocLock(docId: String, block: suspend () -> T): T {
        val mutex = locks.getOrPut(docId) { Mutex() }
        return mutex.withLock { block() }
    }
}
```

## 修改 2：ZipExtract 增强安全（大小限制 + 更稳的 copy）

### `engines/epub/.../internal/open/ZipExtract.kt`（新增参数 + 安全限制）

把 `extractTo` 改为下面逻辑（重点是 entry 大小限制与总量限制，避免 zip bomb）：

```kotlin
package com.example.reader.engines.epub.internal.open

import com.example.reader.source.DocumentSource
import java.io.BufferedInputStream
import java.io.File
import java.io.FileOutputStream
import java.util.zip.ZipInputStream
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

internal object ZipExtract {

    private const val MAX_ENTRY_BYTES = 50L * 1024 * 1024   // 50MB/entry
    private const val MAX_TOTAL_BYTES = 300L * 1024 * 1024  // 300MB total

    suspend fun extractTo(source: DocumentSource, outDir: File) = withContext(Dispatchers.IO) {
        var total = 0L

        source.openInputStream().use { raw ->
            ZipInputStream(BufferedInputStream(raw)).use { zis ->
                while (true) {
                    val entry = zis.nextEntry ?: break
                    val name = entry.name

                    val target = File(outDir, name)
                    val outCanon = outDir.canonicalFile
                    val targetCanon = target.canonicalFile

                    // 防 Zip Slip
                    if (!targetCanon.path.startsWith(outCanon.path)) {
                        zis.closeEntry()
                        continue
                    }

                    if (entry.isDirectory) {
                        targetCanon.mkdirs()
                        zis.closeEntry()
                        continue
                    }

                    // entry.size 可能为 -1（未知），仍然需要边写边限制
                    targetCanon.parentFile?.mkdirs()

                    FileOutputStream(targetCanon).use { fos ->
                        val buf = ByteArray(DEFAULT_BUFFER_SIZE)
                        var written = 0L
                        while (true) {
                            val n = zis.read(buf)
                            if (n <= 0) break
                            fos.write(buf, 0, n)
                            written += n
                            total += n

                            if (written > MAX_ENTRY_BYTES) error("EPUB entry too large: $name")
                            if (total > MAX_TOTAL_BYTES) error("EPUB too large (zip bomb?)")
                        }
                    }

                    zis.closeEntry()
                }
            }
        }
    }
}
```

## 修改 3：OPF 解析必须正确处理 namespace + opfDir 计算必须用 opfPath

### `engines/epub/.../internal/parser/OpfParser.kt`（修改：函数签名 + opfDir + localName）

把 `parse(opfFile: File)` 改成 `parse(opfFile: File, opfPath: String)`，并替换以下核心部分：

```kotlin
internal object OpfParser {

    fun parse(opfFile: File, opfPath: String): OpfPackage {
        val opfDirRel = opfPath.substringBeforeLast('/', missingDelimiterValue = "").trim('/')

        val manifest = linkedMapOf<String, ManifestItem>()
        val spineIdRefs = mutableListOf<String>()

        var title: String? = null
        var creator: String? = null
        var language: String? = null
        var identifier: String? = null

        FileInputStream(opfFile).use { input ->
            val p = Xml.newPullParser()
            p.setInput(input, "UTF-8")

            var event = p.eventType
            while (event != XmlPullParser.END_DOCUMENT) {
                if (event == XmlPullParser.START_TAG) {
                    val local = p.name.substringAfter(':') // ✅ 兼容 dc:title / title

                    when (local) {
                        "title" -> title = readText(p)
                        "creator", "author" -> creator = readText(p)
                        "language" -> language = readText(p)
                        "identifier" -> identifier = readText(p)
                        "item" -> {
                            val id = p.getAttributeValue(null, "id") ?: ""
                            val href = p.getAttributeValue(null, "href") ?: ""
                            val mediaType = p.getAttributeValue(null, "media-type")
                            val properties = p.getAttributeValue(null, "properties")
                            if (id.isNotBlank() && href.isNotBlank()) {
                                manifest[id] = ManifestItem(id, href, mediaType, properties)
                            }
                        }
                        "itemref" -> {
                            val idref = p.getAttributeValue(null, "idref") ?: ""
                            if (idref.isNotBlank()) spineIdRefs += idref
                        }
                    }
                }
                event = p.next()
            }
        }

        val nav = manifest.values.firstOrNull { it.properties?.contains("nav") == true }?.href
        val ncx = manifest.values.firstOrNull { it.mediaType == "application/x-dtbncx+xml" }?.href

        val metadata = DocumentMetadata(
            title = title?.takeIf { it.isNotBlank() },
            author = creator?.takeIf { it.isNotBlank() },
            language = language?.takeIf { it.isNotBlank() },
            identifier = identifier?.takeIf { it.isNotBlank() }
        )

        val spine = spineIdRefs.mapNotNull { idref ->
            val item = manifest[idref] ?: return@mapNotNull null
            val resolved = normalizePath(
                (if (opfDirRel.isBlank()) "" else "$opfDirRel/") + item.href
            )
            SpineItem(idref = idref, href = resolved, mediaType = item.mediaType)
        }

        return OpfPackage(
            metadata = metadata,
            manifest = manifest.toMap(),
            spine = spine,
            opfDir = opfDirRel,
            navPath = nav?.let { normalizePath((if (opfDirRel.isBlank()) "" else "$opfDirRel/") + it) },
            ncxPath = ncx?.let { normalizePath((if (opfDirRel.isBlank()) "" else "$opfDirRel/") + it) }
        )
    }

    // readText/normalizePath 保持你已有实现即可
}
```

## 新增 2：Locator 解析工具（统一 spine:/href: 解析逻辑）

### `engines/epub/src/main/java/.../internal/locator/EpubLocator.kt`（新增）

```kotlin
package com.example.reader.engines.epub.internal.locator

import com.example.reader.engines.epub.internal.open.EpubContainer
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes

internal object EpubLocator {

    fun spineIndexOf(container: EpubContainer, locator: Locator?): Int? {
        if (locator == null) return null
        if (locator.scheme != LocatorSchemes.EPUB_CFI) return null
        val v = locator.value

        return when {
            v.startsWith("spine:") ->
                v.removePrefix("spine:").toIntOrNull()

            v.startsWith("href:") -> {
                val href = v.removePrefix("href:").substringBefore('#')
                container.spineIndexOfHref(href)
            }

            // 预留：未来支持 epubcfi(...)
            else -> null
        }
    }
}
```

## 修改 4：EpubContainer 预计算 spineIndexByHref + TOC 标题映射

### `engines/epub/.../internal/open/EpubContainer.kt`（新增字段与方法）

在 `EpubContainer` 里加这些（不需要动你原有字段）：

```kotlin
internal data class EpubContainer(...){

    private val spineIndexByHref: Map<String, Int> =
        opf.spine.mapIndexed { i, s -> s.href to i }.toMap()

    private val titleByHref: Map<String, String> =
        buildMap {
            fun walk(nodes: List<OutlineNode>) {
                for (n in nodes) {
                    val href = n.locator.value
                        .takeIf { it.startsWith("href:") }
                        ?.removePrefix("href:")
                        ?.substringBefore('#')
                    if (href != null && !containsKey(href)) put(href, n.title)
                    if (n.children.isNotEmpty()) walk(n.children)
                }
            }
            walk(outline)
        }

    fun spineIndexOfHref(href: String): Int? = spineIndexByHref[href]

    fun titleForSpine(index: Int): String? {
        val href = opf.spine.getOrNull(index)?.href ?: return null
        return titleByHref[href]
    }
}
```
