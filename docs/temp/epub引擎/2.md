## 新增：`core/reader/runtime/annotation/InMemoryAnnotationStore.kt`

```kotlin
package com.example.reader.runtime.annotation

import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.AnnotationQuery
import com.example.reader.api.provider.AnnotationStore
import com.example.reader.model.DocumentId
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.annotation.Annotation
import com.example.reader.model.annotation.AnnotationAnchor
import com.example.reader.model.annotation.AnnotationDraft
import com.example.reader.model.annotation.AnnotationId
import java.util.UUID
import java.util.concurrent.ConcurrentHashMap
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.asStateFlow

class InMemoryAnnotationStore : AnnotationStore {

    private val flows = ConcurrentHashMap<String, MutableStateFlow<List<Annotation>>>()

    private fun flowFor(documentId: DocumentId): MutableStateFlow<List<Annotation>> =
        flows.computeIfAbsent(documentId.value) { MutableStateFlow(emptyList()) }

    override fun observe(documentId: DocumentId): Flow<List<Annotation>> =
        flowFor(documentId).asStateFlow()

    override suspend fun list(documentId: DocumentId): ReaderResult<List<Annotation>> =
        ReaderResult.Ok(flowFor(documentId).value)

    override suspend fun query(
        documentId: DocumentId,
        query: AnnotationQuery
    ): ReaderResult<List<Annotation>> {
        val all = flowFor(documentId).value
        val filtered = all.filter { it.matches(query) }
        return ReaderResult.Ok(filtered)
    }

    override suspend fun create(
        documentId: DocumentId,
        draft: AnnotationDraft
    ): ReaderResult<Annotation> {
        return try {
            val now = System.currentTimeMillis()
            val created = Annotation(
                id = AnnotationId(UUID.randomUUID().toString()),
                type = draft.type,
                anchor = draft.anchor,
                content = draft.content,
                style = draft.style,
                createdAtEpochMs = now,
                updatedAtEpochMs = now,
                extra = draft.extra
            )

            val flow = flowFor(documentId)
            flow.value = flow.value + created
            ReaderResult.Ok(created)
        } catch (t: Throwable) {
            ReaderResult.Err(ReaderError.Internal(cause = t))
        }
    }

    override suspend fun update(documentId: DocumentId, annotation: Annotation): ReaderResult<Unit> {
        val flow = flowFor(documentId)
        val old = flow.value
        val idx = old.indexOfFirst { it.id == annotation.id }
        if (idx < 0) return ReaderResult.Err(ReaderError.NotFound("Annotation not found"))

        val now = System.currentTimeMillis()
        val updated = annotation.copy(updatedAtEpochMs = now)

        val newList = old.toMutableList().apply { set(idx, updated) }
        flow.value = newList
        return ReaderResult.Ok(Unit)
    }

    override suspend fun delete(documentId: DocumentId, id: AnnotationId): ReaderResult<Unit> {
        val flow = flowFor(documentId)
        val old = flow.value
        val newList = old.filterNot { it.id == id }
        flow.value = newList
        return ReaderResult.Ok(Unit)
    }

    private fun Annotation.matches(query: AnnotationQuery): Boolean {
        // fixed page query
        query.page?.let { page ->
            val a = anchor as? AnnotationAnchor.FixedRects ?: return false
            return sameLocator(a.page, page)
        }

        // reflow range query（这里做“宽松匹配”：同起点 / 同 scheme+value）
        query.range?.let { r ->
            val a = anchor as? AnnotationAnchor.ReflowRange ?: return false
            return sameLocator(a.range.start, r.start) || sameLocator(a.range.end, r.end)
        }

        // query 全空 => 返回全部
        return true
    }

    private fun sameLocator(a: Locator, b: Locator): Boolean =
        a.scheme == b.scheme && a.value == b.value
}
```

## 1) 新增：`engines/epub/internal/render/EpubPreferencesMapper.kt`

把你的 `RenderConfig.ReflowText` 映射到 Readium 的 `EpubPreferences`，并交给 Navigator 动态更新。

```kotlin
package com.example.reader.epub.internal.render

import com.example.reader.api.render.RenderConfig
import org.readium.r2.navigator.epub.EpubPreferences
import org.readium.r2.navigator.epub.FontFamily

internal object EpubPreferencesMapper {

    /**
     * Readium 的 fontSize 是倍率（示例里 2.0 = 200%）。
     * 你的 RenderConfig 用 sp：这里用 16sp 当 1.0 的基准。
     */
    private const val BASE_FONT_SP = 16.0

    fun toEpubPreferences(config: RenderConfig.ReflowText): EpubPreferences {
        val fontScale = (config.fontSizeSp.toDouble() / BASE_FONT_SP).coerceIn(0.5, 4.0)

        // lineHeight 在 Readium 里同样是倍率
        val lineHeight = config.lineHeightMult.toDouble().coerceIn(1.0, 3.0)

        // pageMargins / paragraphSpacing 在 Readium 是 Double（单位抽象）
        // 这里用“dp -> 相对值”的策略：16dp ≈ 1.0
        val pageMargins = (config.pagePaddingDp.toDouble() / 16.0).coerceIn(0.0, 6.0)
        val paragraphSpacing = (config.paragraphSpacingDp.toDouble() / 16.0).coerceIn(0.0, 6.0)

        return EpubPreferences(
            fontSize = fontScale,
            lineHeight = lineHeight,
            pageMargins = pageMargins,
            paragraphSpacing = paragraphSpacing,
            hyphens = config.hyphenation,
            fontFamily = mapFontFamily(config.fontFamilyName)
        )
    }

    private fun mapFontFamily(name: String?): FontFamily? {
        val key = name?.trim()?.lowercase() ?: return null
        return when (key) {
            "serif" -> FontFamily.SERIF
            "sans-serif", "sans", "sansserif" -> FontFamily.SANS_SERIF
            "monospace" -> FontFamily.MONOSPACE
            "opendyslexic" -> FontFamily.OPEN_DYSLEXIC
            "accessibledfa", "accessible dfa" -> FontFamily.ACCESSIBLE_DFA
            "iawriterduospace", "i a writer duospace", "duospace" -> FontFamily.IA_WRITER_DUOSPACE
            else -> null
        }
    }
}
```

## 2) 新增：`engines/epub/internal/render/EpubDecorationsHost.kt`

负责：**在 Fragment 未绑定时先缓存 decorations，绑定后一次性 apply**（避免丢标注）。

```kotlin
package com.example.reader.epub.internal.render

import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import org.readium.r2.navigator.DecorableNavigator
import org.readium.r2.navigator.Decoration

internal class EpubDecorationsHost(
    private val mainDispatcher: CoroutineDispatcher = Dispatchers.Main.immediate
) {
    private var navigator: DecorableNavigator? = null
    private val pending: MutableMap<String, List<Decoration>> = linkedMapOf()

    fun bind(navigator: DecorableNavigator) {
        this.navigator = navigator
        val snapshot = pending.toMap()
        pending.clear()

        // applyDecorations 必须在主线程
        snapshot.forEach { (group, decos) ->
            navigator.applyDecorations(decos, group)
        }
    }

    fun unbind() {
        navigator = null
    }

    suspend fun apply(group: String, decorations: List<Decoration>) {
        val nav = navigator
        if (nav == null) {
            pending[group] = decorations
            return
        }
        withContext(mainDispatcher) {
            nav.applyDecorations(decorations, group)
        }
    }
}
```

```kotlin
package com.example.reader.epub.internal.provider

import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.ResourceProvider
import java.io.InputStream
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import org.readium.r2.shared.publication.Publication
import org.readium.r2.shared.util.Url
import org.readium.r2.shared.util.resource.asInputStream

internal class EpubResourceProvider(
    private val publication: Publication,
    private val ioDispatcher: CoroutineDispatcher = Dispatchers.IO
) : ResourceProvider {

    override suspend fun openResource(path: String): ReaderResult<InputStream> =
        withContext(ioDispatcher) {
            try {
                val href = parseHref(path)
                    ?: return@withContext ReaderResult.Err(
                        ReaderError.CorruptOrInvalid("Invalid href: $path")
                    )

                val resource = publication.get(href)
                    ?: return@withContext ReaderResult.Err(ReaderError.NotFound("Resource not found: $path"))

                ReaderResult.Ok(resource.asInputStream())
            } catch (t: Throwable) {
                ReaderResult.Err(ReaderError.Io(cause = t))
            }
        }

    override suspend fun getMimeType(path: String): ReaderResult<String?> =
        withContext(ioDispatcher) {
            try {
                val href = parseHref(path)
                    ?: return@withContext ReaderResult.Ok(null)

                val link = publication.linkWithHref(href)
                val type = link?.type?.toString()
                ReaderResult.Ok(type)
            } catch (t: Throwable) {
                ReaderResult.Err(ReaderError.Internal(cause = t))
            }
        }

    private fun parseHref(path: String): Url? {
        // EPUB 的 href 可能包含空格等“非严格 URL”，优先用 fromEpubHref/fromLegacyHref
        return Url.fromEpubHref(path)
            ?: Url.fromLegacyHref(path)
            ?: runCatching { Url(path) }.getOrNull()
    }
}
```

## 4) 新增：`engines/epub/internal/provider/EpubTextProvider.kt`

用 Readium ContentService（可用于复制、TTS、做索引/搜索）。

```kotlin
package com.example.reader.epub.internal.provider

import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.TextProvider
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.epub.internal.locator.ReadiumLocatorAdapter
import kotlinx.coroutines.CoroutineDispatcher
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext
import org.readium.r2.shared.publication.Publication
import org.readium.r2.shared.publication.services.content.Content
import org.readium.r2.shared.publication.services.content.content

internal class EpubTextProvider(
    private val publication: Publication,
    private val ioDispatcher: CoroutineDispatcher = Dispatchers.IO
) : TextProvider {

    override suspend fun getText(range: LocatorRange): ReaderResult<String> =
        withContext(ioDispatcher) {
            try {
                val start = ReadiumLocatorAdapter.fromAppLocator(range.start)
                    ?: return@withContext ReaderResult.Err(
                        ReaderError.CorruptOrInvalid("Invalid locator: ${range.start.scheme}")
                    )

                // 如果 locator 自带 highlight（例如 selection/search 的 locator），优先返回
                val hl = start.text?.highlight
                if (!hl.isNullOrBlank()) {
                    return@withContext ReaderResult.Ok(hl)
                }

                val end = ReadiumLocatorAdapter.fromAppLocator(range.end)
                val endPos = end?.locations?.position

                val content = publication.content(start)
                    ?: return@withContext ReaderResult.Err(ReaderError.Internal("ContentService unavailable"))

                val it = content.iterator()
                val sb = StringBuilder()
                val limit = 8_192

                while (sb.length < limit) {
                    val el = it.nextOrNull() ?: break

                    // 尝试按 positions 截断（如果有）
                    val elPos = el.locator.locations.position
                    if (endPos != null && elPos != null && elPos > endPos) break

                    val txt = (el as? Content.TextualElement)?.text?.trim()
                    if (!txt.isNullOrEmpty()) {
                        if (sb.isNotEmpty()) sb.append('\n')
                        sb.append(txt)
                    }
                }

                ReaderResult.Ok(sb.toString())
            } catch (t: Throwable) {
                ReaderResult.Err(ReaderError.Io(cause = t))
            }
        }

    override suspend fun getTextAround(locator: Locator, maxChars: Int): ReaderResult<String> =
        withContext(ioDispatcher) {
            try {
                val start = ReadiumLocatorAdapter.fromAppLocator(locator)
                    ?: return@withContext ReaderResult.Err(
                        ReaderError.CorruptOrInvalid("Invalid locator: ${locator.scheme}")
                    )

                // 如果有 before/highlight/after，直接拼（常见于 selection/search）
                val before = start.text?.before.orEmpty()
                val highlight = start.text?.highlight.orEmpty()
                val after = start.text?.after.orEmpty()
                val rich = (before + highlight + after).trim()
                if (rich.isNotEmpty()) {
                    return@withContext ReaderResult.Ok(rich.take(maxChars))
                }

                val content = publication.content(start)
                    ?: return@withContext ReaderResult.Err(ReaderError.Internal("ContentService unavailable"))

                val it = content.iterator()
                val sb = StringBuilder()

                while (sb.length < maxChars) {
                    val el = it.nextOrNull() ?: break
                    val txt = (el as? Content.TextualElement)?.text?.trim()
                    if (!txt.isNullOrEmpty()) {
                        if (sb.isNotEmpty()) sb.append('\n')
                        sb.append(txt)
                    }
                }

                ReaderResult.Ok(sb.toString().take(maxChars))
            } catch (t: Throwable) {
                ReaderResult.Err(ReaderError.Io(cause = t))
            }
        }
}
```

## 6) 新增：`engines/epub/internal/provider/EpubAnnotationProvider.kt`

* 监听 `AnnotationStore.observe(documentId)`
* 把 `Annotation` 映射成 Readium `Decoration`
* 通过 `EpubDecorationsHost` 应用到 navigator（支持 “未绑定先缓存”）

```kotlin
package com.example.reader.epub.internal.provider

import com.example.reader.api.annotation.Decoration as AppDecoration
import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.AnnotationProvider
import com.example.reader.api.provider.AnnotationQuery
import com.example.reader.api.provider.AnnotationStore
import com.example.reader.epub.internal.locator.ReadiumLocatorAdapter
import com.example.reader.epub.internal.render.EpubDecorationsHost
import com.example.reader.model.DocumentId
import com.example.reader.model.annotation.Annotation
import com.example.reader.model.annotation.AnnotationAnchor
import com.example.reader.model.annotation.AnnotationDraft
import com.example.reader.model.annotation.AnnotationId
import kotlinx.coroutines.CoroutineScope
import kotlinx.coroutines.Job
import kotlinx.coroutines.flow.catch
import kotlinx.coroutines.flow.collectLatest
import kotlinx.coroutines.flow.distinctUntilChanged
import kotlinx.coroutines.launch
import org.readium.r2.navigator.Decoration

internal class EpubAnnotationProvider(
    private val documentId: DocumentId,
    private val store: AnnotationStore,
    private val decorationsHost: EpubDecorationsHost,
    private val scope: CoroutineScope
) : AnnotationProvider {

    companion object {
        private const val GROUP_ANNOTATIONS = "user.annotations"
        private const val DEFAULT_HIGHLIGHT = 0x66FFF59D // semi yellow
    }

    private val job: Job = scope.launch {
        store.observe(documentId)
            .distinctUntilChanged()
            .catch {
                // 不让异常炸到 UI
                // 你也可以考虑把它转成 ReaderEvent.Error 发给 controller
            }
            .collectLatest { list ->
                val decos = list.mapNotNull { it.toReadiumDecorationOrNull() }
                decorationsHost.apply(GROUP_ANNOTATIONS, decos)
            }
    }

    override fun observeAll() = store.observe(documentId)

    override suspend fun listAll(): ReaderResult<List<Annotation>> =
        store.list(documentId)

    override suspend fun query(query: AnnotationQuery): ReaderResult<List<Annotation>> =
        store.query(documentId, query)

    override suspend fun create(draft: AnnotationDraft): ReaderResult<Annotation> =
        store.create(documentId, draft)

    override suspend fun update(annotation: Annotation): ReaderResult<Unit> =
        store.update(documentId, annotation)

    override suspend fun delete(id: AnnotationId): ReaderResult<Unit> =
        store.delete(documentId, id)

    override suspend fun decorationsFor(query: AnnotationQuery): ReaderResult<List<AppDecoration>> {
        val items = when (val r = store.query(documentId, query)) {
            is ReaderResult.Ok -> r.value
            is ReaderResult.Err -> return r
        }

        val mapped = items.mapNotNull { anno ->
            when (val anchor = anno.anchor) {
                is AnnotationAnchor.ReflowRange ->
                    AppDecoration.Reflow(anchor.range, anno.style)

                is AnnotationAnchor.FixedRects ->
                    AppDecoration.Fixed(anchor.page, anchor.rects, anno.style)
            }
        }
        return ReaderResult.Ok(mapped)
    }

    fun closeInternal() {
        job.cancel()
    }

    private fun Annotation.toReadiumDecorationOrNull(): Decoration? {
        val range = (anchor as? AnnotationAnchor.ReflowRange)?.range ?: return null
        val readiumLocator = ReadiumLocatorAdapter.fromAppLocator(range.start) ?: return null

        val tint = applyOpacity(
            argb = style.colorArgb ?: DEFAULT_HIGHLIGHT,
            opacity = style.opacity
        )

        val readiumStyle = when (type) {
            com.example.reader.model.annotation.AnnotationType.HIGHLIGHT ->
                Decoration.Style.Highlight(tint = tint)

            com.example.reader.model.annotation.AnnotationType.UNDERLINE ->
                Decoration.Style.Underline(tint = tint)

            com.example.reader.model.annotation.AnnotationType.NOTE ->
                Decoration.Style.Highlight(tint = tint)

            com.example.reader.model.annotation.AnnotationType.BOOKMARK ->
                // 书签在 reflow 里通常不是“文本块高亮”，但先用低透明 highlight 表示（可再升级成 margin icon）
                Decoration.Style.Highlight(tint = tint)
        }

        return Decoration(
            id = id.value,
            locator = readiumLocator,
            style = readiumStyle,
            extras = mapOf(
                "type" to type.name
            )
        )
    }

    private fun applyOpacity(argb: Int, opacity: Float?): Int {
        if (opacity == null) return argb
        val clamped = opacity.coerceIn(0f, 1f)
        val alpha = ((clamped * 255f).toInt() and 0xFF) shl 24
        return (argb and 0x00FFFFFF) or alpha
    }
}
```

```diff
+        navigatorProvider = { controllerImpl.navigatorOrNull() }
+    )
+
+    private val decorationsHost = controllerImpl.decorationsHost // 见下一节 controller 改动
+    private val annotationProviderInternal: EpubAnnotationProvider? =
+        annotationStore?.let { store ->
+            EpubAnnotationProvider(
+                documentId = controllerImpl.documentId,
+                store = store,
+                decorationsHost = decorationsHost,
+                scope = sessionScope
+            )
+        }
+
+    override val annotations: AnnotationProvider? = annotationProviderInternal

     override fun close() {
+        annotationProviderInternal?.closeInternal()
         controller.close()
     }
 }
```

> 关键点：
> `EpubAnnotationProvider` 需要 `documentId`、`AnnotationStore`、`EpubDecorationsHost`、`scope`。
> 其中 `decorationsHost` 和 `navigatorOrNull()` 在 controller 里补齐（下一节）。

# 你现在补齐后的能力清单（EPUB）

✅ 已补齐（专业必备）

* **TextProvider**：基于 ContentService，支持复制/导出/TTS 的基础能力
* **ResourceProvider**：能按 href 打开资源 InputStream（图片/CSS 等）
* **SelectionProvider**：统一拿到当前选择 locator（+ 可选 bounds）
* **AnnotationProvider**：

  * CRUD 委托 `AnnotationStore`（可 Room 持久化）
  * 自动监听数据变化并通过 Readium Decorator API **渲染高亮/下划线**

⏭️ 你接下来若要“更完整的专业版”，建议按优先级继续补：

1. **Annotation 的“精确 range”**：把 Readium selection locator 转成 `LocatorRange(start,end)`（需要 CFI range 支持）
2. **Decoration 点击回调**：把 `DecorableNavigator.Listener` 接到 `ReaderEvent`（点高亮打开笔记）
3. **全文搜索**：用 ContentService 做索引（WorkManager + SQLite FTS5/Room FTS）而不是逐章 regex
