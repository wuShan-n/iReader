#### 1.3 新增：构建/更新全书页表（prefix sum）

```kotlin
private suspend fun ensureGlobalPages() {
    val s = sig()
    val existing = globalPages
    if (!globalDirty && existing != null && existing.sig == s) return

    // 在 IO 上做，避免卡 UI（renderCurrent 本来就在 IO）
    val spineCount = container.spineCount.coerceAtLeast(1)
    val pagesBySpine = IntArray(spineCount)

    for (i in 0 until spineCount) {
        // 优先真实页数（metricsStore），否则估算
        pagesBySpine[i] = metricsStore.getPages(i, s)
            ?: paginator.pageCount(i, constraints, config)
    }

    val prefix = IntArray(spineCount + 1)
    var acc = 0
    for (i in 0 until spineCount) {
        prefix[i] = acc
        acc += pagesBySpine[i].coerceAtLeast(1)
    }
    prefix[spineCount] = acc.coerceAtLeast(1)

    globalPages = GlobalPages(sig = s, pagesBySpine = pagesBySpine, prefix = prefix, total = prefix[spineCount])
    globalDirty = false
}

private fun markGlobalDirty() {
    globalDirty = true
}
```

```kotlin
private fun parseLinkBounds(json: String): Map<String, List<NormalizedRect>> {
    return runCatching {
        val arr = JSONArray(json)
        val out = HashMap<String, MutableList<NormalizedRect>>(arr.length())
        for (i in 0 until arr.length()) {
            val o = arr.getJSONObject(i)
            val k = o.optString("k")
            if (k.isNullOrBlank()) continue

            val l = o.optDouble("l", -1.0).toFloat()
            val t = o.optDouble("t", -1.0).toFloat()
            val r = o.optDouble("r", -1.0).toFloat()
            val b = o.optDouble("b", -1.0).toFloat()

            // clamp 0..1
            val rect = NormalizedRect(
                left = l.coerceIn(0f, 1f),
                top = t.coerceIn(0f, 1f),
                right = r.coerceIn(0f, 1f),
                bottom = b.coerceIn(0f, 1f)
            )

            if (rect.right <= rect.left || rect.bottom <= rect.top) continue

            out.getOrPut(k) { mutableListOf() }.add(rect)
        }
        out
    }.getOrDefault(emptyMap())
}

private fun applyBounds(links: List<com.example.reader.model.DocumentLink>, map: Map<String, List<NormalizedRect>>)
    : List<com.example.reader.model.DocumentLink> {
    if (map.isEmpty()) return links
    return links.map { link ->
        val key = linkKeyOf(link)
        val rects = map[key]
        if (rects != null) link.copy(bounds = rects) else link
    }
}
```

2. 新增 `reportLinkBounds(page)`（注意：只取当前 viewport 内的链接，最多 40 条；并做小数压缩）

```js
function clamp01(x){ return Math.max(0, Math.min(1, x)); }
function r3(x){ return Math.round(x * 1000) / 1000; }

function reportLinkBounds(page){
  try{
    const list = [];
    const nodes = document.querySelectorAll('a[href]');
    const max = 40;
    for(let i=0;i<nodes.length && list.length<max;i++){
      const a = nodes[i];
      const rect = a.getBoundingClientRect();
      // 只要在当前 viewport 内
      if(rect.right <= 0 || rect.bottom <= 0 || rect.left >= VW || rect.top >= window.innerHeight) continue;

      const href = a.getAttribute('href') || '';
      if(!href) continue;

      let k = '';
      if(isAbsoluteScheme(href)){
        k = 'E|' + href;
      }else{
        const frag = href.includes('#') ? href.split('#')[1] : '';
        const resolved = resolveFrom(BASE_REL, href);
        const value = 'href:' + resolved + (frag ? ('#' + frag) : '');
        k = 'I|' + value;
      }

      const l = clamp01(rect.left / VW);
      const t = clamp01(rect.top / window.innerHeight);
      const rr = clamp01(rect.right / VW);
      const bb = clamp01(rect.bottom / window.innerHeight);

      if(rr <= l || bb <= t) continue;

      list.push({k:k, l:r3(l), t:r3(t), r:r3(rr), b:r3(bb)});
    }

    if(list.length === 0) return;

    const data = encodeURIComponent(JSON.stringify(list));
    window.location.href = 'reader://linkbounds?spine=' + SPINE + '&sig=' + SIG + '&page=' + page + '&data=' + data;
  }catch(e){}
}
```

3. 把 `reportMetrics()` 改成返回 `{pages, page}`，并在页变化时调用 reportLinkBounds：

```js
function reportMetrics(){
  if(!VW || VW <= 0) return null;
  const scroller = document.scrollingElement || document.documentElement;
  const sw = scroller.scrollWidth || 0;
  const pages = Math.max(1, Math.ceil(sw / VW));
  const page = Math.max(0, Math.round((scroller.scrollLeft || 0) / VW));
  window.location.href = 'reader://metrics?spine=' + SPINE + '&sig=' + SIG + '&pages=' + pages + '&page=' + page;
  return {pages: pages, page: page};
}
```

4. debounce handler 内：只有当 page 变了才上报 bounds：

```js
function reportAll(){
  const m = reportMetrics();
  if(!m) return;
  if(m.page !== lastReportedPage){
    lastReportedPage = m.page;
    reportLinkBounds(m.page);
  }
}

function debounceReport(){
  if(__t) clearTimeout(__t);
  __t = setTimeout(reportAll, 120);
}

window.addEventListener('scroll', function(){ debounceReport(); }, {passive:true});

window.addEventListener('load', function(){
  goToPage();
  goToAnchor();
  setTimeout(reportAll, 0);
});
```

"linkbounds" -> {
    val spine = uri.getQueryParameter("spine")?.toIntOrNull() ?: return true

    val page = uri.getQueryParameter("page")?.toIntOrNull() ?: return true

                    "linkBounds" to Uri.decode(data)
                )

        )
    }
    true

## 现在新增/优化的能力（这轮完成）

* ✅ **真实全书页进度**：`Progression.percent` 与 `label/current/total` 基于“全书真实页表（prefix sum）”，体验会比章节比例稳定很多。
* ✅ **链接热区 bounds 可用**：`RenderPage.links[i].bounds` 会带 `NormalizedRect`（0..1）坐标（仅当前页可见链接、最多 40 个），可用于做可视化热区/辅助功能/自绘 overlay。
* ✅ **不会触发重渲染风暴**：linkbounds 与 metrics 都走 `metricsOnly=1`，只更新 store/state/lastRendered，不强制重新 render。
