下面继续把“章内分页 + 链接热区/跳转”再提升一档（**不改 core API**），只给**新增/修改**代码。

这轮目标：

* ✅ **锚点跳转更准**：`href:...#frag` 不再总落到章节第一页；会先估算 frag 所在页，再滚到锚点
* ✅ **进度跳转更准**：`goToProgress()` 改为按“全书页”映射（按当前 layout/config 估算全书总页数）
* ✅ **分页参数更稳定**：`reflow.page` 的 value 增强，支持带 `cfg`（config hash）
* ✅ **链接更完整**：`mailto:`/`tel:` 也当外链走统一拦截

# 新增 1：章节锚点索引（frag -> 章节内字符偏移）

### `engines/epub/.../internal/anchor/AnchorIndexer.kt`（新增）

```kotlin
package com.example.reader.engines.epub.internal.anchor

import android.util.Xml
import com.example.reader.engines.epub.internal.cache.SimpleLruCache
import com.example.reader.engines.epub.internal.open.EpubContainer
import java.io.File
import java.io.FileInputStream
import org.xmlpull.v1.XmlPullParser

internal class AnchorIndexer(
    private val container: EpubContainer
) {
    // spineIndex -> (anchorId -> charOffset)
    private val cache = SimpleLruCache<Int, Map<String, Int>>(maxSize = 48)

    fun charOffsetFor(spineIndex: Int, anchorId: String): Int? {
        val map = cache.getOrPut(spineIndex) { buildIndex(spineIndex) }
        return map[anchorId]
    }

    private fun buildIndex(spineIndex: Int): Map<String, Int> {
        val rel = container.spinePath(spineIndex)
        val file = File(container.rootDir, rel)
        if (!file.exists()) return emptyMap()

        val out = HashMap<String, Int>(128)
        var charCount = 0
        var inBody = false

        FileInputStream(file).use { input ->
            val p = Xml.newPullParser()
            p.setInput(input, "UTF-8")

            var event = p.eventType
            while (event != XmlPullParser.END_DOCUMENT) {
                when (event) {
                    XmlPullParser.START_TAG -> {
                        val name = p.name.substringAfter(':')
                        if (name.equals("body", ignoreCase = true)) inBody = true

                        if (inBody) {
                            // 任意元素 id / name 都可能成为 fragment 目标
                            val id = p.getAttributeValue(null, "id")?.trim()
                            val aName = if (name.equals("a", true)) p.getAttributeValue(null, "name")?.trim() else null
                            if (!id.isNullOrBlank()) out.putIfAbsent(id, charCount)
                            if (!aName.isNullOrBlank()) out.putIfAbsent(aName, charCount)
                        }
                    }

                    XmlPullParser.TEXT -> {
                        if (inBody) {
                            val t = p.text ?: ""
                            // 粗略计数：文本长度 + 1 空格
                            charCount += t.length + 1
                        }
                    }

                    XmlPullParser.END_TAG -> {
                        val name = p.name.substringAfter(':')
                        if (name.equals("body", ignoreCase = true)) inBody = false
                    }
                }
                event = p.next()
            }
        }
        return out
    }
}
```

# 修改 1：ReflowPaginator 暴露 charsPerPage（保证 anchor->page 与 pageCount 同口径）

### `engines/epub/.../internal/pagination/ReflowPaginator.kt`（修改/新增方法）

在类里新增一个公开方法，并让 `pageCount()` 使用它（保证一致）：

```kotlin
// 新增：同口径 charsPerPage（anchor 跳页会用）
fun charsPerPage(constraints: LayoutConstraints?, config: RenderConfig.ReflowText, isCjk: Boolean): Int {
    if (constraints == null) return Int.MAX_VALUE

    val c = constraints
    val fontPx = config.fontSizeSp * c.density * c.fontScale
    val linePx = kotlin.math.max(1f, fontPx * config.lineHeightMult)
    val paddingPx = config.pagePaddingDp * c.density

    val usableW = (c.viewportWidthPx - (paddingPx * 2)).coerceAtLeast(200f)
    val usableH = (c.viewportHeightPx - (paddingPx * 2)).coerceAtLeast(200f)

    val linesPerPage = (usableH / linePx).toInt().coerceAtLeast(8)
    val charWidth = if (isCjk) kotlin.math.max(6f, fontPx * 0.95f) else kotlin.math.max(6f, fontPx * 0.55f)
    val charsPerLine = (usableW / charWidth).toInt().coerceAtLeast(12)

    return (charsPerLine * linesPerPage * 0.9).toInt().coerceAtLeast(400)
}
```

然后把你原先 `pageCount()` 内的 charsPerPage 计算替换为：

```kotlin
val sampleIsCjk = guessCjk(textLen) // 你原来的推断保留即可
val cpp = charsPerPage(constraints, config, sampleIsCjk).coerceAtLeast(300)
val pages = ceil(textLen.toDouble() / cpp.toDouble()).toInt().coerceAtLeast(1)
```

# 新增 2：全书页映射（用于 goToProgress 更准）

### `engines/epub/.../internal/pagination/GlobalPageMapper.kt`（新增）

```kotlin
package com.example.reader.engines.epub.internal.pagination

import com.example.reader.api.render.LayoutConstraints
import com.example.reader.api.render.RenderConfig
import com.example.reader.engines.epub.internal.open.EpubContainer
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

internal class GlobalPageMapper(
    private val container: EpubContainer,
    private val paginator: ReflowPaginator
) {
    suspend fun totalPages(constraints: LayoutConstraints?, config: RenderConfig.ReflowText): Int =
        withContext(Dispatchers.IO) {
            var sum = 0
            for (i in 0 until container.spineCount) {
                sum += paginator.pageCount(i, constraints, config)
            }
            sum.coerceAtLeast(1)
        }

    suspend fun locateByPercent(
        percent: Double,
        constraints: LayoutConstraints?,
        config: RenderConfig.ReflowText
    ): Pair<Int, Int> = withContext(Dispatchers.IO) {
        val p = percent.coerceIn(0.0, 1.0)
        val total = totalPages(constraints, config)
        val targetGlobal = (p * (total - 1)).toInt().coerceIn(0, total - 1)

        var acc = 0
        for (spine in 0 until container.spineCount) {
            val pc = paginator.pageCount(spine, constraints, config)
            if (targetGlobal < acc + pc) {
                val page = (targetGlobal - acc).coerceIn(0, pc - 1)
                return@withContext spine to page
            }
            acc += pc
        }
        // fallback
        (container.spineCount - 1).coerceAtLeast(0) to 0
    }
}
```

### `engines/epub/.../internal/render/EpubController.kt`（新增字段 + 修改关键逻辑）

## 3.2 新增字段（放属性区）

```kotlin

private fun cfgHash(): Int = config.hashCode()
```

## 3.3 修改：reflow.page value 编码增强（兼容旧的 "s:p"）

把你 buildState / RenderPage locator 的 value 从：

* `"${spineIndex}:${pageIndex}"`
  改为：
* `"${spineIndex}:${pageIndex}:${cfgHash()}"`

例如（两处都改）：

```kotlin
Locator(LocatorSchemes.REFLOW_PAGE, "$spineIndex:$pageIndex:${cfgHash()}")
```

并把 `parseReflowPage` 改成兼容 2/3 段：

```kotlin
private fun parseReflowPage(value: String): Pair<Int, Int>? {
    val parts = value.split(':')
    if (parts.size < 2) return null
    val s = parts[0].toIntOrNull() ?: return null
    val p = parts[1].toIntOrNull() ?: return null
    return s to p
}
```

> 第 3 段 cfgHash 你暂时不用来校验也行（后续可用来做“配置变化后定位降级”为 percent）。

## 3.4 修改：goTo(epub.cfi href:#frag) 先算页再滚锚点

在 `LocatorSchemes.EPUB_CFI` 分支里，把 `pageIndex = 0` 的逻辑改成：

```kotlin
val frag = v.substringAfter('#', "")
val cleanFrag = frag.takeIf { it.isNotBlank() }

pageIndex = if (cleanFrag != null && constraints != null) {
    val offset = anchorIndexer.charOffsetFor(spineIndex, cleanFrag)
    if (offset != null) {
        val isCjk = true // 先沿用 paginator 的粗判（你也可做更精：采样文本算比例）
        val cpp = paginator.charsPerPage(constraints, config, isCjk).coerceAtLeast(300)
        (offset / cpp).coerceAtLeast(0)
            .coerceAtMost((currentPageCount() - 1).coerceAtLeast(0))
    } else 0
} else 0

pendingAnchorId = cleanFrag
```

这样 TOC/搜索命中/链接跳转到 `#frag` 时，先跳到大概页，再 `scrollIntoView()` 精确落点。

## 3.5 修改：goToProgress 用全书页映射（更准）

把你原来的 `goToProgress` 实现替换为：

```kotlin
override suspend fun goToProgress(percent: Double, policy: RenderPolicy): ReaderResult<RenderPage> {
    return try {
        val (s, p) = globalPageMapper.locateByPercent(percent, constraints, config)
        spineIndex = s.coerceIn(0, container.spineCount - 1)
        val pc = currentPageCount()
        pageIndex = p.coerceIn(0, pc - 1)
        pendingAnchorId = null

        _state.value = buildState()
        _events.tryEmit(ReaderEvent.PageChanged(_state.value.locator))
        renderCurrent(policy, emittedAsNavigation = true)
    } catch (t: Throwable) {
        ReaderResult.Err(ReaderError.Internal(cause = t))
    }
}
```

## 这轮优化后的体验变化

* **目录/搜索/正文链接**命中带 `#frag` 时：
  先跳到**更接近的页**（不总是第一页），然后滚到锚点（更像专业阅读器）。
* `goToProgress()`：
  不再只按章节比例粗跳，而是按“全书估算总页数”做映射，体验明显更稳定。
* `reflow.page` 更可扩展：
  value 带上 cfgHash，为后面做“配置变化时定位降级/迁移”留口子。
* mailto/tel：
  和 http 一样走 external 拦截（你可统一用 Intent 打开）。
