### `engines/epub/.../internal/parser/PathResolver.kt`（新增）

```kotlin
package com.example.reader.engines.epub.internal.parser

internal object PathResolver {

    /** baseFileRel: 比如 "OEBPS/nav.xhtml"，href: "Text/ch1.xhtml#p1" */
    fun resolveFrom(baseFileRel: String, href: String): String {
        val baseDir = baseFileRel.substringBeforeLast('/', missingDelimiterValue = "").trim('/')
        val raw = href.substringBefore('#').trim()

        // EPUB 内一般都是相对路径；如果是空，表示当前文档
        val joined = when {
            raw.isBlank() -> baseFileRel
            baseDir.isBlank() -> raw
            else -> "$baseDir/$raw"
        }
        return normalizePath(joined)
    }

    fun fragmentOf(href: String): String =
        href.substringAfter('#', "")

    /** 复用你已有 normalizePath 实现（如果在别处，就把它挪到一个公共文件） */
    fun normalizePath(path: String): String = com.example.reader.engines.epub.internal.parser.normalizePath(path)
}
```

# 新增 2：简单 LRU（替代无限 ConcurrentHashMap）

```kotlin
package com.example.reader.engines.epub.internal.cache

internal class SimpleLruCache<K, V>(private val maxSize: Int) {

    private val map = object : LinkedHashMap<K, V>(16, 0.75f, true) {
        override fun removeEldestEntry(eldest: MutableMap.MutableEntry<K, V>?): Boolean =
            size > maxSize
    }

    @Synchronized fun get(key: K): V? = map[key]

    @Synchronized fun put(key: K, value: V) {
        map[key] = value
    }

    @Synchronized fun getOrPut(key: K, create: () -> V): V {
        val existing = map[key]
        if (existing != null) return existing
        val v = create()
        map[key] = v
        return v
    }

    @Synchronized fun clear() = map.clear()
}
```

# 新增 3：XHTML 文本抽取器（更稳的 Text/Search 基础）

### `engines/epub/.../internal/text/XhtmlTextExtractor.kt`（新增）

```kotlin
package com.example.reader.engines.epub.internal.text

import android.util.Xml
import java.io.File
import java.io.FileInputStream
import org.xmlpull.v1.XmlPullParser

internal object XhtmlTextExtractor {

    /**
     * 尝试用 XML 方式抽 XHTML 文本：
     * - 失败（非严格 XHTML）再由调用方 fallback。
     */
    fun extract(file: File): String {
        FileInputStream(file).use { input ->
            val p = Xml.newPullParser()
            p.setInput(input, "UTF-8")

            val sb = StringBuilder(8192)
            var inBody = false

            var event = p.eventType
            while (event != XmlPullParser.END_DOCUMENT) {
                when (event) {
                    XmlPullParser.START_TAG -> {
                        val name = p.name.substringAfter(':')
                        if (name.equals("body", ignoreCase = true)) inBody = true
                        // 适当把块级元素转空格，避免粘连
                        if (inBody && isBlockLike(name)) sb.append(' ')
                    }
                    XmlPullParser.TEXT -> {
                        if (inBody) {
                            val t = p.text
                            if (!t.isNullOrBlank()) {
                                sb.append(t).append(' ')
                            }
                        }
                    }
                    XmlPullParser.END_TAG -> {
                        val name = p.name.substringAfter(':')
                        if (name.equals("body", ignoreCase = true)) inBody = false
                        if (inBody && isBlockLike(name)) sb.append(' ')
                    }
                }
                event = p.next()
            }

            return sb.toString()
                .replace(Regex("\\s+"), " ")
                .trim()
        }
    }

    private fun isBlockLike(name: String): Boolean = when (name.lowercase()) {
        "p", "div", "br", "li", "h1", "h2", "h3", "h4", "h5", "h6", "section", "article",
        "blockquote", "hr", "tr", "td" -> true
        else -> false
    }
}
```

## 2.2 修改：`NavParser.kt` 改为“只取 toc nav + 构建层级”

把 `parse(navFile, opf)` 改签名并替换实现为：

```kotlin
package com.example.reader.engines.epub.internal.parser

import android.util.Xml
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.OutlineNode
import java.io.File
import java.io.FileInputStream
import org.xmlpull.v1.XmlPullParser

internal object NavParser {

    fun parse(navFile: File, opf: OpfPackage, navRelPath: String): List<OutlineNode> {
        FileInputStream(navFile).use { input ->
            val p = Xml.newPullParser()
            p.setInput(input, "UTF-8")

            var event = p.eventType

            // 只解析 epub:type="toc" 的 nav；找不到就 fallback 解析第一个 nav
            var inNav = false
            var navDepth = 0
            var navIsToc: Boolean? = null

            // 用栈构建 <ol><li> 层级
            data class Frame(val nodes: MutableList<OutlineNode>)
            val stack = ArrayDeque<Frame>()
            stack.addLast(Frame(mutableListOf()))

            var pendingHref: String? = null
            var pendingText: String? = null

            while (event != XmlPullParser.END_DOCUMENT) {
                if (event == XmlPullParser.START_TAG) {
                    val name = p.name.substringAfter(':')
                    when (name.lowercase()) {
                        "nav" -> {
                            inNav = true
                            navDepth = p.depth
                            val type = (p.getAttributeValue(null, "type")
                                ?: p.getAttributeValue("http://www.idpf.org/2007/ops", "type"))
                                ?.lowercase()
                            navIsToc = (type == "toc")
                        }
                        "ol" -> if (inNav && acceptNav(navIsToc)) {
                            // 新层级
                            stack.addLast(Frame(mutableListOf()))
                        }
                        "a" -> if (inNav && acceptNav(navIsToc)) {
                            pendingHref = p.getAttributeValue(null, "href")
                            pendingText = readInnerText(p)?.trim()
                        }
                    }
                } else if (event == XmlPullParser.END_TAG) {
                    val name = p.name.substringAfter(':')
                    when (name.lowercase()) {
                        "nav" -> {
                            if (inNav && p.depth == navDepth) {
                                // 优先 toc nav，若不是 toc 则继续扫描后续 nav
                                if (navIsToc == true) break
                                inNav = false
                                navIsToc = null
                            }
                        }
                        "ol" -> if (inNav && acceptNav(navIsToc)) {
                            // 收拢一层
                            val finished = stack.removeLast().nodes
                            val parent = stack.last().nodes
                            if (finished.isNotEmpty()) {
                                // 把这一层挂到 parent 的最后一个节点 children（符合 ol/li）
                                val last = parent.lastOrNull()
                                if (last != null) {
                                    parent[parent.lastIndex] = last.copy(children = finished)
                                } else {
                                    // 极端：ol 直接在 nav 下，没有 li/a，忽略
                                }
                            }
                        }
                        "li" -> {
                            // li 结束时，如果刚刚读到了 a，就落一个节点到当前层
                            if (inNav && acceptNav(navIsToc)) {
                                val href = pendingHref
                                val text = pendingText
                                if (!href.isNullOrBlank() && !text.isNullOrBlank()) {
                                    val resolved = PathResolver.resolveFrom(navRelPath, href)
                                    val frag = PathResolver.fragmentOf(href)
                                    val locator = Locator(
                                        LocatorSchemes.EPUB_CFI,
                                        "href:$resolved#${frag}".trimEnd('#')
                                    )
                                    stack.last().nodes += OutlineNode(title = text, locator = locator)
                                }
                                pendingHref = null
                                pendingText = null
                            }
                        }
                    }
                }

                event = p.next()
            }

            return stack.first().nodes
        }
    }

    private fun acceptNav(navIsToc: Boolean?): Boolean =
        navIsToc != false // null 表示还不知道：先解析着，遇到 toc 时会提前 break

    private fun readInnerText(p: XmlPullParser): String? {
        var event = p.next()
        if (event == XmlPullParser.TEXT) {
            val t = p.text
            while (!(event == XmlPullParser.END_TAG && p.name.substringAfter(':').equals("a", true))) {
                event = p.next()
            }
            return t
        }
        return null
    }
}
```

## 2.3 修改：`NcxParser.kt` 构建层级 + 相对路径基准正确

把 `parse(ncxFile, opf)` 改签名并替换核心实现为：

```kotlin
package com.example.reader.engines.epub.internal.parser

import android.util.Xml
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.OutlineNode
import java.io.File
import java.io.FileInputStream
import org.xmlpull.v1.XmlPullParser

internal object NcxParser {

    fun parse(ncxFile: File, opf: OpfPackage, ncxRelPath: String): List<OutlineNode> {
        FileInputStream(ncxFile).use { input ->
            val p = Xml.newPullParser()
            p.setInput(input, "UTF-8")

            data class NodeBuilder(
                var title: String? = null,
                var src: String? = null,
                val children: MutableList<OutlineNode> = mutableListOf()
            )

            val stack = ArrayDeque<NodeBuilder>()
            val roots = mutableListOf<OutlineNode>()

            var event = p.eventType
            while (event != XmlPullParser.END_DOCUMENT) {
                if (event == XmlPullParser.START_TAG) {
                    val name = p.name.substringAfter(':')
                    when (name) {
                        "navPoint" -> stack.addLast(NodeBuilder())
                        "text" -> stack.lastOrNull()?.title = readText(p)
                        "content" -> stack.lastOrNull()?.src = p.getAttributeValue(null, "src")
                    }
                } else if (event == XmlPullParser.END_TAG) {
                    val name = p.name.substringAfter(':')
                    if (name == "navPoint") {
                        val b = stack.removeLast()
                        val t = b.title?.trim().orEmpty()
                        val src = b.src?.trim().orEmpty()

                        val node = if (t.isNotBlank() && src.isNotBlank()) {
                            val resolved = PathResolver.resolveFrom(ncxRelPath, src)
                            val frag = PathResolver.fragmentOf(src)
                            val locator = Locator(
                                LocatorSchemes.EPUB_CFI,
                                "href:$resolved#${frag}".trimEnd('#')
                            )
                            OutlineNode(t, locator, b.children.toList())
                        } else {
                            null
                        }

                        if (node != null) {
                            if (stack.isEmpty()) roots += node
                            else stack.last().children += node
                        }
                    }
                }
                event = p.next()
            }

            return roots
        }
    }

    private fun readText(p: XmlPullParser): String? {
        var event = p.next()
        if (event == XmlPullParser.TEXT) {
            val t = p.text
            while (!(event == XmlPullParser.END_TAG && p.name.substringAfter(':').equals("text", true))) {
                event = p.next()
            }
            return t
        }
        return null
    }
}
```
