下面给你一套**可直接落地的 EPUB 引擎实现代码（MVP：章节级渲染 + WebView HTML 注入 + TOC + 搜索/文本/资源 + 内存标注）**，完全对齐你当前的 `core/reader/api` 契约。

> 这版实现的“渲染”输出是 `RenderContent.Html(inlineHtml, baseUri)`：
> feature 的 WebView 只要用 `loadDataWithBaseURL(baseUri, inlineHtml, ...)` 即可。
> 资源（图片/css/font）通过一个 `ContentProvider` 走 `content://` 访问解压目录。

## 2) ContentProvider：让 WebView 能加载图片/CSS/字体

### `engines/epub/src/main/java/com/example/reader/engines/epub/EpubContentProvider.kt`

```kotlin
package com.example.reader.engines.epub

import android.content.ContentProvider
import android.content.ContentValues
import android.content.UriMatcher
import android.database.Cursor
import android.net.Uri
import android.os.ParcelFileDescriptor
import android.webkit.MimeTypeMap
import java.io.File
import java.io.FileNotFoundException

/**
 * URI 形如：
 * content://<authority>/epub/<docId>/file/<relativePath...>
 */
class EpubContentProvider : ContentProvider() {

    override fun onCreate(): Boolean = true

    override fun getType(uri: Uri): String? {
        val file = resolveFile(uri) ?: return null
        val ext = file.extension.lowercase()
        return MimeTypeMap.getSingleton().getMimeTypeFromExtension(ext)
            ?: when (ext) {
                "xhtml", "html", "htm" -> "application/xhtml+xml"
                "css" -> "text/css"
                "svg" -> "image/svg+xml"
                "ttf" -> "font/ttf"
                "otf" -> "font/otf"
                "woff" -> "font/woff"
                "woff2" -> "font/woff2"
                else -> "application/octet-stream"
            }
    }

    override fun openFile(uri: Uri, mode: String): ParcelFileDescriptor {
        val file = resolveFile(uri) ?: throw FileNotFoundException("No such file: $uri")
        if (!file.exists() || !file.isFile) throw FileNotFoundException("No such file: $uri")
        return ParcelFileDescriptor.open(file, ParcelFileDescriptor.MODE_READ_ONLY)
    }

    private fun resolveFile(uri: Uri): File? {
        // content://authority/epub/<docId>/file/<relative...>
        val seg = uri.pathSegments
        if (seg.size < 4) return null
        if (seg[0] != "epub") return null
        val docId = seg[1]
        if (seg[2] != "file") return null

        val rel = seg.subList(3, seg.size).joinToString("/")
        val root = EpubStore.rootDir(docId) ?: return null

        // 防 zip-slip/路径穿越：必须落在 root 下
        val target = File(root, rel)
        val rootCanon = root.canonicalFile
        val targetCanon = target.canonicalFile
        if (!targetCanon.path.startsWith(rootCanon.path)) return null

        return targetCanon
    }

    // Not used
    override fun query(uri: Uri, projection: Array<out String>?, selection: String?, selectionArgs: Array<out String>?, sortOrder: String?): Cursor? = null
    override fun insert(uri: Uri, values: ContentValues?): Uri? = null
    override fun delete(uri: Uri, selection: String?, selectionArgs: Array<out String>?): Int = 0
    override fun update(uri: Uri, values: ContentValues?, selection: String?, selectionArgs: Array<out String>?): Int = 0

    companion object {
        /**
         * 你可以在 app 里用 manifestPlaceholders 覆盖它：
         * authority = "${applicationId}.epub"
         */
        const val DEFAULT_AUTHORITY = "com.example.reader.epub"

        fun buildFileUri(authority: String, docId: String, relativePath: String): Uri {
            val clean = relativePath.trimStart('/')

            val builder = Uri.Builder()
                .scheme("content")
                .authority(authority)
                .appendPath("epub")
                .appendPath(docId)
                .appendPath("file")

            clean.split('/').filter { it.isNotBlank() }.forEach { builder.appendPath(it) }
            return builder.build()
        }
    }
}
```

## 3) 打开/解压/解析（container.xml + opf + toc）

### `engines/epub/src/main/java/com/example/reader/engines/epub/internal/open/EpubOpener.kt`

```kotlin
package com.example.reader.engines.epub.internal.open

import android.content.Context
import com.example.reader.api.engine.ReaderDocument
import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.open.OpenOptions
import com.example.reader.engines.epub.EpubContentProvider
import com.example.reader.engines.epub.EpubStore
import com.example.reader.engines.epub.internal.parser.ContainerParser
import com.example.reader.engines.epub.internal.parser.NcxParser
import com.example.reader.engines.epub.internal.parser.NavParser
import com.example.reader.engines.epub.internal.parser.OpfParser
import com.example.reader.engines.epub.internal.render.EpubDocument
import com.example.reader.model.DocumentId
import com.example.reader.source.DocumentSource
import java.io.File
import java.io.IOException
import java.security.MessageDigest
import java.util.zip.ZipException
import kotlinx.coroutines.CancellationException
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

internal class EpubOpener(
    private val context: Context,
    private val authority: String = EpubContentProvider.DEFAULT_AUTHORITY,
) {
    suspend fun open(source: DocumentSource, options: OpenOptions): ReaderResult<ReaderDocument> =
        withContext(Dispatchers.IO) {
            try {
                val docId = DocumentId(hashId(source))
                val baseDir = File(context.cacheDir, "epub/${docId.value}")
                val extractedDir = File(baseDir, "extracted")
                val marker = File(baseDir, ".complete")

                if (!marker.exists()) {
                    baseDir.mkdirs()
                    extractedDir.mkdirs()
                    ZipExtract.extractTo(source, extractedDir)
                    marker.writeText("ok")
                }

                // 解析 container.xml -> opf path
                val containerXml = File(extractedDir, "META-INF/container.xml")
                if (!containerXml.exists()) {
                    return@withContext ReaderResult.Err(ReaderError.CorruptOrInvalid("Missing container.xml"))
                }
                val opfPath = ContainerParser.parseOpfPath(containerXml)
                    ?: return@withContext ReaderResult.Err(ReaderError.CorruptOrInvalid("Invalid container.xml"))

                val opfFile = File(extractedDir, opfPath)
                if (!opfFile.exists()) {
                    return@withContext ReaderResult.Err(ReaderError.CorruptOrInvalid("Missing OPF: $opfPath"))
                }

                val opf = OpfParser.parse(opfFile)

                // toc：优先 nav.xhtml，其次 ncx
                val toc =
                    opf.navPath?.let { navRel ->
                        val navFile = File(extractedDir, navRel)
                        if (navFile.exists()) NavParser.parse(navFile, opf) else emptyList()
                    }.orEmpty().ifEmpty {
                        opf.ncxPath?.let { ncxRel ->
                            val ncxFile = File(extractedDir, ncxRel)
                            if (ncxFile.exists()) NcxParser.parse(ncxFile, opf) else emptyList()
                        }.orEmpty()
                    }

                val container = EpubContainer(
                    id = docId,
                    rootDir = extractedDir,
                    authority = authority,
                    opf = opf,
                    outline = toc
                )

                // 注册给 ContentProvider
                EpubStore.register(docId.value, extractedDir)

                ReaderResult.Ok(EpubDocument(container, options))
            } catch (t: Throwable) {
                ReaderResult.Err(t.toReaderError())
            }
        }

    private fun hashId(source: DocumentSource): String {
        val raw = buildString {
            append(source.uri.toString())
            append("|")
            append(source.sizeBytes ?: -1)
            append("|")
            append(source.displayName ?: "")
        }
        val md = MessageDigest.getInstance("SHA-256").digest(raw.toByteArray())
        return md.joinToString("") { "%02x".format(it) }.take(32)
    }

    private fun Throwable.toReaderError(): ReaderError =
        when (this) {
            is ReaderError -> this
            is CancellationException -> ReaderError.Cancelled(cause = this)
            is SecurityException -> ReaderError.PermissionDenied(cause = this)
            is ZipException -> ReaderError.CorruptOrInvalid(cause = this)
            is IOException -> ReaderError.Io(cause = this)
            else -> ReaderError.Internal(cause = this)
        }
}
```

### `engines/epub/src/main/java/com/example/reader/engines/epub/internal/open/ZipExtract.kt`

```kotlin
package com.example.reader.engines.epub.internal.open

import com.example.reader.source.DocumentSource
import java.io.BufferedInputStream
import java.io.File
import java.io.FileOutputStream
import java.util.zip.ZipInputStream
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

internal object ZipExtract {

    suspend fun extractTo(source: DocumentSource, outDir: File) = withContext(Dispatchers.IO) {
        source.openInputStream().use { raw ->
            ZipInputStream(BufferedInputStream(raw)).use { zis ->
                while (true) {
                    val entry = zis.nextEntry ?: break
                    val name = entry.name

                    // 防 Zip Slip
                    val target = File(outDir, name)
                    val outCanon = outDir.canonicalFile
                    val targetCanon = target.canonicalFile
                    if (!targetCanon.path.startsWith(outCanon.path)) continue

                    if (entry.isDirectory) {
                        targetCanon.mkdirs()
                    } else {
                        targetCanon.parentFile?.mkdirs()
                        FileOutputStream(targetCanon).use { fos ->
                            zis.copyTo(fos)
                        }
                    }
                    zis.closeEntry()
                }
            }
        }
    }
}
```

```kotlin
import android.util.Xml
import java.io.File
import java.io.FileInputStream
import org.xmlpull.v1.XmlPullParser

internal object ContainerParser {
    fun parseOpfPath(containerXml: File): String? {
        FileInputStream(containerXml).use { input ->
            val p = Xml.newPullParser()
            p.setInput(input, "UTF-8")

            var event = p.eventType
            while (event != XmlPullParser.END_DOCUMENT) {
                if (event == XmlPullParser.START_TAG && p.name == "rootfile") {
                    val fullPath = p.getAttributeValue(null, "full-path")
                    if (!fullPath.isNullOrBlank()) return fullPath
                }
                event = p.next()
            }
            return null
        }
    }
}
```

### `internal/parser/OpfParser.kt`

```kotlin
package com.example.reader.engines.epub.internal.parser

import android.util.Xml
import com.example.reader.model.DocumentMetadata
import java.io.File
import java.io.FileInputStream
import org.xmlpull.v1.XmlPullParser

internal data class ManifestItem(
    val id: String,
    val href: String,
    val mediaType: String?,
    val properties: String?
)

internal data class SpineItem(
    val idref: String,
    val href: String,
    val mediaType: String?
)

internal data class OpfPackage(
    val metadata: DocumentMetadata,
    val manifest: Map<String, ManifestItem>,
    val spine: List<SpineItem>,
    val opfDir: String,
    val navPath: String?,
    val ncxPath: String?
) {
    fun resolveHref(href: String): String {
        val base = if (opfDir.isBlank()) "" else opfDir.trimEnd('/') + "/"
        return normalizePath(base + href)
    }
}

internal object OpfParser {

    fun parse(opfFile: File): OpfPackage {
        val opfDir = opfFile.parentFile?.let { it.relativeToOrSelf(opfFile.parentFile).path } ?: ""
        // 上面这行在 extractedDir 里路径不可靠，这里改成“相对 extracted 根”的 dir：
        val dirRel = opfFile.parentFile?.path?.substringAfterLast("/extracted/") ?: ""
        val opfDirRel = dirRel.trim('/')

        val manifest = linkedMapOf<String, ManifestItem>()
        val spine = mutableListOf<SpineItem>()

        var title: String? = null
        var creator: String? = null
        var language: String? = null
        var identifier: String? = null

        FileInputStream(opfFile).use { input ->
            val p = Xml.newPullParser()
            p.setInput(input, "UTF-8")

            var event = p.eventType
            while (event != XmlPullParser.END_DOCUMENT) {
                when {
                    event == XmlPullParser.START_TAG && p.name == "title" -> title = readText(p)
                    event == XmlPullParser.START_TAG && (p.name == "creator" || p.name == "author") -> creator = readText(p)
                    event == XmlPullParser.START_TAG && p.name == "language" -> language = readText(p)
                    event == XmlPullParser.START_TAG && p.name == "identifier" -> identifier = readText(p)

                    event == XmlPullParser.START_TAG && p.name == "item" -> {
                        val id = p.getAttributeValue(null, "id") ?: ""
                        val href = p.getAttributeValue(null, "href") ?: ""
                        val mediaType = p.getAttributeValue(null, "media-type")
                        val properties = p.getAttributeValue(null, "properties")
                        if (id.isNotBlank() && href.isNotBlank()) {
                            manifest[id] = ManifestItem(id, href, mediaType, properties)
                        }
                    }

                    event == XmlPullParser.START_TAG && p.name == "itemref" -> {
                        val idref = p.getAttributeValue(null, "idref") ?: ""
                        if (idref.isNotBlank()) {
                            // 等下二次用 manifest 解析 href
                            // 这里只保存 idref
                            spine += SpineItem(idref = idref, href = "", mediaType = null)
                        }
                    }
                }
                event = p.next()
            }
        }

        // 计算 nav / ncx
        val nav = manifest.values.firstOrNull { it.properties?.contains("nav") == true }?.let { it.href }
        val ncx = manifest.values.firstOrNull { it.mediaType == "application/x-dtbncx+xml" }?.let { it.href }

        val metadata = DocumentMetadata(
            title = title?.takeIf { it.isNotBlank() },
            author = creator?.takeIf { it.isNotBlank() },
            language = language?.takeIf { it.isNotBlank() },
            identifier = identifier?.takeIf { it.isNotBlank() }
        )

        // 二次补全 spine href
        val spine2 = spine.mapNotNull { s ->
            val item = manifest[s.idref] ?: return@mapNotNull null
            val resolved = normalizePath(
                (if (opfDirRel.isBlank()) "" else "$opfDirRel/") + item.href
            )
            SpineItem(idref = s.idref, href = resolved, mediaType = item.mediaType)
        }

        val pkg = OpfPackage(
            metadata = metadata,
            manifest = manifest.toMap(),
            spine = spine2,
            opfDir = opfDirRel,
            navPath = nav?.let { normalizePath((if (opfDirRel.isBlank()) "" else "$opfDirRel/") + it) },
            ncxPath = ncx?.let { normalizePath((if (opfDirRel.isBlank()) "" else "$opfDirRel/") + it) }
        )

        return pkg
    }

    private fun readText(p: XmlPullParser): String? {
        // 读取当前 START_TAG 的文本
        var event = p.next()
        if (event == XmlPullParser.TEXT) {
            val text = p.text
            // 消费到 END_TAG
            while (event != XmlPullParser.END_TAG) event = p.next()
            return text
        }
        return null
    }
}

internal fun normalizePath(path: String): String {
    val parts = mutableListOf<String>()
    path.split('/').forEach { seg ->
        when (seg) {
            "", "." -> Unit
            ".." -> if (parts.isNotEmpty()) parts.removeAt(parts.size - 1)
            else -> parts.add(seg)
        }
    }
    return parts.joinToString("/")
}
```

### `internal/parser/NavParser.kt`（简化：抓 TOC 的 `<a href>`）

```kotlin
package com.example.reader.engines.epub.internal.parser

import android.util.Xml
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.OutlineNode
import java.io.File
import java.io.FileInputStream
import org.xmlpull.v1.XmlPullParser

/**
 * 简化实现：
 * - 扫描所有 <a href="...">Text</a>
 * - 作为扁平 TOC（children 为空）
 *
 * 生产可加强：只取 epub:type="toc" 的 nav、并构建层级（ol/li）。
 */
internal object NavParser {

    fun parse(navFile: File, opf: OpfPackage): List<OutlineNode> {
        val nodes = mutableListOf<OutlineNode>()

        FileInputStream(navFile).use { input ->
            val p = Xml.newPullParser()
            p.setInput(input, "UTF-8")

            var event = p.eventType
            while (event != XmlPullParser.END_DOCUMENT) {
                if (event == XmlPullParser.START_TAG && p.name == "a") {
                    val href = p.getAttributeValue(null, "href")
                    val title = readInnerText(p)?.trim().orEmpty()
                    if (!href.isNullOrBlank() && title.isNotBlank()) {
                        val rel = opf.resolveHref(href.substringBefore('#'))
                        val locator = Locator(
                            scheme = LocatorSchemes.EPUB_CFI,
                            value = "href:$rel#${href.substringAfter('#', "")}".trimEnd('#')
                        )
                        nodes += OutlineNode(title = title, locator = locator)
                    }
                }
                event = p.next()
            }
        }
        return nodes
    }

    private fun readInnerText(p: XmlPullParser): String? {
        var event = p.next()
        if (event == XmlPullParser.TEXT) {
            val t = p.text
            // 消费到 </a>
            while (!(event == XmlPullParser.END_TAG && p.name == "a")) event = p.next()
            return t
        }
        return null
    }
}
```

### `internal/parser/NcxParser.kt`（简化：抓 navPoint）

```kotlin
package com.example.reader.engines.epub.internal.parser

import android.util.Xml
import com.example.reader.model.Locator
import com.example.reader.model.LocatorSchemes
import com.example.reader.model.OutlineNode
import java.io.File
import java.io.FileInputStream
import org.xmlpull.v1.XmlPullParser

internal object NcxParser {

    fun parse(ncxFile: File, opf: OpfPackage): List<OutlineNode> {
        val nodes = mutableListOf<OutlineNode>()

        FileInputStream(ncxFile).use { input ->
            val p = Xml.newPullParser()
            p.setInput(input, "UTF-8")

            var currentTitle: String? = null
            var currentSrc: String? = null

            var event = p.eventType
            while (event != XmlPullParser.END_DOCUMENT) {
                if (event == XmlPullParser.START_TAG) {
                    when (p.name) {
                        "text" -> currentTitle = readText(p)
                        "content" -> currentSrc = p.getAttributeValue(null, "src")
                    }
                }
                if (event == XmlPullParser.END_TAG && p.name == "navPoint") {
                    val title = currentTitle?.trim().orEmpty()
                    val src = currentSrc
                    if (title.isNotBlank() && !src.isNullOrBlank()) {
                        val href = src.substringBefore('#')
                        val frag = src.substringAfter('#', "")
                        val rel = opf.resolveHref(href)
                        val locator = Locator(LocatorSchemes.EPUB_CFI, "href:$rel#${frag}".trimEnd('#'))
                        nodes += OutlineNode(title, locator)
                    }
                    currentTitle = null
                    currentSrc = null
                }
                event = p.next()
            }
        }
        return nodes
    }

    private fun readText(p: XmlPullParser): String? {
        var event = p.next()
        if (event == XmlPullParser.TEXT) {
            val t = p.text
            while (!(event == XmlPullParser.END_TAG && p.name == "text")) event = p.next()
            return t
        }
        return null
    }
}
```

```kotlin
import kotlinx.coroutines.Dispatchers

import kotlinx.coroutines.flow.Flow

import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asSharedFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.withContext

internal class EpubController(
    private val container: EpubContainer,
    initialLocator: Locator?,
    initialConfig: RenderConfig.ReflowText,
    private val annotationProvider: AnnotationProvider?
) : ReaderController, Closeable {

    private val scope = CoroutineScope(SupervisorJob() + Dispatchers.Main.immediate)

    private var constraints: LayoutConstraints? = null
    private var config: RenderConfig.ReflowText = initialConfig

    private var spineIndex: Int = resolveInitialIndex(initialLocator)

    private val _events = MutableSharedFlow<ReaderEvent>(extraBufferCapacity = 16)
    override val events: Flow<ReaderEvent> = _events.asSharedFlow()

    private val _state = MutableStateFlow(buildState())
    override val state: StateFlow<RenderState> = _state.asStateFlow()

    override suspend fun setLayoutConstraints(constraints: LayoutConstraints): ReaderResult<Unit> {
        this.constraints = constraints
        _state.value = buildState()
        return ReaderResult.Ok(Unit)
    }

    override suspend fun setConfig(config: RenderConfig): ReaderResult<Unit> {
        val reflow = (config as? RenderConfig.ReflowText)
            ?: return ReaderResult.Err(ReaderError.Internal("EPUB only supports ReflowText config"))
        this.config = reflow
        _state.value = buildState()
        return ReaderResult.Ok(Unit)
    }

    override suspend fun render(policy: RenderPolicy): ReaderResult<RenderPage> =
        renderCurrent(policy, emittedAsNavigation = false)

    override suspend fun next(policy: RenderPolicy): ReaderResult<RenderPage> {
        if (spineIndex >= container.spineCount - 1) return render(policy)
        spineIndex++
        _state.value = buildState()
        _events.tryEmit(ReaderEvent.PageChanged(_state.value.locator))
        return renderCurrent(policy, emittedAsNavigation = true)
    }

    override suspend fun prev(policy: RenderPolicy): ReaderResult<RenderPage> {
        if (spineIndex <= 0) return render(policy)
        spineIndex--
        _state.value = buildState()
        _events.tryEmit(ReaderEvent.PageChanged(_state.value.locator))
        return renderCurrent(policy, emittedAsNavigation = true)
    }

    override suspend fun goTo(locator: Locator, policy: RenderPolicy): ReaderResult<RenderPage> {
        val idx = resolveLocatorToIndex(locator)
            ?: return ReaderResult.Err(ReaderError.CorruptOrInvalid("Invalid locator: $locator"))
        spineIndex = idx.coerceIn(0, container.spineCount - 1)
        _state.value = buildState()
        _events.tryEmit(ReaderEvent.PageChanged(_state.value.locator))
        return renderCurrent(policy, emittedAsNavigation = true)
    }

    override suspend fun goToProgress(percent: Double, policy: RenderPolicy): ReaderResult<RenderPage> {
        val p = percent.coerceIn(0.0, 1.0)
        val idx = (p * (container.spineCount - 1)).roundToInt().coerceIn(0, container.spineCount - 1)
        spineIndex = idx
        _state.value = buildState()
        _events.tryEmit(ReaderEvent.PageChanged(_state.value.locator))
        return renderCurrent(policy, emittedAsNavigation = true)
    }

    override suspend fun prefetchNeighbors(count: Int): ReaderResult<Unit> {
        // MVP：不做。增强版可以预读/预解析相邻章节
        return ReaderResult.Ok(Unit)
    }

    override suspend fun invalidate(reason: InvalidateReason): ReaderResult<Unit> {
        _state.value = buildState()
        return ReaderResult.Ok(Unit)
    }

    private suspend fun renderCurrent(policy: RenderPolicy, emittedAsNavigation: Boolean): ReaderResult<RenderPage> {
        return withContext(Dispatchers.IO) {
            try {
                val t0 = System.nanoTime()

                val chapterUri = container.spineUri(spineIndex)
                val inline = HtmlComposer.compose(
                    container = container,
                    spineIndex = spineIndex,
                    config = config
                )

                val decorations: List<Decoration> = annotationProvider?.let { ap ->
                    // 章节级：用一个“章节范围”查询，后续分页/CFI 再细化
                    val range = LocatorRange(
                        start = Locator(LocatorSchemes.EPUB_CFI, "spine:$spineIndex"),
                        end = Locator(LocatorSchemes.EPUB_CFI, "spine:$spineIndex")
                    )
                    ap.decorationsFor(AnnotationQuery(range = range)).fold(
                        onOk = { it },
                        onErr = { emptyList() }
                    )
                } ?: emptyList()

                val t1 = System.nanoTime()
                val metrics = RenderMetrics(
                    renderTimeMs = ((t1 - t0) / 1_000_000L).toLong(),
                    cacheHit = false
                )

                val page = RenderPage(
                    id = PageId("epub:${container.id.value}:spine:$spineIndex"),
                    locator = Locator(LocatorSchemes.EPUB_CFI, "spine:$spineIndex"),
                    content = RenderContent.Html(
                        inlineHtml = inline,
                        baseUri = chapterUri, // 关键：相对资源从章节 url 解析
                        contentUri = null
                    ),
                    links = emptyList(),
                    decorations = decorations,
                    metrics = metrics
                )

                _events.tryEmit(ReaderEvent.Rendered(page.id, metrics))
                ReaderResult.Ok(page)
            } catch (t: Throwable) {
                _events.tryEmit(ReaderEvent.Error(t))
                ReaderResult.Err(ReaderError.Internal(cause = t))
            }
        }
    }

    private fun buildState(): RenderState {
        val total = container.spineCount.coerceAtLeast(1)
        val current = spineIndex.coerceIn(0, total - 1) + 1
        val percent = (current - 1).toDouble() / (total - 1).coerceAtLeast(1)

        return RenderState(
            locator = Locator(LocatorSchemes.EPUB_CFI, "spine:$spineIndex"),
            progression = Progression(
                percent = percent,
                label = "$current/$total",
                current = current,
                total = total
            ),
            nav = NavigationAvailability(
                canGoPrev = spineIndex > 0,
                canGoNext = spineIndex < total - 1
            ),
            titleInView = null,
            config = config
        )
    }

    private fun resolveInitialIndex(locator: Locator?): Int {
        if (locator == null) return 0
        return resolveLocatorToIndex(locator) ?: 0
    }

    private fun resolveLocatorToIndex(locator: Locator): Int? {
        if (locator.scheme != LocatorSchemes.EPUB_CFI) return null
        val v = locator.value
        return when {
            v.startsWith("spine:") -> v.removePrefix("spine:").toIntOrNull()
            v.startsWith("href:") -> {
                val href = v.removePrefix("href:").substringBefore('#')
                container.opf.spine.indexOfFirst { it.href == href }.takeIf { it >= 0 }
            }
            else -> null // 这里可扩展解析 epubcfi(...)
        }
    }

    override fun close() {
        runCatching { scope.coroutineContext[SupervisorJob]?.cancel() }
    }
}
```

### `internal/render/HtmlComposer.kt`

```kotlin
package com.example.reader.engines.epub.internal.render

import com.example.reader.api.render.RenderConfig
import com.example.reader.engines.epub.internal.open.EpubContainer
import java.io.File

internal object HtmlComposer {

    fun compose(
        container: EpubContainer,
        spineIndex: Int,
        config: RenderConfig.ReflowText
    ): String {
        val path = container.spinePath(spineIndex)
        val file = File(container.rootDir, path)
        val raw = file.readTextSafely()

        val body = extractBodyInner(raw) ?: raw

        val css = buildCss(config)

        // 用 base href 解决相对资源路径
        val baseHref = container.spineUri(spineIndex).toString()

        return """
            <!doctype html>
            <html>
              <head>
                <meta charset="utf-8"/>
                <meta name="viewport" content="width=device-width, initial-scale=1.0"/>
                <base href="$baseHref"/>
                <style>$css</style>
              </head>
              <body>
                <div class="reader-content">
                  $body
                </div>
              </body>
            </html>
        """.trimIndent()
    }

    private fun buildCss(cfg: RenderConfig.ReflowText): String {
        // 你可以把 dp/sp 换算放到 feature 侧更准确，这里做 MVP 近似即可
        val fontSizePx = cfg.fontSizeSp * 1.0f
        val lineHeight = cfg.lineHeightMult
        val padding = cfg.pagePaddingDp

        return """
            html, body { height: 100%; }
            body {
              margin: 0;
              padding: ${padding}px;
              font-size: ${fontSizePx}px;
              line-height: ${lineHeight};
              word-break: break-word;
              overflow-wrap: anywhere;
            }
            img { max-width: 100%; height: auto; }
            a { color: inherit; }
        """.trimIndent()
    }

    private fun extractBodyInner(html: String): String? {
        val lower = html.lowercase()
        val i = lower.indexOf("<body")
        if (i < 0) return null
        val start = lower.indexOf(">", i)
        if (start < 0) return null
        val end = lower.lastIndexOf("</body>")
        if (end < 0 || end <= start) return null
        return html.substring(start + 1, end)
    }

    private fun File.readTextSafely(): String =
        runCatching { readText() }.getOrDefault("")
}
```

```kotlin
package com.example.reader.engines.epub.internal.provider

import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.TextProvider
import com.example.reader.engines.epub.internal.open.EpubContainer
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import java.io.File
import java.util.concurrent.ConcurrentHashMap
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

internal class EpubTextProvider(
    private val container: EpubContainer
) : TextProvider {

    private val cache = ConcurrentHashMap<Int, String>()

    override suspend fun getText(range: LocatorRange): ReaderResult<String> {
        val idx = resolveIndex(range.start) ?: return ReaderResult.Err(ReaderError.CorruptOrInvalid("Invalid range"))
        val text = loadChapterText(idx)
        return ReaderResult.Ok(text)
    }

    override suspend fun getTextAround(locator: Locator, maxChars: Int): ReaderResult<String> {
        val idx = resolveIndex(locator) ?: return ReaderResult.Err(ReaderError.CorruptOrInvalid("Invalid locator"))
        val text = loadChapterText(idx)
        return ReaderResult.Ok(text.take(maxChars))
    }

    private fun resolveIndex(locator: Locator): Int? {
        if (locator.scheme != LocatorSchemes.EPUB_CFI) return null
        val v = locator.value
        return when {
            v.startsWith("spine:") -> v.removePrefix("spine:").toIntOrNull()
            v.startsWith("href:") -> {
                val href = v.removePrefix("href:").substringBefore('#')
                container.opf.spine.indexOfFirst { it.href == href }.takeIf { it >= 0 }
            }
            else -> null
        }
    }

    private suspend fun loadChapterText(idx: Int): String = withContext(Dispatchers.IO) {
        cache[idx] ?: run {
            val path = container.spinePath(idx)
            val file = File(container.rootDir, path)
            val raw = runCatching { file.readText() }.getOrDefault("")
            val plain = stripHtml(raw)
            cache[idx] = plain
            plain
        }
    }

    private fun stripHtml(html: String): String {
        // MVP：简单去标签。生产可用更强的 HTML/XHTML 解析。
        return html
            .replace(Regex("<script[\\s\\S]*?</script>", RegexOption.IGNORE_CASE), " ")
            .replace(Regex("<style[\\s\\S]*?</style>", RegexOption.IGNORE_CASE), " ")
            .replace(Regex("<[^>]+>"), " ")
            .replace("&nbsp;", " ")
            .replace(Regex("\\s+"), " ")
            .trim()
    }
}
```

### `internal/provider/EpubSearchProvider.kt`

```kotlin
package com.example.reader.engines.epub.internal.provider

import com.example.reader.api.provider.SearchHit
import com.example.reader.api.provider.SearchOptions
import com.example.reader.api.provider.SearchProvider
import com.example.reader.engines.epub.internal.open.EpubContainer
import com.example.reader.model.Locator
import com.example.reader.model.LocatorRange
import com.example.reader.model.LocatorSchemes
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.flow
import kotlinx.coroutines.withContext

internal class EpubSearchProvider(
    private val container: EpubContainer,
    private val textProvider: EpubTextProvider
) : SearchProvider {

    override fun search(query: String, options: SearchOptions): Flow<SearchHit> = flow {
        if (query.isBlank()) return@flow
        val q = if (options.caseSensitive) query else query.lowercase()

        val startIndex = options.startFrom?.let { loc ->
            if (loc.scheme == LocatorSchemes.EPUB_CFI && loc.value.startsWith("spine:")) {
                loc.value.removePrefix("spine:").toIntOrNull()
            } else null
        } ?: 0

        var hits = 0

        for (i in startIndex until container.spineCount) {
            val text = withContext(Dispatchers.IO) {
                val t = textProvider.getTextAround(
                    Locator(LocatorSchemes.EPUB_CFI, "spine:$i"),
                    maxChars = Int.MAX_VALUE
                )
                t.fold(onOk = { it }, onErr = { "" })
            }

            val hay = if (options.caseSensitive) text else text.lowercase()

            var from = 0
            while (hits < options.maxHits) {
                val idx = hay.indexOf(q, startIndex = from)
                if (idx < 0) break

                val end = idx + q.length
                val excerpt = buildExcerpt(text, idx, end)

                emit(
                    SearchHit(
                        range = LocatorRange(
                            start = Locator(LocatorSchemes.EPUB_CFI, "spine:$i", extras = mapOf("charStart" to idx.toString())),
                            end = Locator(LocatorSchemes.EPUB_CFI, "spine:$i", extras = mapOf("charEnd" to end.toString()))
                        ),
                        excerpt = excerpt,
                        sectionTitle = null
                    )
                )

                hits++
                from = end
            }

            if (hits >= options.maxHits) break
        }
    }

    private fun buildExcerpt(text: String, start: Int, end: Int): String {
        val left = (start - 40).coerceAtLeast(0)
        val right = (end + 40).coerceAtMost(text.length)
        return text.substring(left, right).trim()
    }
}
```

```kotlin
package com.example.reader.engines.epub.internal.provider

import com.example.reader.api.annotation.Decoration
import com.example.reader.api.error.ReaderError
import com.example.reader.api.error.ReaderResult
import com.example.reader.api.provider.AnnotationProvider
import com.example.reader.api.provider.AnnotationQuery
import com.example.reader.model.LocatorRange
import com.example.reader.model.annotation.Annotation
import com.example.reader.model.annotation.AnnotationDraft
import com.example.reader.model.annotation.AnnotationId
import com.example.reader.model.annotation.AnnotationAnchor
import java.util.UUID
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.asStateFlow

internal class EpubAnnotationProvider : AnnotationProvider {

    private val state = MutableStateFlow<List<Annotation>>(emptyList())

    override fun observeAll(): Flow<List<Annotation>> = state.asStateFlow()

    override suspend fun listAll(): ReaderResult<List<Annotation>> = ReaderResult.Ok(state.value)

    override suspend fun query(query: AnnotationQuery): ReaderResult<List<Annotation>> {
        val all = state.value
        val filtered = when {
            query.range != null -> all.filter { it.anchor is AnnotationAnchor.ReflowRange }
            else -> all
        }
        return ReaderResult.Ok(filtered)
    }

    override suspend fun create(draft: AnnotationDraft): ReaderResult<Annotation> {
        val now = System.currentTimeMillis()
        val ann = Annotation(
            id = AnnotationId(UUID.randomUUID().toString()),
            type = draft.type,
            anchor = draft.anchor,
            content = draft.content,
            style = draft.style,
            createdAtEpochMs = now,
            updatedAtEpochMs = now,
            extra = draft.extra
        )
        state.value = state.value + ann
        return ReaderResult.Ok(ann)
    }

    override suspend fun update(annotation: Annotation): ReaderResult<Unit> {
        val list = state.value.toMutableList()
        val i = list.indexOfFirst { it.id == annotation.id }
        if (i < 0) return ReaderResult.Err(ReaderError.NotFound("Annotation not found"))
        list[i] = annotation.copy(updatedAtEpochMs = System.currentTimeMillis())
        state.value = list
        return ReaderResult.Ok(Unit)
    }

    override suspend fun delete(id: AnnotationId): ReaderResult<Unit> {
        state.value = state.value.filterNot { it.id == id }
        return ReaderResult.Ok(Unit)
    }

    override suspend fun decorationsFor(query: AnnotationQuery): ReaderResult<List<Decoration>> {
        val decos = state.value.mapNotNull { ann ->
            when (val a = ann.anchor) {
                is AnnotationAnchor.ReflowRange ->
                    Decoration.Reflow(range = a.range, style = ann.style)
                is AnnotationAnchor.FixedRects ->
                    null
            }
        }
        return ReaderResult.Ok(decos)
    }
}
```
