## 2.5 `domain/usecase/OpenReaderSession.kt`

> **关键：不用 `runtime.openSession`，而是 `openDocument + capabilities -> 选 config -> createSession`**
> 这样能在 session 创建前拿到正确 RenderConfig（避免先分页后再变 config 的开销/闪动）。

```kotlin
package com.ireader.feature.reader.domain.usecase

import com.ireader.core.files.source.DocumentSource
import com.ireader.feature.reader.domain.ReaderSettingsRepository
import com.ireader.reader.api.error.ReaderResult
import com.ireader.reader.api.error.flatMap
import com.ireader.reader.api.open.OpenOptions
import com.ireader.reader.api.render.RenderConfig
import com.ireader.reader.runtime.ReaderSessionHandle
import com.ireader.reader.runtime.ReaderRuntime
import javax.inject.Inject
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

class OpenReaderSession @Inject constructor(
    private val runtime: ReaderRuntime,
    private val settings: ReaderSettingsRepository
) {
    suspend operator fun invoke(
        source: DocumentSource,
        options: OpenOptions,
        initialLocator: com.ireader.reader.model.Locator?
    ): ReaderResult<ReaderSessionHandle> = withContext(Dispatchers.IO) {
        runtime.openDocument(source, options).flatMap { document ->
            val config: RenderConfig = if (document.capabilities.fixedLayout) {
                settings.getFixedConfig()
            } else {
                settings.getReflowConfig()
            }

            val sessionResult = document.createSession(
                initialLocator = initialLocator,
                initialConfig = config
            )

            when (sessionResult) {
                is ReaderResult.Ok -> ReaderResult.Ok(ReaderSessionHandle(document, sessionResult.value))
                is ReaderResult.Err -> {
                    runCatching { document.close() }
                    ReaderResult.Err(sessionResult.error)
                }
            }
        }
    }
}
```

## 3.1 `navigation/ReaderRoute.kt`

```kotlin
package com.ireader.feature.reader.navigation

import android.net.Uri
import androidx.navigation.NavBackStackEntry
import androidx.navigation.NavType
import androidx.navigation.navArgument

object ReaderRoute {
    const val route = "reader"

    const val ARG_BOOK_ID = "bookId"
    const val ARG_LOCATOR = "locator"

    // reader/{bookId}?locator={locator}
    const val pattern = "$route/{$ARG_BOOK_ID}?$ARG_LOCATOR={$ARG_LOCATOR}"

    val arguments = listOf(
        navArgument(ARG_BOOK_ID) { type = NavType.StringType },
        navArgument(ARG_LOCATOR) {
            type = NavType.StringType
            nullable = true
            defaultValue = null
        }
    )

    fun create(bookId: String, locator: String? = null): String {
        val encoded = locator?.let(Uri::encode)
        return if (encoded == null) {
            "$route/$bookId"
        } else {
            "$route/$bookId?$ARG_LOCATOR=$encoded"
        }
    }

    fun bookId(entry: NavBackStackEntry): String =
        requireNotNull(entry.arguments?.getString(ARG_BOOK_ID)) { "Missing $ARG_BOOK_ID" }

    fun locatorArg(entry: NavBackStackEntry): String? =
        entry.arguments?.getString(ARG_LOCATOR)?.let(Uri::decode)
}
```

## `presentation/ReaderViewModel.kt`

```kotlin

import androidx.lifecycle.ViewModel
import androidx.lifecycle.viewModelScope
import com.ireader.feature.reader.domain.LocatorCodec
import com.ireader.feature.reader.domain.ReaderBookRepository

import com.ireader.feature.reader.domain.ReaderSettingsRepository
import com.ireader.feature.reader.domain.usecase.ObserveEffectiveConfig
import com.ireader.feature.reader.domain.usecase.OpenReaderSession
import com.ireader.feature.reader.domain.usecase.SaveReadingProgress
import com.ireader.reader.api.error.ReaderError
import com.ireader.reader.api.error.ReaderResult
import com.ireader.reader.api.render.RenderPolicy
import com.ireader.reader.api.render.ReaderEvent
import com.ireader.reader.model.DocumentCapabilities

import com.ireader.reader.api.render.RenderConfig
import com.ireader.reader.runtime.ReaderSessionHandle
import com.ireader.reader.runtime.flow.asReaderResult
import dagger.hilt.android.lifecycle.HiltViewModel
import javax.inject.Inject
import java.util.concurrent.atomic.AtomicLong
import kotlinx.coroutines.CancellationException
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.Job
import kotlinx.coroutines.channels.Channel
import kotlinx.coroutines.flow.MutableSharedFlow
import kotlinx.coroutines.flow.MutableStateFlow
import kotlinx.coroutines.flow.StateFlow
import kotlinx.coroutines.flow.asSharedFlow
import kotlinx.coroutines.flow.asStateFlow
import kotlinx.coroutines.flow.collectLatest
import kotlinx.coroutines.flow.debounce
import kotlinx.coroutines.flow.distinctUntilChanged
import kotlinx.coroutines.flow.filterIsInstance
import kotlinx.coroutines.flow.launchIn
import kotlinx.coroutines.flow.map
import kotlinx.coroutines.flow.onEach
import kotlinx.coroutines.launch
import kotlinx.coroutines.supervisorScope
import kotlinx.coroutines.sync.Mutex
import kotlinx.coroutines.sync.withLock
import kotlinx.coroutines.withContext

@HiltViewModel
class ReaderViewModel @Inject constructor(
    private val bookRepo: ReaderBookRepository,
    private val progressRepo: ReaderProgressRepository,
    private val settingsRepo: ReaderSettingsRepository,
    private val locatorCodec: LocatorCodec,
    private val openReaderSession: OpenReaderSession,
    private val observeEffectiveConfig: ObserveEffectiveConfig,
    private val saveProgress: SaveReadingProgress,
    private val errorMapper: ReaderUiErrorMapper
) : ViewModel() {

    private val _state = MutableStateFlow(ReaderUiState())
    val state: StateFlow<ReaderUiState> = _state.asStateFlow()

    private val _effects = MutableSharedFlow<ReaderEffect>(extraBufferCapacity = 16)
    val effects = _effects.asSharedFlow()

    private val intents = Channel<ReaderIntent>(capacity = Channel.UNLIMITED)

    private var handle: ReaderSessionHandle? = null

    private var sessionJob: Job? = null
    private var finalRenderJob: Job? = null

    private var lastLayout: com.ireader.reader.api.render.LayoutConstraints? = null

    private val navMutex = Mutex()

    private val renderToken = AtomicLong(0L)

    init {
        viewModelScope.launch {
            for (intent in intents) {
                handleIntent(intent)
            }
        }
    }

    fun dispatch(intent: ReaderIntent) {
        intents.trySend(intent)
    }

    // ---------- Intent handling ----------

    private suspend fun handleIntent(intent: ReaderIntent) {
        when (intent) {
            is ReaderIntent.Start -> {
                if (state.value.bookId == intent.bookId && handle != null) return
                open(bookId = intent.bookId, locatorArg = intent.locatorArg, password = null)
            }

            is ReaderIntent.SubmitPassword -> {
                val bookId = state.value.bookId
                open(bookId = bookId, locatorArg = null, password = intent.password)
            }

            ReaderIntent.CancelPassword -> {
                _state.update { it.copy(passwordPrompt = null) }
                _effects.tryEmit(ReaderEffect.Back)
            }

            is ReaderIntent.LayoutChanged -> {
                lastLayout = intent.constraints
                handle?.let { h ->
                    // layout 改变时：通常需要重新 render
                    viewModelScope.launch {
                        h.controller.setLayoutConstraints(intent.constraints)
                        renderCurrent()
                    }
                }
            }

            ReaderIntent.ToggleChrome -> {
                _state.update { it.copy(chromeVisible = !it.chromeVisible) }
            }

            ReaderIntent.CloseSheet -> {
                _state.update { it.copy(sheet = ReaderSheet.None) }
            }

            ReaderIntent.OpenToc -> {
                _state.update { it.copy(sheet = ReaderSheet.Toc) }
                loadTocIfNeeded()
            }

            ReaderIntent.OpenSearch -> {
                _state.update { it.copy(sheet = ReaderSheet.Search) }
            }

            ReaderIntent.OpenSettings -> {
                _state.update { it.copy(sheet = ReaderSheet.Settings) }
                // currentConfig 已在 renderState / settings flow 中维护
            }

            ReaderIntent.Next -> navigate { controller, policy -> controller.next(policy) }
            ReaderIntent.Prev -> navigate { controller, policy -> controller.prev(policy) }

            is ReaderIntent.GoTo -> navigate { controller, policy -> controller.goTo(intent.locator, policy) }

            is ReaderIntent.GoToProgress -> navigate { controller, policy -> controller.goToProgress(intent.percent, policy) }

            is ReaderIntent.UpdateConfig -> {
                applyConfig(intent.config, persist = intent.persist)
            }

            ReaderIntent.RetryOpen -> {
                open(bookId = state.value.bookId, locatorArg = null, password = null)
            }

            ReaderIntent.RefreshPage -> {
                renderCurrent()
            }
        }
    }

    // ---------- Open / Session lifecycle ----------

    private suspend fun open(bookId: String, locatorArg: String?, password: String?) {
        // 取消旧 session
        closeSession()

        _state.update {
            it.copy(
                bookId = bookId,
                isOpening = true,
                error = null,
                passwordPrompt = null,
                page = null,
                renderState = null,
                capabilities = null,
                title = null,
                sheet = ReaderSheet.None
            )
        }

        val source = withContext(Dispatchers.IO) { bookRepo.getSource(bookId) }
        if (source == null) {
            _state.update {
                it.copy(
                    isOpening = false,
                    error = ReaderUiError(
                        message = UiText.Dynamic("找不到书籍源（bookId=$bookId）"),
                        actionLabel = UiText.Dynamic("返回"),
                        action = ReaderErrorAction.Back,
                        debugCode = "NO_SOURCE"
                    )
                )
            }
            return
        }

        val initialLocator: Locator? = withContext(Dispatchers.IO) {
            locatorArg?.let(locatorCodec::decode) ?: progressRepo.getLastLocator(bookId)
        }

        val options = com.ireader.reader.api.open.OpenOptions(
            password = password
        )

        when (val r = openReaderSession(source, options, initialLocator)) {
            is ReaderResult.Err -> {
                _state.update { it.copy(isOpening = false) }
                handleOpenError(r.error, lastTriedPassword = password)
            }

            is ReaderResult.Ok -> {
                handle = r.value
                _state.update {
                    it.copy(
                        isOpening = false,
                        capabilities = r.value.document.capabilities
                    )
                }

                // title：优先 document.metadata，其次 bookRepo displayTitle
                loadTitle(r.value)

                // collectors
                startSessionCollectors(r.value)

                // apply layout if already known
                lastLayout?.let { layout ->
                    viewModelScope.launch {
                        r.value.controller.setLayoutConstraints(layout)
                    }
                }

                // bind surface: 由 UI 负责调用 bindSurfaceIfPossible()
                // 初次渲染
                renderCurrent()
            }
        }
    }

    private fun handleOpenError(error: ReaderError, lastTriedPassword: String?) {
        if (error is ReaderError.InvalidPassword) {
            _state.update {
                it.copy(passwordPrompt = PasswordPrompt(lastTried = lastTriedPassword))
            }
            return
        }

        _state.update {
            it.copy(error = errorMapper.map(error))
        }
    }

    private fun startSessionCollectors(h: ReaderSessionHandle) {
        sessionJob?.cancel()
        finalRenderJob?.cancel()

        sessionJob = viewModelScope.launch {
            supervisorScope {
                // 1) controller.state
                launch {
                    h.controller.state.collect { st ->
                        _state.update { s ->
                            s.copy(
                                renderState = st,
                                currentConfig = st.config
                            )
                        }
                    }
                }

                // 2) controller.events -> progress save (debounce)
                val locatorFlow = h.controller.events
                    .filterIsInstance<ReaderEvent.PageChanged>()
                    .map { it.locator }
                    .distinctUntilChanged()

                launch {
                    locatorFlow
                        .debounce(600)
                        .collectLatest { locator ->
                            val st = h.controller.state.value
                            try {
                                saveProgress(
                                    bookId = state.value.bookId,
                                    documentId = h.document.id,
                                    locator = locator,
                                    progression = st.progression
                                )
                            } catch (_: CancellationException) {
                                // ignore
                            } catch (t: Throwable) {
                                _effects.tryEmit(ReaderEffect.Snackbar(UiText.Dynamic("保存进度失败")))
                            }
                        }
                }

                // 3) controller.events -> errors
                launch {
                    h.controller.events
                        .filterIsInstance<ReaderEvent.Error>()
                        .collect { ev ->
                            _state.update { it.copy(error = ReaderUiError(UiText.Dynamic("渲染错误：${ev.throwable.message ?: "unknown"}"))) }
                        }
                }

                // 4) settings flow -> apply config + render
                launch {
                    observeEffectiveConfig(h.document.capabilities)
                        .distinctUntilChanged()
                        .collectLatest { cfg ->
                            // 只要 DataStore 推送变化，就应用并重渲染
                            applyConfig(cfg, persist = false)
                        }
                }
            }
        }
    }

    private fun loadTitle(h: ReaderSessionHandle) {
        viewModelScope.launch(Dispatchers.IO) {
            val metadataTitle = runCatching {
                when (val r = h.document.metadata()) {
                    is ReaderResult.Ok -> {
                        // 这里假设 DocumentMetadata 有 title 字段；如果你模型字段名不同，改这里即可
                        val m = r.value
                        val titleProp = runCatching {
                            val k = m::class.members.firstOrNull { it.name == "title" }
                            k?.call(m) as? String
                        }.getOrNull()
                        titleProp
                    }
                    is ReaderResult.Err -> null
                }
            }.getOrNull()

            val fallback = runCatching { bookRepo.getDisplayTitle(state.value.bookId) }.getOrNull()

            val title = metadataTitle ?: fallback
            if (!title.isNullOrBlank()) {
                _state.update { it.copy(title = title) }
            }
        }
    }

    private suspend fun closeSession() {
        finalRenderJob?.cancel()
        finalRenderJob = null

        sessionJob?.cancel()
        sessionJob = null

        val old = handle
        handle = null
        if (old != null) {
            withContext(Dispatchers.IO) {
                runCatching { old.close() }
            }
        }
    }

    // ---------- Rendering ----------

    private suspend fun renderCurrent() {
        val h = handle ?: return
        val policy = policyFor(h.document.capabilities, quality = RenderPolicy.Quality.FINAL)
        // fixedLayout -> 两阶段：先 DRAFT 再 FINAL
        if (h.document.capabilities.fixedLayout) {
            twoPhaseRender(currentLocator = null)
        } else {
            when (val r = h.controller.render(policy)) {
                is ReaderResult.Ok -> _state.update { it.copy(page = r.value, error = null) }
                is ReaderResult.Err -> _state.update { it.copy(error = errorMapper.map(r.error)) }
            }
        }
    }

    private suspend fun navigate(
        block: suspend (controller: com.ireader.reader.api.render.ReaderController, policy: RenderPolicy) -> ReaderResult<com.ireader.reader.api.render.RenderPage>
    ) {
        val h = handle ?: return
        navMutex.withLock {
            val caps = h.document.capabilities
            if (caps.fixedLayout) {
                // fixed: 先 next/prev/goTo 用 DRAFT，随后再 FINAL 精渲
                val draftPolicy = policyFor(caps, quality = RenderPolicy.Quality.DRAFT)
                when (val r = block(h.controller, draftPolicy)) {
                    is ReaderResult.Ok -> {
                        _state.update { it.copy(page = r.value, error = null) }
                        twoPhaseRender(currentLocator = h.controller.state.value.locator)
                    }
                    is ReaderResult.Err -> _state.update { it.copy(error = errorMapper.map(r.error)) }
                }
            } else {
                val policy = policyFor(caps, quality = RenderPolicy.Quality.FINAL)
                when (val r = block(h.controller, policy)) {
                    is ReaderResult.Ok -> _state.update { it.copy(page = r.value, error = null) }
                    is ReaderResult.Err -> _state.update { it.copy(error = errorMapper.map(r.error)) }
                }
            }
        }
    }

    private fun policyFor(caps: DocumentCapabilities, quality: RenderPolicy.Quality): RenderPolicy {
        val prefetch = if (caps.fixedLayout) 2 else 1
        return RenderPolicy(
            quality = quality,
            allowCache = true,
            prefetchNeighbors = prefetch
        )
    }

    private fun twoPhaseRender(currentLocator: Locator?) {
        val h = handle ?: return
        val token = renderToken.incrementAndGet()

        finalRenderJob?.cancel()
        finalRenderJob = viewModelScope.launch {
            // 先 DRAFT 立即出
            _state.update { it.copy(isRenderingFinal = true) }
            val draftPolicy = policyFor(h.document.capabilities, RenderPolicy.Quality.DRAFT)
            val draftRes = h.controller.render(draftPolicy)
            if (renderToken.get() != token) return@launch
            if (draftRes is ReaderResult.Ok) {
                _state.update { it.copy(page = draftRes.value, error = null) }
            }

            // 再 FINAL 精渲（如果期间 locator 变化则丢弃）
            val expected = currentLocator ?: h.controller.state.value.locator
            val finalPolicy = policyFor(h.document.capabilities, RenderPolicy.Quality.FINAL)

            val finalRes = h.controller.render(finalPolicy)
            if (renderToken.get() != token) return@launch

            val nowLocator = h.controller.state.value.locator
            if (nowLocator != expected) return@launch

            when (finalRes) {
                is ReaderResult.Ok -> _state.update { it.copy(page = finalRes.value, error = null, isRenderingFinal = false) }
                is ReaderResult.Err -> _state.update { it.copy(error = errorMapper.map(finalRes.error), isRenderingFinal = false) }
            }
        }
    }

    private suspend fun applyConfig(config: RenderConfig, persist: Boolean) {
        val h = handle ?: run {
            // 还未打开时也可以存设置（例如从别处进入）
            if (persist) persistConfig(config)
            _state.update { it.copy(currentConfig = config) }
            return
        }

        // persist first（避免“应用了但没保存”）
        if (persist) persistConfig(config)

        // apply to controller
        val setRes = h.controller.setConfig(config)
        when (setRes) {
            is ReaderResult.Ok -> {
                _state.update { it.copy(currentConfig = config) }
                // config change usually needs rerender
                renderCurrent()
            }
            is ReaderResult.Err -> _effects.tryEmit(ReaderEffect.Snackbar(UiText.Dynamic("应用设置失败")))
        }
    }

    private suspend fun persistConfig(config: RenderConfig) {
        withContext(Dispatchers.IO) {
            when (config) {
                is RenderConfig.ReflowText -> settingsRepo.updateReflowConfig(config)
                is RenderConfig.FixedPage -> settingsRepo.updateFixedConfig(config)
            }
        }
    }

    // ---------- TOC / Search ----------

    private fun loadTocIfNeeded() {
        val h = handle ?: return
        if (state.value.toc.items.isNotEmpty()) return

        val outlineProvider = h.outline
        if (outlineProvider == null) {
            _state.update { it.copy(toc = it.toc.copy(error = UiText.Dynamic("该文档不支持目录"))) }
            return
        }

        viewModelScope.launch {
            _state.update { it.copy(toc = it.toc.copy(isLoading = true, error = null)) }

            when (val r = outlineProvider.getOutline()) {
                is ReaderResult.Err -> {
                    _state.update { it.copy(toc = it.toc.copy(isLoading = false, error = UiText.Dynamic("加载目录失败"))) }
                }
                is ReaderResult.Ok -> {
                    val items = flattenOutline(r.value, depth = 0)
                    _state.update { it.copy(toc = it.toc.copy(isLoading = false, items = items, error = null)) }
                }
            }
        }
    }

    private fun flattenOutline(nodes: List<OutlineNode>, depth: Int): List<TocItem> {
        val out = mutableListOf<TocItem>()
        for (n in nodes) {
            val title = runCatching {
                val k = n::class.members.firstOrNull { it.name == "title" }
                (k?.call(n) as? String).orEmpty()
            }.getOrElse { "" }

            val locator = runCatching {
                val k = n::class.members.firstOrNull { it.name == "locator" }
                k?.call(n) as? Locator
            }.getOrNull()

            if (locator != null) {
                out += TocItem(
                    title = title.ifBlank { "Untitled" },
                    locatorEncoded = locatorCodec.encode(locator),
                    depth = depth
                )
            }

            val children = runCatching {
                val k = n::class.members.firstOrNull { it.name == "children" }
                @Suppress("UNCHECKED_CAST")
                (k?.call(n) as? List<OutlineNode>).orEmpty()
            }.getOrElse { emptyList() }

            if (children.isNotEmpty()) {
                out += flattenOutline(children, depth + 1)
            }
        }
        return out
    }

    fun startSearch(query: String) {
        val h = handle ?: return
        val searchProvider = h.search
        if (searchProvider == null) {
            _state.update { it.copy(search = it.search.copy(error = UiText.Dynamic("该文档不支持搜索"))) }
            return
        }

        viewModelScope.launch {
            _state.update {
                it.copy(search = it.search.copy(query = query, isSearching = true, results = emptyList(), error = null))
            }

            val options = SearchOptions(
                maxHits = 300,
                startFrom = null
            )

            // Flow<SearchHit> 可能抛异常，用 asReaderResult 包一层
            searchProvider.search(query, options)
                .asReaderResult()
                .onEach { rr ->
                    when (rr) {
                        is ReaderResult.Err -> {
                            _state.update { it.copy(search = it.search.copy(isSearching = false, error = UiText.Dynamic("搜索失败"))) }
                        }
                        is ReaderResult.Ok -> {
                            val hit = rr.value
                            val locator = runCatching {
                                val r = hit.range
                                val k = r::class.members.firstOrNull { it.name == "start" || it.name == "from" }
                                k?.call(r) as? Locator
                            }.getOrNull() ?: return@onEach

                            val item = SearchResultItem(
                                title = hit.sectionTitle,
                                excerpt = hit.excerpt,
                                locatorEncoded = locatorCodec.encode(locator)
                            )

                            _state.update { st ->
                                val list = st.search.results + item
                                st.copy(search = st.search.copy(results = list))
                            }
                        }
                    }
                }
                .launchIn(this)
        }
    }

    // ---------- Public helpers for UI ----------

    fun decodeLocatorOrNull(encoded: String): Locator? = locatorCodec.decode(encoded)

    override fun onCleared() {
        // 最后再保存一次进度（尽量）
        val h = handle
        if (h != null) {
            val st = h.controller.state.value
            viewModelScope.launch {
                runCatching {
                    saveProgress(
                        bookId = state.value.bookId,
                        documentId = h.document.id,
                        locator = st.locator,
                        progression = st.progression
                    )
                }
            }
        }

        viewModelScope.launch {
            closeSession()
        }
        super.onCleared()
    }

    // StateFlow update helper
    private inline fun <T> MutableStateFlow<T>.update(transform: (T) -> T) {
        value = transform(value)
    }
}
```

> 这 VM 不是“点两下就完”的 MVP：
>
> * session 生命周期严格管理（cancel collectors + close handle）
> * fixedLayout 两阶段渲染（DRAFT→FINAL）
> * progress 保存节流（debounce）
> * settings flow 自动生效
> * TOC / search 全实现（search 具备异常保护）
> * 错误统一映射 + 密码弹窗流程
> * 所有操作串行（Mutex）避免 next/prev/goTo 重入导致状态错乱

```kotlin
fun ReaderScreen(
    bookId: String,
    locatorArg: String?,
    onBack: () -> Unit,
    onOpenAnnotations: (bookId: String) -> Unit,
    vm: ReaderViewModel = hiltViewModel()
) {
    val state by vm.state.collectAsStateWithLifecycle()

    // start
    LaunchedEffect(bookId, locatorArg) {
        vm.dispatch(ReaderIntent.Start(bookId = bookId, locatorArg = locatorArg))
    }

    // effects
    LaunchedEffect(Unit) {
        vm.effects.collectLatest { eff ->
            when (eff) {
                is ReaderEffect.Back -> onBack()
                is ReaderEffect.OpenAnnotations -> onOpenAnnotations(eff.bookId)
                is ReaderEffect.Snackbar -> {
                    // 交给 Scaffold 内部 snackbar host
                }
            }
        }
    }

    // Back behavior: sheet > chrome
    BackHandler(enabled = state.sheet != ReaderSheet.None) {
        vm.dispatch(ReaderIntent.CloseSheet)
    }
    BackHandler(enabled = state.sheet == ReaderSheet.None && !state.chromeVisible) {
        vm.dispatch(ReaderIntent.ToggleChrome)
    }

    BoxWithConstraints(Modifier.fillMaxSize()) {
        val density = LocalDensity.current
        val wPx = with(density) { maxWidth.toPx().toInt() }
        val hPx = with(density) { maxHeight.toPx().toInt() }

        LaunchedEffect(wPx, hPx, density.density, density.fontScale) {
            vm.dispatch(
                ReaderIntent.LayoutChanged(
                    com.ireader.reader.api.render.LayoutConstraints(
                        viewportWidthPx = wPx,
                        viewportHeightPx = hPx,
                        density = density.density,
                        fontScale = density.fontScale
                    )
                )
            )
        }

        // Surface container for Embedded engines
        AndroidView(
            factory = { ctx -> ReaderSurfaceLayout(ctx) },
            modifier = Modifier.fillMaxSize(),
            update = { /* surface stays */ }
        )

        ReaderScaffold(
            state = state,
            onBack = onBack,
            onOpenAnnotations = { onOpenAnnotations(state.bookId) },
            onIntent = vm::dispatch,
            onStartSearch = vm::startSearch,
            decodeLocator = vm::decodeLocatorOrNull
        )
    }
}
```

## 7.1 `ui/components/ErrorPane.kt`

```kotlin
package com.ireader.feature.reader.ui.components

import androidx.compose.foundation.layout.Arrangement
```

```kotlin
import androidx.compose.foundation.gestures.detectTapGestures
import androidx.compose.foundation.layout.Box

import com.ireader.feature.reader.presentation.ReaderUiState
import com.ireader.reader.api.render.RenderContent

@Composable
fun PageRenderer(
    state: ReaderUiState,
    onToggleChrome: () -> Unit,
    onNext: () -> Unit,
    onPrev: () -> Unit
) {
    val page = state.page

    Box(Modifier.fillMaxSize()) {
        if (page == null) return

        when (val content = page.content) {
            is RenderContent.Text -> TextPage(
                page = page,
                content = content
            )
            is RenderContent.BitmapPage -> BitmapPage(
                page = page,
                bitmap = content.bitmap
            )

                page = page,
                tiles = content
            )
            is RenderContent.Html -> HtmlPage(content)
            RenderContent.Embedded -> {
                // Embedded：引擎自己渲染到 bindSurface 的 ViewGroup 中
                // 这里不用画任何东西。
            }
        }

        // 仅对非 Html/Tiles/Embedded 启用 “点击分区翻页 + 中间切菜单”
        val enableTapOverlay = when (page.content) {
            is RenderContent.Text,
            is RenderContent.BitmapPage -> true
            else -> false
        }

        if (enableTapOverlay) {
            Box(
                Modifier
                    .fillMaxSize()
                    .pointerInput(Unit) {
                        detectTapGestures { offset ->
                            val w = size.width
                            val x = offset.x
                            when {
                                x < w * 0.25f -> onPrev()
                                x > w * 0.75f -> onNext()
                                else -> onToggleChrome()
                            }
                        }
                    }
            )
        }
    }
```

```kotlin
import androidx.compose.foundation.layout.Box
import androidx.compose.foundation.layout.fillMaxSize
import androidx.compose.foundation.rememberScrollState
import androidx.compose.foundation.text.selection.SelectionContainer
import androidx.compose.foundation.verticalScroll
import androidx.compose.material3.MaterialTheme
import androidx.compose.material3.Text
import androidx.compose.runtime.Composable
import androidx.compose.ui.Modifier
import androidx.compose.ui.text.SpanStyle
import androidx.compose.ui.text.buildAnnotatedString
import com.ireader.reader.api.annotation.Decoration
import com.ireader.reader.api.render.RenderContent
import com.ireader.reader.api.render.RenderPage

@Composable
fun TextPage(
    page: RenderPage,
    content: RenderContent.Text
) {
    val mapping = content.mapping
    val annotated = if (mapping != null && page.decorations.isNotEmpty()) {
        val base = content.text.toString()
        val highlights = page.decorations.filterIsInstance<Decoration.Reflow>()
        buildAnnotatedString {
            append(base)
            for (h in highlights) {
                val range = mapping.charRangeFor(h.range) ?: continue
                val start = range.first.coerceAtLeast(0)
                val end = (range.last + 1).coerceAtMost(base.length)
                if (start >= end) continue
                addStyle(
                    SpanStyle(background = MaterialTheme.colorScheme.secondary.copy(alpha = 0.25f)),
                    start = start,
                    end = end
                )
            }
        }
    } else {
        null
    }

    SelectionContainer {
        Box(
            modifier = Modifier
                .fillMaxSize()
                .verticalScroll(rememberScrollState())
        ) {
            Text(
                text = annotated ?: content.text,
                style = MaterialTheme.typography.bodyLarge
            )
        }
    }
}
```

## 7.7 `ui/components/TilesPage.kt`（重点：真正可用的 tile 渲染）

> 这是非 MVP 的关键：
>
> * 手势缩放/平移
> * 可见区域计算
> * tile grid（输出像素大小恒定，性能稳定）
> * LRU 缓存 + in-flight 去重
> * gesture 期间 DRAFT，停止后 FINAL

```kotlin
package com.ireader.feature.reader.ui.components

import android.graphics.Bitmap
import androidx.collection.LruCache
import androidx.compose.foundation.gestures.detectTransformGestures
import androidx.compose.foundation.layout.BoxWithConstraints
import androidx.compose.foundation.layout.fillMaxSize
import androidx.compose.runtime.*
import androidx.compose.ui.Modifier
import androidx.compose.ui.geometry.Offset
import androidx.compose.ui.graphics.asImageBitmap
import androidx.compose.ui.graphics.drawscope.drawImage
import androidx.compose.ui.input.pointer.pointerInput
import androidx.compose.ui.platform.LocalDensity
import androidx.compose.ui.unit.IntSize
import com.ireader.reader.api.annotation.Decoration
import com.ireader.reader.api.render.RenderContent
import com.ireader.reader.api.render.RenderPage
import com.ireader.reader.api.render.RenderPolicy
import com.ireader.reader.api.render.TileRequest
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.Job
import kotlinx.coroutines.delay
import kotlinx.coroutines.launch
import kotlin.math.ceil
import kotlin.math.floor
import kotlin.math.max
import kotlin.math.min

@Composable
fun TilesPage(
    page: RenderPage,
    tiles: RenderContent.Tiles
) {
    val provider = tiles.tileProvider

    // 当页面变化时清理资源
    DisposableEffect(page.id) {
        onDispose {
            runCatching { provider.close() }
        }
    }

    BoxWithConstraints(Modifier.fillMaxSize()) {
        val viewport = IntSize(
            width = with(LocalDensity.current) { maxWidth.toPx().toInt() },
            height = with(LocalDensity.current) { maxHeight.toPx().toInt() }
        )

        val pageW = tiles.pageWidthPx.toFloat()
        val pageH = tiles.pageHeightPx.toFloat()

        // 初始 fit width
        val baseScale = remember(viewport, pageW) {
            if (pageW <= 0f) 1f else (viewport.width / pageW).coerceIn(0.1f, 10f)
        }

        var zoom by remember { mutableStateOf(1f) }
        var offset by remember { mutableStateOf(Offset.Zero) }
        var isGesturing by remember { mutableStateOf(false) }

        // 让页面初始居中
        LaunchedEffect(viewport, baseScale) {
            val s = baseScale * zoom
            val contentW = pageW * s
            val contentH = pageH * s
            val dx = (viewport.width - contentW) / 2f
            val dy = (viewport.height - contentH) / 2f
            offset = Offset(dx, dy)
        }

        val scope = rememberCoroutineScope()

        // tile cache
        val cache = remember {
            object : LruCache<TileKey, Bitmap>(80) {
                override fun sizeOf(key: TileKey, value: Bitmap): Int = value.byteCount / 1024
            }
        }
        val bitmaps = remember { mutableStateMapOf<TileKey, Bitmap>() }
        val inFlight = remember { mutableStateMapOf<TileKey, Job>() }

        fun requestTile(key: TileKey) {
            if (bitmaps.containsKey(key)) return
            val cached = cache.get(key)
            if (cached != null) {
                bitmaps[key] = cached
                return
            }
            if (inFlight[key]?.isActive == true) return

            val job = scope.launch(Dispatchers.IO) {
                val bmp = provider.renderTile(
                    TileRequest(
                        leftPx = key.left,
                        topPx = key.top,
                        widthPx = key.width,
                        heightPx = key.height,
                        scale = key.scale,
                        quality = key.quality
                    )
                )
                cache.put(key, bmp)
                bitmaps[key] = bmp
            }
            inFlight[key] = job
        }

        // gesture -> update scale/offset
        val minZoom = 0.7f
        val maxZoom = 6.0f
        val tileOutputPx = 512f // 恒定输出像素 tile，性能稳定

        val scale = baseScale * zoom

        // 计算可见区域（page coords）
        fun visiblePageRect(): RectF {
            val left = (-offset.x / scale).coerceIn(0f, pageW)
            val top = (-offset.y / scale).coerceIn(0f, pageH)
            val right = ((viewport.width - offset.x) / scale).coerceIn(0f, pageW)
            val bottom = ((viewport.height - offset.y) / scale).coerceIn(0f, pageH)
            return RectF(left, top, right, bottom)
        }

        // gesture idle -> 切 FINAL
        LaunchedEffect(isGesturing) {
            if (isGesturing) return@LaunchedEffect
            // 停止 150ms 后刷新 FINAL tile
            delay(150)
            // 触发 recomposition -> tiles quality change
        }

        // render tiles list
        val quality = if (isGesturing) RenderPolicy.Quality.DRAFT else RenderPolicy.Quality.FINAL

        // 需要的 tile 集合
        val neededKeys = remember(offset, zoom, viewport, isGesturing, page.id) {
            val rect = visiblePageRect()
            val tilePageSize = (tileOutputPx / scale).coerceIn(64f, 2048f)

            val startX = floor(rect.left / tilePageSize).toInt()
            val endX = ceil(rect.right / tilePageSize).toInt()
            val startY = floor(rect.top / tilePageSize).toInt()
            val endY = ceil(rect.bottom / tilePageSize).toInt()

            val keys = mutableListOf<TileKey>()
            val s = (baseScale * zoom).coerceIn(0.05f, 20f)
            for (ty in startY..endY) {
                for (tx in startX..endX) {
                    val l = (tx * tilePageSize).toInt()
                    val t = (ty * tilePageSize).toInt()
                    val w = min(tilePageSize.toInt(), (pageW - l).toInt().coerceAtLeast(0))
                    val h = min(tilePageSize.toInt(), (pageH - t).toInt().coerceAtLeast(0))
                    if (w <= 0 || h <= 0) continue
                    keys += TileKey(
                        left = l,
                        top = t,
                        width = w,
                        height = h,
                        scale = s,
                        quality = quality
                    )
                }
            }
            keys
        }

        // 请求 tiles
        LaunchedEffect(neededKeys) {
            neededKeys.forEach { key -> requestTile(key) }
        }

        // 清理 inFlight
        LaunchedEffect(Unit) {
            while (true) {
                delay(800)
                val it = inFlight.iterator()
                while (it.hasNext()) {
                    val e = it.next()
                    if (e.value.isCompleted) it.remove()
                }
            }
        }

        androidx.compose.foundation.Canvas(
            modifier = Modifier
                .fillMaxSize()
                .pointerInput(Unit) {
                    detectTransformGestures(
                        onGestureStart = { isGesturing = true },
                        onGestureEnd = { isGesturing = false },
                        onGestureCancel = { isGesturing = false }
                    ) { centroid, pan, gestureZoom, _ ->
                        val oldZoom = zoom
                        val newZoom = (zoom * gestureZoom).coerceIn(minZoom, maxZoom)

                        // 以 centroid 为锚点缩放：保持缩放前后 centroid 对应的 page 点不变
                        val oldScale = baseScale * oldZoom
                        val newScale = baseScale * newZoom

                        val before = (centroid - offset) / oldScale
                        zoom = newZoom
                        val afterOffset = centroid - before * newScale

                        offset = afterOffset + pan
                    }
                }
        ) {
            // draw tiles
            for (key in neededKeys) {
                val bmp = bitmaps[key] ?: continue
                val dx = offset.x + key.left * scale
                val dy = offset.y + key.top * scale
                drawImage(
                    image = bmp.asImageBitmap(),
                    topLeft = Offset(dx, dy)
                )
            }

            // draw fixed decorations overlay (simple)
            val fixedDecos = page.decorations.filterIsInstance<Decoration.Fixed>()
            for (d in fixedDecos) {
                for (r in d.rects) {
                    // 假设 NormalizedRect 0..1
                    val left = offset.x + (r.left * pageW) * scale
                    val top = offset.y + (r.top * pageH) * scale
                    val right = offset.x + (r.right * pageW) * scale
                    val bottom = offset.y + (r.bottom * pageH) * scale
                    drawRect(
                        color = androidx.compose.ui.graphics.Color.Yellow.copy(alpha = 0.25f),
                        topLeft = Offset(left, top),
                        size = androidx.compose.ui.geometry.Size(
                            width = max(0f, right - left),
                            height = max(0f, bottom - top)
                        )
                    )
                }
            }
        }
    }
}

private data class TileKey(
    val left: Int,
    val top: Int,
    val width: Int,
    val height: Int,
    val scale: Float,
    val quality: RenderPolicy.Quality
)

private data class RectF(
    val left: Float,
    val top: Float,
    val right: Float,
    val bottom: Float
)
```

> TileRequest 的语义（这套实现的假设）：
>
> * `left/top/width/height` 是 **page 坐标**（以 pageWidthPx/pageHeightPx 为范围）
> * `scale` 是 **屏幕像素 / page 像素**（即当前显示倍率）
> * provider 返回 bitmap 的像素尺寸应等于 `width*scale` × `height*scale`（或非常接近），这样绘制 1:1 最清晰
>
> 这是 PDF tile 渲染最常见的设计，性能/清晰度都最好。

```kotlin
import com.ireader.reader.model.DocumentCapabilities

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun SettingsSheet(
    capabilities: DocumentCapabilities?,
    config: RenderConfig?,
    onClose: () -> Unit,
    onApply: (RenderConfig, persist: Boolean) -> Unit
) {
    val fixed = capabilities?.fixedLayout == true

    var persist by remember { mutableStateOf(true) }

    ModalBottomSheet(onDismissRequest = onClose) {
        Column(
            Modifier.fillMaxWidth().padding(16.dp),
            verticalArrangement = Arrangement.spacedBy(12.dp)
        ) {
            Text("阅读设置")

            Row(horizontalArrangement = Arrangement.spacedBy(8.dp)) {
                Checkbox(checked = persist, onCheckedChange = { persist = it })
                Text("保存为默认设置")
            }

            if (!fixed) {
                val cfg = (config as? RenderConfig.ReflowText) ?: RenderConfig.ReflowText()
                var font by remember(cfg.fontSizeSp) { mutableStateOf(cfg.fontSizeSp) }
                var line by remember(cfg.lineHeightMult) { mutableStateOf(cfg.lineHeightMult) }
                var pad by remember(cfg.pagePaddingDp) { mutableStateOf(cfg.pagePaddingDp) }

                Text("字体大小：${font.toInt()}sp")
                Slider(
                    value = font,
                    onValueChange = {
                        font = it
                        onApply(cfg.copy(fontSizeSp = it), persist = false) // 即时预览不持久化
                    },
                    valueRange = 12f..30f
                )

                Text("行距：${"%.2f".format(line)}")
                Slider(
                    value = line,
                    onValueChange = {
                        line = it
                        onApply(cfg.copy(lineHeightMult = it), persist = false)
                    },
                    valueRange = 1.1f..2.2f
                )

                Text("页边距：${pad.toInt()}dp")
                Slider(
                    value = pad,
                    onValueChange = {
                        pad = it
                        onApply(cfg.copy(pagePaddingDp = it), persist = false)
                    },
                    valueRange = 8f..36f
                )

                Row(horizontalArrangement = Arrangement.spacedBy(8.dp)) {
                    OutlinedButton(onClick = { onApply(cfg.copy(fontSizeSp = font, lineHeightMult = line, pagePaddingDp = pad), persist) }) {
                        Text("应用")
                    }
                    TextButton(onClick = onClose) { Text("关闭") }
                }
            } else {
                val cfg = (config as? RenderConfig.FixedPage) ?: RenderConfig.FixedPage()

                var zoom by remember(cfg.zoom) { mutableStateOf(cfg.zoom) }
                var rotation by remember(cfg.rotationDegrees) { mutableStateOf(cfg.rotationDegrees) }
                var fit by remember(cfg.fitMode) { mutableStateOf(cfg.fitMode) }

                Text("缩放：${"%.2f".format(zoom)}")
                Slider(
                    value = zoom,
                    onValueChange = {
                        zoom = it
                        onApply(cfg.copy(zoom = it), persist = false)
                    },
                    valueRange = 0.6f..4.0f
                )

                Text("旋转：$rotation°")
                Row(horizontalArrangement = Arrangement.spacedBy(8.dp)) {
                    listOf(0, 90, 180, 270).forEach { deg ->
                        TextButton(onClick = {
                            rotation = deg
                            onApply(cfg.copy(rotationDegrees = deg), persist = false)
                        }) { Text("$deg") }
                    }
                }

                var fitMenu by remember { mutableStateOf(false) }
                OutlinedButton(onClick = { fitMenu = true }) { Text("适配模式：$fit") }
                DropdownMenu(expanded = fitMenu, onDismissRequest = { fitMenu = false }) {
                    RenderConfig.FitMode.entries.forEach { m ->
                        DropdownMenuItem(
                            text = { Text(m.name) },
                            onClick = {
                                fit = m
                                fitMenu = false
                                onApply(cfg.copy(fitMode = m), persist = false)
                            }
                        )
                    }
                }

                Row(horizontalArrangement = Arrangement.spacedBy(8.dp)) {
                    OutlinedButton(onClick = {
                        onApply(cfg.copy(zoom = zoom, rotationDegrees = rotation, fitMode = fit), persist)
                    }) { Text("应用") }
                    TextButton(onClick = onClose) { Text("关闭") }
                }
            }
        }
    }
}
```

# 9) 你需要在 app 做的“适配绑定”（示例）

你很可能已经有：

* `core/data/BookRepo`
* `core/data/ProgressRepo`
* `core/datastore/ReaderSettingsStore`
* `Locator` 的序列化（或你愿意用 kotlinx.serialization）

你只要在 `app` 写类似：

```kotlin
@Module
@InstallIn(SingletonComponent::class)
abstract class ReaderAdaptersModule {

    @Binds abstract fun bindBookRepo(impl: BookRepoAdapter): ReaderBookRepository
    @Binds abstract fun bindProgressRepo(impl: ProgressRepoAdapter): ReaderProgressRepository
    @Binds abstract fun bindSettingsRepo(impl: SettingsRepoAdapter): ReaderSettingsRepository
    @Binds abstract fun bindLocatorCodec(impl: JsonLocatorCodec): LocatorCodec
}
```

Adapter 里直接委托 core/data 的真实 repo 即可（这不是 feature/reader 的事情）。

# 10) 这份实现已经满足“非 MVP / 专业级”的点

* **生命周期**：session/document 成对关闭；collector 取消；重入打开会先 close
* **并发控制**：翻页/跳转串行（Mutex），避免 render 状态错乱
* **固定版式（PDF）体验**：DRAFT→FINAL 两阶段渲染；Tiles 级渲染可用
* **性能**：Tiles 输出像素恒定 + LRU 缓存 + in-flight 去重
* **设置体系**：reflow/fixed 分开配置；即时预览 + 可持久化
* **目录/搜索**：完整 UI + 数据流；异常保护（Flow.asReaderResult）
* **错误体系**：ReaderError → UI 映射；InvalidPassword 走专用弹窗流程
