import androidx.datastore.preferences.core.Preferences
import androidx.datastore.preferences.core.booleanPreferencesKey
import androidx.datastore.preferences.core.edit
import androidx.datastore.preferences.core.floatPreferencesKey

import com.ireader.reader.api.render.RenderConfig
import javax.inject.Inject
import javax.inject.Singleton
import kotlinx.coroutines.flow.Flow
import kotlinx.coroutines.flow.distinctUntilChanged
import kotlinx.coroutines.flow.first
import kotlinx.coroutines.flow.map

@Singleton
class DatastoreReaderSettingsStore @Inject constructor(
    private val dataStore: DataStore<Preferences>
) : ReaderSettingsStore {

    private object Keys {
        object Reflow {
            val fontSizeSp = floatPreferencesKey("reader.reflow.fontSizeSp")
            val lineHeightMult = floatPreferencesKey("reader.reflow.lineHeightMult")
            val paragraphSpacingDp = floatPreferencesKey("reader.reflow.paragraphSpacingDp")
            val pagePaddingDp = floatPreferencesKey("reader.reflow.pagePaddingDp")
            val fontFamilyName = stringPreferencesKey("reader.reflow.fontFamilyName")
            val hyphenation = booleanPreferencesKey("reader.reflow.hyphenation")
        }

        object Fixed {
            val fitMode = stringPreferencesKey("reader.fixed.fitMode")
            val zoom = floatPreferencesKey("reader.fixed.zoom")
            val rotationDegrees = intPreferencesKey("reader.fixed.rotationDegrees")
        }
    }

    private val defaultReflow = RenderConfig.ReflowText()
    private val defaultFixed = RenderConfig.FixedPage()

    override val reflowConfig: Flow<RenderConfig.ReflowText> =
        dataStore.data.map { prefs ->
            defaultReflow.copy(
                fontSizeSp = prefs[Keys.Reflow.fontSizeSp] ?: defaultReflow.fontSizeSp,
                lineHeightMult = prefs[Keys.Reflow.lineHeightMult] ?: defaultReflow.lineHeightMult,
                paragraphSpacingDp = prefs[Keys.Reflow.paragraphSpacingDp] ?: defaultReflow.paragraphSpacingDp,
                pagePaddingDp = prefs[Keys.Reflow.pagePaddingDp] ?: defaultReflow.pagePaddingDp,
                fontFamilyName = prefs[Keys.Reflow.fontFamilyName] ?: defaultReflow.fontFamilyName,
                hyphenation = prefs[Keys.Reflow.hyphenation] ?: defaultReflow.hyphenation
            )
        }.distinctUntilChanged()

    override val fixedConfig: Flow<RenderConfig.FixedPage> =
        dataStore.data.map { prefs ->
            val fitName = prefs[Keys.Fixed.fitMode] ?: defaultFixed.fitMode.name
            val fit = runCatching { RenderConfig.FitMode.valueOf(fitName) }
                .getOrElse { defaultFixed.fitMode }

            defaultFixed.copy(
                fitMode = fit,
                zoom = prefs[Keys.Fixed.zoom] ?: defaultFixed.zoom,
                rotationDegrees = prefs[Keys.Fixed.rotationDegrees] ?: defaultFixed.rotationDegrees
            )
        }.distinctUntilChanged()

    override suspend fun getReflowConfig(): RenderConfig.ReflowText = reflowConfig.first()
    override suspend fun getFixedConfig(): RenderConfig.FixedPage = fixedConfig.first()

    override suspend fun setReflowConfig(config: RenderConfig.ReflowText) {
        dataStore.edit { prefs ->
            prefs[Keys.Reflow.fontSizeSp] = config.fontSizeSp
            prefs[Keys.Reflow.lineHeightMult] = config.lineHeightMult
            prefs[Keys.Reflow.paragraphSpacingDp] = config.paragraphSpacingDp
            prefs[Keys.Reflow.pagePaddingDp] = config.pagePaddingDp
            if (config.fontFamilyName == null) prefs.remove(Keys.Reflow.fontFamilyName)
            else prefs[Keys.Reflow.fontFamilyName] = config.fontFamilyName
            prefs[Keys.Reflow.hyphenation] = config.hyphenation
        }
    }

    override suspend fun setFixedConfig(config: RenderConfig.FixedPage) {
        dataStore.edit { prefs ->
            prefs[Keys.Fixed.fitMode] = config.fitMode.name
            prefs[Keys.Fixed.zoom] = config.zoom
            prefs[Keys.Fixed.rotationDegrees] = config.rotationDegrees
        }
    }
}

object ReaderRoute {
    const val route = "reader"
    const val ARG_BOOK_ID = "bookId"
    const val ARG_LOCATOR = "locator"

    const val pattern = "$route/{$ARG_BOOK_ID}?$ARG_LOCATOR={$ARG_LOCATOR}"

    val arguments = listOf(
        navArgument(ARG_BOOK_ID) { type = NavType.LongType },
        navArgument(ARG_LOCATOR) {
            type = NavType.StringType
            nullable = true
            defaultValue = null

        return if (encoded == null) "$route/$bookId" else "$route/$bookId?$ARG_LOCATOR=$encoded"
    }

        entry.arguments?.getLong(ARG_BOOK_ID)
            ?: error("Missing $ARG_BOOK_ID")

    fun locatorArg(entry: NavBackStackEntry): String? =
        entry.arguments?.getString(ARG_LOCATOR)?.let(Uri::decode)
}

## 3) BookDocumentSourceResolver：用 BookRepo + BookEntity 生成 DocumentSource（真正可打开）

这里是**唯一需要你按 BookEntity 实际字段名微调的地方**。
我按你 repo 方法签名合理假设 BookEntity 有：`sourceUri: String?`、`canonicalPath: String`、`format: BookFormat`、`title: String?`。
（基本所有阅读器都会有）

### `feature/reader/domain/BookDocumentSourceResolver.kt`

```kotlin
package com.ireader.feature.reader.domain

import android.content.Context
import android.net.Uri
import com.ireader.core.files.source.DocumentSource
import com.ireader.reader.model.BookFormat
import dagger.hilt.android.qualifiers.ApplicationContext
import java.io.File
import java.io.FileInputStream
import java.io.InputStream
import java.io.FileNotFoundException
import javax.inject.Inject
import javax.inject.Singleton

/**
 * 把 BookEntity -> DocumentSource。
 *
 * ✅ 专业建议：最终应迁移到 core/files 或 core/data 做统一工厂，这里先让 feature/reader 可运行。
 */
@Singleton
class BookDocumentSourceResolver @Inject constructor(
    @ApplicationContext private val context: Context
) {

    /**
     * book: BookEntity（类型由 BookRepo.getById 推断，不用 import）
     */
    fun resolve(
        sourceUri: String?,
        canonicalPath: String,
        displayName: String?,
        format: BookFormat?
    ): DocumentSource? {
        val uri = when {
            !sourceUri.isNullOrBlank() -> Uri.parse(sourceUri)
            canonicalPath.isNotBlank() -> Uri.fromFile(File(canonicalPath))
            else -> null
        } ?: return null

        val mimeType = when (format) {
            BookFormat.PDF -> "application/pdf"
            BookFormat.EPUB -> "application/epub+zip"
            BookFormat.TXT -> "text/plain"
            null -> null
        }

        return object : DocumentSource {
            override val displayName: String? = displayName
            override val mimeType: String? = mimeType

            override fun openInputStream(): InputStream {
                return when (uri.scheme) {
                    "content" -> context.contentResolver.openInputStream(uri)
                        ?: throw FileNotFoundException("Cannot open input stream: $uri")
                    "file" -> FileInputStream(File(requireNotNull(uri.path)))
                    else -> context.contentResolver.openInputStream(uri)
                        ?: throw FileNotFoundException("Cannot open input stream: $uri")
                }
            }
        }
    }
}
```

> 如果你的 DocumentSource 接口除了 `displayName/mimeType/openInputStream()` 还有别的方法，你在这个匿名对象里补齐即可（实现都可基于 uri + contentResolver）。

import kotlinx.coroutines.withContext

class OpenReaderSession @Inject constructor(
    private val runtime: ReaderRuntime,
    private val settings: ReaderSettingsStore
) {
    suspend operator fun invoke(
        source: DocumentSource,
        options: OpenOptions,
        initialLocator: com.ireader.reader.model.Locator?
    ): ReaderResult<ReaderSessionHandle> = withContext(Dispatchers.IO) {
        runtime.openDocument(source, options).flatMap { document ->
            val config: RenderConfig =
                if (document.capabilities.fixedLayout) settings.getFixedConfig()
                else settings.getReflowConfig()

                initialLocator = initialLocator,
                initialConfig = config
            )

            when (sessionResult) {
                is ReaderResult.Ok -> ReaderResult.Ok(ReaderSessionHandle(document, sessionResult.value))
                is ReaderResult.Err -> {
                    runCatching { document.close() }
                    ReaderResult.Err(sessionResult.error)
                }
            }
        }
    }
}

            ReaderIntent.OpenSettings -> _state.update { it.copy(sheet = ReaderSheet.Settings) }

            ReaderIntent.Next -> navigate { c, p -> c.next(p) }
            ReaderIntent.Prev -> navigate { c, p -> c.prev(p) }

            is ReaderIntent.GoTo -> navigate { c, p -> c.goTo(intent.locator, p) }
            is ReaderIntent.GoToProgress -> navigate { c, p -> c.goToProgress(intent.percent, p) }

            is ReaderIntent.UpdateConfig -> applyConfig(intent.config, persist = intent.persist)

            ReaderIntent.RetryOpen -> open(bookId = state.value.bookId, locatorArg = null, password = null)
            ReaderIntent.RefreshPage -> renderCurrent()
        }
    }

    // ---------- Open / Session lifecycle ----------

    private suspend fun open(bookId: Long, locatorArg: String?, password: String?) {
        closeSession()

        _state.update {
            it.copy(
                bookId = bookId,
                isOpening = true,
                error = null,
                passwordPrompt = null,
                page = null,
                renderState = null,
                capabilities = null,
                title = null,
                sheet = ReaderSheet.None
            )
        }

        val book = withContext(Dispatchers.IO) { bookRepo.getById(bookId) }
        if (book == null) {
            _state.update { it.copy(isOpening = false, error = ReaderUiError(UiText.Dynamic("书籍不存在：$bookId"))) }
            return
        }

        // ✅ 这里按你 BookEntity 字段名推断：sourceUri/canonicalPath/format/title
        val source = sourceResolver.resolve(
            sourceUri = book.sourceUri,
            canonicalPath = book.canonicalPath,
            displayName = book.title,
            format = book.format
        )

        if (source == null) {
            _state.update { it.copy(isOpening = false, error = ReaderUiError(UiText.Dynamic("书籍源无效"))) }
            return
        }

        val initialLocator: Locator? = withContext(Dispatchers.IO) {
            locatorArg?.let(locatorCodec::decode)
                ?: progressRepo.getByBookId(bookId)?.locatorJson?.let(locatorCodec::decode)
        }

        val options = OpenOptions(password = password)

        when (val r = openReaderSession(source, options, initialLocator)) {
            is ReaderResult.Err -> {
                _state.update { it.copy(isOpening = false) }
                handleOpenError(r.error, lastTriedPassword = password)
            }

            is ReaderResult.Ok -> {
                handle = r.value
                _state.update {
                    it.copy(
                        isOpening = false,
                        capabilities = r.value.document.capabilities
                    )
                }

                // touch lastOpened（专业：进入阅读就刷新）
                viewModelScope.launch(Dispatchers.IO) { runCatching { bookRepo.touchLastOpened(bookId) } }

                // title from metadata（强类型）
                loadTitle(r.value)

                // collectors
                startSessionCollectors(r.value)

                // apply layout if known
                lastLayout?.let { layout ->
                    viewModelScope.launch { r.value.controller.setLayoutConstraints(layout) }
                }

                // bind surface if already provided by UI
                boundSurface?.let { surface -> bindSurface(surface) }

                // first render
                renderCurrent()
            }
        }
    }

    private fun handleOpenError(error: ReaderError, lastTriedPassword: String?) {
        if (error is ReaderError.InvalidPassword) {
            _state.update { it.copy(passwordPrompt = PasswordPrompt(lastTried = lastTriedPassword)) }
            return
        }
        _state.update { it.copy(error = errorMapper.map(error)) }
    }

    private fun startSessionCollectors(h: ReaderSessionHandle) {
        sessionJob?.cancel()
        finalRenderJob?.cancel()
        searchJob?.cancel()

        sessionJob = viewModelScope.launch {
            supervisorScope {
                // 1) state
                launch {
                    h.controller.state.collect { st ->
                        _state.update { s -> s.copy(renderState = st, currentConfig = st.config) }
                    }
                }

                // 2) events -> progress save
                val locatorFlow = h.controller.events
                    .filterIsInstance<ReaderEvent.PageChanged>()
                    .map { it.locator }
                    .distinctUntilChanged()

                launch {
                    locatorFlow
                        .debounce(600)
                        .collectLatest { locator ->
                            val st = h.controller.state.value
                            try {
                                saveProgress(bookId = state.value.bookId, locator = locator, progression = st.progression)
                            } catch (_: CancellationException) {
                            } catch (_: Throwable) {
                                _effects.tryEmit(ReaderEffect.Snackbar(UiText.Dynamic("保存进度失败")))
                            }
                        }
                }

                // 3) events -> error
                launch {
                    h.controller.events
                        .filterIsInstance<ReaderEvent.Error>()
                        .collect { ev ->
                            _state.update { it.copy(error = ReaderUiError(UiText.Dynamic("渲染错误：${ev.throwable.message ?: "unknown"}"))) }
                        }
                }

                // 4) settings flow -> apply config
                launch {
                    observeEffectiveConfig(h.document.capabilities)
                        .distinctUntilChanged()
                        .collectLatest { cfg ->
                            applyConfig(cfg, persist = false)
                        }
                }

                // 5) annotations -> invalidate decorations
                val ann = h.annotations
                if (ann != null) {
                    launch {
                        ann.observeAll()
                            .debounce(200)
                            .collectLatest {
                                h.controller.invalidate(InvalidateReason.CONTENT_CHANGED)
                            }
                    }
                }
            }
        }
    }

    private fun loadTitle(h: ReaderSessionHandle) {
        viewModelScope.launch(Dispatchers.IO) {
            val title: String? = when (val r = h.document.metadata()) {
                is ReaderResult.Ok -> r.value.title
                is ReaderResult.Err -> null
            }
            if (!title.isNullOrBlank()) {
                _state.update { it.copy(title = title) }
            }
        }
    }

    private suspend fun closeSession() {
        finalRenderJob?.cancel(); finalRenderJob = null
        sessionJob?.cancel(); sessionJob = null
        searchJob?.cancel(); searchJob = null

        // unbind surface first
        runCatching { handle?.controller?.unbindSurface() }

        val old = handle
        handle = null
        if (old != null) {
            withContext(Dispatchers.IO) { runCatching { old.close() } }
        }
    }

    // ---------- Rendering ----------

    private suspend fun renderCurrent() {
        val h = handle ?: return
        if (h.document.capabilities.fixedLayout) {
            twoPhaseRender()
            return
        }
        when (val r = h.controller.render(policyFor(h.document.capabilities, RenderPolicy.Quality.FINAL))) {
            is ReaderResult.Ok -> _state.update { it.copy(page = r.value, error = null) }
            is ReaderResult.Err -> _state.update { it.copy(error = errorMapper.map(r.error)) }
        }
    }

    private suspend fun navigate(
        block: suspend (ReaderController, RenderPolicy) -> ReaderResult<RenderPage>
    ) {
        val h = handle ?: return
        navMutex.withLock {
            val caps = h.document.capabilities
            if (caps.fixedLayout) {
                val draft = policyFor(caps, RenderPolicy.Quality.DRAFT)
                when (val r = block(h.controller, draft)) {
                    is ReaderResult.Ok -> {
                        _state.update { it.copy(page = r.value, error = null) }
                        twoPhaseRender()
                    }
                    is ReaderResult.Err -> _state.update { it.copy(error = errorMapper.map(r.error)) }
                }
            } else {
                val p = policyFor(caps, RenderPolicy.Quality.FINAL)
                when (val r = block(h.controller, p)) {
                    is ReaderResult.Ok -> _state.update { it.copy(page = r.value, error = null) }
                    is ReaderResult.Err -> _state.update { it.copy(error = errorMapper.map(r.error)) }
                }
            }

            // 专业：翻页后预取邻页（忽略错误）
            runCatching { h.controller.prefetchNeighbors(count = 1) }
        }
    }

    private fun policyFor(caps: DocumentCapabilities, quality: RenderPolicy.Quality): RenderPolicy =
        RenderPolicy(
            quality = quality,
            allowCache = true,
            prefetchNeighbors = if (caps.fixedLayout) 2 else 1
        )

    private fun twoPhaseRender() {
        val h = handle ?: return
        val token = renderToken.incrementAndGet()
        finalRenderJob?.cancel()
        finalRenderJob = viewModelScope.launch {
            _state.update { it.copy(isRenderingFinal = true) }

            val draft = h.controller.render(policyFor(h.document.capabilities, RenderPolicy.Quality.DRAFT))
            if (renderToken.get() != token) return@launch
            if (draft is ReaderResult.Ok) _state.update { it.copy(page = draft.value, error = null) }

            val final = h.controller.render(policyFor(h.document.capabilities, RenderPolicy.Quality.FINAL))
            if (renderToken.get() != token) return@launch

            when (final) {
                is ReaderResult.Ok -> _state.update { it.copy(page = final.value, error = null, isRenderingFinal = false) }
                is ReaderResult.Err -> _state.update { it.copy(error = errorMapper.map(final.error), isRenderingFinal = false) }
            }
        }
    }

    private suspend fun applyConfig(config: RenderConfig, persist: Boolean) {
        val h = handle
        if (persist) persistConfig(config)

        if (h == null) {
            _state.update { it.copy(currentConfig = config) }
            return
        }

        when (val r = h.controller.setConfig(config)) {
            is ReaderResult.Ok -> {
                _state.update { it.copy(currentConfig = config) }
                renderCurrent()
            }
            is ReaderResult.Err -> _effects.tryEmit(ReaderEffect.Snackbar(UiText.Dynamic("应用设置失败")))
        }
    }

    private suspend fun persistConfig(config: RenderConfig) {
        // 持久化已经在 DataStore Store 内部做了，OpenReaderSession 会读取当前配置
        // 这里不直接写 store，是因为 UpdateConfig intent 应在 SettingsSheet 直接调用 store（更清晰）
        // 但为了保持行为一致，你也可以在这里注入 ReaderSettingsStore 并写入。
        // 当前实现：由 ObserveEffectiveConfig 自动从 store 推送；persist 交由 SettingsSheet 调用 setReflow/setFixed。
    }

    // ---------- TOC / Search（强类型） ----------

    private fun loadTocIfNeeded() {
        val h = handle ?: return
        if (state.value.toc.items.isNotEmpty()) return

        val outline = h.outline ?: run {
            _state.update { it.copy(toc = it.toc.copy(error = UiText.Dynamic("该文档不支持目录"))) }
            return
        }

        viewModelScope.launch {
            _state.update { it.copy(toc = it.toc.copy(isLoading = true, error = null)) }
            when (val r = outline.getOutline()) {
                is ReaderResult.Err -> _state.update { it.copy(toc = it.toc.copy(isLoading = false, error = UiText.Dynamic("加载目录失败"))) }
                is ReaderResult.Ok -> {
                    val items = flattenOutline(r.value, depth = 0)
                    _state.update { it.copy(toc = it.toc.copy(isLoading = false, items = items)) }
                }
            }
        }
    }

    private fun flattenOutline(nodes: List<OutlineNode>, depth: Int): List<TocItem> {
        val out = ArrayList<TocItem>(nodes.size)
        for (n in nodes) {
            out += TocItem(
                title = n.title,
                locatorEncoded = locatorCodec.encode(n.locator),
                depth = depth
            )
            if (n.children.isNotEmpty()) {
                out += flattenOutline(n.children, depth + 1)
            }
        }
        return out
    }

    fun startSearch(query: String) {
        val h = handle ?: return
        val provider = h.search ?: run {
            _state.update { it.copy(search = it.search.copy(error = UiText.Dynamic("该文档不支持搜索"))) }
            return
        }

        searchJob?.cancel()
        searchJob = viewModelScope.launch {
            _state.update {
                it.copy(search = it.search.copy(query = query, isSearching = true, results = emptyList(), error = null))
            }

            val options = SearchOptions(maxHits = 300)

            provider.search(query, options)
                .asReaderResult()
                .onEach { rr ->
                    when (rr) {
                        is ReaderResult.Err -> {
                            _state.update { it.copy(search = it.search.copy(isSearching = false, error = UiText.Dynamic("搜索失败"))) }
                        }
                        is ReaderResult.Ok -> {
                            val hit: SearchHit = rr.value
                            // 这里假设 LocatorRange 有 start 字段；若你是 startLocator 改这里
                            val locator = hit.range.start
                            val item = SearchResultItem(
                                title = hit.sectionTitle,
                                excerpt = hit.excerpt,
                                locatorEncoded = locatorCodec.encode(locator)
                            )
                            _state.update { st ->
                                st.copy(search = st.search.copy(results = st.search.results + item))
                            }
                        }
                    }
                }
                .launchIn(this)
        }
    }

    fun decodeLocatorOrNull(encoded: String): Locator? = locatorCodec.decode(encoded)

    // ---------- helpers ----------

    override fun onCleared() {
        viewModelScope.launch { closeSession() }
        super.onCleared()
    }

    private inline fun <T> MutableStateFlow<T>.update(block: (T) -> T) {
        value = block(value)
    }
}

import androidx.compose.foundation.layout.BoxWithConstraints
import androidx.compose.foundation.layout.fillMaxSize
import androidx.compose.material3.SnackbarHostState

import com.ireader.feature.reader.presentation.ReaderViewModel
import com.ireader.feature.reader.presentation.asString
import com.ireader.feature.reader.ui.surface.ReaderSurfaceLayout
import kotlinx.coroutines.flow.collectLatest

@Composable
fun ReaderScreen(
    bookId: Long,
    locatorArg: String?,
    onBack: () -> Unit,
    onOpenAnnotations: (bookId: Long) -> Unit,
    vm: ReaderViewModel = hiltViewModel()
) {
    val state by vm.state.collectAsStateWithLifecycle()
    val snackbarHostState = remember { SnackbarHostState() }

    LaunchedEffect(bookId, locatorArg) {
        vm.dispatch(ReaderIntent.Start(bookId = bookId, locatorArg = locatorArg))
    }

    // effects
    LaunchedEffect(Unit) {
        vm.effects.collectLatest { eff ->
            when (eff) {
                is ReaderEffect.Back -> onBack()
                is ReaderEffect.Snackbar -> snackbarHostState.showSnackbar(eff.message.asString())
                is ReaderEffect.OpenAnnotations -> onOpenAnnotations(eff.bookId)
            }
        }
    }

    // back behavior
    BackHandler(enabled = state.sheet != ReaderSheet.None) { vm.dispatch(ReaderIntent.CloseSheet) }
    BackHandler(enabled = state.sheet == ReaderSheet.None && !state.chromeVisible) { vm.dispatch(ReaderIntent.ToggleChrome) }

    BoxWithConstraints(Modifier.fillMaxSize()) {
        val density = LocalDensity.current
        val wPx = with(density) { maxWidth.toPx().toInt() }
        val hPx = with(density) { maxHeight.toPx().toInt() }

        LaunchedEffect(wPx, hPx, density.density, density.fontScale) {
            vm.dispatch(
                ReaderIntent.LayoutChanged(
                    com.ireader.reader.api.render.LayoutConstraints(
                        viewportWidthPx = wPx,
                        viewportHeightPx = hPx,
                        density = density.density,
                        fontScale = density.fontScale
                    )
                )
            )
        }

        // ✅ Surface + bind/unbind
        var surface: ReaderSurfaceLayout? by remember { mutableStateOf(null) }

        AndroidView(
            factory = { ctx -> ReaderSurfaceLayout(ctx).also { surface = it } },
            modifier = Modifier.fillMaxSize(),
            update = { view -> surface = view }
        )

        DisposableEffect(surface) {
            val s = surface
            if (s != null) vm.bindSurface(s)
            onDispose { vm.unbindSurface() }
        }

        ReaderScaffold(
            state = state,
            snackbarHostState = snackbarHostState,
            onBack = onBack,
            onOpenAnnotations = { onOpenAnnotations(state.bookId) },
            onIntent = vm::dispatch,
            onStartSearch = vm::startSearch,
            decodeLocator = vm::decodeLocatorOrNull
        )
    }
}
