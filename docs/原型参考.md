// --- 导航控制 ---
@Composable
fun AppNavigation() {
var currentScreen by remember { mutableStateOf("bookshelf") }

    if (currentScreen == "bookshelf") {
        BookshelfScreen(onBookClick = { currentScreen = "reader" })
    } else {
        ReadingScreen(onBackToBookshelf = { currentScreen = "bookshelf" })
    }
}

@OptIn(ExperimentalMaterial3Api::class)
@Composable
fun ReadingScreen(onBackToBookshelf: () -> Unit) {
var isUIVisible by remember { mutableStateOf(false) }
var activePanel by remember { mutableStateOf(PanelState.NONE) }
var isNightMode by remember { mutableStateOf(false) }

    var sysBrightness by remember { mutableStateOf(true) }
    var eyeProtection by remember { mutableStateOf(false) }

    val bgColor = if (isNightMode) ColorReaderBgNight else ColorReaderBgDay
    val textColor = if (isNightMode) ColorTextNight else ColorTextDay
    val panelBgColor = if (isNightMode) ColorPanelBgNight else ColorPanelBgDay
    val panelTextColor = if (isNightMode) Color.LightGray else Color.Black

    // 处理安卓物理返回键
    BackHandler(enabled = isUIVisible || activePanel != PanelState.NONE) {
        if (activePanel != PanelState.NONE && activePanel != PanelState.FULL_SETTINGS) {
            activePanel = PanelState.NONE
        } else if (activePanel == PanelState.FULL_SETTINGS) {
            activePanel = PanelState.SETTINGS
        } else {
            isUIVisible = false
        }
    }

    Box(modifier = Modifier.fillMaxSize().background(bgColor)) {

        // 模拟分页内容 (利用 Pager 完美适配任何屏幕宽度)
        val pages = listOf(
            "来就能有一把真剑，就忍不住兴奋起来，丢了树枝，开始边跑边喊。\n\n刘羡阳想着阮师傅私下传授的那几个拳架子，就开始练习起来，倒也有模有样，虎虎生风。\n\n刘羡阳与廊桥越来越近。廊桥北端的台阶上，坐着四个人：姿态婀娜的丰腴美妇，怀里抱着一个身穿大红袍子的男孩，男孩高高扬起下巴，像是一位刚刚获得大捷的将军；",
            "台阶那一头，坐着个满头霜雪的高大老人，老人正在小声安慰一个气鼓鼓的小女孩。小女孩粉雕玉琢，宛如世上最精巧的瓷娃娃，她的稚嫩肌肤在阳光照耀下晶莹剔透，以至于能够清晰看到皮肤下的一条条青筋脉络。\n\n两个孩子刚刚吵完架，小女孩泫然欲泣，小男孩愈发得意。",
            "陈平安坐在小板凳上，望着那个名叫宋集薪的同龄人，他有着一双极为好看的眼眸，哪怕是生气的时候，也让人觉得他在笑。陈平安低头看了看自己的草鞋，又抬头看了看宋集薪脚上那双纤尘不染的绸缎靴子，突然觉得很有意思。\n\n“看什么看？泥腿子！”宋集薪恶狠狠地瞪了一眼陈平安。陈平安也不恼，只是咧嘴一笑，露出雪白的牙齿。这让宋集薪更生气了，仿佛一拳打在了棉花上。",
            "老人看着这两个孩子，无奈地摇了摇头。这世道，最不缺的就是心高气傲的少年郎，可真能走得长远的，往往是那些懂得低头看路的人。"
        )
        val pagerState = rememberPagerState(pageCount = { pages.size })
        val coroutineScope = rememberCoroutineScope()

        HorizontalPager(
            state = pagerState,
            modifier = Modifier.fillMaxSize().pointerInput(Unit) {
                detectTapGestures { offset ->
                    val width = size.width
                    if (activePanel != PanelState.NONE) {
                        activePanel = PanelState.NONE
                        return@detectTapGestures
                    }
                    if (isUIVisible) {
                        isUIVisible = false
                        return@detectTapGestures
                    }

                    if (offset.x < width * 0.3f) {
                        // 左侧点击，上一页
                        coroutineScope.launch { if (pagerState.currentPage > 0) pagerState.animateScrollToPage(pagerState.currentPage - 1) }
                    } else if (offset.x > width * 0.7f) {
                        // 右侧点击，下一页
                        coroutineScope.launch { if (pagerState.currentPage < pages.size - 1) pagerState.animateScrollToPage(pagerState.currentPage + 1) }
                    } else {
                        // 中间点击，唤出UI
                        isUIVisible = true
                    }
                }
            }
        ) { page ->
            Column(modifier = Modifier.fillMaxSize().padding(horizontal = 24.dp).statusBarsPadding().padding(top = 20.dp, bottom = 40.dp)) {
                Text("第二章 稗草", fontSize = 11.sp, color = Color.Gray, letterSpacing = 2.sp, modifier = Modifier.padding(bottom = 32.dp))
                Text(
                    text = "\u3000\u3000${pages[page]}", // 首行缩进
                    fontSize = 20.sp,
                    color = textColor,
                    lineHeight = 38.sp,
                    letterSpacing = 1.2.sp
                )
            }
        }

        // 底部阅读进度条
        Row(
            modifier = Modifier.align(Alignment.BottomCenter).fillMaxWidth().navigationBarsPadding()
                .padding(start = 24.dp, end = 24.dp, bottom = 12.dp),
            horizontalArrangement = Arrangement.SpaceBetween,
            verticalAlignment = Alignment.CenterVertically
        ) {
            Text("${String.format("%.2f", (pagerState.currentPage + 1).toFloat() / pages.size * 100)}%", fontSize = 10.sp, color = Color.Gray)
            Row(verticalAlignment = Alignment.CenterVertically) {
                Text("21:12", fontSize = 10.sp, color = Color.Gray, modifier = Modifier.padding(end = 4.dp))
                // 简易电池图标
                Box(modifier = Modifier.width(20.dp).height(10.dp).border(1.dp, Color.Gray, RoundedCornerShape(2.dp)).padding(1.dp)) {
                    Box(modifier = Modifier.fillMaxHeight().fillMaxWidth(0.57f).background(Color.Gray))
                }
            }
        }

        // --- 顶部和底部操作栏 ---
        AnimatedVisibility(visible = isUIVisible, enter = fadeIn(), exit = fadeOut()) {
            Box(modifier = Modifier.fillMaxSize()) {
                // 顶部动作栏
                Row(
                    modifier = Modifier.align(Alignment.TopCenter).fillMaxWidth().background(panelBgColor.copy(alpha = 0.98f)).statusBarsPadding().padding(horizontal = 12.dp, vertical = 12.dp),
                    horizontalArrangement = Arrangement.SpaceBetween,
                    verticalAlignment = Alignment.CenterVertically
                ) {
                    Icon(Icons.Default.ArrowBack, contentDescription = "Back", tint = panelTextColor, modifier = Modifier.size(28.dp).clickable { onBackToBookshelf() })
                    Row(horizontalArrangement = Arrangement.spacedBy(16.dp)) {
                        Icon(Icons.Outlined.MoreVert, contentDescription = "More", tint = panelTextColor, modifier = Modifier.size(24.dp).clickable { activePanel = PanelState.READER_MORE })
                    }
                }

                // 底部导航栏
                Row(
                    modifier = Modifier.align(Alignment.BottomCenter).fillMaxWidth().background(panelBgColor.copy(alpha = 0.98f)).navigationBarsPadding().padding(vertical = 12.dp, horizontal = 16.dp),
                    horizontalArrangement = Arrangement.SpaceAround
                ) {
                    BottomNavItemCustom(label = "目录", isNightMode = isNightMode, isSelected = activePanel == PanelState.MENU, onClick = { activePanel = PanelState.MENU }) { MenuIcon() }
                    BottomNavItemCustom(label = "亮度", isNightMode = isNightMode, isSelected = activePanel == PanelState.BRIGHTNESS, onClick = { activePanel = PanelState.BRIGHTNESS }) { BrightnessIcon() }
                    BottomNavItemCustom(label = if(isNightMode) "日间" else "夜间", isNightMode = isNightMode, isSelected = false, onClick = { isNightMode = !isNightMode }) { MoonSunIcon(isNightMode) }
                    BottomNavItemCustom(label = "设置", isNightMode = isNightMode, isSelected = activePanel == PanelState.SETTINGS, onClick = { activePanel = PanelState.SETTINGS }) { SettingsIcon() }
                }
            }
        }

        // --- BottomSheet 面板 ---
        if (activePanel != PanelState.NONE && activePanel != PanelState.FULL_SETTINGS) {
            ModalBottomSheet(
                onDismissRequest = { activePanel = PanelState.NONE },
                containerColor = panelBgColor,
                dragHandle = { BottomSheetDefaults.DragHandle(color = Color.LightGray) }
            ) {
                when (activePanel) {
                    PanelState.MENU -> MenuPanel(panelTextColor)
                    PanelState.BRIGHTNESS -> BrightnessPanel(sysBrightness, { sysBrightness = it }, eyeProtection, { eyeProtection = it }, panelTextColor)
                    PanelState.SETTINGS -> SettingsPanel({ activePanel = it }, { activePanel = PanelState.FULL_SETTINGS }, panelTextColor, isNightMode)
                    PanelState.SUB_FONT -> SubFontPanel({ activePanel = PanelState.SETTINGS }, panelTextColor)
                    PanelState.SUB_SPACING -> SubSpacingPanel({ activePanel = PanelState.SETTINGS }, panelTextColor)
                    PanelState.SUB_PAGETURN -> SubPageTurnPanel({ activePanel = PanelState.SETTINGS }, panelTextColor)
                    PanelState.SUB_MOREBG -> SubMoreBgPanel({ activePanel = PanelState.SETTINGS }, panelTextColor)
                    PanelState.READER_MORE -> ReaderMorePanel({ activePanel = PanelState.NONE }, panelTextColor)
                    else -> {}
                }
                Spacer(modifier = Modifier.navigationBarsPadding())
            }
        }

        // --- 全屏设置 ---
        AnimatedVisibility(visible = activePanel == PanelState.FULL_SETTINGS, enter = slideInVertically(initialOffsetY = { it }) + fadeIn(), exit = slideOutVertically(targetOffsetY = { it }) + fadeOut()) {
            FullSettingsScreen(onBack = { activePanel = PanelState.SETTINGS }, panelBgColor, panelTextColor)
        }
    }
}

@Composable
fun MoonSunIcon(isNightMode: Boolean) {
Icon(if (isNightMode) Icons.Outlined.WbSunny else Icons.Outlined.NightlightRound, contentDescription = null, modifier = Modifier.size(24.dp))
}

@Composable
fun SettingsIcon() {
val iconColor = LocalContentColor.current
Canvas(modifier = Modifier.size(24.dp)) {
val path = Path().apply {
val cx = size.width/2; val cy = size.height/2; val r = 9.dp.toPx()
for (i in 0..5) {
val angle = (i * 60 - 30) * Math.PI / 180
val x = cx + r * kotlin.math.cos(angle).toFloat()
val y = cy + r * kotlin.math.sin(angle).toFloat()
if (i == 0) moveTo(x, y) else lineTo(x, y)
}
close()
}
drawPath(path = path, color = iconColor, style = Stroke(width = 1.5.dp.toPx(), join = StrokeJoin.Round))
drawCircle(color = iconColor, radius = 3.5.dp.toPx(), center = androidx.compose.ui.geometry.Offset(size.width/2, size.height/2), style = Stroke(width = 1.5.dp.toPx()))
}
}

@Composable

Column(horizontalAlignment = Alignment.CenterHorizontally, modifier = Modifier.clickable(onClick = onClick).padding(8.dp)) {
CompositionLocalProvider(LocalContentColor provides tint) { iconDrawing() }
Spacer(modifier = Modifier.height(4.dp))
Text(text = label, fontSize = 10.sp, color = tint)
}
}

}
}
